package main

import (
	"fmt"
	"math/rand"
	"time"
)

func main() {
	rand.Seed(time.Now().UnixNano())
	num := rand.Intn(100)

	fmt.Println("Guess a number between 0 and 99!")
	for i := 1; i <= 10; i++ {
		var guess int
		fmt.Printf("Guess %v: ", i)
		fmt.Scan(&guess)

		if guess == num {
			fmt.Println("Congratulations, you guessed it!")
			return
		} else if guess < num {
			fmt.Println("Too low. Try again.")
		} else {
			fmt.Println("Too high. Try again.")
		}
	}

	fmt.Println("Sorry, you ran out of tries. The number was", num)
}

注：此代码实现了一个简单的猜数字游戏，随机生成一个0到99之间的数字，然后用户最多可以猜10次，如果在此之内猜中，则胜利。
