1. 写一个程序，输入一个字符串，然后将字符串中的大写字母转成小写字母，小写字母转成大写字母，其他字符保持不变。
package main

import (
    "fmt"
    "strings"
)

func main() {
    var input string
    fmt.Print("Enter a string: ")
    fmt.Scanln(&input)
    output := ""
    for _, ch := range input {
        if ch >= 'A' && ch <= 'Z' {
            output += strings.ToLower(string(ch))
        } else if ch >= 'a' && ch <= 'z' {
            output += strings.ToUpper(string(ch))
        } else {
            output += string(ch)
        }
    }
    fmt.Println(output)
}
  1. 写一个程序，将给定的整数数组反转。
package main

import "fmt"

func main() {
    arr := []int{1, 2, 3, 4, 5}
    reverse(arr)
    fmt.Println(arr)
}

func reverse(arr []int) {
    for i := 0; i < len(arr)/2; i++ {
        j := len(arr) - 1 - i
        arr[i], arr[j] = arr[j], arr[i]
    }
}
  1. 写一个程序，输入一个数字n，输出1到n的所有素数。
package main

import "fmt"

func main() {
    var n int
    fmt.Print("Enter a number: ")
    fmt.Scanln(&n)
    primes := []int{}
    for i := 2; i <= n; i++ {
        isPrime := true
        for j := 2; j*j <= i; j++ {
            if i%j == 0 {
                isPrime = false
                break
            }
        }
        if isPrime {
            primes = append(primes, i)
        }
    }
    fmt.Println(primes)
}
  1. 写一个程序，输入一个数字n，输出斐波那契数列的前n项。
package main

import "fmt"

func main() {
    var n int
    fmt.Print("Enter a number: ")
    fmt.Scanln(&n)
    fib := []int{0, 1}
    for i := 2; i < n; i++ {
        fib = append(fib, fib[i-1]+fib[i-2])
    }
    fmt.Println(fib)
}
  1. 写一个程序，将一个字符串按照单词反转顺序输出。
package main

import (
    "fmt"
    "strings"
)

func main() {
    input := "hello world"
    words := strings.Split(input, " ")
    output := ""
    for i := len(words) - 1; i >= 0; i-- {
        output += words[i] + " "
    }
    fmt.Println(strings.TrimSpace(output))
}
