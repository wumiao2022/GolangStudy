这是一个Go语言实现的冒泡排序算法：

package main

import "fmt"

func bubbleSort(arr []int) {
    n := len(arr)
    for i := 0; i < n-1; i++ {
        for j := 0; j < n-i-1; j++ {
            if arr[j] > arr[j+1] {
                arr[j], arr[j+1] = arr[j+1], arr[j]
            }
        }
    }
}

func main() {
    arr := []int{10, 7, 8, 9, 1, 5}
    fmt.Println("Original array is:", arr)
    bubbleSort(arr)
    fmt.Println("Sorted array is:", arr)
}
