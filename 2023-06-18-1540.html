以下是一个简单的例子：

package main

import "fmt"

func main() {
    nums := []int{2, 3, 4}
    sum := 0
    for _, num := range nums {
        sum += num
    }
    fmt.Println("Sum:", sum)

    for i, num := range nums {
        if num == 3 {
            fmt.Println("Index of 3:", i)
        }
    }

    m := make(map[string]int)
    m["a"] = 1
    m["b"] = 2
    fmt.Println("Map:", m)
}
