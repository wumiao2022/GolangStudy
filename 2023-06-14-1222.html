1. 反转字符串
package main

import (
	"fmt"
)

func main() {
	str := "Hello World!"
	reverseStr := []rune(str)

	for i, j := 0, len(reverseStr)-1; i < j; i, j = i+1, j-1 {
		reverseStr[i], reverseStr[j] = reverseStr[j], reverseStr[i]
	}

	fmt.Println(string(reverseStr))
}
  1. 阶乘
package main

import (
	"fmt"
)

func main() {
	fmt.Println(factorial(5))
}

func factorial(n int) int {
	if n <= 0 {
		return 1
	}
	return n * factorial(n-1)
}
  1. 斐波那契数列
package main

import (
	"fmt"
)

func main() {
	fmt.Println(fibonacci(10))
}

func fibonacci(n int) []int {
	f := make([]int, n)

	if n == 0 {
		return f
	}

	f[0] = 0
	if n == 1 {
		return f
	}

	f[1] = 1
	if n == 2 {
		return f
	}

	for i := 2; i < n; i++ {
		f[i] = f[i-1] + f[i-2]
	}

	return f
}

  1. 质数
package main

import (
	"fmt"
)

func main() {
	primes := sieveOfEratosthenes(50)
	fmt.Println(primes)
}

func sieveOfEratosthenes(n int) []int {
	primes := make([]bool, n+1)

	for i := 2; i <= n; i++ {
		primes[i] = true
	}

	for i := 2; i*i <= n; i++ {
		if primes[i] {
			for j := i * i; j <= n; j += i {
				primes[j] = false
			}
		}
	}

	var result []int
	for i := 2; i <= n; i++ {
		if primes[i] {
			result = append(result, i)
		}
	}

	return result
}
