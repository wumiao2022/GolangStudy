  1. 判断某个数是不是素数
package main

import (
    "fmt"
    "math"
)

func isPrime(num int) bool {
    if num <= 1 {
        return false
    }
    for i := 2; i <= int(math.Sqrt(float64(num))); i++ {
        if num%i == 0 {
            return false
        }
    }
    return true
}

func main() {
    num := 7
    if isPrime(num) {
        fmt.Println(num, "is a prime number")
    } else {
        fmt.Println(num, "is not a prime number")
    }
}
  1. 将一个字符串反转输出
package main

import "fmt"

func reverse(str string) string {
    result := ""
    for i := len(str)-1; i >= 0; i-- {
        result += string(str[i])
    }
    return result
}

func main() {
    str := "Hello, world!"
    fmt.Println(reverse(str))
}
  1. 实现一个冒泡排序算法
package main

import "fmt"

func bubbleSort(arr []int) {
    for i := 0; i < len(arr)-1; i++ {
        for j := 0; j < len(arr)-i-1; j++ {
            if arr[j] > arr[j+1] {
                arr[j], arr[j+1] = arr[j+1], arr[j]
            }
        }
    }
}

func main() {
    arr := []int{2, 1, 5, 4, 3}
    bubbleSort(arr)
    fmt.Println(arr)
}
  1. 实现一个查找字符串中出现次数最多的字符和次数的函数
package main

import "fmt"

func findMostFrequentChar(str string) (string, int) {
    charCounts := make(map[rune]int)
    for _, ch := range str {
        charCounts[ch]++
    }
    maxChar := ""
    maxCount := 0
    for ch, count := range charCounts {
        if count > maxCount {
            maxChar = string(ch)
            maxCount = count
        }
    }
    return maxChar, maxCount
}

func main() {
    str := "hello, world!"
    char, count := findMostFrequentChar(str)
    fmt.Println("The most frequent char is", char, "with", count, "occurrences")
}
