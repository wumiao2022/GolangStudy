package main

import (
	"fmt"
	"math/rand"
	"time"
)

func main() {
	rand.Seed(time.Now().UnixNano())

	// Generate a random number between 1 and 100
	target := rand.Intn(100) + 1

	fmt.Println("Welcome to the Guessing Game!")
	fmt.Println("I have chosen a number between 1 and 100.")
	fmt.Println("Can you guess it?")

	guessesTaken := 0
	for {
		fmt.Print("Take a guess: ")
		var guess int
		_, err := fmt.Scanf("%d", &guess)
		if err != nil {
			fmt.Println("Invalid input. Please enter a valid number.")
			continue
		}

		guessesTaken++

		if guess < target {
			fmt.Println("Your guess is too low.")
		} else if guess > target {
			fmt.Println("Your guess is too high.")
		} else {
			fmt.Printf("Congratulations! You guessed the number in %d tries.\n", guessesTaken)
			break
		}
	}
}

This program is a simple guessing game where the user needs to guess a randomly chosen number between 1 and 100. It keeps prompting the user for guesses until the correct number is guessed. After each guess, it provides feedback on whether the guess was too low or too high. Once the user guesses the correct number, it displays the number of tries taken.
