package main

import "fmt"

func main() {
	// 第一题：创建一个函数，接收两个整数参数，并返回它们的和
	func1(3, 5)

	// 第二题：创建一个函数，接收一个字符串参数，并打印出字符串的长度
	func2("hello")

	// 第三题：创建一个函数，接收一个整数参数n，打印从1到n的所有偶数
	func3(10)

	// 第四题：创建一个函数，接收一个整数切片参数，返回切片中的最大数和最小数
	slice := []int{1, 5, 3, 8, 2, 6}
	max, min := func4(slice)
	fmt.Println("Max:", max, "Min:", min)

	// 第五题：创建一个函数，接收一个整数切片参数，并将切片中的元素按照从小到大排序
	func5(slice)
	fmt.Println("Sorted slice:", slice)
}

func func1(a, b int) {
	sum := a + b
	fmt.Println("Sum:", sum)
}

func func2(str string) {
	length := len(str)
	fmt.Println("Length:", length)
}

func func3(n int) {
	for i := 1; i <= n; i++ {
		if i%2 == 0 {
			fmt.Println(i)
		}
	}
}

func func4(slice []int) (int, int) {
	max := slice[0]
	min := slice[0]
	for _, num := range slice {
		if num > max {
			max = num
		}
		if num < min {
			min = num
		}
	}
	return max, min
}

func func5(slice []int) {
	for i := 0; i < len(slice)-1; i++ {
		for j := 0; j < len(slice)-i-1; j++ {
			if slice[j] > slice[j+1] {
				slice[j], slice[j+1] = slice[j+1], slice[j]
			}
		}
	}
}
