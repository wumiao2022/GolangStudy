// 练习2：求和函数
package main

import "fmt"

func Sum(a, b int) int {
	return a + b
}

func main() {
	fmt.Println(Sum(2, 3))
}

// 练习3：判断奇偶数
package main

import "fmt"

func IsEven(n int) bool {
	if n%2 == 0 {
		return true
	}
	return false
}

func main() {
	fmt.Println(IsEven(4))
	fmt.Println(IsEven(7))
}

// 练习4：数组遍历
package main

import "fmt"

func main() {
	nums := []int{1, 2, 3, 4, 5}
	for _, num := range nums {
		fmt.Println(num)
	}
}

// 练习5：切片操作
package main

import "fmt"

func main() {
	nums := []int{1, 2, 3, 4, 5}
	slicedNums := nums[1:4]
	fmt.Println(slicedNums)
}

// 练习6：输入输出
package main

import "fmt"

func main() {
	var name string
	fmt.Print("请输入你的名字：")
	fmt.Scan(&name)
	fmt.Printf("你好，%s！", name)
}

// 练习7：循环控制
package main

import "fmt"

func main() {
	for i := 1; i <= 10; i++ {
		if i%2 == 0 {
			fmt.Println(i)
		}
	}
}
