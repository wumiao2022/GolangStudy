1. 判断字符串是否包含某个子串
package main

import (
	"fmt"
	"strings"
)

func main() {
	str1 := "hello world"
	str2 := "world"
	if strings.Contains(str1, str2) {
		fmt.Println("str1 contains str2")
	} else {
		fmt.Println("str1 does not contain str2")
	}
}
  1. 数组中查找某个元素
package main

import "fmt"

func main() {
	arr := [5]int{2, 4, 6, 8, 10}
	key := 8
	found := false
	for i := 0; i < len(arr); i++ {
		if arr[i] == key {
			found = true
			break
		}
	}
	if found {
		fmt.Printf("key %d found in array\n", key)
	} else {
		fmt.Printf("key %d not found in array\n", key)
	}
}
  1. 判断一个数是否是质数
package main

import "fmt"

func isPrime(n int) bool {
	if n <= 1 {
		return false
	}
	for i := 2; i*i <= n; i++ {
		if n%i == 0 {
			return false
		}
	}
	return true
}

func main() {
	n := 17
	if isPrime(n) {
		fmt.Printf("%d is a prime number\n", n)
	} else {
		fmt.Printf("%d is not a prime number\n", n)
	}
}
  1. 反转字符串
package main

import "fmt"

func reverse(str string) string {
	runes := []rune(str)
	for i, j := 0, len(runes)-1; i < len(runes)/2; i, j = i+1, j-1 {
		runes[i], runes[j] = runes[j], runes[i]
	}
	return string(runes)
}

func main() {
	str := "hello world"
	fmt.Println(reverse(str))
}
  1. 计算两个数的最大公约数
package main

import "fmt"

func gcd(a, b int) int {
	if b == 0 {
		return a
	} else {
		return gcd(b, a%b)
	}
}

func main() {
	a, b := 12, 18
	fmt.Printf("gcd(%d, %d) = %d\n", a, b, gcd(a, b))
}
