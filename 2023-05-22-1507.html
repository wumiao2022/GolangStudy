  2. 计算三角形面积

    package main
import "fmt"
func main() {
var base, height float32
fmt.Println("Enter base and height of the triangle:")
fmt.Scan(&base, &height)
var area = 0.5 * base * height
fmt.Printf("Area of the triangle is %.2f", area)
}

  3. 判断一个数是奇数还是偶数

    package main
import "fmt"
func main() {
var n int
fmt.Println("Enter a number:")
fmt.Scan(&n)
if n%2 == 0 {
    fmt.Printf("%d is even", n)
} else {
    fmt.Printf("%d is odd", n)
}
}

  4. 把一个字符串反转过来

    package main
import "fmt"
func main() {
var str string
fmt.Println("Enter a string:")
fmt.Scan(&str)
var revStr string
for i := len(str) - 1; i >= 0; i-- {
    revStr += string(str[i])
}
fmt.Println(revStr)
}

  5. 数组排序

    package main
import "fmt"
func main() {
var arr = []int{4, 7, 2, 3, 9, 1, 8, 5, 6}
sort(arr)
fmt.Println(arr)
}
func sort(arr []int) {
for i := 0; i < len(arr)-1; i++ {
    for j := i + 1; j < len(arr); j++ {
        if arr[j] < arr[i] {
            arr[i], arr[j] = arr[j], arr[i]
        }
    }
}
}
