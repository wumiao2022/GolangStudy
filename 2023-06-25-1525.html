1. 判断字符串是否包含某个子串
package main

import (
	"fmt"
	"strings"
)

func main() {
	str := "hello world"
	substr := "o w"

	if strings.Contains(str, substr) {
		fmt.Println("包含子串")
	} else {
		fmt.Println("不包含子串")
	}
}
  1. 计算两个数之和
package main

import "fmt"

func add(x int, y int) int {
	return x + y
}

func main() {
	fmt.Println(add(1, 2))
}
  1. 数组遍历
package main

import "fmt"

func main() {
	arr := [4]int{1, 2, 3, 4}

	for i := 0; i < len(arr); i++ {
		fmt.Println(arr[i])
	}
}
  1. 切片操作
package main

import "fmt"

func main() {
	slice := []int{1, 2, 3, 4, 5}

	// 对切片进行截取，只取第2、3个元素
	slice2 := slice[1:3]
	fmt.Println(slice2)

	// 对切片进行追加元素
	slice = append(slice, 6)
	fmt.Println(slice)
}
  1. 结构体定义和使用
package main

import "fmt"

type Person struct {
	name string
