以下是一个简单的Go程序，它打印出 “Hello, World!“：

package main

import "fmt"

func main() {
    fmt.Println("Hello, World!")
}

以下是一个Golang多线程并发实现：

package main

import (
    "fmt"
    "sync"
)

func main() {
    var wg sync.WaitGroup
    for i := 0; i < 5; i++ {
        wg.Add(1)
        go func(num int) {
            fmt.Println("This is goroutine", num)
            wg.Done()
        }(i)
    }
    wg.Wait()
}
```
