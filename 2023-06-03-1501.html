1. 判断一个数是否为质数

    func isPrime(n int) bool {
if n < 2 {
    return false
}
for i := 2; i <= int(math.Sqrt(float64(n))); i++ {
    if n%i == 0 {
        return false
    }
}
return true
}

  2. 求斐波那契数列第n项

    func fibonacci(n int) int {
if n <= 1 {
    return n
}
return fibonacci(n-1) + fibonacci(n-2)
}

  3. 实现一个简单的计算器

    func calculate(op string, a, b float64) float64 {
switch op {
case "+":
    return a + b
case "-":
    return a - b
case "*":
    return a * b
case "/":
    if b == 0 {
        return math.Inf(1)
    }
    return a / b
default:
    return math.NaN()
}
}

  4. 使用二分法查找一个数在有序数组中的位置

    func binarySearch(arr []int, target int) int {
left, right := 0, len(arr)-1
for left <= right {
    mid := (left + right) / 2
    if arr[mid] == target {
        return mid
    }
    if arr[mid] < target {
        left = mid + 1
    } else {
        right = mid - 1
    }
}
return -1
}

  5. 使用冒泡排序对数组进行升序排序

    func bubbleSort(arr []int) {
for i := 0; i < len(arr)-1; i++ {
    for j := 0; j < len(arr)-1-i; j++ {
        if arr[j] > arr[j+1] {
            arr[j], arr[j+1] = arr[j+1], arr[j]
        }
    }
}
