1. 打印出1到1000内所有的奇数：
package main

import "fmt"

func main() {
    for i := 1; i <= 1000; i++ {
        if i%2 != 0 {
            fmt.Println(i)
        }
    }
}
  1. 打印出斐波那契数列前20个数字：
package main

import "fmt"

func main() {
    var num1, num2 int = 0, 1
    for i := 0; i < 20; i++ {
        fmt.Println(num2)
        num1, num2 = num2, num1+num2
    }
}
  1. 将一个句子中的每个单词首字母大写，其他字母小写：
package main

import (
    "bufio"
    "fmt"
    "os"
    "strings"
)

func main() {
    fmt.Print("Enter a sentence: ")
    scanner := bufio.NewScanner(os.Stdin)
    scanner.Scan()
    sentence := scanner.Text()

    words := strings.Fields(sentence)
    for _, word := range words {
        fmt.Printf("%s ", strings.Title(strings.ToLower(word)))
    }
    fmt.Println()
}
  1. 计算两个数的最大公约数：
package main

import "fmt"

func gcd(a, b int) int {
    if b == 0 {
        return a
    }
    return gcd(b, a%b)
}

func main() {
    fmt.Println(gcd(12, 30))
}
  1. 判断一个字符串是否是回文字符串：
package main

import (
    "fmt"
    "strings"
)

func isPalindrome(s string) bool {
    s = strings.ToLower(s)
    s = strings.ReplaceAll(s, " ", "")

    for i := 0; i < len(s)/2; i++ {
        if s[i] != s[len(s)-i-1] {
            return false
        }
    }
    return true
}

func main() {
    fmt.Println(isPalindrome("racecar"))
}
