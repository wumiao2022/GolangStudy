以下是一些Golang的练习实例代码：

  1. 输出Hello, World!：

    package main
import "fmt"
func main() {
   fmt.Println("Hello, World!")
}

  2. 输入两个整数并计算它们的和：

    package main
import "fmt"
func main() {
   var num1, num2 int
   fmt.Println("请输入两个整数：")
   fmt.Scan(&num1, &num2)
   fmt.Printf("它们的和是：%d", num1+num2)
}

  3. 判断一个数是否为质数：

    package main
import "fmt"
func main() {
   var num int
   fmt.Println("请输入一个整数：")
   fmt.Scan(&num)
   prime := true // 假设num是质数
   for i := 2; i <= num/2; i++ {
       if num%i == 0 {
           prime = false // 不是质数
           break
       }
   }
   if prime {
       fmt.Printf("%d是质数。", num)
   } else {
       fmt.Printf("%d不是质数。", num)
   }
}

  4. 计算斐波那契数列的前n项：

    package main
import "fmt"
func main() {
   var n int
   fmt.Println("请输入要计算的斐波那契数列的项数：")
   fmt.Scan(&n)
   a, b := 0, 1
   for i := 0; i < n; i++ {
       fmt.Printf("%d ", a)
       a, b = b, a+b
   }
}

  5. 实现冒泡排序：

    package main
import "fmt"
func main() {
   nums := []int{5, 2, 9, 1, 5, 6}
   for i := 0; i < len(nums)-1; i++ {
       for j := 0; j < len(nums)-i-1; j++ {
           if nums[j] > nums[j+1] {
               nums[j], nums[j+1] = nums[j+1], nums[j]
           }
       }
   }
   fmt.Println(nums)
}
