package main

import "fmt"

func main() {
	// 1. 输出Hello, World!
	fmt.Println("Hello, World!")

	// 2. 计算2个整数的和
	a := 10
	b := 5
	sum := a + b
	fmt.Println("Sum:", sum)

	// 3. 判断一个数是奇数还是偶数
	num := 7
	if num%2 == 0 {
		fmt.Println(num, "is even")
	} else {
		fmt.Println(num, "is odd")
	}

	// 4. 循环打印1到10的数字
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}

	// 5. 定义一个函数，接收一个字符串并将其反转
	str := "Hello, Go!"
	reversedStr := reverseString(str)
	fmt.Println(reversedStr)
}

func reverseString(str string) string {
	runes := []rune(str)
	for i, j := 0, len(str)-1; i < j; i, j = i+1, j-1 {
		runes[i], runes[j] = runes[j], runes[i]
	}
	return string(runes)
}
