package main

import "fmt"

func main() {
    // 练习：打印Hello, World!
    fmt.Println("Hello, World!")

    // 练习：打印1到10之间的数字
    for i := 1; i <= 10; i++ {
        fmt.Println(i)
    }

    // 练习：计算1到10之间的数字的和
    sum := 0
    for i := 1; i <= 10; i++ {
        sum += i
    }
    fmt.Println("Sum:", sum)

    // 练习：定义一个函数，传入两个整数参数并返回它们的和
    add := func(a, b int) int {
        return a + b
    }
    result := add(2, 3)
    fmt.Println("Result:", result)

    // 练习：定义一个结构体表示一个人，包含姓名和年龄字段，并创建一个该类型的实例
    type Person struct {
        Name string
        Age  int
    }
    person := Person{Name: "Alice", Age: 25}
    fmt.Println(person)
}
