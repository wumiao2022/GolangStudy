package main

import "fmt"

func main() {
    // 练习1: 打印"Hello, World!"
    fmt.Println("Hello, World!")

    // 练习2: 交换两个变量的值
    a := 5
    b := 10
    a, b = b, a
    fmt.Println("a =", a, "b =", b)

    // 练习3: 计算两个数的和并打印结果
    num1 := 3
    num2 := 7
    sum := num1 + num2
    fmt.Println("sum =", sum)

    // 练习4: 判断一个数是否为偶数，并打印结果
    num := 6
    if num%2 == 0 {
        fmt.Println(num, "is even")
    } else {
        fmt.Println(num, "is odd")
    }
}
