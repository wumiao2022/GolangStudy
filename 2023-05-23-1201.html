以下是一个简单的Golang程序，实现输入一个数字，判断其正负性：

package main

import "fmt"

func main() {
    var num int
    fmt.Print("请输入一个整数：")
    fmt.Scanln(&num)
    if num > 0 {
        fmt.Println("该数字是正数")
    } else if num < 0 {
        fmt.Println("该数字是负数")
    } else {
        fmt.Println("该数字是零")
    }
}
