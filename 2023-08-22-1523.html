package main

import "fmt"

func main() {
	// 练习1：打印Hello, World!
	fmt.Println("Hello, World!")

	// 练习2：计算1+2的结果并打印
	sum := 1 + 2
	fmt.Println("Sum =", sum)

	// 练习3：声明一个整数变量x，并将其初始化为10，打印x的值
	x := 10
	fmt.Println("x =", x)

	// 练习4：声明一个字符串变量message，并将其赋值为"Hello, Golang!"，打印message的值
	message := "Hello, Golang!"
	fmt.Println("message =", message)
}
