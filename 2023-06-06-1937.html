这是一个简单的计算斐波那契数列的 Golang 代码：

package main

import "fmt"

func main() {
    fmt.Println("Fibonacci series:")
    var a, b, c int = 0, 1, 0
    for i := 0; i < 10; i++ {
        fmt.Println(c)
        a = b
        b = c
        c = a + b
    }
}

这是一个简单的实现冒泡排序的 Golang 代码：

package main

import "fmt"

func main() {
    var arr = []int{5, 2, 7, 3, 9, 1, 4}
    fmt.Println("Unsorted array:", arr)
    for i := 0; i < len(arr)-1; i++ {
        for j := i + 1; j < len(arr); j++ {
            if arr[i] > arr[j] {
                arr[i], arr[j] = arr[j], arr[i]
            }
        }
    }
    fmt.Println("Sorted array:", arr)
}
