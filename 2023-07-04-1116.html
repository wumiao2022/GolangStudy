package main

import "fmt"

func main() {
	// 1. 打印 Hello, World!
	fmt.Println("Hello, World!")

	// 2. 交换两个变量的值
	a := 10
	b := 20
	a, b = b, a
	fmt.Println("a =", a, "b =", b)

	// 3. 判断一个数是否为偶数
	num := 12
	if num%2 == 0 {
		fmt.Println(num, "is even")
	} else {
		fmt.Println(num, "is odd")
	}

	// 4. 求一个正整数的阶乘
	fact := 1
	n := 5
	for i := 1; i <= n; i++ {
		fact *= i
	}
	fmt.Println(n, "! =", fact)

	// 5. 反转字符串
	str := "GoLang"
	reversedStr := ""
	for i := len(str) - 1; i >= 0; i-- {
		reversedStr += string(str[i])
	}
	fmt.Println("Reversed string:", reversedStr)
}
