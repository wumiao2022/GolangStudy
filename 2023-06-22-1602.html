以下是一个简单的Golang练习实例：

package main

import "fmt"

func main() {
	// 1. 打印Hello World
	fmt.Println("Hello World")

	// 2. 输出1 ~ 10之间的偶数
	for i := 1; i <= 10; i++ {
		if i%2 == 0 {
			fmt.Println(i)
		}
	}

	// 3. 计算1 ~ 100之间所有的奇数之和
	sum := 0
	for i := 1; i <= 100; i++ {
		if i%2 != 0 {
			sum += i
		}
	}
	fmt.Println(sum)

	// 4. 判断一个数是否为质数
	num := 7
	flag := true
	for i := 2; i < num; i++ {
		if num%i == 0 {
			flag = false
			break
		}
	}
	if flag {
		fmt.Println(num, "是质数")
	} else {
		fmt.Println(num, "不是质数")
	}
}
