package main

import "fmt"

// 练习1：打印出从1到100的数字，但是遇到可以被3整除的数字用"Fizz"代替，遇到可以被5整除的数字用"Buzz"代替，同时可以被3和5整除的数字用"FizzBuzz"代替。
func fizzBuzz() {
    for i := 1; i <= 100; i++ {
        if i%3 == 0 && i%5 == 0 {
            fmt.Println("FizzBuzz")
        } else if i%3 == 0 {
            fmt.Println("Fizz")
        } else if i%5 == 0 {
            fmt.Println("Buzz")
        } else {
            fmt.Println(i)
        }
    }
}

// 练习2：编写一个函数，接受整数数组和目标整数，返回数组中两个数字加和等于目标整数的下标。
func twoSum(nums []int, target int) []int {
    hashMap := make(map[int]int)
    for idx, num := range nums {
        complement := target - num
        if val, ok := hashMap[complement]; ok {
            return []int{val, idx}
        }
        hashMap[num] = idx
    }
    return nil
}

// 练习3：实现斐波那契数列的递归函数。
func fibonacci(n int) int {
    if n <= 1 {
        return n
    }
    return fibonacci(n-1) + fibonacci(n-2)
}

func main() {
    fizzBuzz()

    nums := []int{2, 7, 11, 15}
    target := 9
    result := twoSum(nums, target)
    fmt.Println("Indices:", result)

    n := 10
    fmt.Println("Fibonacci result:", fibonacci(n))
}
