package main

import "fmt"

func main() {
	// 1. 定义一个整型变量num，并初始化为10
	num := 10
	fmt.Println(num)

	// 2. 定义一个字符串变量str，并初始化为"Hello, World!"
	str := "Hello, World!"
	fmt.Println(str)

	// 3. 定义一个浮点数变量pi，并初始化为3.14
	pi := 3.14
	fmt.Println(pi)

	// 4. 定义一个布尔型变量isTrue，并初始化为true
	isTrue := true
	fmt.Println(isTrue)

	// 5. 定义一个整型数组numbers，包含5个元素，并分别初始化为1、2、3、4、5
	numbers := [5]int{1, 2, 3, 4, 5}
	fmt.Println(numbers)

	// 6. 定义一个切片slice，包含3个元素，并分别初始化为"apple"、"banana"、"orange"
	slice := []string{"apple", "banana", "orange"}
	fmt.Println(slice)

	// 7. 定义一个映射map，包含3个键值对，并分别初始化为"red": 1, "green": 2, "blue": 3
	mymap := map[string]int{"red": 1, "green": 2, "blue": 3}
	fmt.Println(mymap)
}

注意：以上代码只是展示了一些基本的变量类型的定义和初始化，可以运行并查看结果。练习中还包含了数组、切片和映射的定义和初始化，以及使用fmt包输出变量的值。你可以在此基础上进行更多的练习和实践。
