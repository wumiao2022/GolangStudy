package main

import "fmt"

func main() {
	// 练习1：打印Hello, World!
	fmt.Println("Hello, World!")

	// 练习2：将两个数相加并输出结果
	num1 := 10
	num2 := 5
	sum := num1 + num2
	fmt.Println("Sum:", sum)

	// 练习3：计算面积并输出结果
	width := 8
	height := 6
	area := width * height
	fmt.Println("Area:", area)

	// 练习4：使用for循环打印0到10之间的偶数
	for i := 0; i <= 10; i += 2 {
		fmt.Print(i, " ")
	}
	fmt.Println()

	// 练习5：使用switch语句根据数字输出对应的英文单词
	number := 3
	switch number {
	case 1:
		fmt.Println("One")
	case 2:
		fmt.Println("Two")
	case 3:
		fmt.Println("Three")
	default:
		fmt.Println("Unknown")
	}
}
