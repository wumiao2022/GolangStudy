这是一个简单的Golang练习，可以用来练习基础的数据类型和流程控制语句：

package main

import "fmt"

func main() {
    // 打印1到10之间的整数
    for i := 1; i <= 10; i++ {
        fmt.Println(i)
    }
    
    // 创建一个字符串数组，并打印出数组中的每一项
    strs := []string{"apple", "banana", "orange"}
    for _, s := range strs {
        fmt.Println(s)
    }
    
    // 创建一个整型数组，并计算数组中所有元素的和
    nums := []int{1, 2, 3, 4, 5}
    sum := 0
    for _, n := range nums {
        sum += n
    }
    fmt.Println("Sum:", sum)
    
    // 创建一个映射类型，存储几个人的名字和年龄，并打印出所有人名字和年龄的对应关系
    ages := map[string]int{
        "Alice":  25,
        "Bob":    30,
        "Charlie": 35,
    }
    for name, age := range ages {
        fmt.Printf("%s is %d years old.\n", name, age)
    }
}
