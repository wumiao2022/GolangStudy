以下是一个简单的Golang练习示例，用于学习循环和条件语句：

package main

import "fmt"

func main() {
    // 打印1到10之间所有的偶数
    for i := 1; i <= 10; i++ {
        if i%2 == 0 {
            fmt.Println(i)
        }
    }

    // 计算1到10之间所有奇数的和
    sum := 0
    for i := 1; i <= 10; i++ {
        if i%2 != 0 {
            sum += i
        }
    }
    fmt.Println(sum)

    // 打印一个5行乘法表
    for i := 1; i <= 5; i++ {
        for j := 1; j <= i; j++ {
            fmt.Printf("%d*%d=%d ", i, j, i*j)
        }
        fmt.Println()
    }
}
