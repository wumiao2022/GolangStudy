func main() { // 1. 打印Hello, World! fmt.Println(“Hello, World!”)

// 2. 计算1到10的和
sum := 0
for i := 1; i <= 10; i++ {
	sum += i
}
fmt.Println("Sum:", sum)

// 3. 判断一个数是否为偶数
num := 8
if num%2 == 0 {
	fmt.Println(num, "is even")
} else {
	fmt.Println(num, "is odd")
}

// 4. 反转字符串
str := "Hello, World!"
reversedStr := ""
for i := len(str)-1; i >= 0; i-- {
	reversedStr += string(str[i])
}
fmt.Println("Reversed String:", reversedStr)

// 5. 找出切片中的最大值
slice := []int{2, 5, 1, 10, 7}
max := slice[0]
for _, num := range slice {
	if num > max {
		max = num
	}
}
fmt.Println("Max:", max)

// 6. 冒泡排序
sortSlice := []int{7, 2, 9, 1, 5}
for i := 0; i < len(sortSlice)-1; i++ {
	for j := 0; j < len(sortSlice)-i-1; j++ {
		if sortSlice[j] > sortSlice[j+1] {
			temp := sortSlice[j]
			sortSlice[j] = sortSlice[j+1]
			sortSlice[j+1] = temp
		}
	}
}
fmt.Println("Sorted Slice:", sortSlice)

}
