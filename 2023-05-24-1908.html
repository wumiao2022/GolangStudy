  1. 获取用户输入并打印
package main

import "fmt"

func main() {
	var name string
	fmt.Print("请输入您的姓名：")
	fmt.Scan(&name)
	fmt.Println("欢迎你，" + name + "!")
}
  1. 数组排序
package main

import (
	"fmt"
	"sort"
)

func main() {
	numbers := []int{9, 4, 6, 2, 1, 8, 5, 7, 3}
	sort.Ints(numbers)
	fmt.Println(numbers)
}
  1. 结构体
package main

import "fmt"

type person struct {
	name string
	age  int
}

func main() {
	p := person{name: "张三", age: 25}
	fmt.Println("姓名：" + p.name)
	fmt.Println("年龄：" + string(p.age))
}
  1. 并发
package main

import (
	"fmt"
	"time"
)

func main() {
	go printNumbers()
	go printLetters()
	time.Sleep(2000 * time.Millisecond)
}

func printNumbers() {
	for i := 1; i <= 5; i++ {
		time.Sleep(250 * time.Millisecond)
		fmt.Print(i)
	}
}

func printLetters() {
	for i := 'a'; i <= 'e'; i++ {
		time.Sleep(400 * time.Millisecond)
		fmt.Print(string(i))
	}
}
