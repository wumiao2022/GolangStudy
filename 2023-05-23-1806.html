  1. 判断一个数是不是质数：
func isPrime(num int) bool {
	if num < 2 {
		return false
	}
	for i := 2; i <= int(math.Sqrt(float64(num))); i++ {
		if num%i == 0 {
			return false
		}
	}
	return true
}
  1. 打印斐波那契数列前20项：
func fibonacci(n int) int {
	if n <= 1 {
		return n
	}
	return fibonacci(n-1) + fibonacci(n-2)
}

for i := 0; i < 20; i++ {
	fmt.Println(fibonacci(i))
}
  1. 反转一个字符串：
func reverseString(str string) string {
	result := ""
	for _, s := range str {
		result = string(s) + result
	}
	return result
}

fmt.Println(reverseString("hello"))
  1. 判断一个字符串是不是回文串：
func isPalindrome(str string) bool {
	for i := 0; i < len(str)/2; i++ {
		if str[i] != str[len(str)-i-1] {
			return false
		}
	}
	return true
}

fmt.Println(isPalindrome("abccba"))
  1. 统计一个字符串中出现次数最多的字符及其出现次数：
func maxFrequency(str string) (rune, int) {
	frequency := make(map[rune]int)
	for _, s := range str {
		frequency[s]++
	}

	maxFreq := 0
	maxChar := rune(0)
	for char, freq := range frequency {
		if freq > maxFreq {
			maxFreq = freq
			maxChar = char
		}
	}

	return maxChar, maxFreq
}

maxChar, maxFreq := maxFrequency("abccbaab")
fmt.Printf("maxChar:%c, maxFreq:%d\n", maxChar, maxFreq)
