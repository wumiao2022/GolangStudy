1. 打印”Hello, world!“：

    package main
import "fmt"
func main() {
fmt.Println("Hello, world!")
}

  2. 计算并输出1+2+3+…+100的值：

    package main
import "fmt"
func main() {
sum := 0
for i := 1; i <= 100; i++ {
	sum += i
}
fmt.Println(sum)
}

  3. 定义一个名为Person的结构体，包含name和age两个属性，并创建一个实例p：

    package main
import "fmt"
type Person struct {
name string
age  int
}
func main() {
p := Person{
	name: "John",
	age:  30,
}
fmt.Println(p)
}

  4. 在数组中查找指定的元素，返回其下标，如果不存在返回-1：

    package main
import "fmt"
func main() {
arr := []int{1, 2, 3, 4, 5}
target := 3
index := -1
for i, val := range arr {
	if val == target {
		index = i
		break
	}
}
fmt.Println(index)
}

  5. 实现一个简单的HTTP服务器，接收请求并返回响应：

    package main
import (
"fmt"
"net/http"
)
func handler(w http.ResponseWriter, r *http.Request) {
fmt.Fprintln(w, "Hello, client!")
}
func main() {
http.HandleFunc("/", handler)
http.ListenAndServe(":8080", nil)
}
