以下是一个简单的计算1到100的和的Golang程序示例：

package main

import "fmt"

func main() {
    sum := 0
    for i := 1; i <= 100; i++ {
        sum += i
    }
    fmt.Println("1到100的和为:", sum)
}

以下是一个简单的判断一个整数是否为质数的Golang程序示例：

package main

import "fmt"

func isPrime(n int) bool {
    if n <= 1 {
        return false
    }
    for i := 2; i <= n/2; i++ {
        if n%i == 0 {
            return false
        }
    }
    return true
}

func main() {
    num := 13
    if isPrime(num) {
        fmt.Println(num, "是质数")
    } else {
        fmt.Println(num, "不是质数")
    }
}
