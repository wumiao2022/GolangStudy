1. 判断一个数是否为质数
func isPrime(n int) bool {
    if n <= 1 {
        return false
    }

    for i := 2; i*i <= n; i++ {
        if n%i == 0 {
            return false
        }
    }

    return true
}
  1. 判断一个字符串是否为回文串
func isPalindrome(s string) bool {
    for i := 0; i < len(s)/2; i++ {
        if s[i] != s[len(s)-1-i] {
            return false
        }
    }

    return true
}
  1. 计算一个整数数组中的平均值
func average(nums []int) float64 {
    sum := 0
    for _, num := range nums {
        sum += num
    }

    return float64(sum) / float64(len(nums))
}
  1. 将一个数字转化为二进制表示
func toBinary(n int) string {
    if n == 0 {
        return "0"
    }

    res := ""

    for n != 0 {
        res = strconv.Itoa(n%2) + res
        n /= 2
    }

    return res
}
  1. 使用冒泡排序对一个整数数组进行排序
func bubbleSort(nums []int) {
    n := len(nums)

    for i := 0; i < n-1; i++ {
        for j := 0; j < n-1-i; j++ {
            if nums[j] > nums[j+1] {
                nums[j], nums[j+1] = nums[j+1], nums[j]
            }
        }
    }
}
