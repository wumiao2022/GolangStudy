    func BubbleSort(arr []int) {
for i := 0; i < len(arr)-1; i++ {
    for j := 0; j < len(arr)-i-1; j++ {
        if arr[j] > arr[j+1] {
            arr[j], arr[j+1] = arr[j+1], arr[j]
        }
    }
}
}

  4. 编写一个函数，将一个整数转换为罗马数字。

    func IntToRoman(num int) string {
var (
    result string
    romanNumeral = map[int]string{
        1000: "M",
        900:  "CM",
        500:  "D",
        400:  "CD",
        100:  "C",
        90:   "XC",
        50:   "L",
        40:   "XL",
        10:   "X",
        9:    "IX",
        5:    "V",
        4:    "IV",
        1:    "I",
    }
)
for k, v := range romanNumeral {
    for num >= k {
        num -= k
        result += v
    }
}
return result
}
