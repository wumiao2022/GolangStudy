package main

import (
	"fmt"
)

func main() {
	// 练习：变量声明与打印
	var name string = "John"
	age := 25
	fmt.Println("Name:", name)
	fmt.Println("Age:", age)

	// 练习：循环打印奇数
	for i := 1; i <= 10; i += 2 {
		fmt.Println(i)
	}

	// 练习：切片操作
	numbers := []int{1, 2, 3, 4, 5}
	fmt.Println(numbers[1:4])

	// 练习：函数调用
	result := add(3, 5)
	fmt.Println("Result:", result)
}

func add(a, b int) int {
	return a + b
}
