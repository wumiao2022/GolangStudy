1. 在终端中输入一个数字，判断该数是否为偶数。
package main

import "fmt"

func main() {
    var num int
    fmt.Print("请输入一个数字：")
    fmt.Scan(&num)
    if num%2 == 0 {
        fmt.Println(num, "是偶数")
    } else {
        fmt.Println(num, "是奇数")
    }
}

  1. 计算数组中所有元素之和。
package main

import "fmt"

func main() {
    arr := []int{1, 2, 3, 4, 5}
    sum := 0
    for _, val := range arr {
        sum += val
    }
    fmt.Println("数组元素之和为：", sum)
}
  1. 实现一个简单的计算器程序，支持加、减、乘、除四种运算。
package main

import "fmt"

func main() {
    var num1, num2, result float64
    var operator string
    fmt.Print("请输入第一个数字：")
    fmt.Scan(&num1)
    fmt.Print("请输入第二个数字：")
    fmt.Scan(&num2)
    fmt.Print("请输入运算符：")
    fmt.Scan(&operator)
    switch operator {
    case "+":
        result = num1 + num2
    case "-":
        result = num1 - num2
    case "*":
        result = num1 * num2
    case "/":
        result = num1 / num2
    default:
        fmt.Println("不支持的运算符")
        return
    }
    fmt.Println("计算结果为：", result)
}
  1. 编写一个函数，用于判断一个字符串是否为回文字符串。
package main

import "fmt"

func main() {
    str := "abba"
    if isPalindrome(str) {
        fmt.Println(str, "是回文字符串")
    } else {
        fmt.Println(str, "不是回文字符串")
    }
}

func isPalindrome(str string) bool {
    l := len(str)
    for i := 0; i < l/2; i++ {
        if str[i] != str[l-1-i] {
            return false
        }
    }
    return true
}
