1. 将字符串 “hello world” 反转输出。

    package main
import "fmt"
func main() {
str := "hello world"
for i := len(str) - 1; i >= 0; i-- {
	fmt.Printf("%c", str[i])
}
fmt.Println()
}

  3. 实现一个函数，将字符串切片中的所有字符串连接成一个字符串，并在每个字符串之间添加指定的分隔符。

    package main
import (
"fmt"
"strings"
)
func joinWithSep(strs []string, sep string) string {
return strings.Join(strs, sep)
}
func main() {
strs := []string{"hello", "world", "go"}
sep := "-"
res := joinWithSep(strs, sep)
fmt.Println(res)
}

  4. 实现一个函数，计算两个正整数的最大公约数。

    package main
import "fmt"
func gcd(a, b int) int {
if a%b == 0 {
	return b
} else {
	return gcd(b, a%b)
}
}
func main() {
a := 24
b := 32
res := gcd(a, b)
fmt.Printf("%d 和 %d 的最大公约数为 %d\n", a, b, res)
}

  5. 实现一个函数，查找一个字符串中出现次数最多的字符以及出现次数。

    package main
import (
"fmt"
)
func findMaxFreq(str string) (byte, int) {
var freqList [256]int
for i := 0; i < len(str); i++ {
	freqList[str[i]]++
}
maxFreq := 0
var maxChar byte
for i := 0; i < len(freqList); i++ {
	if freqList[i] > maxFreq {
		maxFreq = freqList[i]
		maxChar = byte(i)
	}
}
return maxChar, maxFreq
}
func main() {
str := "abccbaadd"
maxChar, maxFreq := findMaxFreq(str)
fmt.Printf("%c 出现次数最多，出现了 %d 次\n", maxChar, maxFreq)
}
