  1. 打印9*9乘法表
for i := 1; i <= 9; i++ {
    for j := 1; j <= i; j++ {
        fmt.Printf("%d*%d=%d\t", j, i, i*j)
    }
    fmt.Println()
}
  1. 计算1+2+3+…+100的和
sum := 0
for i := 1; i <= 100; i++ {
    sum += i
}
fmt.Println(sum)
  1. 判断一个数是否是素数
func isPrime(n int) bool {
    if n <= 1 {
        return false
    }
    for i := 2; i*i <= n; i++ {
        if n%i == 0 {
            return false
        }
    }
    return true
}
