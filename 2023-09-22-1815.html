package main

import "fmt"

func main() {
	// 多练习一：打印Hello, World!
	fmt.Println("Hello, World!")

	// 多练习二：计算两个整数的和并打印结果
	a := 10
	b := 20
	sum := a + b
	fmt.Println("The sum of", a, "and", b, "is", sum)

	// 多练习三：判断一个整数是否为偶数
	num := 15
	if num%2 == 0 {
		fmt.Println(num, "is an even number")
	} else {
		fmt.Println(num, "is an odd number")
	}

	// 多练习四：判断一个年份是否为闰年
	year := 2022
	if year%4 == 0 && (year%100 != 0 || year%400 == 0) {
		fmt.Println(year, "is a leap year")
	} else {
		fmt.Println(year, "is not a leap year")
	}

	// 多练习五：计算圆的周长和面积
	radius := 5.0
	pi := 3.14159
	circumference := 2 * pi * radius
	area := pi * radius * radius
	fmt.Printf("The circumference of the circle with radius %.2f is %.2f\n", radius, circumference)
	fmt.Printf("The area of the circle with radius %.2f is %.2f\n", radius, area)
}

这是一些简单的练习示例代码，每个练习包含一个小任务。你可以将它们复制到一个新的Go文件中，然后运行查看结果。每个练习的代码块中已经给出相应的注释，说明了具体的任务要求。加油，多练习才能提高！
