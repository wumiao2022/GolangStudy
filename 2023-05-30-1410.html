1. 实现一个计算器，能够进行加减乘除操作

    package main
import "fmt"
func main() {
var num1, num2 float64
var operator string
fmt.Println("请输入两个数和运算符，中间用空格隔开：")
fmt.Scanln(&num1, &num2, &operator)
var result float64
switch operator {
case "+":
    result = num1 + num2
case "-":
    result = num1 - num2
case "*":
    result = num1 * num2
case "/":
    result = num1 / num2
default:
    fmt.Println("无效的运算符")
    return
}
fmt.Printf("%.2f %s %.2f = %.2f\n", num1, operator, num2, result)
}

  2. 实现一个生成斐波那契数列的函数

    package main
import "fmt"
func main() {
n := 10
fmt.Println(fibonacci(n))
}
func fibonacci(n int) []int {
if n <= 0 {
    return []int{}
}
nums := make([]int, n)
nums[0] = 0
if n > 1 {
    nums[1] = 1
    for i := 2; i < n; i++ {
        nums[i] = nums[i-1] + nums[i-2]
    }
}
return nums
}

  3. 实现一个查找给定整数切片中的最大值和最小值的函数

    package main
import "fmt"
func main() {
nums := []int{7, 5, 3, 9, 2}
max, min := findMaxAndMin(nums)
fmt.Printf("最大值：%d, 最小值：%d", max, min)
}
func findMaxAndMin(nums []int) (int, int) {
if len(nums) == 0 {
    return 0, 0
}
max, min := nums[0], nums[0]
for _, num := range nums {
    if num > max {
        max = num
    }
    if num < min {
        min = num
    }
}
return max, min
}
