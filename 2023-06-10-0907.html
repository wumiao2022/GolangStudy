package main

import "fmt"

func main() {
    var num int = 8
    if num%2 == 0 {
        fmt.Println(num, "is even")
    } else {
        fmt.Println(num, "is odd")
    }
}
  1. 求一个slice中所有元素之和并输出
package main

import "fmt"

func main() {
    var nums = []int{2, 4, 6, 8, 10}
    var sum int
    for _, num := range nums {
        sum += num
    }
    fmt.Println("The sum of all numbers in the slice is", sum)
}
  1. 实现一个函数，将一个字符串反转并返回新的字符串
package main

import "fmt"

func main() {
    var str = "hello world"
    revStr := reverse(str)
    fmt.Println("The reversed string is", revStr)
}

func reverse(str string) string {
    runes := []rune(str)
    for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 {
        runes[i], runes[j] = runes[j], runes[i]
    }
    return string(runes)
}
