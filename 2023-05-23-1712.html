这是一个Golang的练习示例：

package main

import "fmt"

func main() {
	// 打印九九乘法表
	for i := 1; i <= 9; i++ {
		for j := 1; j <= i; j++ {
			fmt.Printf("%d*%d=%-2d ", j, i, i*j)
		}
		fmt.Println()
	}

	// 冒泡排序
	arr := []int{9, 6, 3, 4, 1, 2, 8, 7, 5}
	for i := 0; i < len(arr)-1; i++ {
		for j := 0; j < len(arr)-i-1; j++ {
			if arr[j] > arr[j+1] {
				arr[j], arr[j+1] = arr[j+1], arr[j]
			}
		}
	}
	fmt.Println(arr)
}

该示例中包括了打印九九乘法表和冒泡排序两个练习题的代码。
