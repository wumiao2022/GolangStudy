package main

import (
	"fmt"
)

func main() {
	// 练习1：打印1到10的所有整数
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}

	// 练习2：计算1到100的和
	sum := 0
	for i := 1; i <= 100; i++ {
		sum += i
	}
	fmt.Println("1到100的和为:", sum)

	// 练习3：判断一个数是否为偶数
	num := 6
	if num%2 == 0 {
		fmt.Println(num, "是偶数")
	} else {
		fmt.Println(num, "是奇数")
	}

	// 练习4：输出斐波那契数列的前20个数
	n := 20
	fmt.Println("斐波那契数列的前", n, "个数为:")
	fibonacci := make([]int, n)
	fibonacci[0] = 0
	fibonacci[1] = 1
	for i := 2; i < n; i++ {
		fibonacci[i] = fibonacci[i-1] + fibonacci[i-2]
	}
	for _, num := range fibonacci {
		fmt.Println(num)
	}
}
