  1. 将一个字符串数组进行倒序排列。
func reverseStringSlice(strs []string) []string {
    for i := 0; i < len(strs)/2; i++ {
        j := len(strs) - i - 1
        strs[i], strs[j] = strs[j], strs[i]
    }
    return strs
}
  1. 删除一个字符串切片中所有的重复元素。
func removeDuplicateStrings(strs []string) []string {
    set := make(map[string]bool)
    for _, str := range strs {
        set[str] = true
    }
    result := make([]string, 0, len(set))
    for str := range set {
        result = append(result, str)
    }
    return result
}
  1. 计算一个字符串中各个字符出现的次数。
func countChars(str string) map[rune]int {
    counts := make(map[rune]int)
    for _, char := range str {
        counts[char]++
    }
    return counts
}
  1. 实现一个Fibonacci数列的函数，返回前n个数。
func fibonacci(n int) []int {
    if n < 2 {
        return make([]int, 0, n)
    }
    sequence := make([]int, n)
    sequence[0], sequence[1] = 1, 1
    for i := 2; i < n; i++ {
        sequence[i] = sequence[i-1] + sequence[i-2]
    }
    return sequence
}
  1. 实现一个函数，用于将一个整型数组进行去或，也就是将所有小于等于0的数删除。
func removeNegatives(nums []int) []int {
    result := make([]int, 0, len(nums))
    for _, num := range nums {
        if num > 0 {
            result = append(result, num)
        }
    }
    return result
}
