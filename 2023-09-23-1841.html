package main

import "fmt"

func main() {
    // 1. 声明一个整型变量x，并赋值为10
    x := 10

    // 2. 打印变量x的值
    fmt.Println(x)

    // 3. 声明一个字符串变量name，并赋值为"Go语言"
    name := "Go语言"

    // 4. 打印变量name的值
    fmt.Println(name)

    // 5. 声明一个切片变量numbers，包含3个整型元素：1, 2, 3
    numbers := []int{1, 2, 3}

    // 6. 打印切片变量numbers的值
    fmt.Println(numbers)

    // 7. 将切片变量numbers的第2个元素修改为4
    numbers[1] = 4

    // 8. 打印修改后的切片变量numbers的值
    fmt.Println(numbers)
}
