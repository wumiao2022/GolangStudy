package main

import "fmt"

func main() {
	// 打印"Hello, World!"到控制台
	fmt.Println("Hello, World!")

	// 声明一个字符串变量，并将其赋值为"Go is awesome!"
	message := "Go is awesome!"

	// 打印字符串变量到控制台
	fmt.Println(message)

	// 声明一个整数变量，并将其赋值为10
	number := 10

	// 打印整数变量到控制台
	fmt.Println(number)

	// 声明一个布尔变量，并将其赋值为true
	isTrue := true

	// 打印布尔变量到控制台
	fmt.Println(isTrue)
}
