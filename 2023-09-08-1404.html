package main

import (
	"fmt"
	"time"
)

func main() {
	// 今日日期
	today := time.Now().Format("2006-01-02")

	// 打印今日日期
	fmt.Println("今日日期：", today)

	// 输出1到10的平方数
	for i := 1; i <= 10; i++ {
		square := i * i
		fmt.Printf("%d的平方是：%d\n", i, square)
	}

	// 找出数组中的最大值
	numbers := []int{10, 30, 50, 20, 40}
	max := numbers[0]
	for _, num := range numbers {
		if num > max {
			max = num
		}
	}
	fmt.Println("数组中的最大值是：", max)
}
