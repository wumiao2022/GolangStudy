这是一个Golang练习题，目的是实现对字典的基本操作。

package main

import "fmt"

func main() {
    // 创建一个字典
    dict := make(map[string]string)

    // 添加键值对
    dict["apple"] = "苹果"
    dict["orange"] = "橘子"

    // 打印字典
    fmt.Println(dict)

    // 获取键对应的值
    fmt.Println(dict["apple"])

    // 删除键值对
    delete(dict, "orange")

    // 检查键是否存在
    value, exists := dict["orange"]
    fmt.Println(value, exists)

    // 遍历字典
    for key, value := range dict {
        fmt.Println(key, ":", value)
    }
}

以上代码实现了对字典的创建、添加、获取、删除、检查存在以及遍历等基本操作。
