这是一个简单的Golang练习实例代码：

package main

import "fmt"

func main() {
	// 1. 输出 "Hello, World!"
	fmt.Println("Hello, World!")

	// 2. 判断一个数字是否为偶数
	num := 10
	if num%2 == 0 {
		fmt.Printf("%d is even\n", num)
	} else {
		fmt.Printf("%d is odd\n", num)
	}

	// 3. 计算一个数组的平均值
	arr := [5]int{1, 2, 3, 4, 5}
	var sum int
	for _, num := range arr {
		sum += num
	}
	avg := float64(sum) / float64(len(arr))
	fmt.Printf("Average of %v is %v\n", arr, avg)

	// 4. 翻转一个字符串
	str := "hello"
	reversedStr := ""
	for i := len(str) - 1; i >= 0; i-- {
		reversedStr += string(str[i])
	}
	fmt.Printf("Reversed string of %s is %s\n", str, reversedStr)
}

这个程序包含了以下四个练习：

  1. 输出 “Hello, World!”
  2. 判断一个数字是否为偶数
  3. 计算一个数组的平均值
  4. 翻转一个字符串
