// 练习4：给定一个字符串，将其反转并打印结果
package main

import "fmt"

func main() {
	str := "Hello, World!"
	reversed := ""
	for i := len(str)-1; i >= 0; i-- {
		reversed += string(str[i])
	}
	fmt.Println("Reversed:", reversed)
}

// 练习5：计算斐波那契数列的第n项，并打印结果
// 斐波那契数列：0, 1, 1, 2, 3, 5, 8, 13, 21, ...
package main

import "fmt"

func main() {
	n := 10
	fib := []int{0, 1}
	for i := 2; i <= n; i++ {
		fib = append(fib, fib[i-1] + fib[i-2])
	}
	fmt.Printf("Fibonacci sequence up to %dth term: %v", n, fib)
}
