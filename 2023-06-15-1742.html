下面是一些练习实例代码：

  1. Hello World
package main

import "fmt"

func main() {
    fmt.Println("Hello World!")
}
  1. 变量和常量
package main

import "fmt"

func main() {
    // 使用var关键字定义变量
    var a int
    a = 1
    fmt.Println(a)

    // 使用:=定义变量
    b := "hello"
    fmt.Println(b)

    // 定义常量
    const c = 3.14
    fmt.Println(c)
}
  1. 条件语句
package main

import "fmt"

func main() {
    age := 18

    // if语句
    if age >= 18 {
        fmt.Println("You are an adult.")
    } else {
        fmt.Println("You are a teenager.")
    }

    // switch语句
    switch age {
    case 0:
        fmt.Println("You are a baby.")
    case 10:
        fmt.Println("You are a child.")
    case 18:
        fmt.Println("You are an adult.")
    default:
        fmt.Println("You are of an unknown age.")
    }
}
  1. 循环语句
package main

import "fmt"

func main() {
    // for循环
    for i:=0; i<5; i++ {
        fmt.Println(i)
    }

    // while循环
    num := 0
    for num < 10 {
        fmt.Println(num)
        num++
    }

    // 无限循环
    for {
        fmt.Println("This is an infinite loop.")
    }
}
  1. 数组和切片
package main

import "fmt"

func main() {
    // 数组
    var arr [5]int
    arr[0] = 1
    arr[1] = 2
    arr[2] = 3
    arr[3] = 4
    arr[4] = 5
    fmt.Println(arr)

    // 切片
    slice := []int{1, 2, 3, 4, 5}
    fmt.Println(slice)
}
  1. 函数
package main

import "fmt"

func main() {
    // 函数调用
    result := add(2, 3)
    fmt.Println("The result is:", result)

    // 函数作为变量
    calculator := func(a, b int) int {
        return a * b
    }
    fmt.Println(calculator(3, 4))
}

// 定义函数
func add(x, y int) int {
    return x + y
}
  1. 结构体
package main

import "fmt"

type Person struct {
    name string
    age int
}

func main() {
    // 结构体定义
    var person1 Person
    person1.name = "Alice"
    person1.age = 18
    fmt.Println(person1)

    // 结构体字面量
    person2 := Person{name: "Bob", age: 20}
    fmt.Println(person2)

    // 结构体指针
    person3 := &Person{name: "Charlie", age: 22}
    fmt.Println(*person3)
}
