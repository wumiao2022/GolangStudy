1. 获取当前时间戳：
package main

import (
    "fmt"
    "time"
)

func main() {
    timestamp := time.Now().Unix()
    fmt.Println(timestamp)
}
  1. 判断切片中是否包含某个元素：
package main

import "fmt"

func main() {
    nums := []int{1, 2, 3, 4, 5}
    target := 3

    exists := false
    for _, num := range nums {
        if num == target {
            exists = true
            break
        }
    }
    fmt.Println(exists)
}
  1. 将字符串转换为数字：
package main

import (
    "fmt"
    "strconv"
)

func main() {
    str := "1234"
    num, _ := strconv.Atoi(str)
    fmt.Println(num)
}
  1. 获取命令行参数：
package main

import (
    "fmt"
    "os"
)

func main() {
    args := os.Args[1:]
    fmt.Println(args)
}
  1. 将结构体转换为JSON字符串：
package main

import (
    "encoding/json"
    "fmt"
)

type Person struct {
    Name string `json:"name"`
    Age  int    `json:"age"`
}

func main() {
    person := Person{Name: "Alice", Age: 30}
    jsonStr, _ := json.Marshal(person)
    fmt.Println(string(jsonStr))
}
