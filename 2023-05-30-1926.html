1. 计算阶乘
func factorial(n int) int {
    if n == 0 {
        return 1
    } else {
        return n * factorial(n-1)
    }
}
  1. 求斐波那契数列
func fibonacci(n int) int {
    if n == 0 {
        return 0
    } else if n == 1 {
        return 1
    } else {
        return fibonacci(n-1) + fibonacci(n-2)
    }
}
  1. 判断一个数是否为素数
func isPrime(n int) bool {
    if n <= 1 {
        return false
    } else if n == 2 {
        return true
    } else {
        for i := 2; i < int(math.Sqrt(float64(n)))+1; i++ {
            if n%i == 0 {
                return false
            }
        }
        return true
    }
}
  1. 计算两个数的最大公约数
func gcd(a, b int) int {
    for b != 0 {
        a, b = b, a%b
    }
    return a
}
  1. 计算一个数的平方根
func sqrt(x float64) float64 {
    z := 1.0
    for i := 0; i < 10; i++ {
        z = z - (z*z-x)/(2*z)
    }
    return z
}
