  1. 实现一个简单的计算器
package main

import "fmt"

func main() {
    var num1, num2 float64
    var operator string
    fmt.Print("Enter first number: ")
    fmt.Scanln(&num1)
    fmt.Print("Enter operator (+, -, *, /): ")
    fmt.Scanln(&operator)
    fmt.Print("Enter second number: ")
    fmt.Scanln(&num2)
    var result float64
    switch operator {
    case "+":
        result = num1 + num2
    case "-":
        result = num1 - num2
    case "*":
        result = num1 * num2
    case "/":
        if num2 == 0 {
            fmt.Println("Cannot divide by zero")
            return
        } else {
            result = num1 / num2
        }
    default:
        fmt.Println("Invalid operator")
        return
    }
    fmt.Printf("%g %s %g = %g", num1, operator, num2, result)
}
  1. 判断一个字符串是否为回文字符串
package main

import (
    "fmt"
    "strings"
)

func main() {
    var str string = "racecar"
    var reversedStr string = ""
    for i := len(str) - 1; i >= 0; i-- {
        reversedStr += string(str[i])
    }
    if strings.ToLower(str) == strings.ToLower(reversedStr) {
        fmt.Println("Palindrome")
    } else {
        fmt.Println("Not a palindrome")
    }
}
  1. 找出一个数组中的最大和最小值
package main

import (
    "fmt"
    "math"
)

func main() {
    var nums []float64 = []float64{3.5, -5.2, 10.1, 7.8, 2.3}
    var min float64 = math.Inf(1)
    var max float64 = math.Inf(-1)
    for _, num := range nums {
        if num < min {
            min = num
        }
        if num > max {
            max = num
        }
    }
    fmt.Printf("Min: %g, Max: %g", min, max)
}
