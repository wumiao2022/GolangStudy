package main

import (
    "fmt"
    "time"
)

func main() {
    // 1. 输出当前时间的年、月、日、时、分、秒
    now := time.Now()
    fmt.Println("Year:", now.Year())
    fmt.Println("Month:", now.Month())
    fmt.Println("Day:", now.Day())
    fmt.Println("Hour:", now.Hour())
    fmt.Println("Minute:", now.Minute())
    fmt.Println("Second:", now.Second())

    // 2. 根据当前时间，判断今天是星期几
    weekday := now.Weekday()
    fmt.Println("Today is", weekday)

    // 3. 使用for循环输出0-9的数字
    for i := 0; i < 10; i++ {
        fmt.Println(i)
    }

    // 4. 输出1-100之间所有能被3整除的数字
    for i := 1; i <= 100; i++ {
        if i%3 == 0 {
            fmt.Println(i)
        }
    }
}
