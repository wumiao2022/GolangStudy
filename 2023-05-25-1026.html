package main

import (
	"fmt"
	"math/rand"
)

func main() {
	// 生成10个随机数
	var nums [10]int
	for i := 0; i < 10; i++ {
		nums[i] = rand.Intn(100)
	}

	// 输出随机数
	fmt.Println("随机数：", nums)

	// 冒泡排序
	for i := 0; i < len(nums)-1; i++ {
		for j := 0; j < len(nums)-i-1; j++ {
			if nums[j] > nums[j+1] {
				nums[j], nums[j+1] = nums[j+1], nums[j]
			}
		}
	}

	// 输出排序结果
	fmt.Println("排序后：", nums)
}
