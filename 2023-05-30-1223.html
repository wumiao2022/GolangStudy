1. 找到整型切片中最大值
func findMax(s []int) int {
    max := s[0]
    for _, val := range s {
        if val > max {
            max = val
        }
    }
    return max
}

  1. 数组反转
func reverseArray(s []int) {
    for i := 0; i < len(s)/2; i++ {
        j := len(s) - 1 - i
        s[i], s[j] = s[j], s[i]
    }
}
  1. 计算两个数的最大公约数
func gcd(a, b int) int {
    for b != 0 {
        a, b = b, a%b
    }
    return a
}
  1. 将map按照键名排序并输出
func sortMap(m map[string]int) {
    keys := make([]string, 0, len(m))
    for k := range m {
        keys = append(keys, k)
    }
    sort.Strings(keys)
    for _, k := range keys {
        fmt.Printf("%s: %d\n", k, m[k])
    }
}
