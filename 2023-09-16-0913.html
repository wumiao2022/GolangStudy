package main

import "fmt"

// Example1 is a simple Hello World program
func Example1() {
	fmt.Println("Hello, World!")
}

// Example2 is a program to print the sum of two numbers
func Example2() {
	num1 := 10
	num2 := 20
	sum := num1 + num2
	fmt.Println("Sum:", sum)
}

// Example3 is a program to find the maximum of three numbers
func Example3() {
	num1 := 10
	num2 := 20
	num3 := 15

	max := num1
	if num2 > max {
		max = num2
	}
	if num3 > max {
		max = num3
	}

	fmt.Println("Maximum:", max)
}

// Example4 is a program to print the Fibonacci series
func Example4() {
	n := 10
	fmt.Print("Fibonacci Series: 0 1")

	a, b := 0, 1
	for i := 2; i < n; i++ {
		next := a + b
		fmt.Print(" ", next)
		a, b = b, next
	}

	fmt.Println()
}

// Example5 is a program to calculate the factorial of a number
func Example5() {
	n := 5
	factorial := 1

	for i := 1; i <= n; i++ {
		factorial *= i
	}

	fmt.Println("Factorial:", factorial)
}

func main() {
	Example1()
	Example2()
	Example3()
	Example4()
	Example5()
}
