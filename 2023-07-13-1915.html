package main

import "fmt"

func main() {
	// 1. 打印Hello, World!
	fmt.Println("Hello, World!")

	// 2. 计算并打印1+2的结果
	sum := 1 + 2
	fmt.Println(sum)

	// 3. 声明一个整数变量num，并给它赋值为5，然后打印出变量num的值
	num := 5
	fmt.Println(num)

	// 4. 声明一个字符串变量name，并给它赋值为"John"，然后打印出变量name的值
	name := "John"
	fmt.Println(name)

	// 5. 使用for循环打印出1到10的整数
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}
}

注意：以上代码只是一些简单的练习示例，每日多练习才能提高编程能力。
