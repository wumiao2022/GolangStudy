以下是一个简单的Go程序，用于将输入的字符串反转输出：

package main

import (
	"fmt"
	"strings"
)

func main() {
	var str string
	fmt.Print("请输入字符串: ")
	fmt.Scan(&str)

	// 借助golang自带的strings包中的Reverse函数来反转字符串
	var reversedStr = strings.Builder{}
	for i := len(str) - 1; i >= 0; i-- {
		reversedStr.WriteByte(str[i])
	}
	fmt.Println(reversedStr.String())
}
