  1. 计算阶乘
package main

import "fmt"

func main() {
    var num int = 5
    var result int = 1
    for i := 1; i <= num; i++ {
        result *= i;
    }
    fmt.Println(num, "! =", result)
}
  1. 判断是否是素数
package main

import "fmt"

func main() {
    var num int = 11
    var isPrime bool = true
    for i := 2; i < num; i++ {
        if num % i == 0 {
            isPrime = false
            break
        }
    }
    if isPrime {
        fmt.Println(num, "is prime")
    } else {
        fmt.Println(num, "is not prime")
    }
}
  1. 计算平均数
package main

import "fmt"

func main() {
    var nums []int = []int{5, 10, 15, 20}
    var total int
    for _, num := range nums {
       total += num
    }
    var avg float64 = float64(total) / float64(len(nums))
    fmt.Printf("The average is %.2f\n", avg)
}
  1. 反转字符串
package main

import "fmt"

func main() {
    var str string = "Hello World"
    var reversed string
    for i := len(str) - 1; i >= 0; i-- {
        reversed += string(str[i])
    }
    fmt.Println(reversed)
}
  1. 字符串转整数
package main

import (
    "fmt"
    "strconv"
)

func main() {
    var str string = "123"
    num, err := strconv.Atoi(str)
    if err != nil {
        fmt.Println("Conversion failed")
    } else {
        fmt.Println(num)
    }
}
