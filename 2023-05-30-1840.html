题目：统计一个字符串中每个字符出现的次数

func countCharacters(s string) map[rune]int {
    count := make(map[rune]int)
    for _, r := range s {
        count[r]++
    }
    return count
}

题目：判断一个数是否是素数

func isPrime(n int) bool {
    if n < 2 {
        return false
    }
    for i := 2; i*i <= n; i++ {
        if n%i == 0 {
            return false
        }
    }
    return true
}

题目：反转一个整数

func reverseInt(n int) int {
    var reverse int
    for ; n != 0; n = n / 10 {
        reverse = reverse*10 + n%10
    }
    return reverse
}

题目：找出一个整数数组中最大的元素

func findMax(nums []int) int {
    max := nums[0]
    for _, num := range nums {
        if num > max {
            max = num
        }
    }
    return max
}

题目：从一个整数数组中删除指定值的元素

func removeElement(nums []int, val int) []int {
    j := 0
    for i := 0; i < len(nums); i++ {
        if nums[i] != val {
            nums[j] = nums[i]
            j++
        }
    }
    return nums[:j]
}

题目：合并两个有序整数数组

func merge(nums1 []int, m int, nums2 []int, n int) []int {
    i, j := 0, 0
    nums := make([]int, m+n)
    for k := 0; k < m+n; k++ {
        if i >= m {
            nums[k] = nums2[j]
            j++
        } else if j >= n {
            nums[k] = nums1[i]
            i++
        } else if nums1[i] <= nums2[j] {
            nums[k] = nums1[i]
            i++
        } else {
            nums[k] = nums2[j]
            j++
        }
    }
    return nums
}
