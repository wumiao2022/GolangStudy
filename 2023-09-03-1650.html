// 练习3：判断给定数字是否为素数（只能被1和自身整除的正整数）
package main

import "fmt"

func isPrime(num int) bool {
    if num < 2 {
        return false
    }
    for i := 2; i*i <= num; i++ {
        if num%i == 0 {
            return false
        }
    }
    return true
}

func main() {
    num := 17
    if isPrime(num) {
        fmt.Printf("%d is a prime number\n", num)
    } else {
        fmt.Printf("%d is not a prime number\n", num)
    }
}

// 练习4：计算给定字符串中字符的个数（不包括空格和特殊字符）
package main

import (
    "fmt"
    "unicode"
)

func countCharacters(str string) int {
    count := 0
    for _, char := range str {
        if unicode.IsLetter(char) {
            count++
        }
    }
    return count
}

func main() {
    str := "Hello, World!"
    fmt.Printf("Number of characters: %d\n", countCharacters(str))
}

// 练习5：将给定字符串反转
package main

import (
    "fmt"
    "unicode/utf8"
)

func reverseString(str string) string {
    reversed := ""
    for len(str) > 0 {
        runeValue, size := utf8.DecodeLastRuneInString(str)
        reversed += string(runeValue)
        str = str[:len(str)-size]
    }
    return reversed
}

func main() {
    str := "Hello, World!"
    fmt.Printf("Reversed string: %s\n", reverseString(str))
}


// 练习6：生成一个长度为 n 的斐波那契数列切片
package main

import "fmt"

func fibonacci(n int) []int {
    if n <= 0 {
        return []int{}
    }
    fib := make([]int, n)
    fib[0] = 0
    if n > 1 {
        fib[1] = 1
        for i := 2; i < n; i++ {
            fib[i] = fib[i-1] + fib[i-2]
        }
    }
    return fib
}

func main() {
    n := 10
    fmt.Printf("Fibonacci sequence of length %d: %v\n", n, fibonacci(n))
}
