package main

import (
    "fmt"
    "math/rand"
    "time"
)

func main() {
    rand.Seed(time.Now().UnixNano())
    num := rand.Intn(100)

    fmt.Println("猜数字游戏开始！")
    fmt.Println("请输入一个0~99的整数：")

    for {
        var guess int
        count := 0

        for {
            _, err := fmt.Scan(&guess)
            if err == nil {
                break
            }
            fmt.Println("输入有误，请重新输入：")
        }

        count++
        if guess > num {
            fmt.Println("猜大了！继续猜：")
        } else if guess < num {
            fmt.Println("猜小了！继续猜：")
        } else {
            fmt.Printf("恭喜你，猜对了！共猜了 %v 次\n", count)
            break
        }
    }
}

package main

import (
    "fmt"
    "time"
)

func main() {
    fmt.Println("计算斐波那契数列。请输入一个整数：")
    
    var n int
    _, err := fmt.Scan(&n)
    if err != nil {
        fmt.Println("输入有误，请重新运行程序并输入一个整数。")
        return
    }
    
    fmt.Printf("斐波那契数列前 %v 个数字为：", n)
    for i := 0; i < n; i++ {
        fmt.Printf("%v ", fibonacci(i))
    }
}

func fibonacci(n int) int {
    if n <= 1 {
        return n
    }
    
    return fibonacci(n-1) + fibonacci(n-2)
}

package main

import (
    "fmt"
    "math/rand"
    "time"
)

func main() {
    rand.Seed(time.Now().UnixNano())
    target := rand.Intn(100)

    var guess int
    var count int

    fmt.Println("猜数字游戏开始！请输入一个0~99的整数：")

    for {
        _, err := fmt.Scan(&guess)
        count++

        if err != nil {
            fmt.Println("输入有误，请重新输入：")
            continue
        }

        if guess < target {
            fmt.Printf("猜小了！继续猜（已猜：%v次）：", count)
        } else if guess > target {
            fmt.Printf("猜大了！继续猜（已猜：%v次）：", count)
        } else {
            fmt.Printf("猜对了！一共猜了：%v次。\n", count)
            break
        }
    }
}
