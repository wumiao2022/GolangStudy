package main

import "fmt"

func main() {
    // 1. 输出 "Hello, World!"
    fmt.Println("Hello, World!")

    // 2. 使用for循环输出1到10的数
    for i := 1; i <= 10; i++ {
        fmt.Println(i)
    }

    // 3. 计算和输出两个数的和
    num1 := 10
    num2 := 20
    sum := num1 + num2
    fmt.Println("Sum:", sum)

    // 4. 定义并输出一个字符串变量
    message := "Golang is awesome!"
    fmt.Println(message)

    // 5. 使用if语句判断一个数的正负
    num := -5
    if num > 0 {
        fmt.Println("Positive")
    } else if num < 0 {
        fmt.Println("Negative")
    } else {
        fmt.Println("Zero")
    }

    // 6. 使用switch语句判断一个数字对应的星期几
    day := 3
    switch day {
    case 1:
        fmt.Println("Monday")
    case 2:
        fmt.Println("Tuesday")
    case 3:
        fmt.Println("Wednesday")
    case 4:
        fmt.Println("Thursday")
    case 5:
        fmt.Println("Friday")
    case 6:
        fmt.Println("Saturday")
    case 7:
        fmt.Println("Sunday")
    default:
        fmt.Println("Invalid day")
    }

    // 7. 定义和调用一个函数，计算两个数的乘积
    multiply(3, 4)
}

func multiply(num1, num2 int) {
    result := num1 * num2
    fmt.Println("Product:", result)
}
