// 练习2：计算1-100的和 package main

import “fmt”

func main() { sum := 0 for i := 1; i <= 100; i++ { sum += i } fmt.Println(“1-100的和为：”, sum) }

// 练习3：判断一个数是否为质数 package main

import “fmt”

func main() { num := 23 isPrime := true for i := 2; i < num; i++ { if num%i == 0 { isPrime = false break } } if isPrime { fmt.Println(num, “是质数”) } else { fmt.Println(num, “不是质数”) } }

// 练习4：将字符串反转输出 package main

import “fmt”

func main() { str := “Hello, World!” rstr := “” for i := len(str)-1; i >= 0; i– { rstr += string(str[i]) } fmt.Println(rstr) }

// 练习5：将一个int数组中的数按照升序排列 package main

import “fmt”

func main() { nums := []int{5, 2, 8, 1, 9, 4, 3} for i := 0; i < len(nums)-1; i++ { for j := i+1; j < len(nums); j++ { if nums[i] > nums[j] { temp := nums[i] nums[i] = nums[j] nums[j] = temp } } } fmt.Println(nums) }
