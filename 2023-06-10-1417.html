1. 打印九九乘法表
package main

import "fmt"

func main() {
    for i := 1; i <= 9; i++ {
        for j := 1; j <= i; j++ {
            fmt.Printf("%d*%d=%d\t", j, i, i*j)
        }
        fmt.Println()
    }
}
  1. 实现一个简单的数字猜测游戏
package main

import (
    "fmt"
    "math/rand"
    "time"
)

func main() {
    rand.Seed(time.Now().UnixNano())
    answer := rand.Intn(100) + 1
    fmt.Println("数字猜测游戏开始了！")
    for {
        var guess int
        fmt.Print("请输入一个1-100的数字：")
        fmt.Scan(&guess)
        if guess == answer {
            fmt.Println("恭喜你，猜对了！")
            break
        } else if guess < answer {
            fmt.Println("猜小了，再试试？")
        } else {
            fmt.Println("猜大了，再试试？")
        }
    }
}
  1. 实现一个简单的计算器
package main

import "fmt"

func main() {
    var num1, num2 float64
    var operator string
    fmt.Print("请输入一个数字：")
    fmt.Scan(&num1)
    fmt.Print("请输入运算符（+、-、*、/）：")
    fmt.Scan(&operator)
    fmt.Print("请输入另一个数字：")
    fmt.Scan(&num2)
    var result float64
    switch operator {
    case "+":
        result = num1 + num2
    case "-":
        result = num1 - num2
    case "*":
        result = num1 * num2
    case "/":
        result = num1 / num2
    default:
        fmt.Println("无效的运算符！")
        return
    }
    fmt.Printf("%g %s %g = %g\n", num1, operator, num2, result)
}
