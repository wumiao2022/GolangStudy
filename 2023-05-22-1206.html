1. 编写一个程序，计算并输出1至1000的和。
sum := 0
for i := 1; i <= 1000; i++ {
    sum += i
}
fmt.Println(sum)
  1. 编写一个程序，输入一个整数n，输出从1至n的所有奇数。
var n int
fmt.Scan(&n)
for i := 1; i <= n; i += 2 {
    fmt.Println(i)
}
  1. 编写一个程序，输入两个整数x和y，计算它们的乘积并输出。
var x, y int
fmt.Scan(&x, &y)
product := x * y
fmt.Println(product)
  1. 编写一个程序，实现一个简单的计算器功能，支持加法、减法、乘法和除法，用户通过命令行输入两个数和操作符，输出计算结果。
var num1, num2 float64
var operator string
fmt.Scan(&num1, &num2, &operator)

switch operator {
case "+":
    fmt.Println(num1 + num2)
case "-":
    fmt.Println(num1 - num2)
case "*":
    fmt.Println(num1 * num2)
case "/":
    if num2 == 0 {
        fmt.Println("除数不能为0")
    } else {
        fmt.Println(num1 / num2)
    }
default:
    fmt.Println("无效的操作符")
}
  1. 编写一个程序，输入一个字符串，反转字符串并输出。
var str string
fmt.Scan(&str)

runes := []rune(str)

for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 {
    runes[i], runes[j] = runes[j], runes[i]
}

fmt.Println(string(runes))
