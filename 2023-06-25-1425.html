这是一个判断一个数是奇数还是偶数的简单程序：

package main

import (
	"fmt"
)

func main() {
	num := 10
	if num%2 == 0 {
		fmt.Println("偶数")
	} else {
		fmt.Println("奇数")
	}
}

这是一个将字符串反转的简单程序：

package main

import "fmt"

func main() {
	str := "Hello World"
	runes := []rune(str)
	for i, j := 0, len(runes)-1; i < len(runes)/2; i, j = i+1, j-1 {
		runes[i], runes[j] = runes[j], runes[i]
	}
	fmt.Println(string(runes))
}

这是一个判断一个字符串是否是回文的简单程序：

package main

import "fmt"

func isPalindrome(str string) bool {
	runes := []rune(str)
	for i, j := 0, len(runes)-1; i < len(runes)/2; i, j = i+1, j-1 {
		if runes[i] != runes[j] {
			return false
		}
	}
	return true
}

func main() {
	str := "racecar"
	if isPalindrome(str) {
		fmt.Printf("%s是回文字符串\n", str)
	} else {
		fmt.Printf("%s不是回文字符串\n", str)
	}
}
