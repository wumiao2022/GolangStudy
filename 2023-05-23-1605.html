2. 计算两个整数的和 
```go
package main

import "fmt"

func main() {
    var num1, num2, sum int
    num1 = 5
    num2 = 7
    sum = num1 + num2
    fmt.Printf("The sum of %d and %d is %d", num1, num2, sum)
}

4. 判断某一年是否是闰年 
```go
package main

import "fmt"

func main() {
    var year int = 2020
    isLeapYear := (year%4 == 0 && year%100 != 0) || (year%400 == 0)
    if isLeapYear {
        fmt.Printf("%d is a leap year", year)
    } else {
        fmt.Printf("%d is not a leap year", year)
    }
}
