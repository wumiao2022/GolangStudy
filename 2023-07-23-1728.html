package main

import "fmt"

func main() {
    // 1. 声明一个整数变量x，并初始化为10
    var x int = 10
    fmt.Println(x)

    // 2. 声明一个字符串变量message，并赋值为"Hello, world!"
    var message string = "Hello, world!"
    fmt.Println(message)

    // 3. 声明一个浮点数切片numbers，并初始化为包含4个元素的切片 [1.2, 3.4, 5.6, 7.8]
    numbers := []float64{1.2, 3.4, 5.6, 7.8}
    fmt.Println(numbers)

    // 4. 声明一个布尔型变量isTrue，并赋值为true
    isTrue := true
    fmt.Println(isTrue)

    // 5. 声明一个空接口类型变量data，并赋值为整数值100
    var data interface{} = 100
    fmt.Println(data)
}

注意：以上代码仅为示例，请确保你已正确安装了go环境，并在命令行中运行上述代码。
