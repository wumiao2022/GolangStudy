以下是一个Golang练习题的代码实现，可以测试学习者对Golang基础语法和控制流程的掌握程度：

package main

import "fmt"

func main() {
	// 练习题 1: 输出 1 到 10 的所有整数，每行一个
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}

	// 练习题 2: 输出 1 到 100 的所有奇数，每行一个
	for i := 1; i <= 100; i += 2 {
		fmt.Println(i)
	}

	// 练习题 3: 输出 1 到 100 的所有偶数，每行一个
	for i := 2; i <= 100; i += 2 {
		fmt.Println(i)
	}

	// 练习题 4: 打印乘法口诀表
	for i := 1; i <= 9; i++ {
		for j := 1; j <= i; j++ {
			fmt.Printf("%d*%d=%d ", j, i, i*j)
		}
		fmt.Println()
	}

	// 练习题 5: 判断一个数是不是素数
	num := 23
	isPrime := true
	for i := 2; i < num; i++ {
		if num%i == 0 {
			isPrime = false
			break
		}
	}
	if isPrime {
		fmt.Printf("%d is a prime number\n", num)
	} else {
		fmt.Printf("%d is not a prime number\n", num)
	}

	// 练习题 6: 实现冒泡排序（升序）
	arr := []int{3, 2, 4, 5, 1}
	for i := 0; i < len(arr)-1; i++ {
		for j := 0; j < len(arr)-1-i; j++ {
			if arr[j] > arr[j+1] {
				arr[j], arr[j+1] = arr[j+1], arr[j]
			}
		}
	}
	fmt.Println("Sorted array:", arr)
}
