1. 打印1到100的数，但是如果这个数可以被3整除，就打印Fizz；如果可以被5整除，就打印Buzz；如果可以同时被3和5整除，就打印FizzBuzz。

    package main
import "fmt"
func main() {
for i := 1; i <= 100; i++ {
    if i%3 == 0 && i%5 == 0 {
        fmt.Println("FizzBuzz")
    } else if i%3 == 0 {
        fmt.Println("Fizz")
    } else if i%5 == 0 {
        fmt.Println("Buzz")
    } else {
        fmt.Println(i)
    }
}
}

  2. 判断一个整数是否为素数。

    package main
import (
"fmt"
"math"
)
func main() {
num := 29
prime := true
for i := 2; i <= int(math.Sqrt(float64(num))); i++ {
    if num%i == 0 {
        prime = false
        break
    }
}
if prime && num > 1 {
    fmt.Println(num, "is a prime number")
} else {
    fmt.Println(num, "is not a prime number")
}
}

  3. 求斐波那契数列的前20个数。

    package main
import "fmt"
func main() {
var (
    a, b, c int = 0, 1, 0
)
for i := 0; i < 20; i++ {
    fmt.Printf("%d ", a)
    c = a + b
    a = b
    b = c
}
}

  4. 实现一个求阶乘的函数。

    package main
import "fmt"
func factorial(num int) int {
if num == 0 {
    return 1
}
return num * factorial(num-1)
}
func main() {
num := 5
fmt.Printf("%d! = %d\n", num, factorial(num))
}

  5. 判断一个字符串是否是回文串。

    package main
import "fmt"
func isPalindrome(str string) bool {
for i := 0; i < len(str)/2; i++ {
    if str[i] != str[len(str)-1-i] {
        return false
    }
}
return true
}
func main() {
str1 := "racecar"
str2 := "hello"
fmt.Printf("%s is palindrome? %t\n", str1, isPalindrome(str1))
fmt.Printf("%s is palindrome? %t\n", str2, isPalindrome(str2))
}
