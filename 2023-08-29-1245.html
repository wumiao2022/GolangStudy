package main

import "fmt"

func main() {
	// Exercise 1: Print "Hello, World!" to the console

	fmt.Println("Hello, World!")

	// Exercise 2: Calculate the sum of two numbers and print the result

	num1 := 10
	num2 := 5
	sum := num1 + num2
	fmt.Println("Sum:", sum)

	// Exercise 3: Create a function that takes two integers as parameters and returns their product

	product := multiply(3, 4)
	fmt.Println("Product:", product)

	// Exercise 4: Create a slice of strings with names of your favorite fruits and print the length of the slice

	fruits := []string{"apple", "banana", "orange"}
	fmt.Println("Number of fruits:", len(fruits))

	// Exercise 5: Iterate over the slice of fruits and print each fruit

	for _, fruit := range fruits {
		fmt.Println(fruit)
	}
}

func multiply(a, b int) int {
	return a * b
}
