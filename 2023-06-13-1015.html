以下是一些练习代码：

  1. 输出“Hello World”:

    package main
import "fmt"
func main() {
fmt.Println("Hello World")
}

  2. 求1到100的和:

    package main
import "fmt"
func main() {
sum := 0
for i := 0; i <= 100; i++ {
    sum += i
}
fmt.Println("Sum of 1 to 100 is:", sum)
}

  3. 判断一个数是否为素数:

    package main
import (
"fmt"
"math"
)
func isPrime(n int) bool {
for i := 2; i <= int(math.Sqrt(float64(n))); i++ {
    if n%i == 0 {
        return false
    }
}
return true
}
func main() {
n := 17
if isPrime(n) {
    fmt.Println(n, "is a prime number")
} else {
    fmt.Println(n, "is not a prime number")
}
}

  4. 在一个已排序的数组中搜索一个元素:

    package main
import "fmt"
func binarySearch(arr []int, target int) int {
low := 0
high := len(arr) - 1
for low <= high {
    mid := (low + high) / 2
    if arr[mid] < target {
        low = mid + 1
    } else if arr[mid] > target {
        high = mid - 1
    } else {
        return mid
    }
}
return -1
}
func main() {
arr := []int{1, 2, 3, 4, 5, 6, 7, 8, 9}
target := 7
result := binarySearch(arr, target)
if result == -1 {
    fmt.Println(target, "is not in the array")
} else {
    fmt.Println(target, "is at index", result)
}
}

  5. 反转一个字符串:

    package main
import "fmt"
func reverseString(s string) string {
runes := []rune(s)
for i, j := 0, len(runes)-1; i < len(runes)/2; i, j = i+1, j-1 {
    runes[i], runes[j] = runes[j], runes[i]
}
return string(runes)
}
func main() {
s := "Hello World"
fmt.Println(reverseString(s))
}
