1. 打印出1~1000之间所有能够被3整除的数。

    package main
import "fmt"
func main() {
for i := 1; i <= 1000; i++ {
	if i%3 == 0 {
		fmt.Println(i)
	}
}
}

  2. 编写一个程序，输入一个字符串，反转字符串并输出。

    package main
import "fmt"
func main() {
var s string
fmt.Println("请输入一个字符串：")
fmt.Scanln(&s)
r := []rune(s)
for i, j := 0, len(r)-1; i < j; i, j = i+1, j-1 {
	r[i], r[j] = r[j], r[i]
}
fmt.Println("反转后的字符串为：", string(r))
}

  3. 编写一个程序，输入一个正整数n，求n的阶乘并输出。

    package main
import "fmt"
func main() {
var n int
fmt.Println("请输入一个正整数：")
fmt.Scanln(&n)
factorial := 1
for i := 1; i <= n; i++ {
	factorial *= i
}
fmt.Println(n, "的阶乘是：", factorial)
}

  4. 编写一个程序，输入一个字符串，统计其中的英文字母、数字和其他字符的个数并输出。

    package main
import (
"fmt"
"unicode"
)
func main() {
var s string
fmt.Println("请输入一个字符串：")
fmt.Scanln(&s)
var letter, number, other int
for _, r := range s {
	if unicode.IsLetter(r) {
		letter++
	} else if unicode.IsNumber(r) {
		number++
	} else {
		other++
	}
}
fmt.Printf("英文字母：%d，数字：%d，其他字符：%d\n", letter, number, other)
}

  5. 编写一个程序，输入一个字符串，判断它是否是回文串并输出。

    package main
import "fmt"
func main() {
var s string
fmt.Println("请输入一个字符串：")
fmt.Scanln(&s)
r := []rune(s)
for i, j := 0, len(r)-1; i < j; i, j = i+1, j-1 {
	if r[i] != r[j] {
		fmt.Println(s, "不是回文串")
		return
	}
}
fmt.Println(s, "是回文串")
}
