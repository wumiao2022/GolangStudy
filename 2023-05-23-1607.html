  2. 冒泡排序：

    arr := []int{49, 38, 65, 97, 76, 13, 27}
length := len(arr)
for i := 0; i < length-1; i++ {
for j := 0; j < length-i-1; j++ {
    if arr[j] > arr[j+1] {
        arr[j], arr[j+1] = arr[j+1], arr[j]
    }
}
}
fmt.Println(arr)

  4. 判断是否是质数：

    func isPrime(num int) bool {
if num <= 1 {
    return false
}
for i := 2; i*i <= num; i++ {
    if num%i == 0 {
        return false
    }
}
return true
}

  5. 斐波那契数列：

    func fibonacci(n int) int {
if n <= 1 {
    return n
}
return fibonacci(n-1) + fibonacci(n-2)
}
