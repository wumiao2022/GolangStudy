package main

import "fmt"

func main() {
	// 练习1：打印Hello, World!
	fmt.Println("Hello, World!")

	// 练习2：计算两个数的和并输出
	a := 5
	b := 3
	sum := a + b
	fmt.Println("Sum:", sum)

	// 练习3：计算5的阶乘并输出
	factorial := 1
	num := 5
	for i := 1; i <= num; i++ {
		factorial *= i
	}
	fmt.Println("Factorial:", factorial)

	// 练习4：判断一个数是否为偶数，并输出结果
	num2 := 7
	if num2%2 == 0 {
		fmt.Println(num2, "is even")
	} else {
		fmt.Println(num2, "is odd")
	}

	// 练习5：遍历打印1到10的整数
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}
}
