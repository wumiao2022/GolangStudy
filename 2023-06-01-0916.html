1. 找到一个切片中的最大值和最小值，并将它们打印出来。
package main

import (
    "fmt"
)

func main() {
    numbers := []int{10, 20, 30, 40, 50}
    max := numbers[0]
    min := numbers[0]

    for i := 1; i < len(numbers); i++ {
        if numbers[i] > max {
            max = numbers[i]
        }
        if numbers[i] < min {
            min = numbers[i]
        }
    }

    fmt.Println("Max:", max, "Min:", min)
}
  1. 计算一个整数切片的平均值。
package main

import (
    "fmt"
)

func main() {
    numbers := []int{10, 20, 30, 40, 50}
    sum := 0

    for _, number := range numbers {
        sum += number
    }

    fmt.Println("Average:", float64(sum)/float64(len(numbers)))
}
  1. 编写递归函数，计算斐波那契数列中第n项的值。
package main

import (
    "fmt"
)

func fibonacci(n int) int {
    if n == 0 {
        return 0
    } else if n == 1 {
        return 1
    } else {
        return fibonacci(n-1) + fibonacci(n-2)
    }
}

func main() {
    fmt.Println("Fibonacci(10):", fibonacci(10))
}

  1. 编写函数，将一个字符串反转。
package main

import (
    "fmt"
)

func reverseString(str string) string {
    runes := []rune(str)
    for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 {
        runes[i], runes[j] = runes[j], runes[i]
    }
    return string(runes)
}

func main() {
    fmt.Println(reverseString("hello world"))
}
