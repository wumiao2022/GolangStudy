package main

import (
	"fmt"
	"time"
)

func main() {
	fmt.Println("Hello, Golang!")

	// 练习1：打印1-10之间的数字
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}

	// 练习2：计算1-100之间所有整数的和
	sum := 0
	for i := 1; i <= 100; i++ {
		sum += i
	}
	fmt.Println("Sum:", sum)

	// 练习3：打印当前时间
	fmt.Println("Current time:", time.Now())

	// 练习4：使用切片存储一组字符串，并遍历打印出来
	strings := []string{"Hello", "World", "Golang"}
	for _, s := range strings {
		fmt.Println(s)
	}
}
