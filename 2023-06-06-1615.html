func main() { // 练习1：打印1-10的整数 for i := 1; i <= 10; i++ { fmt.Println(i) }

// 练习2：打印100以内的偶数
for i := 1; i <= 100; i++ {
    if i%2 == 0 {
        fmt.Println(i)
    }
}

// 练习3：计算1+2+...+10的和
sum := 0
for i := 1; i <= 10; i++ {
    sum += i
}
fmt.Println(sum)

// 练习4：打印10以内的乘法口诀表
for i := 1; i <= 10; i++ {
    for j := 1; j <= i; j++ {
        fmt.Printf("%d*%d=%d\t", j, i, i*j)
    }
    fmt.Println()
}

// 练习5：将1-100的整数按照3个一行的格式打印出来
for i := 1; i <= 100; i++ {
    fmt.Printf("%d ", i)
    if i%3 == 0 {
        fmt.Println()
    }
}

}
