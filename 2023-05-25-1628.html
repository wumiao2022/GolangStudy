这是一个简单的示例程序，实现了将一个整数切片反转的功能：

package main

import "fmt"

func reverseIntSlice(nums []int) {
    for i, j := 0, len(nums)-1; i < j; i, j = i+1, j-1 {
        nums[i], nums[j] = nums[j], nums[i]
    }
}

func main() {
    nums := []int{1, 2, 3, 4, 5}
    fmt.Println("原始切片：", nums)
    reverseIntSlice(nums)
    fmt.Println("反转后的切片：", nums)
}
