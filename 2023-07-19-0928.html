package main

import (
	"fmt"
	"math/rand"
	"time"
)

func main() {
	rand.Seed(time.Now().UnixNano())

	// 生成一个随机整数切片
	numbers := make([]int, 10)
	for i := range numbers {
		numbers[i] = rand.Intn(100)
	}

	// 打印切片中的所有元素
	fmt.Println("切片中的元素为:")
	fmt.Println(numbers)

	// 计算切片中的最大值和最小值
	min := numbers[0]
	max := numbers[0]
	for _, num := range numbers {
		if num < min {
			min = num
		}
		if num > max {
			max = num
		}
	}

	// 打印最大值和最小值
	fmt.Println("切片中的最小值为:", min)
	fmt.Println("切片中的最大值为:", max)
}

这个实例生成一个含有10个随机整数的切片，并计算出切片中的最大值和最小值，然后将它们打印出来。你可以每天运行这个实例来练习Golang的切片操作和随机数生成。
