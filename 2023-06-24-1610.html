这是一个非常好的实践方法，不仅可以帮助你巩固所学的知识，还可以提升你的编程技能。以下是一些示例：

  1. 打印出1到100中不能被3整除的所有数字：
for i := 1; i <= 100; i++ {
    if i%3 != 0 {
        fmt.Println(i)
    }
}
  1. 从切片中移除指定值：
func remove(slice []int, value int) []int {
    for i := 0; i < len(slice); i++ {
        if slice[i] == value {
            slice = append(slice[:i], slice[i+1:]...)
        }
    }
    return slice
}
  1. 计算一个数的阶乘：
func factorial(n int) int {
    if n == 0 {
        return 1
    }
    return n * factorial(n-1)
}
  1. 模拟实现一个栈（先进后出）：
type Stack struct {
    items []int
}

func (s *Stack) Push(item int) {
    s.items = append(s.items, item)
}

func (s *Stack) Pop(item int) int {
    last := len(s.items) - 1
    item := s.items[last]
    s.items = s.items[:last]
    return item
}
  1. 对切片进行排序：
slice := []int{3, 7, 1, 8, 4, 2, 9, 6, 5}
sort.Slice(slice, func(i, j int) bool {
    return slice[i] < slice[j]
})
