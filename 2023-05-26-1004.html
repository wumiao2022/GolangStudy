1. 定义一个数组，将数组中的每个元素加1，并输出数组
package main

import "fmt"

func main() {
    arr := [5]int{1, 2, 3, 4, 5}
    for i := 0; i < len(arr); i++ {
        arr[i]++
    }
    fmt.Println(arr)
}
  1. 输入一个数字n，输出n个随机数
package main

import (
    "fmt"
    "math/rand"
    "time"
)

func main() {
    rand.Seed(time.Now().UnixNano()) // 初始化随机数种子
    var n int
    fmt.Print("请输入数字n：")
    fmt.Scan(&n) // 用户输入
    for i := 0; i < n; i++ {
        fmt.Print(rand.Intn(100), " ") // 生成随机数
    }
}
  1. 定义一个结构体表示人员信息，输入n个人员信息并输出
package main

import "fmt"

type Person struct {
    Name    string
    Age     int
    Gender  string
    Address string
}

func main() {
    var n int
    fmt.Print("请输入人员数量n：")
    fmt.Scan(&n) // 用户输入
    people := make([]Person, n)
    for i := 0; i < n; i++ {
        fmt.Printf("请输入第%d个人员信息\n", i+1)
        fmt.Print("姓名：")
        fmt.Scan(&people[i].Name)
        fmt.Print("年龄：")
        fmt.Scan(&people[i].Age)
        fmt.Print("性别：")
        fmt.Scan(&people[i].Gender)
        fmt.Print("地址：")
        fmt.Scan(&people[i].Address)
    }
    fmt.Println(people)
}
  1. 定义一个函数，将传入的字符串反转并输出
package main

import "fmt"

func reverseString(s string) string {
    runes := []rune(s) // 将字符串转换为rune切片
    for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 {
        runes[i], runes[j] = runes[j], runes[i] // 交换元素
    }
    return string(runes) // 将rune切片转换为字符串
}

func main() {
    s := "hello, world"
    fmt.Println(reverseString(s))
}
