package main

import "fmt"

func main() {
	// 示例1: 输出Hello, World!
	fmt.Println("Hello, World!")
	
	// 示例2: 计算整数相加
	num1 := 5
	num2 := 10
	sum := num1 + num2
	fmt.Printf("The sum of %d and %d is %d\n", num1, num2, sum)
	
	// 示例3: 判断数值是否为偶数
	num := 7
	if num%2 == 0 {
		fmt.Printf("%d is even\n", num)
	} else {
		fmt.Printf("%d is odd\n", num)
	}
	
	// 示例4: 循环打印乘法表
	for i := 1; i <= 9; i++ {
		for j := 1; j <= i; j++ {
			fmt.Printf("%d * %d = %d\t", j, i, i*j)
		}
		fmt.Println()
	}
	
	// 示例5: 切片的基本操作
	numbers := []int{1, 2, 3, 4, 5}
	fmt.Println("Slice before modification:", numbers)
	numbers[2] = 10
	numbers = append(numbers, 6)
	fmt.Println("Slice after modification:", numbers)
	
	// 示例6: 函数的调用
	result := add(3, 4)
	fmt.Println("Result:", result)
}

func add(a, b int) int {
	return a + b
}
