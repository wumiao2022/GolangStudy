// 练习3: 判断一个数是否为素数 package main

import “fmt”

func main() { num := 17 isPrime := true for i := 2; i < num; i++ { if num%i == 0 { isPrime = false break } } if isPrime { fmt.Println(num, “is a prime number”) } else { fmt.Println(num, “is not a prime number”) } }

// 练习4: 实现斐波那契数列 package main

import “fmt”

func fibonacci(n int) { a, b := 0, 1 for i := 0; i < n; i++ { fmt.Print(a, “ “) a, b = b, a+b } fmt.Println() }

func main() { fmt.Println(“Fibonacci Series:”) fibonacci(10) }

// 练习5: 计算圆的面积和周长 package main

import ( “fmt” “math” )

func calculateCircle(radius float64) (float64, float64) { area := math.Pi * radius * radius perimeter := 2 * math.Pi * radius return area, perimeter }

func main() { radius := 5.0 area, perimeter := calculateCircle(radius) fmt.Println(“Radius:”, radius) fmt.Println(“Area:”, area) fmt.Println(“Perimeter:”, perimeter) }
