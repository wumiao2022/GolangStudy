2. 实现一个命令行程序，输入两个数值，输出它们的和。
```go
package main

import (
	"bufio"
	"fmt"
	"os"
	"strconv"
)

func main() {
	reader := bufio.NewReader(os.Stdin)
	fmt.Print("Enter a number: ")
	num1, _ := reader.ReadString('\n')
	num1 = num1[:len(num1)-1]

	fmt.Print("Enter another number: ")
	num2, _ := reader.ReadString('\n')
	num2 = num2[:len(num2)-1]

	number1, _ := strconv.Atoi(num1)
	number2, _ := strconv.Atoi(num2)

	sum := number1 + number2
	fmt.Println("The sum of", num1, "and", num2, "is", sum)
}

4. 实现一个计算N的阶乘的程序。
```go
package main

import (
	"fmt"
)

func factorial(n int) int {
	if n == 0 {
		return 1
	}
	return n * factorial(n-1)
}

func main() {
	n := 5
	fmt.Printf("%d! = %d", n, factorial(n))
}
``` 

5. 实现一个函数，判断一个整数是否是素数。
```go
package main

import (
	"fmt"
	"math"
)

func isPrime(number int) bool {
	if number <= 1 {
		return false
	}
	for i := 2; i <= int(math.Sqrt(float64(number))); i++ {
		if number%i == 0 {
			return false
		}
	}
	return true
}

func main() {
	number1 := 29
	number2 := 30
	fmt.Println(number1, "is a prime number? ", isPrime(number1))
	fmt.Println(number2, "is a prime number? ", isPrime(number2))
}
