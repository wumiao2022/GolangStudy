这是一个简单的递归函数程序代码，用于计算斐波那契数列：

package main

import "fmt"

func fibonacci(n int) int {
    if n <= 1 {
        return 1
    }
    return fibonacci(n-1) + fibonacci(n-2)
}

func main() {
    for i := 0; i < 10; i++ {
        fmt.Println(fibonacci(i))
    }
}

这是一个简单的命令行参数处理程序代码：

package main

import (
    "fmt"
    "os"
)

func main() {
    args := os.Args
    fmt.Println("Your command-line arguments are:")
    for i := 0; i < len(args); i++ {
        fmt.Println(args[i])
    }
}

这是一个简单的协程程序代码，用于打印数字：

package main

import (
    "fmt"
    "time"
)

func printNumbers() {
    for i := 1; i <= 100; i++ {
        fmt.Println(i)
        time.Sleep(time.Millisecond * 100)
    }
}

func main() {
    go printNumbers()
    time.Sleep(time.Second * 2)
}
