1. 翻转字符串
func reverseString(s string) string {
    runes := []rune(s)
    for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 {
        runes[i], runes[j] = runes[j], runes[i]
    }
    return string(runes)
}
  1. 判断素数
func isPrime(n int) bool {
    if n < 2 {
        return false
    }
    for i := 2; i*i <= n; i++ {
        if n%i == 0 {
            return false
        }
    }
    return true
}
  1. 给定一个切片，求其中的最大值和最小值
func findMaxMin(arr []int) (int, int) {
    if len(arr) < 1 {
        return 0, 0
    }
    min := arr[0]
    max := arr[0]
    for i := 1; i < len(arr); i++ {
        if arr[i] < min {
            min = arr[i]
        }
        if arr[i] > max {
            max = arr[i]
        }
    }
    return max, min
}
  1. 实现简单的计算器
func calculator(a int, b int, op byte) int {
    switch op {
    case '+':
        return a + b
    case '-':
        return a - b
    case '*':
        return a * b
    case '/':
        if b == 0 {
            return 0
        }
        return a / b
    default:
        return 0
    }
}
  1. 实现Fibonacci数列
func fibonacci(n int) []int {
    if n < 1 {
        return nil
    }
    if n == 1 {
        return []int{1}
    }
    if n == 2 {
        return []int{1, 1}
    }

    arr := make([]int, n)
    arr[0] = 1
    arr[1] = 1
    for i := 2; i < n; i++ {
        arr[i] = arr[i-1] + arr[i-2]
    }
    return arr
}
