1. 求一个整数数组的最大值
package main

import "fmt"

func main() {
    arr := []int{5, 1, 8, 3, 10, 2}
    max := arr[0]
    for i := 1; i < len(arr); i++ {
        if arr[i] > max {
            max = arr[i]
        }
    }
    fmt.Println("最大值为:", max)
}
  1. 判断一个字符串是否为回文串
package main

import "fmt"

func main() {
    str := "abcba"
    isPalindromic := true
    for i := 0; i < len(str)/2; i++ {
        if str[i] != str[len(str)-i-1] {
            isPalindromic = false
            break
        }
    }
    if isPalindromic {
        fmt.Println(str, "是回文串")
    } else {
        fmt.Println(str, "不是回文串")
    }
}
  1. 求一个正整数的阶乘
package main

import "fmt"

func main() {
    n := 5
    fact := 1
    for i := 1; i <= n; i++ {
        fact *= i
    }
    fmt.Printf("%d的阶乘为%d\n", n, fact)
}
  1. 反转一个字符串
package main

import "fmt"

func main() {
    s := "hello, world!"
    r := []rune(s)
    for i, j := 0, len(r)-1; i < j; i, j = i+1, j-1 {
        r[i], r[j] = r[j], r[i]
    }
    fmt.Println(string(r))
}
  1. 计算斐波那契数列的第n项
package main

import "fmt"

func main() {
    n := 10
    a, b := 0, 1
    for i := 0; i < n; i++ {
        a, b = b, a+b
    }
    fmt.Printf("斐波那契数列的第%d项为%d\n", n, a)
}
