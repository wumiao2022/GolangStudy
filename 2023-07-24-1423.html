package main

import "fmt"

func main() {
	// 练习1: 打印"Hello, Golang!"
	fmt.Println("Hello, Golang!")

	// 练习2: 声明一个整数变量num并赋值为10，打印num的值
	num := 10
	fmt.Println(num)

	// 练习3: 声明一个字符串变量name并赋值为"Jack"，打印name的值
	name := "Jack"
	fmt.Println(name)

	// 练习4: 声明一个浮点数变量height并赋值为1.75，打印height的值
	height := 1.75
	fmt.Println(height)

	// 练习5: 定义一个切片slice，包含整数1、2、3、4，打印slice的值
	slice := []int{1, 2, 3, 4}
	fmt.Println(slice)

	// 练习6: 定义一个空的映射map，键是字符串类型，值是整数类型，打印map的值
	mapping := make(map[string]int)
	fmt.Println(mapping)
}
