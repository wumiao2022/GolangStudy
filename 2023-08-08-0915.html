package main

import (
	"fmt"
	"time"
)

func main() {
	// 1. 打印当前时间
	fmt.Println(time.Now())

	// 2. 计算1到10的累加和
	sum := 0
	for i := 1; i <= 10; i++ {
		sum += i
	}
	fmt.Println("Sum:", sum)

	// 3. 判断一个数是否为偶数
	num := 7
	if num%2 == 0 {
		fmt.Println(num, "is even")
	} else {
		fmt.Println(num, "is odd")
	}

	// 4. 将字符串转换为整数
	str := "123"
	num, _ = strconv.Atoi(str)
	fmt.Println("String to int:", num)

	// 5. 反转一个字符串
	str2 := "Hello, World!"
	reverse := ""
	for i := len(str2) - 1; i >= 0; i-- {
		reverse += string(str2[i])
	}
	fmt.Println("Reversed string:", reverse)
}
