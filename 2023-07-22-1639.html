package main

import "fmt"

func main() {
	// 练习1: 打印出Hello, World!
	fmt.Println("Hello, World!")

	// 练习2: 打印出数字1到10
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}

	// 练习3: 判断一个数是奇数还是偶数
	num := 7
	if num%2 == 0 {
		fmt.Println("偶数")
	} else {
		fmt.Println("奇数")
	}

	// 练习4: 计算斐波那契数列的第n项
	n := 10
	fibonacci := make([]int, n)
	fibonacci[0] = 0
	fibonacci[1] = 1
	for i := 2; i < n; i++ {
		fibonacci[i] = fibonacci[i-1] + fibonacci[i-2]
	}
	fmt.Println(fibonacci[n-1])

	// 练习5: 打印出一个矩形
	width := 5
	height := 3
	for i := 0; i < height; i++ {
		for j := 0; j < width; j++ {
			fmt.Print("*")
		}
		fmt.Println()
	}
}
