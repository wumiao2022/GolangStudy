  2. 实现一个函数，用于判断一个字符串是否为回文字符串，并输出结果。

    package main
import "fmt"
func isPalindrome(s string) bool {
for i := 0; i < len(s)/2; i++ {
    if s[i] != s[len(s)-i-1] {
        return false
    }
}
return true
}
func main() {
s1 := "abcba"
s2 := "hello"
fmt.Printf("%s is a palindrome: %t\n", s1, isPalindrome(s1))
fmt.Printf("%s is a palindrome: %t\n", s2, isPalindrome(s2))
}

  3. 实现一个函数，用于统计一个字符串中每个字符出现的次数，并输出结果。

    package main
import (
"fmt"
)
func charCount(s string) map[rune]int {
counts := make(map[rune]int)
for _, r := range s {
    counts[r]++
}
return counts
}
func main() {
s := "abacabadabacaba"
counts := charCount(s)
for r, count := range counts {
    fmt.Printf("%c: %d\n", r, count)
}
}

  4. 实现一个函数，用于获取一个整数切片中的最大值和最小值，并输出结果。

    package main
import "fmt"
func maxMin(nums []int) (int, int) {
max, min := nums[0], nums[0]
for _, v := range nums {
    if v > max {
        max = v
    }
    if v < min {
        min = v
    }
}
return max, min
}
func main() {
nums := []int{3, 7, 1, 5, 9, 2}
max, min := maxMin(nums)
fmt.Printf("Max: %d, Min: %d\n", max, min)
}

  5. 实现一个函数，用于求一个整数切片的总和、平均值和中位数，并输出结果。

    package main
import (
"fmt"
"sort"
)
func sumAvgMedian(nums []int) (int, float64, float64) {
sum := 0
for _, v := range nums {
    sum += v
}
avg := float64(sum) / float64(len(nums))
sort.Ints(nums)
midIndex := len(nums) / 2
var median float64
if len(nums)%2 == 0 {
    median = float64(nums[midIndex-1]+nums[midIndex]) / 2
} else {
    median = float64(nums[midIndex])
}
return sum, avg, median
}
func main() {
nums := []int{3, 7, 1, 5, 9, 2}
sum, avg, median := sumAvgMedian(nums)
fmt.Printf("Sum: %d, Avg: %f, Median: %f\n", sum, avg, median)
}
