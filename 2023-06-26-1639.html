1. 判断一个数是否为质数：
func isPrime(num int) bool {
    if num <= 1 {
        return false
    }
    for i := 2; i*i <= num; i++ {
        if num%i == 0 {
            return false
        }
    }
    return true
}
  1. 交换两个变量的值：
func swap(a, b *int) {
    *a, *b = *b, *a
}
  1. 反转字符串：
func reverseString(s string) string {
    runes := []rune(s)
    for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 {
        runes[i], runes[j] = runes[j], runes[i]
    }
    return string(runes)
}
  1. 判断字符串是否为回文串：
func isPalindrome(s string) bool {
    runes := []rune(s)
    for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 {
        if runes[i] != runes[j] {
            return false
        }
    }
    return true
}
  1. 查找一个字符串中出现次数最多的字符：
func findMaxChar(s string) rune {
    counts := make(map[rune]int)
    for _, r := range s {
        counts[r]++
    }
    var maxChar rune
    maxCount := -1
    for r, c := range counts {
        if c > maxCount {
            maxChar = r
            maxCount = c
        }
    }
    return maxChar
}
