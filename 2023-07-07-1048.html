package main

import (
	"fmt"
)

func main() {
	// Exercise 1: Print "Hello, World!"
	fmt.Println("Hello, World!")

	// Exercise 2: Find the sum of two integers
	sum := 3 + 5
	fmt.Println("Sum:", sum)

	// Exercise 3: Find the product of two floating-point numbers
	num1 := 2.5
	num2 := 1.5
	product := num1 * num2
	fmt.Println("Product:", product)

	// Exercise 4: Concatenate two strings
	string1 := "Hello"
	string2 := "GoLang"
	result := string1 + " " + string2
	fmt.Println("Result:", result)

	// Exercise 5: Find the remainder of a division
	dividend := 20
	divisor := 3
	remainder := dividend % divisor
	fmt.Println("Remainder:", remainder)

	// Exercise 6: Display the type of a variable
	var variable string = "Golang"
	fmt.Printf("Type: %T\n", variable)

	// Exercise 7: Use a for loop to print numbers from 1 to 10
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}

	// Exercise 8: Create a function that returns the absolute value of a number
	fmt.Println(absoluteValue(-5))
}

func absoluteValue(num int) int {
	if num < 0 {
		return -num
	}
	return num
}
