package main

import "fmt"

func main() {
	// 练习1：打印出"Hello, World!"
	fmt.Println("Hello, World!")

	// 练习2：声明一个字符串变量并赋值"Go is awesome!"
	str := "Go is awesome!"
	fmt.Println(str)

	// 练习3：声明一个整数变量并赋值10，然后声明一个浮点数变量并赋值3.14
	num1 := 10
	num2 := 3.14
	fmt.Println(num1, num2)

	// 练习4：声明一个布尔变量并赋值true，然后声明一个空的切片
	flag := true
	slice := []int{}
	fmt.Println(flag, slice)

	// 练习5：定义一个函数add，接收两个整数参数并返回它们的和
	add := func(a, b int) int {
		return a + b
	}

	// 练习6：调用add函数并打印结果
	result := add(5, 3)
	fmt.Println(result)
}
