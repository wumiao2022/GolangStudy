以下是一个 Golang 程序，用于计算并输出斐波那契数列的前20个数。

package main

import "fmt"

func main() {
    var a, b int = 0, 1
    for i := 0; i < 20; i++ {
        fmt.Println(a)
        a, b = b, a+b
    }
}

以下是一个 Golang 程序，用于计算并输出1到100之间的素数。

package main

import "fmt"

func main() {
    for i := 2; i <= 100; i++ {
        flag := true
        for j := 2; j*j <= i; j++ {
            if i%j == 0 {
                flag = false
                break
            }
        }
        if flag == true {
            fmt.Println(i)
        }
    }
}
