以下是一个简单的练习实例代码，用于打印出1到10之间的奇数：

package main

import "fmt"

func main() {
    for i := 1; i <= 10; i++ {
        if i%2 != 0 {
            fmt.Println(i)
        }
    }
}

以下是一个练习实例代码，用于将一个切片中的所有元素倒序排列：

package main

import "fmt"

func reverse(slice []int) []int {
    for i, j := 0, len(slice)-1; i < j; i, j = i+1, j-1 {
        slice[i], slice[j] = slice[j], slice[i]
    }
    return slice
}

func main() {
    numbers := []int{1, 2, 3, 4, 5}
    reversedNumbers := reverse(numbers)
    fmt.Println(reversedNumbers)
}

以下是一个练习实例代码，用于计算斐波那契数列的前n项：

package main

import "fmt"

func fibonacci(n int) []int {
    sequence := make([]int, n)
    sequence[0], sequence[1] = 1, 1
    for i := 2; i < n; i++ {
        sequence[i] = sequence[i-1] + sequence[i-2]
    }
    return sequence
}

func main() {
    n := 10
    sequence := fibonacci(n)
    fmt.Println(sequence)
}
