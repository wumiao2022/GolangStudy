package main

import (
	"fmt"
)

func main() {
	// 练习1：打印Hello, World!
	fmt.Println("Hello, World!")

	// 练习2：声明一个整数变量并赋值，然后将其打印出来
	var num int = 10
	fmt.Println(num)

	// 练习3：声明一个字符串变量并赋值，然后将其打印出来
	var str string = "Hello, Golang!"
	fmt.Println(str)

	// 练习4：声明一个浮点数变量并赋值，然后将其打印出来
	var f float64 = 3.14
	fmt.Println(f)

	// 练习5：声明一个布尔值变量并赋值，然后将其打印出来
	var b bool = true
	fmt.Println(b)
}
