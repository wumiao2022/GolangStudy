这是一个简单的练习，用于打印出从1到100的数字，但是当数字被3整除时，打印Fizz，当数字被5整除时，打印Buzz，当数字同时被3和5整除时，打印FizzBuzz。

package main

import "fmt"

func main() {
    for i := 1; i <= 100; i++ {
        if i%3 == 0 && i%5 == 0 {
            fmt.Println("FizzBuzz")
        } else if i%3 == 0 {
            fmt.Println("Fizz")
        } else if i%5 == 0 {
            fmt.Println("Buzz")
        } else {
            fmt.Println(i)
        }
    }
}
