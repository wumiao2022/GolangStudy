1. 打印一个 “Hello, World!” 的程序：

    package main
import "fmt"
func main() {
fmt.Println("Hello, World!")
}

  2. 输入一个数，求出它的平方：

    package main
import "fmt"
func main() {
var num int
fmt.Print("请输入一个数：")
fmt.Scan(&num)
square := num * num
fmt.Printf("%d 的平方是 %d\n", num, square)
}

  3. 计算并输出从 1 到 10 的和：

    package main
import "fmt"
func main() {
sum := 0
for i := 1; i <= 10; i++ {
    sum += i
}
fmt.Println("1 到 10 的和为：", sum)
}

  4. 输入两个数，计算它们的和、差、积、商：

    package main
import "fmt"
func main() {
var num1, num2 int
fmt.Print("请输入第一个数：")
fmt.Scan(&num1)
fmt.Print("请输入第二个数：")
fmt.Scan(&num2)
add := num1 + num2
sub := num1 - num2
mul := num1 * num2
div := num1 / num2
fmt.Printf("和：%d，差：%d，积：%d，商：%d\n", add, sub, mul, div)
}

  5. 打印出以下图形：

    *
**
***
****
*****

    package main
import "fmt"
func main() {
for i := 1; i <= 5; i++ {
    for j := 1; j <= i; j++ {
        fmt.Print("*")
    }
    fmt.Println()
}
}
