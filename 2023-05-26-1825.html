1. 反转字符串

    func reverse(s string) string {
runes := []rune(s)
for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 {
    runes[i], runes[j] = runes[j], runes[i]
}
return string(runes)
}

  2. 斐波那契数列

    func fibonacci(n int) int {
if n <= 1 {
    return n
}
return fibonacci(n-1) + fibonacci(n-2)
}

  3. 找出数组中的最大数

    func findMax(arr []int) int {
max := arr[0]
for _, val := range arr {
    if val > max {
        max = val
    }
}
return max
}

  4. 检查字符串是否为回文

    func isPalindrome(s string) bool {
for i := 0; i < len(s)/2; i++ {
    if s[i] != s[len(s)-i-1] {
        return false
    }
}
return true
}

  5. 判断一个数是否为质数

    func isPrime(n int) bool {
if n <= 1 {
    return false
}
for i := 2; i*i <= n; i++ {
    if n%i == 0 {
        return false
    }
}
return true
}
