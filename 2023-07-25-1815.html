package main

import "fmt"

func main() {
    // 打印 "Hello, World!"
    fmt.Println("Hello, World!")

    // 声明一个整数变量 x，并赋值为 10
    x := 10
    fmt.Println(x)

    // 声明一个字符串变量 name，并赋值为 "Gopher"
    name := "Gopher"
    fmt.Println(name)

    // 声明一个浮点数变量 pi，并赋值为 3.14159
    pi := 3.14159
    fmt.Println(pi)

    // 声明一个布尔型变量 isTrue，并赋值为 true
    isTrue := true
    fmt.Println(isTrue)
}
