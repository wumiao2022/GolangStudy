随机生成10个0-100之间的整数，并在终端输出这些数字的最大值、最小值和平均值

package main

import (
	"fmt"
	"math/rand"
)

func main() {
	var nums [10]int
	for i := 0; i < 10; i++ {
		nums[i] = rand.Intn(101)
	}
	fmt.Printf("生成的数字为：%v\n", nums)

	max, min, sum := nums[0], nums[0], 0
	for _, num := range nums {
		sum += num
		if num > max {
			max = num
		}
		if num < min {
			min = num
		}
	}
	avg := float64(sum) / float64(len(nums))

	fmt.Printf("最大值为：%d，最小值为：%d，平均值为：%.2f", max, min, avg)
}
