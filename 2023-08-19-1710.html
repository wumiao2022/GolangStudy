package main

import "fmt"

func main() {
	// 练习1：打印出1至100的所有奇数
	for i := 1; i <= 100; i += 2 {
		fmt.Println(i)
	}

	// 练习2：判断一个数是否为质数
	num := 17
	isPrime := true
	for i := 2; i < num; i++ {
		if num%i == 0 {
			isPrime = false
			break
		}
	}
	if isPrime {
		fmt.Println(num, "是一个质数")
	} else {
		fmt.Println(num, "不是一个质数")
	}

	// 练习3：计算斐波那契数列的第n项
	n := 10
	fib := make([]int, n+1)
	fib[0] = 0
	fib[1] = 1
	for i := 2; i <= n; i++ {
		fib[i] = fib[i-1] + fib[i-2]
	}
	fmt.Println("第", n, "项斐波那契数列的值为", fib[n])

	// 练习4：将字符串进行翻转
	str := "Hello, Go!"
	reversedStr := ""
	for i := len(str) - 1; i >= 0; i-- {
		reversedStr += string(str[i])
	}
	fmt.Println("翻转后的字符串为", reversedStr)
}
