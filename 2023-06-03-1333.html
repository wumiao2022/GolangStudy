package main

import (
	"fmt"
)

func main() {
	// 1. 打印"Hello, world!"
	fmt.Println("Hello, world!")

	// 2. 定义一个整型变量并输出其值
	var num int = 10
	fmt.Println(num)

	// 3. 定义一个字符串变量并输出其值
	var str string = "Hello, Golang!"
	fmt.Println(str)

	// 4. 使用for循环打印1到10的数字
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}

	// 5. 使用if-else语句判断一个数是否为偶数
	num = 5
	if num%2 == 0 {
		fmt.Println(num, "is even")
	} else {
		fmt.Println(num, "is odd")
	}

	// 6. 定义一个结构体并初始化其成员变量
	type Person struct {
		Name string
		Age  int
	}
	p := Person{
		Name: "Alice",
		Age:  20,
	}
	fmt.Println(p)

	// 7. 定义一个切片并添加元素
	s := []int{1, 2, 3}
	s = append(s, 4)
	fmt.Println(s)

	// 8. 定义一个map并添加元素
	m := map[string]int{
		"Alice": 20,
		"Bob":   25,
	}
	m["Charlie"] = 30
	fmt.Println(m)

	// 9. 使用defer延迟执行函数
	defer fmt.Println("Deferred statement")
	fmt.Println("Regular statement")

	// 10. 使用panic抛出异常
	panic("Oops! Something went wrong!")
}
