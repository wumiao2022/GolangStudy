以下是一些Golang的练习实例代码：

  1. 打印Hello World
package main

import "fmt"

func main() {
    fmt.Println("Hello, World!")
}
  1. 求和
package main

import "fmt"

func main() {
    var a, b, sum int
    fmt.Println("Enter two numbers:")
    fmt.Scan(&a, &b)
    sum = a + b
    fmt.Printf("Sum of %d and %d is %d", a, b, sum)
}
  1. 判断奇偶性
package main

import "fmt"

func main() {
    var num int
    fmt.Println("Enter a number:")
    fmt.Scan(&num)
    if num%2 == 0 {
        fmt.Println("The number is even")
    } else {
        fmt.Println("The number is odd")
    }
}
  1. 判断是否为质数
package main

import "fmt"

func main() {
    var num int
    fmt.Println("Enter a number:")
    fmt.Scan(&num)
    isPrime := true
    for i := 2; i <= num/2; i++ {
        if num%i == 0 {
            isPrime = false
            break
        }
    }
    if isPrime {
        fmt.Println("The number is prime")
    } else {
        fmt.Println("The number is not prime")
    }
}
  1. 反转字符串
package main

import "fmt"

func main() {
    str := "Hello, World!"
    strLen := len(str)
    reversed := make([]byte, strLen)
    for i := 0; i < strLen; i++ {
        reversed[i] = str[strLen-i-1]
    }
    fmt.Println(string(reversed))
}
