package main

import "fmt"

func main() {
	// 1. 打印九九乘法表
	for i := 1; i <= 9; i++ {
		for j := 1; j <= i; j++ {
			fmt.Printf("%d*%d=%d\t", j, i, i*j)
		}
		fmt.Println()
	}

	// 2. 实现冒泡排序
	nums := []int{5, 2, 6, 1, 8, 10}
	for i := 0; i < len(nums)-1; i++ {
		for j := 0; j < len(nums)-1-i; j++ {
			if nums[j] > nums[j+1] {
				nums[j], nums[j+1] = nums[j+1], nums[j]
			}
		}
	}
	fmt.Println(nums)

	// 3. 实现一个简易计算器
	var num1, num2 float64
	var op string
	fmt.Print("请输入第一个数：")
	fmt.Scanln(&num1)
	fmt.Print("请输入第二个数：")
	fmt.Scanln(&num2)
	fmt.Print("请输入操作符(+、-、*、/)：")
	fmt.Scanln(&op)
	switch op {
	case "+":
		fmt.Printf("%v + %v = %v\n", num1, num2, num1+num2)
	case "-":
		fmt.Printf("%v - %v = %v\n", num1, num2, num1-num2)
	case "*":
		fmt.Printf("%v * %v = %v\n", num1, num2, num1*num2)
	case "/":
		if num2 == 0 {
			fmt.Println("除数不能为0")
			return
		}
		fmt.Printf("%v / %v = %v\n", num1, num2, num1/num2)
	default:
		fmt.Println("无效的操作符")
	}

	// 4. 实现斐波那契数列
	var n int
	fmt.Print("请输入正整数n：")
	fmt.Scanln(&n)
	res := fibonacci(n)
	fmt.Printf("第%d项的值为%d\n", n, res)
}

func fibonacci(n int) int {
	if n <= 1 {
		return n
	}
	return fibonacci(n-1) + fibonacci(n-2)
}
