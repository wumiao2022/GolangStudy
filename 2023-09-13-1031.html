func main() {
    // 练习1: 输出Hello, World!
    fmt.Println("Hello, World!")

    // 练习2: 定义一个整数变量，输出其值
    var num int
    num = 100
    fmt.Println(num)

    // 练习3: 定义一个字符串变量，输出其值
    var str string
    str = "Go is awesome!"
    fmt.Println(str)

    // 练习4: 定义一个布尔变量，输出其值
    var isTrue bool
    isTrue = true
    fmt.Println(isTrue)

    // 练习5: 使用if语句判断一个整数变量的值是否大于10，如果是则输出"大于10"，否则输出"小于等于10"
    if num > 10 {
        fmt.Println("大于10")
    } else {
        fmt.Println("小于等于10")
    }

    // 练习6: 使用for循环输出1到10之间的整数
    for i := 1; i <= 10; i++ {
        fmt.Println(i)
    }

    // 练习7: 定义一个切片，其中包含5个字符串元素，输出切片中的元素
    slice := []string{"apple", "banana", "orange", "grape", "watermelon"}
    for _, fruit := range slice {
        fmt.Println(fruit)
    }

    // 练习8: 定义一个函数，接收两个整数参数并返回它们的和
    add := func(a, b int) int {
        return a + b
    }
    fmt.Println(add(2, 3))
}
