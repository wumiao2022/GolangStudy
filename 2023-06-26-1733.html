  1. 打印数字0到9
package main
 
import "fmt"
 
func main() {
    for i := 0; i < 10; i++ {
        fmt.Println(i)
    }
}
  1. 使用if-else语句判断奇偶数
package main
 
import "fmt"
 
func main() {
    num := 5
 
    if num%2 == 0 {
        fmt.Println("偶数")
    } else {
        fmt.Println("奇数")
    }
}
  1. 使用switch-case语句实现计算器功能
package main
 
import "fmt"
 
func main() {
    var operator string
    var num1, num2 float32
 
    fmt.Print("请输入运算符(+, -, *, /): ")
    fmt.Scanln(&operator)
 
    fmt.Print("请输入两个数: ")
    fmt.Scanf("%f %f", &num1, &num2)
 
    switch operator {
    case "+":
        fmt.Printf("%.2f + %.2f = %.2f", num1, num2, num1+num2)
    case "-":
        fmt.Printf("%.2f - %.2f = %.2f", num1, num2, num1-num2)
    case "*":
        fmt.Printf("%.2f * %.2f = %.2f", num1, num2, num1*num2)
    case "/":
        if num2 != 0 {
            fmt.Printf("%.2f / %.2f = %.2f", num1, num2, num1/num2)
        } else {
            fmt.Println("除数不能为零")
        }
    default:
        fmt.Println("无效的运算符")
    }
}
  1. 实现一个简单的数组反转
package main
 
import "fmt"
 
func main() {
    arr := []int{1, 2, 3, 4, 5}
    length := len(arr)
 
    for i := 0; i < length/2; i++ {
        arr[i], arr[length-i-1] = arr[length-i-1], arr[i]
    }
 
    fmt.Println(arr)
}
  1. 实现冒泡排序
package main
 
import "fmt"
 
func main() {
    arr := []int{5, 3, 4, 2, 1}
    length := len(arr)
 
    for i := 0; i < length-1; i++ {
        for j := 0; j < length-i-1; j++ {
            if arr[j] > arr[j+1] {
                arr[j], arr[j+1] = arr[j+1], arr[j]
            }
        }
    }
 
    fmt.Println(arr)
}
