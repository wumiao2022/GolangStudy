package main

import "fmt"

func main() {
	// 实例1：打印Hello, World!
	fmt.Println("Hello, World!")

	// 实例2：计算两个数的和
	fmt.Println(add(10, 20))

	// 实例3：判断一个数是奇数还是偶数
	num := 10
	if num%2 == 0 {
		fmt.Println("偶数")
	} else {
		fmt.Println("奇数")
	}

	// 实例4：循环打印数字1到10
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}
}

// 函数：计算两个数的和
func add(a, b int) int {
	return a + b
}
