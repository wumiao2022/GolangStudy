1. 反转字符串

    func reverseString(s string) string {
r := []rune(s)
for i, j := 0, len(r)-1; i < len(r)/2; i, j = i+1, j-1 {
    r[i], r[j] = r[j], r[i]
}
return string(r)
}

  2. 统计单词数量

    func wordCount(s string) map[string]int {
words := strings.Fields(s)
wordMap := make(map[string]int)
for _, word := range words{
    wordMap[word]++
}
return wordMap
}

  3. 求一个数组中的最大值和最小值

    func MaxAndMin(nums []int) (int, int){
max, min := nums[0], nums[0]
for _, num := range nums[1:]{
    if num > max{
        max = num
    }
    if num < min{
        min = num
    }
}
return max, min
}

  4. 判断一个数是否是质数

    func isPrime(num int) bool{
if num < 2{
    return false
}
for i:=2; i*i<=num; i++{
    if num % i == 0{
        return false
    }
}
return true
}

  5. 查找一个字符串中最长的连续重复子串

    func longestRepeatingSubstring(s string) string {
var res string
for i := 0; i < len(s); i++ {
    for j := i+1; j < len(s); j++ {
        if len(res) >= j-i {
            continue
        }
        if s[i:j] == s[j:2*j-i] {
            res = s[i:j]
        }
    }
}
return res
}
