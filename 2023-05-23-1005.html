package main

import (
	"fmt"
)

// 练习1：打印乘法口诀表
func printMultiplicationTable() {
	for i := 1; i <= 9; i++ {
		for j := 1; j <= i; j++ {
			fmt.Printf("%d*%d=%d\t", j, i, i*j)
		}
		fmt.Println()
	}
}

// 练习2：检查一个字符串是否为回文字符串
func isPalindrome(s string) bool {
	for i := 0; i < len(s)/2; i++ {
		if s[i] != s[len(s)-1-i] {
			return false
		}
	}
	return true
}

// 练习3：斐波那契数列（非递归实现）
func fibonacci(n int) int {
	if n <= 1 {
		return n
	}
	first, second := 0, 1
	for i := 2; i <= n; i++ {
		first, second = second, first+second
	}
	return second
}

func main() {
	// 打印乘法口诀表
	printMultiplicationTable()

	// 检查回文字符串
	fmt.Println(isPalindrome("level"))

	// 斐波那契数列
	fmt.Println(fibonacci(10))
}
