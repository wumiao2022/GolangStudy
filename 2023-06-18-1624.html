1. 判断一个年份是否为闰年

    package main
import "fmt"
func main() {
year := 2020
if (year%4 == 0 && year%100 != 0) || (year%400 == 0) {
    fmt.Println(year, "年是闰年")
} else {
    fmt.Println(year, "年不是闰年")
}
}

  2. 打印九九乘法表

    package main
import "fmt"
func main() {
for i := 1; i <= 9; i++ {
    for j := 1; j <= i; j++ {
        fmt.Printf("%d*%d=%d ", j, i, i*j)
    }
    fmt.Println()
}
}

  3. 求一个数的阶乘

    package main
import "fmt"
func main() {
num := 5
factorial := 1
for i := num; i >= 1; i-- {
    factorial *= i
}
fmt.Printf("%d的阶乘为%d", num, factorial)
}

  4. 判断一个数是否为素数

    package main
import (
"fmt"
"math"
)
func main() {
num := 17
isPrime := true
for i := 2; i <= int(math.Sqrt(float64(num))); i++ {
    if num%i == 0 {
        isPrime = false
        break
    }
}
if isPrime {
    fmt.Println(num, "是素数")
} else {
    fmt.Println(num, "不是素数")
}
}

  5. 冒泡排序算法

    package main
import "fmt"
func main() {
nums := []int{2, 4, 1, 6, 8, 3, 5, 7}
for i := 0; i < len(nums)-1; i++ {
    for j := 0; j < len(nums)-i-1; j++ {
        if nums[j] > nums[j+1] {
            nums[j], nums[j+1] = nums[j+1], nums[j]
        }
    }
}
fmt.Println(nums)
}
