package main

import "fmt"

func main() {
    // 1. Hello World
    fmt.Println("Hello, World!")

    // 2. Variable declaration and assignment
    var number int
    number = 10

    // 3. Simple arithmetic operations
    result := number + 5

    // 4. If statement
    if result > 15 {
        fmt.Println("Result is greater than 15")
    } else {
        fmt.Println("Result is less than or equal to 15")
    }

    // 5. For loop
    for i := 0; i < 5; i++ {
        fmt.Println(i)
    }

    // 6. Array
    var numbers = [5]int{1, 2, 3, 4, 5}

    // 7. Slice
    var slice = numbers[1:4]

    // 8. Function
    sum := addNumbers(5, 10)
    fmt.Println("Sum is:", sum)
}

func addNumbers(a, b int) int {
    return a + b
}
