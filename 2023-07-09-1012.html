package main

import "fmt"

func main() {
    // 练习1：声明一个整型变量x，并将其初始化为10
    x := 10
    fmt.Println(x)

    // 练习2：声明一个字符串变量name，并将其初始化为"Hello"
    name := "Hello"
    fmt.Println(name)

    // 练习3：声明一个浮点数变量pi，并将其初始化为3.1415
    pi := 3.1415
    fmt.Println(pi)

    // 练习4：声明一个布尔型变量isTrue，并将其初始化为true
    isTrue := true
    fmt.Println(isTrue)

    // 练习5：声明一个整型切片slices，并将其初始化为空切片
    slices := []int{}
    fmt.Println(slices)
}

输出:
10
Hello
3.1415
true
[]
