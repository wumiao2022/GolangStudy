以下是一个简单的 Golang 练习实例，用于将一个字符串反转：

package main

import "fmt"

func reverseString(str string) string {
    runes := []rune(str)
    for i, j := 0, len(runes)-1; i < len(runes)/2; i, j = i+1, j-1 {
        runes[i], runes[j] = runes[j], runes[i]
    }
    return string(runes)
}

func main() {
    fmt.Println(reverseString("hello world"))
}
