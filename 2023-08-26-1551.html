package main

import "fmt"

func main() {
	// 练习1：打印Hello, World!
	fmt.Println("Hello, World!")

	// 练习2：声明并初始化一个整数变量，然后打印出它的值
	num := 10
	fmt.Println(num)

	// 练习3：声明一个字符串变量并赋值为你的名字，然后打印出它
	name := "Alice"
	fmt.Println(name)

	// 练习4：声明一个切片并添加一些元素，然后打印出它的长度和容量
	slice := []int{1, 2, 3, 4, 5}
	fmt.Println("Length:", len(slice))
	fmt.Println("Capacity:", cap(slice))

	// 练习5：声明一个map并添加一些键值对，然后打印出它的长度
	m := map[string]int{
		"apple":  1,
		"banana": 2,
		"orange": 3,
	}
	fmt.Println("Length:", len(m))
}

以上是一些简单的练习示例代码，每个练习都有一个具体的目标，你可以尝试编写代码来实现这些目标，并运行验证结果。不断地练习和尝试是学习编程语言的重要方法，希望能对你有所帮助！
