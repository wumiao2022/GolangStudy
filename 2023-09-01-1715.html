func main() { // 练习1：输出Hello, World! fmt.Println(“Hello, World!”)

// 练习2：计算两个整数的和
num1 := 10
num2 := 20
sum := num1 + num2
fmt.Println("Sum:", sum)

// 练习3：判断一个数是否为偶数
num := 25
if num%2 == 0 {
    fmt.Println(num, "is even.")
} else {
    fmt.Println(num, "is odd.")
}

// 练习4：判断一个年份是否为闰年
year := 2021
if (year%4 == 0 && year%100 != 0) || year%400 == 0 {
    fmt.Println(year, "is a leap year.")
} else {
    fmt.Println(year, "is not a leap year.")
}

// 练习5：求一个数组中的最大值
arr := []int{5, 8, 2, 9, 3}
max := arr[0]
for i := 1; i < len(arr); i++ {
    if arr[i] > max {
        max = arr[i]
    }
}
fmt.Println("Max:", max)

}
