package main

// 1. 求两个整数的和
func sum(a, b int) int {
	return a + b
}

// 2. 判断一个数是否为偶数
func isEven(num int) bool {
	if num%2 == 0 {
		return true
	}
	return false
}

// 3. 计算斐波那契数列的第n个数
func fibonacci(n int) int {
	if n <= 1 {
		return n
	}
	return fibonacci(n-1) + fibonacci(n-2)
}

// 4. 将字符串中的元音字母替换成其他字符
func replaceVowels(str string, replacement rune) string {
	vowels := map[rune]bool{
		'a': true,
		'e': true,
		'i': true,
		'o': true,
		'u': true,
	}
	result := ""
	for _, ch := range str {
		if vowels[ch] {
			result += string(replacement)
		} else {
			result += string(ch)
		}
	}
	return result
}

// 5. 判断一个字符串是否为回文字符串
func isPalindrome(str string) bool {
	for i := 0; i < len(str)/2; i++ {
		if str[i] != str[len(str)-1-i] {
			return false
		}
	}
	return true
}

func main() {
	// 测试示例
	a, b := 10, 5
	sumResult := sum(a, b)
	println("Sum:", sumResult)

	num := 9
	isEvenResult := isEven(num)
	println("Is", num, "even?", isEvenResult)

	n := 6
	fibonacciResult := fibonacci(n)
	println("Fibonacci at", n, ":", fibonacciResult)

	str := "hello"
	replacement := '*'
	replaceResult := replaceVowels(str, replacement)
	println("Replace vowels:", replaceResult)

	palindromeStr := "racecar"
	isPalindromeResult := isPalindrome(palindromeStr)
	println("Is", palindromeStr, "a palindrome?", isPalindromeResult)
}
