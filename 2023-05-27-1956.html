  2. 字符串反转

    package main
import (
"fmt"
"strings"
)
func main() {
str := "Hello, world!"
revStr := reverseString(str)
fmt.Println(revStr)
}
func reverseString(str string) string {
runes := []rune(str)
for i, j := 0, len(runes)-1; i < len(runes)/2; i, j = i+1, j-1 {
    runes[i], runes[j] = runes[j], runes[i]
}
return string(runes)
}

  3. 判断素数

    package main
import (
"fmt"
)
func main() {
fmt.Println(isPrime(2))
fmt.Println(isPrime(5))
fmt.Println(isPrime(7))
fmt.Println(isPrime(10))
}
func isPrime(num int) bool {
if num < 2 {
    return false
}
for i := 2; i <= num/2; i++ {
    if num%i == 0 {
        return false
    }
}
return true
}

  4. 求平均数

    package main
import (
"fmt"
)
func main() {
nums := []int{1, 2, 3, 4, 5}
avg := average(nums)
fmt.Println(avg)
}
func average(nums []int) float64 {
sum := 0
for _, num := range nums {
    sum += num
}
return float64(sum) / float64(len(nums))
}

  5. 斐波那契数列

    package main
import (
"fmt"
)
func main() {
for i := 0; i < 10; i++ {
    fmt.Print(fibonacci(i), " ")
}
}
func fibonacci(num int) int {
if num < 2 {
    return num
}
return fibonacci(num-1) + fibonacci(num-2)
}
