// 练习3：使用冒泡排序将一个整型数组按从小到大排序
arr := []int{5, 3, 8, 1, 2}
for i := 0; i < len(arr)-1; i++ {
    for j := 0; j < len(arr)-i-1; j++ {
        if arr[j] > arr[j+1] {
            arr[j], arr[j+1] = arr[j+1], arr[j]
        }
    }
}
fmt.Println(arr)

func factorial(n int) int { if n == 0 { return 1 } return n * factorial(n-1) }
