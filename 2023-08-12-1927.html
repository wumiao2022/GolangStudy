package main

import "fmt"

func main() {
	// 练习1：打印出Hello World!
	fmt.Println("Hello World!")

	// 练习2：定义一个字符串变量并赋值为"Go语言是最好的编程语言！"，然后打印出该字符串的长度
	str := "Go语言是最好的编程语言！"
	fmt.Println(len(str))

	// 练习3：使用for循环打印出1到10的所有整数
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}

	// 练习4：定义一个名为add的函数，接收两个整数参数，返回它们的和
	add := func(a, b int) int {
		return a + b
	}
	fmt.Println(add(3, 5))

	// 练习5：定义一个名为isEven的函数，接收一个整数参数，判断该数是否为偶数，是则返回true，否则返回false
	isEven := func(num int) bool {
		if num%2 == 0 {
			return true
		} else {
			return false
		}
	}
	fmt.Println(isEven(7))
}
