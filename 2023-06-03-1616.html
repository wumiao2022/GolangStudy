1. 删除切片中的重复元素

    func removeDuplicate(s []int) []int {
keys := make(map[int]bool)
ans := []int{}
for _, v := range s {
    if _, value := keys[v]; !value {
        keys[v] = true
        ans = append(ans, v)
    }
}
return ans
}

  2. 根据身高重建队列

    func reconstructQueue(people [][]int) [][]int {
n := len(people)
sort.Slice(people, func(i, j int) bool {
    a, b := people[i], people[j]
    if a[0] != b[0] {
        return a[0] > b[0]
    }
    return a[1] < b[1]
})


ans := make([][]int, n)
for _, person := range people {
    spaces := person[1] + 1
    for i := 0; i < n; i++ {
        if ans[i] == nil {
            spaces--
            if spaces == 0 {
                ans[i] = person
                break
            }
        }
    }
}
return ans
}

  3. 冒泡排序

    func BubbleSort(a []int) {
n := len(a)
for i := 0; i < n-1; i++ {
    for j := 0; j < n-1-i; j++ {
        if a[j] > a[j+1] {
            a[j], a[j+1] = a[j+1], a[j]
        }
    }
}
