package main

import "fmt"

func main() {
	// 题目：打印1到10之间的偶数
	for i := 1; i <= 10; i++ {
		if i%2 == 0 {
			fmt.Printf("%d ", i)
		}
	}
}

package main

import "fmt"

func main() {
	// 题目：计算数组中所有元素的平均值
	numbers := []int{3, 5, 2, 8, 1}

	sum := 0
	for _, num := range numbers {
		sum += num
	}

	average := float64(sum) / float64(len(numbers))
	fmt.Println("Average:", average)
}

package main

import "fmt"

func main() {
	// 题目：翻转字符串
	str := "Hello, World!"

	reversedStr := ""
	for i := len(str) - 1; i >= 0; i-- {
		reversedStr += string(str[i])
	}

	fmt.Println("Reversed String:", reversedStr)
}
