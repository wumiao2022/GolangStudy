  1. 判断一个数是否为偶数
func isEven(num int) bool {
    return num%2 == 0
}
  1. 求一个数组的和
func sum(arr []int) int {
    var result int
    for _, value := range arr {
        result += value
    }
    return result
}
  1. 输出乘法口诀表
func printTable() {
    for i := 1; i <= 9; i++ {
        for j := 1; j <= i; j++ {
            fmt.Printf("%d*%d=%d ", j, i, i*j)
        }
        fmt.Println()
    }
}
  1. 判断一个字符串是否为回文字符串
func isPalindrome(str string) bool {
    n := len(str)
    for i := 0; i < n/2; i++ {
        if str[i] != str[n-i-1] {
            return false
        }
    }
    return true
}
