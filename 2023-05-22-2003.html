1. 翻转字符串：
func reverseString(s string) string {
    var res string
    for i := len(s) - 1; i >= 0; i-- {
        res += string(s[i])
    }
    return res
}
  1. 求两个整数的最大公约数：
func gcd(a, b int) int {
    if a == 0 {
        return b
    }
    return gcd(b%a, a)
}
  1. 反转链表：
type ListNode struct {
    Val int
    Next *ListNode
}

func reverseListNode(head *ListNode) *ListNode {
    var prev *ListNode
    curr := head
    for curr != nil {
        next := curr.Next
        curr.Next = prev
        prev = curr
        curr = next
    }
    return prev
}
  1. 按照奇偶性重排数组：
func oddEvenSort(nums []int) []int {
    l, r := 0, len(nums)-1
    for l < r {
        if nums[l]%2 == 0 {
            nums[l], nums[r] = nums[r], nums[l]
            r--
        } else {
            l++
        }
    }
    return nums
}
  1. 实现一个括号匹配函数：
func isValid(s string) bool {
    stack := []byte{}
    for i := 0; i < len(s); i++ {
        if s[i] == '(' || s[i] == '{' || s[i] == '[' {
            stack = append(stack, s[i])
        } else {
            if len(stack) == 0 {
                return false
            }
            top := stack[len(stack)-1]
            if s[i] == ')' && top == '(' ||
                s[i] == '}' && top == '{' ||
                s[i] == ']' && top == '[' {
                stack = stack[:len(stack)-1]
            } else {
                return false
            }
        }
    }
    return len(stack) == 0
}
