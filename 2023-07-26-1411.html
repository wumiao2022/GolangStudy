package main

import "fmt"

func main() {
    // 练习1：打印出从1到10的数字
    for i := 1; i <= 10; i++ {
        fmt.Println(i)
    }

    // 练习2：打印出0到100之间的偶数
    for j := 0; j <= 100; j += 2 {
        fmt.Println(j)
    }

    // 练习3：计算1到100的和
    sum := 0
    for k := 1; k <= 100; k++ {
        sum += k
    }
    fmt.Println("Sum:", sum)

    // 练习4：判断一个数是否为质数
    num := 17
    isPrime := true
    for l := 2; l < num; l++ {
        if num%l == 0 {
            isPrime = false
            break
        }
    }
    if isPrime {
        fmt.Println(num, "is prime")
    } else {
        fmt.Println(num, "is not prime")
    }
}
