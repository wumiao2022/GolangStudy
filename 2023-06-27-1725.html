这是一个简单的Golang程序，可以生成一个1到100之间的随机整数，并询问用户猜数字，最终输出猜测次数。

package main

import (
	"fmt"
	"math/rand"
	"time"
)

func main() {
	rand.Seed(time.Now().UnixNano())
	target := rand.Intn(100) + 1
	count := 0
	var guess int
	for {
		count++
		fmt.Print("请输入你的猜测：")
		fmt.Scan(&guess)
		if guess == target {
			fmt.Println("恭喜你，猜对了！")
			fmt.Printf("你猜了%d次\n", count)
			break
		} else if guess < target {
			fmt.Println("猜小了")
		} else {
			fmt.Println("猜大了")
		}
	}
}
