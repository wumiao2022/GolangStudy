package main

import "fmt"

func main() {
    // 1. 打印出 "Hello, World!"
    fmt.Println("Hello, World!")

    // 2. 将两个整数相加并打印结果
    num1 := 10
    num2 := 5
    result := num1 + num2
    fmt.Println("Result:", result)

    // 3. 输出一个九九乘法表
    for i := 1; i <= 9; i++ {
        for j := 1; j <= i; j++ {
            fmt.Printf("%d * %d = %d\t", j, i, i*j)
        }
        fmt.Println()
    }

    // 4. 编写一个函数，接收一个正整数参数n，计算并返回n的阶乘
    fmt.Println(factorial(5))

    // 5. 实现冒泡排序算法对一个整数数组进行排序
    arr := []int{5, 2, 8, 1, 9}
    bubbleSort(arr)
    fmt.Println("Sorted array:", arr)
}

func factorial(n int) int {
    result := 1
    for i := 1; i <= n; i++ {
        result *= i
    }
    return result
}

func bubbleSort(arr []int) {
    n := len(arr)
    for i := 0; i < n-1; i++ {
        for j := 0; j < n-i-1; j++ {
            if arr[j] > arr[j+1] {
                arr[j], arr[j+1] = arr[j+1], arr[j]
            }
        }
    }
}
