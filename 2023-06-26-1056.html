    "fmt"
    "time"
)

func main() {
    now := time.Now()
    fmt.Println(now.Format("2006-01-02 15:04:05"))
}
  1. 判断一个数是否为质数：
package main

import "fmt"

func isPrime(n int) bool {
    if n <= 1 {
        return false
    }
    for i := 2; i <= n/2; i++ {
        if n%i == 0 {
            return false
        }
    }
    return true
}

func main() {
    num := 23
    if isPrime(num) {
        fmt.Printf("%d is a prime number.\n", num)
    } else {
        fmt.Printf("%d is not a prime number.\n", num)
    }
}
  1. 字符串反转：
package main

import (
    "fmt"
    "strings"
)

func reverse(s string) string {
    r := []rune(s)
    for i, j := 0, len(r)-1; i < len(r)/2; i, j = i+1, j-1 {
        r[i], r[j] = r[j], r[i]
    }
    return string(r)
}

func main() {
    str := "Hello, world!"
    fmt.Println(reverse(str))
    fmt.Println(strings.Repeat("-", len(str)))
}
  1. 生成斐波那契数列：
package main

import "fmt"

func fibonacci(n int) []int {
    f := make([]int, n)
    for i := 0; i < n; i++ {
        if i < 2 {
            f[i] = 1
        } else {
            f[i] = f[i-1] + f[i-2]
        }
    }
    return f
}

func main() {
    fmt.Println(fibonacci(10))
}
  1. 十进制转二进制：
package main

import (
    "fmt"
    "strconv"
)

func decToBin(n int) string {
    return strconv.FormatInt(int64(n), 2)
}

func main() {
    num := 123
    fmt.Println(decToBin(num))
}
