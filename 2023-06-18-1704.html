这是一个计算斐波那契数列的例子：

package main

import "fmt"

func fibonacci() func() int {
    a, b := 0, 1
    return func() int {
        a, b = b, a+b
        return a
    }
}

func main() {
    f := fibonacci()
    for i := 0; i < 10; i++ {
        fmt.Println(f())
    }
}

这是一个简单的HTTP服务器示例：

package main

import (
    "fmt"
    "net/http"
)

func handler(w http.ResponseWriter, r *http.Request) {
    fmt.Fprintf(w, "Hello, World!")
}

func main() {
    http.HandleFunc("/", handler)
    http.ListenAndServe(":8080", nil)
}

这是一个简单的命令行参数解析器：

package main

import (
    "flag"
    "fmt"
)

func main() {
    name := flag.String("name", "World", "Specify your name")
    flag.Parse()
    fmt.Printf("Hello, %s!\n", *name)
}
