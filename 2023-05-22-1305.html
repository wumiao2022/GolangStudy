  1. 使用for循环输出1到10的数字
package main

import "fmt"

func main() {
    fmt.Println("1到10的数字：")
    for i := 1; i <= 10; i++ {
        fmt.Println(i)
    }
}

  1. 使用if语句判断数字的奇偶性
package main

import "fmt"

func main() {
    num := 11
    if num % 2 == 0 {
        fmt.Println(num, "是偶数")
    } else {
        fmt.Println(num, "是奇数")
    }
}
  1. 使用switch语句判断一个数的等级
package main

import "fmt"

func main() {
    score := 85
    switch {
    case score >= 90:
        fmt.Println("优秀")
    case score >= 75:
        fmt.Println("良好")
    case score >= 60:
        fmt.Println("及格")
    default:
        fmt.Println("不及格")
    }
}

  1. 使用defer、recover和panic处理异常
package main

import "fmt"

func test() {
    defer func() {
        err := recover()
        if err != nil {
            fmt.Println("程序出现异常：", err)
        }
    }()
    panic("test函数出现异常")
}

func main() {
    test()
    fmt.Println("程序正常结束")
}
