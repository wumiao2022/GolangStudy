以下是一些练习实例代码：

  1. 打印Hello World

    package main
import "fmt"
func main() {
fmt.Println("Hello World!")
}

  2. 计算两个数的和

    package main
import "fmt"
func main() {
num1 := 10
num2 := 20
sum := num1 + num2
fmt.Println("The sum is:", sum)
}

  3. 判断一个数是否为素数

    package main
import "fmt"
func main() {
num := 17
isPrime := true
for i := 2; i < num; i++ {
    if num%i == 0 {
        isPrime = false
        break
    }
}
if isPrime {
    fmt.Println(num, "is a prime number.")
} else {
    fmt.Println(num, "is not a prime number.")
}
}

  4. 将字符串反转

    package main
import "fmt"
func main() {
str := "Hello World"
reversed := ""
for i := len(str) - 1; i >= 0; i-- {
    reversed += string(str[i])
}
fmt.Println("The reversed string is:", reversed)
}
