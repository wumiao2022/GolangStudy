以下是一个简单的Golang程序，用于检查给定的数字是否是质数：

package main

import "fmt"

func isPrime(number int) bool {
    if number <= 1 {
        return false
    }
    for i := 2; i*i <= number; i++ {
        if number%i == 0 {
            return false
        }
    }
    return true
}

func main() {
    fmt.Println(isPrime(7)) // true
    fmt.Println(isPrime(12)) // false
}
