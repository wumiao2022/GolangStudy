package main

import (
	"fmt"
)

func main() {
	// 练习1：输出"Hello, Golang!"
	fmt.Println("Hello, Golang!")

	// 练习2：定义一个整型变量x，并将其赋初值为10，然后将x的值加1，并输出结果
	var x int = 10
	x = x + 1
	fmt.Println(x)

	// 练习3：定义一个字符串变量name，并将其赋初值为"John"，然后将name的值改为"Tom"，并输出结果
	var name string = "John"
	name = "Tom"
	fmt.Println(name)

	// 练习4：定义一个切片a，其元素为1, 2, 3, 4, 5，并输出a的长度和容量
	a := []int{1, 2, 3, 4, 5}
	fmt.Println("Length of a:", len(a))
	fmt.Println("Capacity of a:", cap(a))

	// 练习5：定义一个map b，其键为字符串类型，值为整型类型，并初始化b，使其包含键值对{"apple": 1, "banana": 2, "orange": 3}，然后输出b的长度
	b := map[string]int{
		"apple":  1,
		"banana": 2,
		"orange": 3,
	}
	fmt.Println("Length of b:", len(b))
}
