package main

import (
	"fmt"
)

func main() {
	// 练习1：输出Hello World
	fmt.Println("Hello World")

	// 练习2：输出1-100之间的所有偶数
	for i := 1; i <= 100; i++ {
		if i%2 == 0 {
			fmt.Println(i)
		}
	}

	// 练习3：输入一个数字n，输出1到n的所有奇数
	var n int
	fmt.Println("请输入一个数字:")
	fmt.Scan(&n)
	for i := 1; i <= n; i++ {
		if i%2 != 0 {
			fmt.Println(i)
		}
	}

	// 练习4：输入一个数字n，输出n的阶乘
	var fac int = 1
	fmt.Println("请输入一个数字:")
	fmt.Scan(&n)
	for i := 1; i <= n; i++ {
		fac *= i
	}
	fmt.Printf("%d的阶乘为%d\n", n, fac)

	// 练习5：输入一个字符串，反转输出
	var str string
	fmt.Println("请输入一个字符串:")
	fmt.Scan(&str)
	for i := len(str) - 1; i >= 0; i-- {
		fmt.Printf("%c", str[i])
	}
	fmt.Println()

	// 练习6：输出一个九九乘法表
	for i := 1; i <= 9; i++ {
		for j := 1; j <= i; j++ {
			fmt.Printf("%d*%d=%d\t", j, i, j*i)
		}
		fmt.Println()
	}
}
