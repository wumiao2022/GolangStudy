以下是一个练习实例代码：

package main

import (
	"fmt"
)

func main() {
	// 1. 定义一个数组并打印出来
	arr := [5]int{1, 2, 3, 4, 5}
	fmt.Println(arr)

	// 2. 定义一个切片并打印出来
	slice := []int{6, 7, 8, 9, 10}
	fmt.Println(slice)

	// 3. 使用 for 循环遍历打印出数组和切片中的每个元素
	for i := 0; i < len(arr); i++ {
		fmt.Println(arr[i])
	}
	for i := 0; i < len(slice); i++ {
		fmt.Println(slice[i])
	}

	// 4. 使用 for range 循环遍历打印出数组和切片中的每个元素
	for _, elem := range arr {
		fmt.Println(elem)
	}
	for _, elem := range slice {
		fmt.Println(elem)
	}

	// 5. 定义一个 map 并打印出来
	m := map[string]int{"a": 1, "b": 2, "c": 3}
	fmt.Println(m)

	// 6. 使用 for range 循环遍历打印出 map 中的每个元素
	for key, value := range m {
		fmt.Printf("%s:%d ", key, value)
	}
	fmt.Println()
}
