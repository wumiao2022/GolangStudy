func main() { // 1. 打印Hello, World! fmt.Println(“Hello, World!”)

// 2. 计算两个整数的和
a := 5
b := 10
sum := a + b
fmt.Println("Sum:", sum)

// 3. 判断一个数是奇数还是偶数
num := 7
if num%2 == 0 {
    fmt.Println("Even")
} else {
    fmt.Println("Odd")
}

// 4. 遍历数组并打印元素
arr := []int{1, 2, 3, 4, 5}
for _, num := range arr {
    fmt.Println(num)
}

// 5. 使用for循环打印九九乘法表
for i := 1; i <= 9; i++ {
    for j := 1; j <= i; j++ {
        fmt.Printf("%d * %d = %d\t", j, i, i*j)
    }
    fmt.Println()
}

}
