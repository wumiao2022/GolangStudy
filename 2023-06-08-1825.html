这是一个简单的 Golang 多线程实现，其中默认创建了10个 goroutine 并发执行，每个 goroutine 执行1000次递增操作：

package main

import (
    "fmt"
    "sync"
)

func main() {
    var wg sync.WaitGroup
    counter := 0
    for i := 0; i < 10; i++ {
        wg.Add(1)
        go func() {
            defer wg.Done()
            for j := 0; j < 1000; j++ {
                counter++
            }
        }()
    }
    wg.Wait()
    fmt.Println("Counter:", counter)
}
