六、结构体与方法

package main

import "fmt"

type Animal struct {
    name string
    age  int
}

func (a Animal) sayHello() {
    fmt.Printf("I am %s, and I'm %d years old.\n", a.name, a.age)
}

func main() {
    cat := Animal{name: "Tom", age: 2}
    cat.sayHello()
}

七、并发与通道

package main

import (
    "fmt"
    "sync"
    "time"
)

func worker(id int, jobs <-chan int, results chan<- int) {
    for j := range jobs {
        fmt.Println("worker", id, "processing job", j)
        time.Sleep(time.Second)
        results <- j * 2
    }
}

func main() {
    jobs := make(chan int, 100)
    results := make(chan int, 100)

    for w := 1; w <= 3; w++ {
        go worker(w, jobs, results)
    }

    for j := 1; j <= 9; j++ {
        jobs <- j
    }
    close(jobs)

    var wg sync.WaitGroup
    wg.Add(1)
    go func() {
        defer wg.Done()
        for r := range results {
            fmt.Println("result:", r)
        }
    }()
    wg.Wait()
}
