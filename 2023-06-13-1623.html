  1. 流程控制
package main

import "fmt"

func main() {
	// if else
	age := 18
	if age >= 18 {
		fmt.Println("adult")
	} else {
		fmt.Println("child")
	}

	// switch case
	dayOfWeek := 3
	switch dayOfWeek {
	case 1:
		fmt.Println("Monday")
	case 2:
		fmt.Println("Tuesday")
	case 3:
		fmt.Println("Wednesday")
	default:
		fmt.Println("Error")
	}

	// for loop
	for i := 0; i < 5; i++ {
		fmt.Println(i)
	}

	// while loop
	j := 0
	for j < 5 {
		fmt.Println(j)
		j++
	}

	// range loop
	nums := []int{1, 2, 3, 4, 5}
	for index, value := range nums {
		fmt.Printf("index: %d, value: %d\n", index, value)
	}
}
  1. 数组和切片
package main

import "fmt"

func main() {
	// array
	var a [5]int
	a[0] = 1
	a[1] = 2
	a[2] = 3
	a[3] = 4
	a[4] = 5
	fmt.Println(a)

	// slice
	b := []int{1, 2, 3, 4, 5}
	fmt.Println(b)
	fmt.Println(b[1:3])
	fmt.Println(b[:3])
	fmt.Println(b[3:])
	fmt.Println(len(b))
	fmt.Println(cap(b))

	c := make([]int, 3, 5)
	fmt.Println(c)
	fmt.Println(len(c))
	fmt.Println(cap(c))
}
  1. 函数
package main

import "fmt"

func main() {
	result := add(3, 5)
	fmt.Println(result)

	result1, result2 := swap(2, 4)
	fmt.Println(result1, result2)
}

func add(a int, b int) int {
	return a + b
}

func swap(a int, b int) (int, int) {
	return b, a
}

}

  1. 接口
package main

import "fmt"

type Shape interface {
	Area() float64
}

type Circle struct {
	radius float64
}

func (c Circle) Area() float64 {
	return 3.14 * c.radius * c.radius
}

type Rectangle struct {
	width  float64
	height float64
}

func (r Rectangle) Area() float64 {
	return r.width * r.height
}

func main() {
	shapes := []Shape{
		Circle{2},
		Rectangle{3, 4},
	}
	for _, shape := range shapes {
		fmt.Println(shape.Area())
	}
}
  1. 并发
package main

import (
	"fmt"
	"time"
)

func main() {
	go printHello()
	time.Sleep(time.Second)

	names := []string{"Alice", "Bob", "Charlie"}
	for _, name := range names {
		go printName(name)
	}
	time.Sleep(time.Second)

	ch := make(chan string)
	go sendHello(ch)
	result := <-ch
	fmt.Println(result)
}

func printHello() {
	fmt.Println("Hello")
}

func printName(name string) {
	fmt.Printf("Hello, my name is %s\n", name)
}

func sendHello(ch chan string) {
	ch <- "Hello"
}
