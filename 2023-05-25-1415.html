package main

import (
    "fmt"
    "math/rand"
    "time"
)

func main() {
    rand.Seed(time.Now().UnixNano())

    // 生成随机数字数组
    nums := make([]int, 10)
    for i := range nums {
        nums[i] = rand.Intn(100)
    }

    // 打印原始数组
    fmt.Println("原始数组：", nums)

    // 统计数组中的元素个数
    count := len(nums)
    fmt.Println("元素个数：", count)

    // 计算数组中所有元素的和
    sum := 0
    for _, num := range nums {
        sum += num
    }
    fmt.Println("所有元素的和：", sum)

    // 查找数组中最大和最小值
    max, min := nums[0], nums[0]
    for _, num := range nums {
        if num > max {
            max = num
        } else if num < min {
            min = num
        }
    }
    fmt.Println("最大值：", max)
    fmt.Println("最小值：", min)

    // 反转数组
    for i := 0; i < count/2; i++ {
        nums[i], nums[count-1-i] = nums[count-1-i], nums[i]
    }
    fmt.Println("反转数组：", nums)

    // 删除指定位置的元素
    removeIndex := 4
    if removeIndex >= 0 && removeIndex < count {
        nums = append(nums[:removeIndex], nums[removeIndex+1:]...)
    }
    fmt.Println("删除指定位置的元素：", nums)

    // 在指定位置插入元素
    insertIndex := 2
    insertValue := 888
    if insertIndex >= 0 && insertIndex <= count {
        nums = append(nums[:insertIndex], append([]int{insertValue}, nums[insertIndex:]...)...)
    }
    fmt.Println("在指定位置插入元素：", nums)
}
