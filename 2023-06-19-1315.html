// Day1:

package main

import “fmt”

func main() { // 翻转字符串 str := “hello world” fmt.Println(reverseString(str))

// 寻找数组中的最大值
nums := []int{1, 5, 3, 8, 2}
fmt.Println(findMax(nums))

}

// 翻转字符串 func reverseString(str string) string { runes := []rune(str) for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 { runes[i], runes[j] = runes[j], runes[i] } return string(runes) }

// 寻找数组中的最大值 func findMax(nums []int) int { max := nums[0] for i := 1; i < len(nums); i++ { if nums[i] > max { max = nums[i] } } return max }

// 求数组的平均值 func average(nums []int) float64 { sum := 0 for _, num := range nums { sum += num } return float64(sum) / float64(len(nums)) }

// 删除字符串中的某个字符 func removeChar(str string, char byte) string { result := “” for i := 0; i < len(str); i++ { if str[i] != char { result += string(str[i]) } } return result }

// 判断两个字符串是否相等 func stringsEqual(str1, str2 string) bool { if len(str1) != len(str2) { return false } for i := 0; i < len(str1); i++ { if str1[i] != str2[i] { return false } } return true }

}

// 将切片转换为字符串 func sliceToString(nums []int, sep string) string { result := “” for i, num := range nums { result += fmt.Sprintf(“%d”, num) if i != len(nums)-1 { result += sep } } return result }

// 判断一个字符串是否是回文串 func isPalindrome(str string) bool { for i, j := 0, len(str)-1; i < j; i, j = i+1, j-1 { if str[i] != str[j] { return false } } return true }

package main

import “fmt”

func main() { // 将字符串转换为浮点数 str := “3.14” fmt.Println(parseFloat(str))

// 判断一个字符串是否是整数
str2 := "123"
fmt.Println(isInteger(str2))

}

// 将字符串转换为浮点数 func parseFloat(str string) float64 { result := 0.0 decimal := false factor := 1.0 for i := 0; i < len(str); i++ { if str[i] == ‘.’ { decimal = true continue } digit := int(str[i] - ‘0’) if digit < 0 || digit > 9 { break } if decimal { factor *= 0.1 result += factor * float64(digit) } else { result *= 10 result += float64(digit) } } return result }

// 判断一个字符串是否是整数 func isInteger(str string) bool { for i := 0; i < len(str); i++ { if str[i] < ‘0’ || str[i] > ‘9’ { return false } } return true }
