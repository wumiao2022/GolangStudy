  1. 判断字符串是否是回文字符串
func isPalindrome(s string) bool {
    for i := 0; i < len(s)/2; i++ {
        if s[i] != s[len(s)-i-1] {
            return false
        }
    }
    return true
}
  1. 打印出1~100中的质数
func isPrime(num int) bool {
    if num <= 1 {
        return false
    }
    for i := 2; i < num; i++ {
        if num%i == 0 {
            return false
        }
    }
    return true
}

func printPrimes() {
    for i := 1; i <= 100; i++ {
        if isPrime(i) {
            fmt.Printf("%d ", i)
        }
    }
}
  1. 反转整数
func reverseInt(num int) int {
    rev := 0
    for num != 0 {
        rev = rev*10 + num%10
        num /= 10
    }
    return rev
}
  1. 判断是否是闰年
func isLeapYear(year int) bool {
    if year%4 == 0 && (year%100 != 0 || year%400 == 0) {
        return true
    }
    return false
}
  1. 实现一个简单的计算器
func calculator(operation string, num1, num2 float64) float64 {
    switch operation {
    case "+":
        return num1 + num2
    case "-":
        return num1 - num2
    case "*":
        return num1 * num2
    case "/":
        return num1 / num2
    default:
        return 0
    }
}
