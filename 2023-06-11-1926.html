一、变量交换

fmt.Println(a, b)

二、判断数字正负

num := -10
if num >= 0 {
    fmt.Println("该数字是正数")
} else {
    fmt.Println("该数字是负数")
}

三、求和

sum := 0
for i := 1; i <= 100; i++ {
    sum += i
}
fmt.Println(sum)

四、求平均数

nums := []int{10, 20, 30, 40, 50}
sum := 0
for _, num := range nums {
    sum += num
}
average := float64(sum) / float64(len(nums))
fmt.Printf("平均数是: %.2f", average)

五、判断是否是闰年

year := 2021
if year%4 == 0 && year%100 != 0 || year%400 == 0 {
    fmt.Printf("%d年是闰年", year)
} else {
    fmt.Printf("%d年不是闰年", year)
}

六、判断一个数是不是素数

num := 17
isPrime := true
for i := 2; i < num; i++ {
    if num%i == 0 {
        isPrime = false
        break
    }
}
if isPrime {
    fmt.Printf("%d是素数", num)
} else {
    fmt.Printf("%d不是素数", num)
}

七、找出slice中的最大数和最小数

nums := []int{10, 20, 30, 40, 50}
max, min := nums[0], nums[0]
for _, num := range nums {
    if num > max {
        max = num
    }
    if num < min {
        min = num
    }
}
fmt.Printf("最大数是%d，最小数是%d", max, min)

八、复制一个slice

nums1 := []int{10, 20, 30, 40, 50}
nums2 := make([]int, len(nums1))
copy(nums2, nums1)
fmt.Println(nums2)

九、用递归计算斐波那契数列

func fib(n int) int {
    if n <= 1 {
        return n
    }
    return fib(n-1) + fib(n-2)
}

fmt.Println(fib(10))

十、并发执行两个任务

var wg sync.WaitGroup
wg.Add(2)
go func() {
    fmt.Println("任务1执行中...")
    time.Sleep(time.Second)
    fmt.Println("任务1执行完成")
    wg.Done()
}()
go func() {
    fmt.Println("任务2执行中...")
    time.Sleep(2 * time.Second)
    fmt.Println("任务2执行完成")
    wg.Done()
}()
wg.Wait()
