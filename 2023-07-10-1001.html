package main

import "fmt"

func main() {
	// Example 1: Print "Hello, World!" to the console
	fmt.Println("Hello, World!")

	// Example 2: Calculate the sum of two numbers and print the result
	a := 5
	b := 7
	sum := a + b
	fmt.Println("Sum:", sum)

	// Example 3: Print the first 10 natural numbers
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}

	// Example 4: Check if a number is even or odd
	number := 9
	if number%2 == 0 {
		fmt.Println("Even")
	} else {
		fmt.Println("Odd")
	}

	// Example 5: Calculate the factorial of a number
	factorial := 1
	n := 5
	for i := 1; i <= n; i++ {
		factorial *= i
	}
	fmt.Println("Factorial of", n, "is", factorial)
}

注意：以上示例代码只是为了提供一些练习的实例，并不涵盖所有Golang编程的方面。
