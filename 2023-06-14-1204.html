以下是一个简单的Golang练习实例代码：

package main

import "fmt"

func main() {
	// 1. 输出“Hello, world!”
	fmt.Println("Hello, world!")

	// 2. 计算并输出两个数的和
	a := 3
	b := 5
	sum := a + b
	fmt.Println("The sum of", a, "and", b, "is", sum)

	// 3. 判断一个数是否为偶数
	c := 4
	if c % 2 == 0 {
		fmt.Println(c, "is even")
	} else {
		fmt.Println(c, "is odd")
	}

	// 4. 计算并输出1到100中所有偶数的和
	sum_even := 0
	for i := 1; i <= 100; i++ {
		if i % 2 == 0 {
			sum_even += i
		}
	}
	fmt.Println("The sum of all even numbers from 1 to 100 is", sum_even)

	// 5. 计算斐波那契数列的前20项
	num1 := 0
	num2 := 1
	fmt.Print(num1, " ")
	fmt.Print(num2, " ")
	for i := 2; i < 20; i++ {
		num3 := num1 + num2
		fmt.Print(num3, " ")
		num1 = num2
		num2 = num3
	}
}

以上代码包括五个练习实例，分别是：

  1. 输出“Hello, world!”
  2. 计算并输出两个数的和
  3. 判断一个数是否为偶数
  4. 计算并输出1到100中所有偶数的和
  5. 计算斐波那契数列的前20项

希望这些练习能帮助你更深入地学习和掌握Golang。
