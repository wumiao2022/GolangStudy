package main

import (
	"fmt"
	"math/rand"
	"time"
)

func main() {
	rand.Seed(time.Now().UnixNano())

	// 随机生成一个10个元素的整数数组
	arr := make([]int, 10)
	for i := 0; i < len(arr); i++ {
		arr[i] = rand.Intn(100)
	}

	// 计算数组的和
	sum := 0
	for _, num := range arr {
		sum += num
	}

	// 打印数组和平均值
	avg := float64(sum) / float64(len(arr))
	fmt.Printf("数组的和为：%d\n", sum)
	fmt.Printf("数组的平均值为：%.2f\n", avg)
}
