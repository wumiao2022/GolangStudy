package main

import "fmt"

func main() {
	// 练习1：打印Hello, World!
	fmt.Println("Hello, World!")

	// 练习2：声明一个整数变量x，并赋值为10，打印出x的值
	x := 10
	fmt.Println(x)

	// 练习3：声明一个字符串变量name，并赋值为"John"，打印出name的值
	name := "John"
	fmt.Println(name)

	// 练习4：同时声明多个变量a, b, c，并分别赋值为3, 4.5, "Golang"，打印出它们的值
	a, b, c := 3, 4.5, "Golang"
	fmt.Println(a, b, c)

	// 练习5：声明一个常量pi，并赋值为3.14159，打印出pi的值
	const pi = 3.14159
	fmt.Println(pi)
}
