1. 反转字符串
package main 

import "fmt"

func main() {
    str := "hello world"
    rev := ""
    for _, s := range str {
        rev = string(s) + rev
    }
    fmt.Println(rev)
}
  1. 求两个数的最大公约数
package main 

import "fmt"

func gcd(a, b int) int {
    for b != 0 {
        a, b = b, a%b
    }
    return a
}

func main() {
    fmt.Println(gcd(12, 18))
}
  1. 求Fibonacci数列的第n项
package main 

import "fmt"

func fib(n int) int {
    a, b := 0, 1
    for i := 0; i < n; i++ {
        a, b = b, a+b
    }
    return a
}

func main() {
    fmt.Println(fib(10))
}
  1. 判断一个字符串是否是回文串
package main 

import "fmt"

func isPalindrome(s string) bool {
    for i := 0; i < len(s)/2; i++ {
        if s[i] != s[len(s)-1-i] {
            return false
        }
    }
    return true
}

func main() {
    fmt.Println(isPalindrome("racecar"))
}
  1. 计算一个整数数组中的平均值
package main 

import "fmt"

func average(nums []int) float64 {
    sum := 0
    for _, n := range nums {
        sum += n
    }
    return float64(sum) / float64(len(nums))
}

func main() {
    nums := []int{1, 2, 3, 4, 5}
    fmt.Println(average(nums))
}
