请参考以下练习实例代码：

  1. 打印出1到10的数字
for i := 1; i <= 10; i++ {
    fmt.Println(i)
}
  1. 判断一个数字是奇数还是偶数
if num % 2 == 0 {
    fmt.Println("这是一个偶数")
} else {
    fmt.Println("这是一个奇数")
}
  1. 打印出1到100之间的所有素数
for i := 2; i <= 100; i++ {
    flag := true
    for j := 2; j < i; j++ {
        if i % j == 0 {
            flag = false
            break
        }
    }
    if flag {
        fmt.Println(i)
    }
}
  1. 反转一个字符串
str := "hello world"
newStr := ""
for i := len(str)-1; i >= 0; i-- {
    newStr += string(str[i])
}
fmt.Println(newStr)
  1. 计算一个数组的平均数
arr := []int{1, 2, 3, 4, 5}
sum := 0
for _, value := range arr {
    sum += value
}
avg := sum / len(arr)
fmt.Println(avg)
