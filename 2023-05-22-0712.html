  1. 判断一个字符串是否为回文字符串
package main

import "fmt"

func main() {
	str := "racecar"
	l := len(str)
	for i := 0; i < l/2; i++ {
		if str[i] != str[l-1-i] {
			fmt.Printf("%v is not a palindrome.", str)
			return
		}
	}
	fmt.Printf("%v is a palindrome.", str)
}
  1. 查找一个字符串中出现次数最多的字符
package main

import (
	"fmt"
	"strings"
)

func main() {
	str := "beautiful"
	counts := make(map[rune]int)
	for _, r := range str {
		counts[r]++
	}
	var maxVal int
	var maxRune rune
	for r, count := range counts {
		if count > maxVal {
			maxVal = count
			maxRune = r
		}
	}
	fmt.Printf("The character '%c' appears %v times in \"%v\".", maxRune, maxVal, str)
}
  1. 删除一个字符串切片中特定的字符串
package main

import "fmt"

func main() {
	strs := []string{"apple", "banana", "orange", "kiwi", "grape", "pear"}
	target := "kiwi"
	for i, str := range strs {
		if str == target {
			strs = append(strs[:i], strs[i+1:]...)
			break
		}
	}
	fmt.Printf("After removing \"%v\": %v", target, strs)
}

  1. 反转一个字符串中的字符顺序
package main

import "fmt"

func main() {
	str := "hello world"
	s := []rune(str)
	l := len(s)
	for i := 0; i < l/2; i++ {
		s[i], s[l-1-i] = s[l-1-i], s[i]
	}
	fmt.Printf("Reverse of \"%v\" is \"%v\".", str, string(s))
}
