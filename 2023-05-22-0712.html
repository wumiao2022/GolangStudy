以下是一道练习实例代码，它实现了一个根据关键字过滤掉数组中不匹配项的函数。

package main

import "fmt"

// FilterArray takes an array and a filter string and returns a new array
// containing only elements that contain the filter string.
func FilterArray(arr []string, filter string) []string {
    var result []string

    for _, s := range arr {
        if contains(s, filter) {
            result = append(result, s)
        }
    }

    return result
}

// contains returns true if s contains the filter string.
func contains(s, filter string) bool {
    return len(filter) == 0 || (len(s) >= len(filter) && s[:len(filter)] == filter)
}

func main() {
    arr := []string{"apple", "banana", "cherry", "date", "eggfruit", "fig"}
    filter := "e"
    fmt.Println(FilterArray(arr, filter))
}

以上代码会过滤掉数组 arr 中不包含关键字 filter 的项，并返回包含关键字 filter 的项组成的新数组。输出为：

[apple date eggfruit]
