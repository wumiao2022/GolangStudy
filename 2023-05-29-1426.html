  1. 编写一个程序，输出1到100之间的所有奇数。
package main

import "fmt"

func main() {
    for i := 1; i <= 100; i++ {
        if i%2 == 1 {
            fmt.Printf("%d ", i)
        }
    }
}
  1. 编写一个函数，计算两个整数的和。
package main

import "fmt"

func sum(a, b int) int {
    return a + b
}

func main() {
    a, b := 10, 20
    fmt.Printf("sum of %d and %d is %d", a, b, sum(a, b))
}
  1. 编写一个程序，求斐波那契数列的第n项（n由用户输入）。
package main

import "fmt"

func fibonacci(n int) int {
    if n <= 1 {
        return n
    }
    return fibonacci(n-1) + fibonacci(n-2)
}

func main() {
    var n int
    fmt.Print("Enter the value of n: ")
    fmt.Scan(&n)
    fmt.Printf("Fibonacci number at position %d is %d", n, fibonacci(n))
}
  1. 编写一个程序，判断一个字符串是否是回文串。
package main

import (
    "fmt"
    "strings"
)

func isPalindrome(str string) bool {
    str = strings.ToLower(str)
    for i := 0; i < len(str)/2; i++ {
        if str[i] != str[len(str)-i-1] {
            return false
        }
    }
    return true
}

func main() {
    str := "level"
    if isPalindrome(str) {
        fmt.Printf("%s is a palindrome", str)
    } else {
        fmt.Printf("%s is not a palindrome", str)
    }
}
  1. 编写一个程序，交换两个整数变量的值。
package main

import "fmt"

func main() {
    a, b := 10, 20
    fmt.Printf("Before swapping a=%d b=%d\n", a, b)
    a, b = b, a // swapping
    fmt.Printf("After swapping a=%d b=%d", a, b)
}
