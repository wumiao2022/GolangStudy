package main

import "fmt"

func main() {
	// Exercise 1: Print "Hello, World!" to the console.
	fmt.Println("Hello, World!")

	// Exercise 2: Create a variable called "name" and assign your name to it. Then, print "Hello, [name]!" to the console.
	name := "John"
	fmt.Println("Hello, " + name + "!")

	// Exercise 3: Create a function called "add" that takes two integers as parameters and returns their sum. Print the result to the console.
	fmt.Println(add(3, 5))

	// Exercise 4: Create a loop that prints numbers from 1 to 10 (inclusive) to the console.
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}

	// Exercise 5: Create a struct called "Person" with fields "name" (string) and "age" (int). Create an instance of "Person" and print its values to the console.
	type Person struct {
		name string
		age  int
	}

	person := Person{
		name: "Alice",
		age:  25,
	}

	fmt.Println(person)
}

func add(a, b int) int {
	return a + b
}
