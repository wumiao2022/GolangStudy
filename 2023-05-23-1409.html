以下是一段打印出所有质数的代码：

package main

import "fmt"

func main() {
    for i := 2; i < 100; i++ {
        prime := true
        for j := 2; j < i; j++ {
            if i%j == 0 {
                prime = false
                break
            }
        }
        if prime {
            fmt.Println(i)
        }
    }
}
