1. 将一个字符串反转
func reverseString(str string) string {
    reversed := ""
    for _, s := range str {
        reversed = string(s) + reversed
    }
    return reversed
}
  1. 打印九九乘法表
func printMultiplicationTable() {
    for i := 1; i <= 9; i++ {
        for j := 1; j <= i; j++ {
            fmt.Printf("%d*%d=%d\t", j, i, i*j)
        }
        fmt.Println()
    }
}
  1. 求两个数的最大公约数
func gcd(a, b int) int {
    if b == 0 {
        return a
    }
    return gcd(b, a%b)
}
  1. 判断一个数是否为质数
func isPrime(n int) bool {
    if n <= 1 {
        return false
    }
    for i := 2; i <= int(math.Sqrt(float64(n))); i++ {
        if n%i == 0 {
            return false
        }
    }
    return true
}
  1. 检查一个字符串是否为回文字符串
func isPalindrome(str string) bool {
    for i := 0; i < len(str)/2; i++ {
        if str[i] != str[len(str)-i-1] {
            return false
        }
    }
    return true
}
