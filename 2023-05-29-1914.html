2. 判断一个数是否为质数
```go
package main

import "fmt"

func isPrime(n int) bool {
    if n <= 1 {
        return false
    }
    for i := 2; i*i <= n; i++ {
        if n%i == 0 {
            return false
        }
    }
    return true
}

func main() {
    n := 7
    if isPrime(n) {
        fmt.Println(n, "is a prime number.")
    } else {
        fmt.Println(n, "is not a prime number.")
    }
}

4. 计算一个字符串中每个字符出现的次数
```go
package main

import "fmt"

func countChars(str string) map[string]int {
    result := make(map[string]int)
    for _, ch := range str {
        result[string(ch)]++
    }
    return result
}

func main() {
    str := "hello world"
    charCounts := countChars(str)
    fmt.Println(charCounts)
}
