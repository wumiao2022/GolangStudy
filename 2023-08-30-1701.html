1. 实现一个简单的计算器程序，输入两个数字和一个运算符（+、-、*、/），输出计算结果。
package main

import (
	"fmt"
)

func main() {
	var num1, num2 float64
	var operator string

	fmt.Print("Enter first number: ")
	fmt.Scanln(&num1)

	fmt.Print("Enter second number: ")
	fmt.Scanln(&num2)

	fmt.Print("Enter operator (+, -, *, /): ")
	fmt.Scanln(&operator)

	result := 0.0

	switch operator {
	case "+":
		result = num1 + num2
	case "-":
		result = num1 - num2
	case "*":
		result = num1 * num2
	case "/":
		result = num1 / num2
	default:
		fmt.Println("Invalid operator")
		return
	}

	fmt.Printf("Result: %f\n", result)
}
  1. 实现一个函数，接收一个切片作为参数，返回一个逆序排列的切片。
package main

import (
	"fmt"
)

func reverseSlice(slice []int) []int {
	reversed := make([]int, len(slice))
	for i := 0; i < len(slice); i++ {
		reversed[i] = slice[len(slice)-1-i]
	}
	return reversed
}

func main() {
	numbers := []int{1, 2, 3, 4, 5}
	reversedNumbers := reverseSlice(numbers)
	fmt.Println(reversedNumbers)
}
  1. 实现一个函数，接收一个整数作为参数，返回该整数的阶乘。
package main

import (
	"fmt"
)

func factorial(n int) int {
	if n <= 1 {
		return 1
	}
	return n * factorial(n-1)
}

func main() {
	num := 5
	result := factorial(num)
	fmt.Printf("%d! = %d\n", num, result)
}
  1. 实现一个函数，接收一个字符串作为参数，返回统计该字符串中字母的个数（不区分大小写）。
package main

import (
	"fmt"
	"strings"
)

func countLetters(str string) int {
	count := 0
	for _, char := range strings.ToLower(str) {
		if char >= 'a' && char <= 'z' {
			count++
		}
	}
	return count
}

func main() {
	str := "Hello, World! 123"
	result := countLetters(str)
	fmt.Printf("Number of letters: %d\n", result)
}
  1. 实现一个函数，接收两个整数作为参数，返回它们的最大公约数。
package main

import (
	"fmt"
)

func gcd(a, b int) int {
	for b != 0 {
		a, b = b, a%b
	}
	return a
}

func main() {
	num1 := 12
	num2 := 18
	result := gcd(num1, num2)
	fmt.Printf("GCD of %d and %d is %d\n", num1, num2, result)
}
