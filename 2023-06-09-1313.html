func main() {

        fmt.Println(i) // print numbers from 0 to 4

    // Example 3:
    x := 3
    if x < 5 {
        fmt.Println("x is less than 5") // print "x is less than 5"
    }

    // Example 4:
    nums := []int{1, 2, 3, 4, 5}
    for i, num := range nums {
        fmt.Printf("nums[%d] = %d\n", i, num) // print each element of nums with its index
    }

    // Example 5:
    var sum int
    for i := 1; i <= 10; i++ {
        if i%2 == 0 {
            sum += i // add even numbers from 1 to 10
        }
    }
    fmt.Println(sum) // print sum of all even numbers from 1 to 10
}
