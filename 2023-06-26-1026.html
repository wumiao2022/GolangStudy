以下是一道简单的 Golang 练习题，实现一个函数，找出数组中的最大值和最小值：

func findMinMax(arr []int) (int, int) {
    min := arr[0]
    max := arr[0]
    for _, v := range arr {
        if v < min {
            min = v
        }
        if v > max {
            max = v
        }
    }
    return min, max
}

使用方法：

arr := []int{2, 4, 1, 8, 5, 3, 9}
min, max := findMinMax(arr)
fmt.Println("Min: ", min, "Max: ", max) // Min: 1 Max: 9
