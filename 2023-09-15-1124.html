package main

import "fmt"

func main() {
	// 练习1：打印从1到10的所有整数
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}

	// 练习2：计算并打印1到10的所有整数的平方值
	for i := 1; i <= 10; i++ {
		fmt.Println(i * i)
	}

	// 练习3：计算并打印1到10的所有整数的和
	sum := 0
	for i := 1; i <= 10; i++ {
		sum += i
	}
	fmt.Println(sum)

	// 练习4：使用for循环打印出一个等腰直角三角形
	for i := 1; i <= 5; i++ {
		for j := 1; j <= i; j++ {
			fmt.Print("*")
		}
		fmt.Println()
	}

	// 练习5：使用嵌套循环打印出一个九九乘法表
	for i := 1; i <= 9; i++ {
		for j := 1; j <= i; j++ {
			fmt.Printf("%d * %d = %d\t", j, i, j*i)
		}
		fmt.Println()
	}
}
