func main() { // 练习1：打印Hello, World! fmt.Println(“Hello, World!”)

// 练习2：判断一个数是否为偶数
num := 6
if num % 2 == 0 {
    fmt.Println("该数是偶数")
} else {
    fmt.Println("该数是奇数")
}

// 练习3：计算1到10的累加和
sum := 0
for i := 1; i <= 10; i++ {
    sum += i
}
fmt.Println("1到10的累加和为:", sum)

// 练习4：求一个整数切片的最大值
nums := []int{5, 2, 9, 1, 7}
max := nums[0]
for _, num := range nums {
    if num > max {
        max = num
    }
}
fmt.Println("整数切片的最大值为:", max)

}
