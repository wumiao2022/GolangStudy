以下是一些Golang的练习题代码：

  1. 打印出九九乘法表
package main

import "fmt"

func main() {
    for i := 1; i <= 9; i++ {
        for j := 1; j <= i; j++ {
            fmt.Printf("%d*%d=%d\t", j, i, i*j)
        }
        fmt.Println()
    }
}
  1. 给定一个字符串，判断是否为回文串
package main

import "fmt"

func isPalindrome(s string) bool {
    for i := 0; i < len(s)/2; i++ {
        if s[i] != s[len(s)-1-i] {
            return false
        }
    }
    return true
}

func main() {
    s1 := "level"
    s2 := "hello"
    fmt.Println(s1, "is a palindrome:", isPalindrome(s1))
    fmt.Println(s2, "is a palindrome:", isPalindrome(s2))
}
  1. 实现一个简单的计算器
package main

import "fmt"

func main() {
    var operator string
    var num1, num2 float64

    fmt.Print("Enter operator (+, -, *, /): ")
    fmt.Scanln(&operator)

    fmt.Print("Enter first number: ")
    fmt.Scanln(&num1)

    fmt.Print("Enter second number: ")
    fmt.Scanln(&num2)

    switch operator {
    case "+":
        fmt.Printf("%v + %v = %v", num1, num2, num1+num2)
    case "-":
        fmt.Printf("%v - %v = %v", num1, num2, num1-num2)
    case "*":
        fmt.Printf("%v * %v = %v", num1, num2, num1*num2)
    case "/":
        fmt.Printf("%v / %v = %v", num1, num2, num1/num2)
    default:
        fmt.Println("Invalid operator")
    }
}
