以下是今天的练习实例代码：

  1. 输出Hello, World!

    package main
import "fmt"
func main() {
fmt.Println("Hello, World!")
}

  2. 计算两个数的和并输出结果

    package main
import "fmt"
func main() {
a := 10
b := 20
sum := a + b
fmt.Println(sum)
}

  3. 判断一个数是否是偶数并输出结果

    package main
import "fmt"
func main() {
num := 4
if num%2 == 0 {
    fmt.Println(num, "is even")
} else {
    fmt.Println(num, "is odd")
}
}

  4. 使用for循环输出1到100之间的所有奇数

    package main
import "fmt"
func main() {
for i := 1; i <= 100; i++ {
    if i%2 != 0 {
        fmt.Println(i)
    }
}
}

  5. 输出斐波那契数列的前20项

    package main
import "fmt"
func main() {
a, b := 0, 1
for i := 0; i < 20; i++ {
    fmt.Println(a)
    a, b = b, a+b
}
}
