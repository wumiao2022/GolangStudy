这是一个Golang的结构体示例：

package main

import "fmt"

type person struct {
    name   string
    age    int
    gender string
}

func main() {
    var p1 person
    p1.name = "Alice"
    p1.age = 26
    p1.gender = "female"
    fmt.Println(p1)

    p2 := person{name: "Bob", age: 32, gender: "male"}
    fmt.Println(p2)

    p3 := person{"Charlie", 45, "male"}
    fmt.Println(p3)
}
