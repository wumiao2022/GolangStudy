这是一个Golang多线程计算并发的简单示例：

package main

import (
    "fmt"
    "sync"
)

func main() {
    // 定义一个等待组
    var wg sync.WaitGroup
    // 设置线程数
    for i:=0; i<10; i++ {
        // 每个线程加入等待组
        wg.Add(1)
        go func(n int) {
            defer wg.Done()
            // 循环和计算
            sum := 0
            for j:=0; j<100000000/n; j++ {
                sum += j * n
            }
            fmt.Println("sum", sum)
        }(i)
    }
    // 等待所有线程完成
    wg.Wait()
    fmt.Println("all done")
}
