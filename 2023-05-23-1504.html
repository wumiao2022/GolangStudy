  1. 判断一个数是否为素数

    func isPrime(num int) bool {
for i := 2; i <= int(math.Sqrt(float64(num))); i++ {
    if num%i == 0 {
        return false
    }
}
return num > 1
}

  2. 找出一个字符串中最长的不重复子串

    func longestSubstr(s string) string {
var res string
for i := 0; i < len(s); i++ {
    subMap := make(map[byte]bool)
    sub := ""
    for j := i; j < len(s); j++ {
        if subMap[s[j]] {
            break
        }
        subMap[s[j]] = true
        sub += string(s[j])
        if len(sub) > len(res) {
            res = sub
        }
    }
}
return res
}

  4. 计算Fibonacci数列的第n个数

    func fibonacci(n int) int {
if n == 0 {
    return 0
}
if n == 1 {
    return 1
}
return fibonacci(n-1) + fibonacci(n-2)
}
