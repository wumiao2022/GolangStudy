func main() { // 实例1：打印Hello, World! fmt.Println(“Hello, World!”)

// 实例2：求和函数
nums := []int{1, 2, 3, 4, 5}
sum := calculateSum(nums)
fmt.Println("Sum:", sum)

// 实例3：判断是否为素数
num := 17
if isPrime(num) {
    fmt.Println(num, "is prime.")
} else {
    fmt.Println(num, "is not prime.")
}

// 实例4：遍历切片
fruits := []string{"apple", "banana", "orange"}
for i, fruit := range fruits {
    fmt.Println("Index:", i, "Fruit:", fruit)
}

}

// 求和函数 func calculateSum(nums []int) int { sum := 0 for _, num := range nums { sum += num } return sum }

// 判断是否为素数 func isPrime(num int) bool { if num < 2 { return false } for i := 2; i*i <= num; i++ { if num%i == 0 { return false } } return true }
