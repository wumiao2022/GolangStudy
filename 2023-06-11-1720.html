  1. 使用递归函数计算斐波那契数列第n个数
package main

import "fmt"

func fib(n int) int {
    if n == 0 {
        return 0
    } else if n == 1 {
        return 1
    } else {
        return fib(n-1) + fib(n-2)
    }
}

func main() {
    fmt.Println(fib(10))
}

  1. 使用map存储学生姓名和成绩，并按照成绩从高到低进行排序
package main

import "fmt"

func sortMap(m map[string]int) map[string]int {
    keys := make([]string, 0, len(m))
    for k := range m {
        keys = append(keys, k)
    }
    sort.Slice(keys, func(i, j int) bool {
        return m[keys[i]] > m[keys[j]]
    })
    sortedMap := make(map[string]int)
    for _, k := range keys {
        sortedMap[k] = m[k]
    }
    return sortedMap
}

func main() {
    scores := map[string]int{
        "Alice": 85,
        "Bob":   77,
        "Cindy": 92,
        "David": 68,
    }
    sortedScores := sortMap(scores)
    fmt.Println(sortedScores)
}
