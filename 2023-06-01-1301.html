1. 反转字符串
func reverseString(s string) string {
    runes := []rune(s)
    for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 {
        runes[i], runes[j] = runes[j], runes[i]
    }
    return string(runes)
}
  1. 判断字符串是否为回文串
func isPalindrome(s string) bool {
    s = strings.ToLower(s)
    for i, j := 0, len(s)-1; i < j; i, j = i+1, j-1 {
        for i < j && !isAlphanumeric(s[i]) {
            i++
        }
        for i < j && !isAlphanumeric(s[j]) {
            j--
        }
        if s[i] != s[j] {
            return false
        }
    }
    return true
}

func isAlphanumeric(c byte) bool {
    return (c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z') || (c >= '0' && c <= '9')
}
  1. 将数组中的0移动到末尾
func moveZeroes(nums []int) {
    i := 0
    for _, num := range nums {
        if num != 0 {
            nums[i] = num
            i++
        }
    }
    for ; i < len(nums); i++ {
        nums[i] = 0
    }
}
  1. 判断一个数是否为完全平方数
func isPerfectSquare(num int) bool {
    if num < 1 {
        return false
    }
    if num == 1 {
        return true
    }
    left, right := 1, num/2
    for left <= right {
        mid := left + (right-left)/2
        if mid*mid == num {
            return true
        } else if mid*mid < num {
            left = mid + 1
        } else {
            right = mid - 1
        }
    }
    return false
}
  1. 判断一个数是否为质数
func isPrime(n int) bool {
    if n < 2 {
        return false
    }
    for i := 2; i*i <= n; i++ {
        if n%i == 0 {
            return false
        }
    }
    return true
}
