package main

import "fmt"

func main() {
	// 练习1：打印Hello, World!
	fmt.Println("Hello, World!")

	// 练习2：计算两个整数的和
	num1 := 10
	num2 := 5
	sum := num1 + num2
	fmt.Println("Sum:", sum)

	// 练习3：判断一个数是否为偶数
	num := 7
	if num%2 == 0 {
		fmt.Println(num, "is even")
	} else {
		fmt.Println(num, "is odd")
	}

	// 练习4：使用for循环打印1到10的数字
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}

	// 练习5：编写一个函数，接收一个字符串作为参数，返回字符串的长度和第一个字符
	str := "Hello, Golang!"
	length, firstChar := getLengthAndFirstChar(str)
	fmt.Println("Length:", length)
	fmt.Println("First Character:", firstChar)
}

func getLengthAndFirstChar(str string) (int, string) {
	return len(str), string(str[0])
}

PS：这是一个小练习集合，包括打印Hello, World!、计算两个整数的和、判断一个数是否为偶数、使用for循环打印1到10的数字以及编写一个函数获取字符串的长度和第一个字符。你可以逐个练习并运行代码来理解每个概念的使用。
