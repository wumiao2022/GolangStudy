完整代码如下：

package main

import "fmt"

type person struct {
    name string
    age int
}

func main() {
    people := []person{}
    for _, name := range []string{"Alice", "Bob", "Charlie"} {
        p := person{name: name, age: 0}
        switch name {
        case "Alice":
            p.age = 25
        case "Bob":
            p.age = 30
        case "Charlie":
            p.age = 35
        }
        people = append(people, p)
    }
    for _, p := range people {
        fmt.Printf("Name: %s, Age: %d\n", p.name, p.age)
    }
}
