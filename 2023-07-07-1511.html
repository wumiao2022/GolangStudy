package main

import "fmt"

func main() {
	// 实例1：交换两个变量的值
	a, b := 10, 20
	a, b = b, a
	fmt.Println(a, b)

	// 实例2：打印1到100之间的所有奇数
	for i := 1; i <= 100; i += 2 {
		fmt.Println(i)
	}

	// 实例3：计算两个数的最大公约数
	fmt.Println(gcd(20, 30))
}

func gcd(a, b int) int {
	if b == 0 {
		return a
	} else {
		return gcd(b, a%b)
	}
}
