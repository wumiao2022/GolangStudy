package main

import (
	"fmt"
	"time"
)

func main() {
	// 练习1：打印当前日期和时间
	fmt.Println(time.Now())

	// 练习2：判断一个数是否为偶数
	num := 6
	if num%2 == 0 {
		fmt.Println(num, "是偶数")
	} else {
		fmt.Println(num, "是奇数")
	}

	// 练习3：计算字符串的长度
	str := "Hello, Golang"
	fmt.Println("字符串长度:", len(str))

	// 练习4：使用 for 循环输出 1-10 的数字
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}

	// 练习5：实现一个函数，返回两个数的和
	fmt.Println("两数之和:", add(3, 5))
}

func add(a, b int) int {
	return a + b
}

注意：以上代码仅供参考，如果需要运行，请在本地的Go环境中进行。
