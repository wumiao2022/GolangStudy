package main

import "fmt"

func main() {
	// 1. 定义一个整型数组，并将数组元素逆序排列
	numbers := []int{1, 2, 3, 4, 5}
	reversedNumbers := make([]int, len(numbers))
	for i, j := 0, len(numbers)-1; i < len(numbers); i, j = i+1, j-1 {
		reversedNumbers[i] = numbers[j]
	}
	fmt.Println(reversedNumbers)

	// 2. 将字符串数组按字母顺序排序
	words := []string{"apple", "banana", "cherry", "date"}
	for i := 0; i < len(words)-1; i++ {
		for j := 0; j < len(words)-i-1; j++ {
			if words[j] > words[j+1] {
				words[j], words[j+1] = words[j+1], words[j]
			}
		}
	}
	fmt.Println(words)

	// 3. 计算斐波那契数列的前 n 个数
	n := 10
	fibNumbers := make([]int, n)
	fibNumbers[0] = 0
	fibNumbers[1] = 1
	for i := 2; i < n; i++ {
		fibNumbers[i] = fibNumbers[i-1] + fibNumbers[i-2]
	}
	fmt.Println(fibNumbers)
}
