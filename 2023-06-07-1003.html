以下是一个简单的Golang程序，用于将给定的二进制字符串转换为整数。

package main

import (
    "fmt"
    "strconv"
)

func main() {
    binaryStr := "111010"
    decimalInt, _ := strconv.ParseInt(binaryStr, 2, 64)
    fmt.Println(decimalInt)
}

以下是一个简单的Golang程序，用于将给定的字符串数组按字母顺序进行排序。

package main

import (
     "fmt"
     "sort"
)

func main() {
     strs := []string{"apple", "banana", "orange", "kiwi", "grape"}
     sort.Strings(strs)
     fmt.Println(strs)
}

以下是一个简单的Golang程序，用于为给定的切片查找最大值。

package main

import (
     "fmt"
     "sort"
)

func main() {
     nums := []int{3, 8, 2, 11, 5, 1}
     sort.Ints(nums)
     max := nums[len(nums) - 1]
     fmt.Println(max)
}
