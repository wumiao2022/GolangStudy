  1. 判断一个数是否为素数
package main

import (
    "fmt"
    "math"
)

func isPrime(n int) bool {
    if n <= 1 {
        return false
    }
    for i := 2; i <= int(math.Sqrt(float64(n))); i++ {
        if n%i == 0 {
            return false
        }
    }
    return true
}

func main() {
    var n int
    fmt.Print("请输入一个整数: ")
    fmt.Scanln(&n)
    if isPrime(n) {
        fmt.Println(n, "是素数")
    } else {
        fmt.Println(n, "不是素数")
    }
}
  1. 求解阶乘
package main

import (
    "fmt"
)

func factorial(n int) int {
    if n == 1 {
        return 1
    }
    return n * factorial(n-1)
}

func main() {
    var n int
    fmt.Print("请输入一个正整数: ")
    fmt.Scanln(&n)
    fmt.Printf("%d! = %d\n", n, factorial(n))
}
  1. 判断一个字符串是否为回文字符串
package main

import (
    "fmt"
    "strings"
)

func isPalindrome(str string) bool {
    var rev strings.Builder
    for i := len(str) - 1; i >= 0; i-- {
        rev.WriteByte(str[i])
    }
    return str == rev.String()
}

func main() {
    var str string
    fmt.Print("请输入一个字符串: ")
    fmt.Scanln(&str)
    if isPalindrome(str) {
        fmt.Println(str, "是回文字符串")
    } else {
        fmt.Println(str, "不是回文字符串")
    }
}
