以下是每日多练的练习实例代码：

  1. 打印斐波那契数列的前n项

    package main
import "fmt"
func main() {
var n int
fmt.Println("请输入n：")
fmt.Scanln(&n)
var a, b int = 0, 1
for i := 0; i < n; i++ {
    fmt.Printf("%d ", a)
    a, b = b, a+b
}
}

  2. 判断一个数是不是素数

    package main
import "fmt"
func main() {
var n int
fmt.Println("请输入一个正整数：")
fmt.Scanln(&n)
if n < 2 {
    fmt.Println(n, "不是素数")
    return
}
for i := 2; i <= n/2; i++ {
    if n%i == 0 {
        fmt.Println(n, "不是素数")
        return
    }
}
fmt.Println(n, "是素数")
}

  3. 冒泡排序

    package main
import "fmt"
func bubbleSort(arr []int) {
for i := 0; i < len(arr)-1; i++ {
    for j := 0; j < len(arr)-1-i; j++ {
        if arr[j] > arr[j+1] {
            arr[j], arr[j+1] = arr[j+1], arr[j]
        }
    }
}
}
func main() {
arr := []int{3, 2, 4, 1, 5}
bubbleSort(arr)
fmt.Println(arr)
}

  4. 计算一个字符串中每个字符出现的次数

    package main
import "fmt"
func main() {
str := "hello world"
m := make(map[rune]int)
for _, c := range str {
    m[c]++
}
for k, v := range m {
    fmt.Printf("%c：%d\n", k, v)
}
}

  5. 使用协程和通道实现多个goroutine间的数据传输

    package main
import "fmt"
func square(c chan int) {
for i := 0; i <= 9; i++ {
    c <- i * i
}
close(c)
}
func main() {
c := make(chan int)
go square(c)
for x := range c {
    fmt.Println(x)
}
}
