package main

import (
	"fmt"
)

func main() {
	// 练习1：打印Hello, World!
	fmt.Println("Hello, World!")

	// 练习2：声明一个整数变量并赋值为10，打印该变量的值
	num := 10
	fmt.Println(num)

	// 练习3：声明一个字符串变量并赋值为"Go programming"，打印该变量的值
	str := "Go programming"
	fmt.Println(str)

	// 练习4：声明一个布尔变量并给予初始值为true，打印该变量的值
	flag := true
	fmt.Println(flag)

	// 练习5：声明一个整型数组并初始化为包含1到5这5个元素，打印数组的值
	arr := [5]int{1, 2, 3, 4, 5}
	fmt.Println(arr)
}
