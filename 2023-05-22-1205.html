  1. 将字符串反转
func reverseString(s string) string {
    runes := []rune(s)
    for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 {
        runes[i], runes[j] = runes[j], runes[i]
    }
    return string(runes)
}
  1. 计算斐波那契数列
func fibonacci(n int) int {
    if n < 2 {
        return n
    }
    return fibonacci(n-1) + fibonacci(n-2)
}
  1. 判断一个数是否是质数
func isPrime(n int) bool {
    if n <= 1 {
        return false
    }
    for i := 2; i*i <= n; i++ {
        if n%i == 0 {
            return false
        }
    }
    return true
}
  1. 实现一个简单的栈
type Stack struct {
    items []int
}

func (s *Stack) Push(item int) {
    s.items = append(s.items, item)
}

func (s *Stack) Pop() int {
    if s.IsEmpty() {
        return -1
    }
    index := len(s.items) - 1
    removed := s.items[index]
    s.items = s.items[:index]
    return removed
}

func (s *Stack) IsEmpty() bool {
    return len(s.items) == 0
}

func (s *Stack) Peek() int {
    if s.IsEmpty() {
        return -1
    }
    return s.items[len(s.items)-1]
}
  1. 实现一个简单的链表
type Node struct {
    value int
    next  *Node
}

type LinkedList struct {
    head *Node
    tail *Node
}

func (ll *LinkedList) Add(value int) {
    node := &Node{value, nil}
    if ll.head == nil {
        ll.head = node
        ll.tail = node
    } else {
        ll.tail.next = node
        ll.tail = node
    }
}

func (ll *LinkedList) Contains(value int) bool {
    node := ll.head
    for node != nil {
        if node.value == value {
            return true
        }
        node = node.next
    }
    return false
}

func (ll *LinkedList) Remove(value int) {
    if ll.head == nil {
        return
    }
    if ll.head.value == value {
        ll.head = ll.head.next
        return
    }
    prev := ll.head
    node := ll.head.next
    for node != nil {
        if node.value == value {
            prev.next = node.next
            if node == ll.tail {
                ll.tail = prev
            }
            return
        }
        prev = node
        node = node.next
    }
}
