// 练习2：判断一个数是否为素数
var num int // 待判断的数
fmt.Scan(&num)
isPrime := true
for i := 2; i < num; i++ {
	if num%i == 0 {
		isPrime = false
		break
	}
}
if isPrime {
	fmt.Println(num, "是素数")
} else {
	fmt.Println(num, "不是素数")
}

// 练习3：统计一个字符串中字母、数字、空格和其他字符出现的次数
str := "Hello 123 !"
var letter, digit, space, other int
for _, v := range str {
	switch {
	case v >= 'a' && v <= 'z' || v >= 'A' && v <= 'Z':
		letter++
	case v >= '0' && v <= '9':
		digit++
	case v == ' ':
		space++
	default:
		other++
	}
}
fmt.Printf("字母：%d，数字：%d，空格：%d，其他：%d\n", letter, digit, space, other)

}
