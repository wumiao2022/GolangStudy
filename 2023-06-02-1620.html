1. 打印100以内所有奇数：
for i := 1; i <= 100; i += 2 {
    fmt.Printf("%d ", i)
}
  1. 计算0到100的和：
var sum int
for i := 0; i <= 100; i++ {
    sum += i
}
fmt.Println(sum)
  1. 打印乘法口诀表：
for i := 1; i <= 9; i++ {
    for j := 1; j <= i; j++ {
        fmt.Printf("%d*%d=%d\t", j, i, j*i)
    }
    fmt.Println()
}
  1. 判断一个数是否是质数：
func isPrime(n int) bool {
    if n <= 1 {
        return false
    }
    for i := 2; i <= int(math.Sqrt(float64(n))); i++ {
        if n%i == 0 {
            return false
        }
    }
    return true
}
  1. 求斐波那契数列的第n项：
func fibonacci(n int) int {
    if n == 0 || n == 1 {
        return n
    }
    return fibonacci(n-1) + fibonacci(n-2)
}
