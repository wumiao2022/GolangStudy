以下是一个实现用户登录功能的Golang代码示例：

package main

import (
	"fmt"
	"strings"
)

func main() {
	users := map[string]string{"admin": "password", "user1": "123456"}

	var username string
	var password string
	fmt.Print("请输入用户名：")
	fmt.Scanln(&username)
	fmt.Print("请输入密码：")
	fmt.Scanln(&password)

	passwordHash := strings.TrimSpace(users[username])
	if passwordHash == "" {
		fmt.Println("无效用户名")
		return
	}

	if passwordHash == password {
		fmt.Println("登录成功")
	} else {
		fmt.Println("密码错误")
	}
}
