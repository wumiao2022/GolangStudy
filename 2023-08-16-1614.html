package main

import "fmt"

func main() {
	// 1. 输出Hello, World!
	fmt.Println("Hello, World!")

	// 2. 计算1+1，并输出结果
	fmt.Println(1 + 1)

	// 3. 声明一个整数变量，并赋初值为10，输出变量的值
	var num int = 10
	fmt.Println(num)

	// 4. 使用for循环输出1到10的所有偶数
	for i := 1; i <= 10; i++ {
		if i%2 == 0 {
			fmt.Println(i)
		}
	}

	// 5. 声明一个切片，包含5个字符串元素，并使用for循环遍历输出所有元素
	strings := []string{"apple", "banana", "cherry", "date", "elderberry"}
	for _, str := range strings {
		fmt.Println(str)
	}
}
