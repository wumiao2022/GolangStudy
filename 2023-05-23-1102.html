题目：写一个函数，实现两个字符串的比较，如果两个字符串相等则返回true，否则返回false。

func CompareStrings(str1, str2 string) bool {
    if str1 == str2 {
        return true
    } else {
        return false
    }
}

题目：编写一个函数，统计一个字符串中每个字符出现的次数。

func CountCharacters(str string) map[rune]int {
    count := make(map[rune]int)
    for _, char := range str {
        count[char]++
    }
    return count
}

题目：写一个函数，统计一个字符串中单词的个数。

func CountWords(str string) int {
    count := 0
    for _, char := range str {
        if char == ' ' {
            count++
        }
    }
    return count + 1
}

题目：编写一个函数，将一个字符串进行反转。

func ReverseString(str string) string {
    runes := []rune(str)
    for i, j := 0, len(runes)-1; i < len(runes)/2; i, j = i+1, j-1 {
        runes[i], runes[j] = runes[j], runes[i]
    }
    return string(runes)
}

题目：编写一个函数，实现一个简单的冒泡排序算法对一个整型切片进行排序。

func BubbleSort(slice []int) []int {
    for i := 0; i < len(slice); i++ {
        for j := 0; j < len(slice)-1-i; j++ {
            if slice[j] > slice[j+1] {
                slice[j], slice[j+1] = slice[j+1], slice[j]
            }
        }
    }
    return slice
}

题目：编写一个函数，将一个字符串中的所有单词进行反转。

func ReverseWords(str string) string {
    words := strings.Fields(str)
    for i, j := 0, len(words)-1; i < j; i, j = i+1, j-1 {
        words[i], words[j] = words[j], words[i]
    }
    return strings.Join(words, " ")
}

题目：编写一个函数，判断一个整数是否是回文数。

func IsPalindrome(num int) bool {
    if num < 0 || (num%10 == 0 && num != 0) {
        return false
    }
    reverse := 0
    for num > reverse {
        reverse = reverse*10 + num%10
        num /= 10
    }
    return num == reverse || num == reverse/10
}
