package main

import "fmt"

func main() {
	// 实现一个函数，将一个字符串反转并返回
	fmt.Println(reverseString("Hello, Go!"))

	// 实现一个函数，接收一个整数数组，返回数组的最大值和最小值
	numbers := []int{1, 4, 2, 6, 3, 9, 5}
	max, min := findMaxAndMin(numbers)
	fmt.Println("Max:", max, "Min:", min)

	// 实现一个函数，接收一个字符串切片，返回切片中的所有元素的长度之和
	strings := []string{"apple", "banana", "cherry"}
	sum := calculateStringLength(strings)
	fmt.Println("Sum:", sum)
}

func reverseString(s string) string {
	r := []rune(s)
	for i, j := 0, len(r)-1; i < j; i, j = i+1, j-1 {
		r[i], r[j] = r[j], r[i]
	}
	return string(r)
}

func findMaxAndMin(numbers []int) (int, int) {
	max := numbers[0]
	min := numbers[0]
	for _, num := range numbers {
		if num > max {
			max = num
		}
		if num < min {
			min = num
		}
	}
	return max, min
}

func calculateStringLength(strings []string) int {
	sum := 0
	for _, str := range strings {
		sum += len(str)
	}
	return sum
}
