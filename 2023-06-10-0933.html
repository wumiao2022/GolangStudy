1. 找出切片中的最大值

    package main
import (
"fmt"
)
func main() {
nums := []int{2, 4, 6, 3, 7, 1}
max := nums[0]
for _, num := range nums {
    if num > max {
        max = num
    }
}
fmt.Println("Max number in slice is:", max)
}

  2. 将字符串中的字符反转

    package main
import (
"fmt"
)
func main() {
str := "hello world"
runes := []rune(str)
for i, j := 0, len(runes)-1; i < len(runes)/2; i, j = i+1, j-1 {
    runes[i], runes[j] = runes[j], runes[i]
}
fmt.Println("Reversed string is:", string(runes))
}

  3. 判断一个数是否为质数

    package main
import (
"fmt"
)
func main() {
num := 17
isPrime := true
for i := 2; i < num; i++ {
    if num%i == 0 {
        isPrime = false
        break
    }
}
if isPrime {
    fmt.Printf("%d is a prime number", num)
} else {
    fmt.Printf("%d is not a prime number", num)
}
}

  4. 将一个有序整数数组转换为二叉搜索树

    package main
import (
"fmt"
)
type TreeNode struct {
Val   int
Left  *TreeNode
Right *TreeNode
}
func main() {
nums := []int{-10, -3, 0, 5, 9}
root := sortedArrayToBST(nums)
fmt.Println("Inorder traversal of constructed BST is:")
printInorder(root)
}
func sortedArrayToBST(nums []int) *TreeNode {
if len(nums) == 0 {
    return nil
}
mid := len(nums) / 2
root := &TreeNode{Val: nums[mid]}
root.Left = sortedArrayToBST(nums[:mid])
root.Right = sortedArrayToBST(nums[mid+1:])
return root
}
func printInorder(node *TreeNode) {
if node == nil {
    return
}
printInorder(node.Left)
fmt.Printf("%d ", node.Val)
printInorder(node.Right)
}

  5. 用递归实现斐波那契数列

    package main
import (
"fmt"
)
func main() {
n := 10
for i := 0; i < n; i++ {
    fmt.Printf("%d ", fib(i))
}
}
func fib(n int) int {
if n < 2 {
    return n
}
return fib(n-1) + fib(n-2)
}
