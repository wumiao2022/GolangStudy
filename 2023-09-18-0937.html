package main

import "fmt"

func main() {
    // 练习1: 定义一个变量age，并赋值为18
    // 将age的值增加2，并打印出来
    age := 18
    age += 2
    fmt.Println(age)

    // 练习2: 定义一个字符串变量name，并赋值为"John"
    // 将name与另一个字符串" Doe" 相加，并打印出来
    name := "John"
    name += " Doe"
    fmt.Println(name)

    // 练习3: 定义一个布尔类型的变量isTrue，并赋值为true
    // 将isTrue取反，并打印出来
    isTrue := true
    isTrue = !isTrue
    fmt.Println(isTrue)
}
