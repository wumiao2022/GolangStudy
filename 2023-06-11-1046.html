package main

import "fmt"

func main() {
	// 练习题1：打印乘法口诀表
	for i := 1; i <= 9; i++ {
		for j := 1; j <= i; j++ {
			fmt.Printf("%d×%d=%d\t", j, i, j*i)
		}
		fmt.Println()
	}

	// 练习题2：找出数组中的最大值和最小值
	nums := []int{2, 5, 1, 7, 3, 9, 4, 6, 8}
	min, max := nums[0], nums[0]
	for _, num := range nums {
		if num < min {
			min = num
		}
		if num > max {
			max = num
		}
	}
	fmt.Printf("最小值为：%d，最大值为：%d\n", min, max)

	// 练习题3：判断一个字符串是否是回文字符串
	str := "level"
	length := len(str)
	for i := 0; i < length/2; i++ {
		if str[i] != str[length-1-i] {
			fmt.Println("不是回文字符串")
			return
		}
	}
	fmt.Println("是回文字符串")
}
