package main

import "fmt"

func main() {
	// 1. 打印 "Hello, World!"
	fmt.Println("Hello, World!")

	// 2. 声明一个变量并赋值为整数类型的数值，然后打印该变量的值
	num := 10
	fmt.Println(num)

	// 3. 声明一个浮点数类型的变量并赋值为小数，然后打印该变量的值
	floatNum := 3.14
	fmt.Println(floatNum)

	// 4. 声明一个布尔类型的变量并赋值为true或false，然后打印该变量的值
	isTrue := true
	fmt.Println(isTrue)

	// 5. 声明一个字符串类型的变量并赋值为一段文本，然后打印该变量的值
	message := "Hello, Golang!"
	fmt.Println(message)
}
