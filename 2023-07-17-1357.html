package main

import "fmt"

func main() {
	// 练习1: 打印1到10的所有整数
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}

	// 练习2: 计算1到100之间所有奇数的和
	sum := 0
	for i := 1; i <= 100; i += 2 {
		sum += i
	}
	fmt.Println("奇数的和为:", sum)

	// 练习3: 打印10以内的乘法表
	for i := 1; i <= 10; i++ {
		for j := 1; j <= 10; j++ {
			fmt.Printf("%d*%d=%d\t", i, j, i*j)
		}
		fmt.Println()
	}

	// 练习4: 判断一个数是否为素数
	num := 17
	isPrime := true
	for i := 2; i <= num/2; i++ {
		if num%i == 0 {
			isPrime = false
			break
		}
	}
	if isPrime {
		fmt.Println(num, "是素数")
	} else {
		fmt.Println(num, "不是素数")
	}

	// 练习5: 求斐波那契数列的前10个数字
	n := 10
	first := 0
	second := 1
	fmt.Print("斐波那契数列: ")
	for i := 1; i <= n; i++ {
		fmt.Print(first, " ")
		temp := first
		first = second
		second = temp + second
	}
	fmt.Println()
}

注意：上述代码仅为示例，仅包含基本的练习题。作为学习者，建议自行扩展和修改练习内容，以便更全面地掌握Golang的知识。
