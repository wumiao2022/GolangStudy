package main

import "fmt"

func main() {
	// 练习1：打印出 "Hello, Golang!" 字符串
	fmt.Println("Hello, Golang!")

	// 练习2：计算两个数的和并打印结果
	num1 := 10
	num2 := 5
	sum := num1 + num2
	fmt.Println("Sum:", sum)

	// 练习3：将一个整数数组按逆序排列并打印结果
	numbers := []int{5, 2, 7, 9, 1}
	for i := len(numbers) - 1; i >= 0; i-- {
		fmt.Println(numbers[i])
	}

	// 练习4：定义一个结构体表示一个人的信息，包括姓名和年龄，并打印结果
	type Person struct {
		Name string
		Age  int
	}
	person := Person{Name: "Alice", Age: 25}
	fmt.Println(person)
}
