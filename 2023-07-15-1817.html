package main

import "fmt"

func main() {
	// 练习1: 打印"Hello, World!"
	fmt.Println("Hello, World!")

	// 练习2: 计算两个数的和
	num1 := 10
	num2 := 5
	sum := num1 + num2
	fmt.Println("Sum:", sum)

	// 练习3: 判断一个数是奇数还是偶数
	number := 21
	if number%2 == 0 {
		fmt.Println(number, "is even")
	} else {
		fmt.Println(number, "is odd")
	}

	// 练习4: 遍历一个整数数组并打印所有元素
	numbers := []int{1, 2, 3, 4, 5}
	for _, num := range numbers {
		fmt.Println(num)
	}

	// 练习5: 根据身高和体重计算BMI指数
	height := 1.75 // 单位：米
	weight := 70   // 单位：千克
	bmi := weight / (height * height)
	fmt.Printf("BMI: %.2f\n", bmi)

	// 练习6: 交换两个变量的值
	a := 10
	b := 5
	fmt.Println("Before swapping: a =", a, "b =", b)
	a, b = b, a
	fmt.Println("After swapping: a =", a, "b =", b)
}
