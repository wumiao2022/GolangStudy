package main

import "fmt"

func main() {
    // Example 1
    fmt.Println("Hello, World!")

    // Example 2
    var x int = 5
    var y float64 = 3.14
    fmt.Println(x, y)

    // Example 3
    const message string = "This is a constant"
    fmt.Println(message)

    // Example 4
    var a, b int = 10, 20
    c := a + b
    fmt.Println(c)

    // Example 5
    var age int
    fmt.Println("Please enter your age:")
    fmt.Scan(&age)
    fmt.Printf("Your age is: %d", age)
}
