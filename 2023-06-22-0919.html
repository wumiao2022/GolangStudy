1. 实现一个简单的HTTP服务器，监听端口并返回”Hello, World!“消息。

    package main
import (
"fmt"
"net/http"
)
func main() {
http.HandleFunc("/", func(w http.ResponseWriter, r *http.Request) {
	fmt.Fprintf(w, "Hello, World!")
})
http.ListenAndServe(":8080", nil)
}

  2. 实现一个生成斐波那契数列的函数，返回一个切片。

    package main
import "fmt"
func Fibonacci(n int) []int {
if n < 2 {
	return []int{0, 1}
}
fib := make([]int, n+1)
fib[0] = 0
fib[1] = 1
for i := 2; i <= n; i++ {
	fib[i] = fib[i-1] + fib[i-2]
}
return fib[0:n]
}
func main() {
fmt.Println(Fibonacci(10))
}

  3. 实现一个计算阶乘的函数，返回计算结果。

    package main
import "fmt"
func Factorial(n int) int {
if n == 0 {
	return 1
}
return n * Factorial(n-1)
}
func main() {
fmt.Println(Factorial(5))
}

  4. 实现一个读取文件并打印文件内容的函数。

    package main
import (
"fmt"
"io/ioutil"
)
func ReadFile(filename string) {
content, err := ioutil.ReadFile(filename)
if err != nil {
	panic(err)
}
fmt.Print(string(content))
}
func main() {
ReadFile("test.txt")
}

  5. 实现一个判断是否是质数（素数）的函数。

    package main
import "fmt"
func IsPrime(n int) bool {
if n < 2 {
	return false
}
for i := 2; i < n; i++ {
	if n%i == 0 {
		return false
	}
}
return true
}
func main() {
fmt.Println(IsPrime(7))
fmt.Println(IsPrime(12))
}
