1. 打印九九乘法表
package main

import "fmt"

func main() {
    for i := 1; i <= 9; i++ {
        for j := 1; j <= i; j++ {
            fmt.Printf("%d*%d=%d\t", j, i, i*j)
        }
        fmt.Println()
    }
}
  1. 求两个数的最大公约数
package main

import "fmt"

func main() {
    var num1, num2 int
    fmt.Println("请输入两个整数：")
    fmt.Scanln(&num1, &num2)

    var gcd int
    for i := 1; i <= num1 && i <= num2; i++ {
        if num1%i == 0 && num2%i == 0 {
            gcd = i
        }
    }

    fmt.Printf("%d和%d的最大公约数为%d\n", num1, num2, gcd)
}
  1. 判断一个数是否为素数
package main

import "fmt"

func main() {
    var num int
    fmt.Println("请输入一个整数：")
    fmt.Scanln(&num)

    isPrime := true
    for i := 2; i < num; i++ {
        if num%i == 0 {
            isPrime = false
            break
        }
    }

    if isPrime {
        fmt.Printf("%d是素数\n", num)
    } else {
        fmt.Printf("%d不是素数\n", num)
    }
}
  1. 实现冒泡排序
package main

import "fmt"

func main() {
    nums := []int{3, 9, 1, 4, 8, 5, 2, 7, 6}

    for i := 0; i < len(nums)-1; i++ {
        for j := 0; j < len(nums)-1-i; j++ {
            if nums[j] > nums[j+1] {
                nums[j], nums[j+1] = nums[j+1], nums[j]
            }
        }
    }

    fmt.Println(nums)
}
  1. 实现快速排序
package main

import "fmt"

func quicksort(nums []int, left, right int) {
    if left >= right {
        return
    }

    i, j := left, right
    pivot := nums[left]

    for i < j {
        for i < j && nums[j] >= pivot {
            j--
        }
        nums[i] = nums[j]

        for i < j && nums[i] <= pivot {
            i++
        }
        nums[j] = nums[i]
    }

    nums[i] = pivot
    quicksort(nums, left, i-1)
    quicksort(nums, i+1, right)
}

func main() {
    nums := []int{3, 9, 1, 4, 8, 5, 2, 7, 6}

    quicksort(nums, 0, len(nums)-1)

    fmt.Println(nums)
}
