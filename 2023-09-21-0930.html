package main

import "fmt"

// 练习1：计算两个整数的和
func add(a, b int) int {
	return a + b
}

// 练习2：判断一个数是否为偶数
func isEven(n int) bool {
	return n%2 == 0
}

// 练习3：找出一个整数切片中的最大值
func findMax(nums []int) int {
	max := nums[0]
	for _, num := range nums {
		if num > max {
			max = num
		}
	}
	return max
}

// 练习4：将一个整数切片中的元素按照从大到小的顺序排序
func sortDescending(nums []int) []int {
	for i := 0; i < len(nums)-1; i++ {
		for j := i + 1; j < len(nums); j++ {
			if nums[i] < nums[j] {
				nums[i], nums[j] = nums[j], nums[i]
			}
		}
	}
	return nums
}

func main() {
	// 以下为测试代码
	fmt.Println(add(3, 5))    // 输出：8
	fmt.Println(isEven(6))    // 输出：true
	fmt.Println(findMax([]int{5, 2, 9, 1, 7}))    // 输出：9
	fmt.Println(sortDescending([]int{5, 2, 9, 1, 7}))    // 输出：[9 7 5 2 1]
}
