1. 找出切片中的最大值和最小值
package main

import "fmt"

func main() {
    slice := []int{5, 2, 7, 3, 9, 1}

    max, min := slice[0], slice[0]

    for _, v := range slice {
        if v > max {
            max = v
        }
        if v < min {
            min = v
        }
    }

    fmt.Printf("Max value: %v\nMin value: %v", max, min)
}

package main

import (
    "fmt"
    "net/http"
)

}

    http.ListenAndServe(":8080", nil)
}
  1. 实现一个函数，将一个字符串反转
package main

import "fmt"

func reverseString(s string) string {
    runes := []rune(s)

    for i, j := 0, len(runes)-1; i < len(runes)/2; i, j = i+1, j-1 {
        runes[i], runes[j] = runes[j], runes[i]
    }

    return string(runes)
}

func main() {
    s := "hello"

    fmt.Println(reverseString(s))
}
  1. 实现一个简单的计算器，支持加、减、乘、除四种运算
package main

import (
    "fmt"
    "strconv"
)

func calculate(num1 string, operator string, num2 string) float64 {
    f1, _ := strconv.ParseFloat(num1, 64)
    f2, _ := strconv.ParseFloat(num2, 64)

    switch operator {
    case "+":
        return f1 + f2
    case "-":
        return f1 - f2
    case "*":
        return f1 * f2
    case "/":
        return f1 / f2
    default:
        return 0
    }
}

func main() {
    result := calculate("5", "+", "3")
    fmt.Println(result)

    result = calculate("5", "-", "3")
    fmt.Println(result)

    result = calculate("5", "*", "3")
    fmt.Println(result)

    result = calculate("5", "/", "3")
    fmt.Println(result)
}
  1. 实现一个函数，判断一个字符串是否为回文
package main

import "fmt"

func isPalindrome(s string) bool {
    runes := []rune(s)

    for i, j := 0, len(runes)-1; i < len(runes)/2; i, j = i+1, j-1 {
        if runes[i] != runes[j] {
            return false
        }
    }

    return true
}

func main() {
    s := "racecar"

    if isPalindrome(s) {
        fmt.Println("Yes, it is a palindrome")
    } else {
        fmt.Println("No, it is not a palindrome")
    }
}
