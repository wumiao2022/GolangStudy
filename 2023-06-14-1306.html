以下是一个Golang程序，实现了一个简单的命令行计算器：

package main

import (
	"fmt"
	"os"
	"strconv"
)

func main() {
	if len(os.Args) != 4 {
		fmt.Println("Usage: <operator> <num1> <num2>")
		os.Exit(1)
	}

	operator := os.Args[1]
	num1, err1 := strconv.Atoi(os.Args[2])
	num2, err2 := strconv.Atoi(os.Args[3])

	if err1 != nil || err2 != nil {
		fmt.Println("Invalid input. Please input numbers only.")
		os.Exit(1)
	}

	result := 0

	switch operator {
	case "+":
		result = num1 + num2
	case "-":
		result = num1 - num2
	case "*":
		result = num1 * num2
	case "/":
		if num2 == 0 {
			fmt.Println("Cannot divide by zero.")
			os.Exit(1)
		}
		result = num1 / num2
	default:
		fmt.Println("Invalid operator. Please choose from +, -, *, /")
		os.Exit(1)
	}

	fmt.Printf("%d %s %d = %d\n", num1, operator, num2, result)
}
