以下是一个简单的Golang程序，可以打印出“Hello, World！”：

package main

import "fmt"

func main() {
    fmt.Println("Hello, World!")
}

以下是一个基本的函数，可以计算两个整数之和：

package main

import "fmt"

func add(x int, y int) int {
    return x + y
}

func main() {
    fmt.Println(add(3, 5))
}

以下是一个简单的Web服务器，可以在本地主机的端口8000上运行：

package main

import (
    "fmt"
    "net/http"
)

func handler(w http.ResponseWriter, r *http.Request) {
    fmt.Fprintf(w, "Hello, World!")
}

func main() {
    http.HandleFunc("/", handler)
    http.ListenAndServe(":8000", nil)
}
