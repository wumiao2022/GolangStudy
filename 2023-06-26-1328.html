  1. 给定一个数组，找出其中的最大元素
arr := [5]int{3, 8, 1, 9, 5}
max := arr[0]

for _, v := range arr {
    if v > max {
        max = v
    }
}

fmt.Println("最大元素为：", max)
  1. 判断一个数是否是素数
func isPrime(n int) bool {
    if n < 2 {
        return false
    }

    for i := 2; i*i <= n; i++ {
        if n%i == 0 {
            return false
        }
    }

    return true
}

fmt.Println(isPrime(7)) // true
fmt.Println(isPrime(10)) // false
