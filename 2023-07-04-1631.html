以下是一道Golang练习题：

package main

import (
	"fmt"
	"math/rand"
	"time"
)

func main() {
	rand.Seed(time.Now().UnixNano())

	// 生成一个长度为10的随机整数数组
	arr := make([]int, 10)
	for i := 0; i < len(arr); i++ {
		arr[i] = rand.Intn(100)
	}

	// 输出数组的内容
	fmt.Println("原始数组：", arr)

	// 使用冒泡排序对数组进行排序
	for i := 0; i < len(arr)-1; i++ {
		for j := 0; j < len(arr)-1-i; j++ {
			if arr[j] > arr[j+1] {
				arr[j], arr[j+1] = arr[j+1], arr[j]
			}
		}
	}

	// 输出排序后的数组
	fmt.Println("排序后的数组：", arr)
}

这个练习题使用冒泡排序算法对一个长度为10的随机整数数组进行排序，然后输出排序后的数组。你可以运行这段代码来查看结果。
