package main

import "fmt"

func main() {
	// 练习1：打印Hello, World!
	fmt.Println("Hello, World!")

	// 练习2：计算1到100的和
	sum := 0
	for i := 1; i <= 100; i++ {
		sum += i
	}
	fmt.Println("Sum:", sum)

	// 练习3：判断一个数是否为质数
	num := 17
	isPrime := true
	for i := 2; i < num; i++ {
		if num%i == 0 {
			isPrime = false
			break
		}
	}
	fmt.Println(num, "is prime:", isPrime)

	// 练习4：输出乘法表
	for i := 1; i <= 9; i++ {
		for j := 1; j <= i; j++ {
			fmt.Printf("%d*%d=%d ", j, i, j*i)
		}
		fmt.Println()
	}

	// 练习5：实现斐波那契数列
	fibonacci(10)
}

func fibonacci(n int) {
	if n <= 0 {
		fmt.Println("Invalid input")
		return
	}
	if n >= 1 {
		fmt.Print(0, " ")
	}
	if n >= 2 {
		fmt.Print(1, " ")
	}
	a, b := 0, 1
	for i := 3; i <= n; i++ {
		a, b = b, a+b
		fmt.Print(b, " ")
	}
	fmt.Println()
}
