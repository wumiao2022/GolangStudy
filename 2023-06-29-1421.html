在这里是一个Golang的每日练习代码示例：

package main

import "fmt"

func main() {
	// 练习1: 输出"Hello, World!"
	fmt.Println("Hello, World!")

	// 练习2: 输出1到10之间的偶数
	for i := 1; i <= 10; i++ {
		if i%2 == 0 {
			fmt.Println(i)
		}
	}

	// 练习3: 计算1到100之间所有整数的和
	sum := 0
	for i := 1; i <= 100; i++ {
		sum += i
	}
	fmt.Println("Sum:", sum)

	// 练习4: 打印九九乘法表
	for i := 1; i <= 9; i++ {
		for j := 1; j <= i; j++ {
			fmt.Printf("%d x %d = %d\t", j, i, i*j)
		}
		fmt.Println()
	}

	// 练习5: 判断一个数是否为素数
	num := 17
	isPrime := true
	for i := 2; i < num/2; i++ {
		if num%i == 0 {
			isPrime = false
			break
		}
	}
	if isPrime {
		fmt.Println(num, "is prime")
	} else {
		fmt.Println(num, "is not prime")
	}
}
