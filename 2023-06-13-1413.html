Sure! 这是一个简单的练习题，可以帮助你练习使用Golang中的条件语句if和循环语句for：

package main

import "fmt"

func main() {
    // 练习1：打印1-10之间的所有偶数
    for i := 1; i <= 10; i++ {
        if i%2 == 0 {
            fmt.Println(i)
        }
    }

    // 练习2：打印1-50之间的所有既是3的倍数又是5的倍数的数字
    for i := 1; i <= 50; i++ {
        if i%3 == 0 && i%5 == 0 {
            fmt.Println(i)
        }
    }
    
    // 练习3：计算一个整数数组的和
    nums := []int{1, 2, 3, 4, 5}
    sum := 0
    for _, num := range nums {
        sum += num
    }
    fmt.Println(sum)
}
