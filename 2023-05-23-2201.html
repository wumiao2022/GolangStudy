package main

import (
	"fmt"
	"math/rand"
	"time"
)

func main() {
	// 生成随机数种子
	rand.Seed(time.Now().UnixNano())

	// 生成 5 个随机数
	var nums [5]int
	for i := 0; i < 5; i++ {
		nums[i] = rand.Intn(10)
	}

	// 输出所有随机数
	fmt.Println("随机数：", nums)

	// 计算随机数平均值
	var sum int
	for _, num := range nums {
		sum += num
	}
	average := float64(sum) / float64(len(nums))

	// 输出平均值
	fmt.Printf("平均值：%.2f", average)
}

该程序生成 5 个随机数并计算它们的平均值。通过使用 rand.Seed 和 rand.Intn 生成随机数，并利用循环计算随机数总和和长度。最后输出这些随机数和计算的平均值。
