这是一个简单的Golang练习代码，用于展示如何使用for循环和切片来统计一个整数切片中的偶数个数。

package main

import "fmt"

func countEvenNumbers(arr []int) int {
    count := 0
    for _, num := range arr {
        if num%2 == 0 {
            count++
        }
    }
    return count
}

func main() {
    nums := []int{1,2,3,4,5,6,7,8,9}
    fmt.Println(countEvenNumbers(nums)) // Output: 4
}
