package main

import (
	"fmt"
	"time"
)

func main() {
	// 练习1：输出当前时间（格式：2006-01-02 15:04:05）
	currentTime := time.Now()
	fmt.Println(currentTime.Format("2006-01-02 15:04:05"))

	// 练习2：判断一个年份是否为闰年
	year := 2022
	if year%4 == 0 && (year%100 != 0 || year%400 == 0) {
		fmt.Println(year, "是闰年")
	} else {
		fmt.Println(year, "不是闰年")
	}

	// 练习3：使用for循环打印1到100的奇数
	for i := 1; i <= 100; i += 2 {
		fmt.Println(i)
	}

	// 练习4：使用switch语句判断一个数字的等级（90-100为优秀，80-89为良好，60-79为及格，0-59为不及格）
	score := 85
	switch {
	case score >= 90 && score <= 100:
		fmt.Println("优秀")
	case score >= 80 && score <= 89:
		fmt.Println("良好")
	case score >= 60 && score <= 79:
		fmt.Println("及格")
	default:
		fmt.Println("不及格")
	}

	// 练习5：使用递归函数计算斐波那契数列的第n项（n由用户输入）
	n := 10
	result := fibonacci(n)
	fmt.Println("第", n, "项的斐波那契数列为：", result)
}

func fibonacci(n int) int {
	if n <= 1 {
		return n
	}
	return fibonacci(n-1) + fibonacci(n-2)
}
