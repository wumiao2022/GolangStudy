package main

import (
	"fmt"
)

func main() {
	// 1. 打印"Hello, World!"到控制台

	fmt.Println("Hello, World!")

	// 2. 定义一个整数变量x，赋予它一个值，并将其打印到控制台

	var x int
	x = 10
	fmt.Println(x)

	// 3. 定义一个字符串变量name，赋予它一个名字，并将其打印到控制台

	var name string
	name = "John"
	fmt.Println(name)

	// 4. 定义一个包含5个整数的数组，并将其打印到控制台

	var arr [5]int
	arr = [5]int{1, 2, 3, 4, 5}
	fmt.Println(arr)

	// 5. 定义一个函数add，接收两个整数参数并返回它们的和，将结果打印到控制台

	add := func(a, b int) int {
		return a + b
	}

	result := add(3, 5)
	fmt.Println(result)
}
