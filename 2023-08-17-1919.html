package main

import "fmt"

func main() {
	// Exercise 1: Hello World
	fmt.Println("Hello, World!")

	// Exercise 2: Variables and Constants
	var name string = "John"
	age := 25
	const pi float64 = 3.14
	fmt.Println("Name:", name)
	fmt.Println("Age:", age)
	fmt.Println("PI:", pi)

	// Exercise 3: Arrays and Slices
	numbers := []int{1, 2, 3, 4, 5}
	fmt.Println("Numbers:", numbers)

	// Exercise 4: Control Flow - If Else
	score := 85
	if score >= 90 {
		fmt.Println("Grade: A")
	} else if score >= 80 {
		fmt.Println("Grade: B")
	} else {
		fmt.Println("Grade: C")
	}

	// Exercise 5: Loops
	for i := 1; i <= 5; i++ {
		fmt.Println(i)
	}

	// Exercise 6: Functions
	sum := add(5, 3)
	fmt.Println("Sum:", sum)
}

func add(a, b int) int {
	return a + b
}
