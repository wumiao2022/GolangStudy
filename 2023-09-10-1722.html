package main

import (
	"fmt"
)

func main() {
	// 1. 打印Hello, World!
	fmt.Println("Hello, World!")

	// 2. 定义两个整数变量并求它们的和
	num1 := 10
	num2 := 20
	sum := num1 + num2
	fmt.Println("Sum:", sum)

	// 3. 定义一个函数，接收一个整数参数并返回该参数的平方
	square := func(num int) int {
		return num * num
	}

	fmt.Println("Square of 5:", square(5))

	// 4. 使用for循环打印1到10之间的所有偶数
	fmt.Println("Even numbers:")
	for i := 1; i <= 10; i++ {
		if i%2 == 0 {
			fmt.Println(i)
		}
	}

	// 5. 定义一个结构体表示汽车，包含品牌和价格字段，并创建一个汽车实例输出其信息
	type Car struct {
		Brand  string
		Price  float64
	}

	car := Car{Brand: "BMW", Price: 50000.50}
	fmt.Println("Car brand:", car.Brand)
	fmt.Println("Car price:", car.Price)
}
