1. 计算字符串中字母出现的次数:
func countLetters(str string) map[rune]int {
    letterCount := make(map[rune]int)
    for _, letter := range str {
        letterCount[letter]++
    }
    return letterCount
}
  1. 判断一个切片中是否包含某个元素：
func contains(slice []string, item string) bool {
    for _, s := range slice {
        if s == item {
            return true
        }
    }
    return false
}
  1. 判断一个数是否为质数：
func isPrime(num int) bool {
    if num < 2 {
        return false
    }
    for i := 2; i*i <= num; i++ {
        if num%i == 0 {
            return false
        }
    }
    return true
}
  1. 找出一个切片中的最大值:
func max(slice []int) int {
    if len(slice) == 0 {
        return 0
    }

    maxVal := slice[0]
    for _, val := range slice {
        if val > maxVal {
            maxVal = val
        }
    }
    return maxVal
}
