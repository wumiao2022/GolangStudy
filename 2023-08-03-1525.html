package main

import (
	"fmt"
)

func main() {
	// 练习1：打印出1到10之间的所有偶数
	for i := 1; i <= 10; i++ {
		if i%2 == 0 {
			fmt.Printf("%v ", i)
		}
	}
	fmt.Println()

	// 练习2：计算1到100之间所有整数的和
	sum := 0
	for i := 1; i <= 100; i++ {
		sum += i
	}
	fmt.Println("Sum:", sum)

	// 练习3：根据用户输入的数字，判断是奇数还是偶数
	var num int
	fmt.Print("Enter a number: ")
	fmt.Scanln(&num)
	if num%2 == 0 {
		fmt.Println("Even")
	} else {
		fmt.Println("Odd")
	}
}
