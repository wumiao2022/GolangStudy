1. 编写一个函数，该函数接受一个整数数组并返回其中最大和最小值的差值。
func diffMinMax(nums []int) int {
    min, max := nums[0], nums[0]
    for i := 1; i < len(nums); i++ {
        if nums[i] < min {
            min = nums[i]
        }
        if nums[i] > max {
            max = nums[i]
        }
    }
    return max - min
}
  1. 编写一个函数，该函数将两个字符串拼接在一起，并确保相邻字符之间至少有一个空格。
func concatStrings(str1, str2 string) string {
    var result strings.Builder
    for i := 0; i < len(str1); i++ {
        result.WriteByte(str1[i])
        if i < len(str1)-1 || len(str2) > 0 {
            result.WriteByte(' ')
        }
    }
    for i := 0; i < len(str2); i++ {
        result.WriteByte(str2[i])
        if i < len(str2)-1 {
            result.WriteByte(' ')
        }
    }
    return result.String()
}
  1. 编写一个函数，该函数将给定的整数按照从小到大的顺序排序，并返回排序后的切片。
func sortInts(nums []int) []int {
    sort.Ints(nums)
    return nums
}
  1. 编写一个函数，该函数接受一个整数并返回它的二进制表示字符串。
func intToBinaryString(num int) string {
    return strconv.FormatInt(int64(num), 2)
}
  1. 编写一个函数，该函数接受一个字符串，找到其中出现次数最多的字符并返回它。
func mostFrequentChar(str string) byte {
    var freq [256]int
    maxFreq := 0
    var maxChar byte
    for i := 0; i < len(str); i++ {
        freq[str[i]]++
        if freq[str[i]] > maxFreq {
            maxFreq = freq[str[i]]
            maxChar = str[i]
        }
    }
    return maxChar
}
