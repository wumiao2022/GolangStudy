package main

import (
	"fmt"
	"time"
)

func main() {
	fmt.Println("Hello, Go learner!")
	fmt.Println("Today's exercise is to print the current time and date.")
	
	currentTime := time.Now()

	fmt.Println("Current time:", currentTime.Format("15:04:05"))
	fmt.Println("Current date:", currentTime.Format("2006-01-02"))
}
