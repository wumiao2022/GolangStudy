  1. 统计字符串中某个字符出现的次数
func countChar(str string, char byte) int {
    count := 0
    for _, c := range str {
        if byte(c) == char {
            count++
        }
    }
    return count
}

// 示例用法
count := countChar("hello world", 'l')
fmt.Println(count)  // 输出：3

func handleRequest(w http.ResponseWriter, r *http.Request) {
    fmt.Fprintf(w, "<h1>Hello, world!</h1>")
}

func main() {
    http.HandleFunc("/", handleRequest)
    http.ListenAndServe(":8080", nil)
}
  1. 实现一个栈
type Stack struct {
    items []int
}

func (s *Stack) Push(item int) {
    s.items = append(s.items, item)
}

func (s *Stack) Pop() int {
    if len(s.items) == 0 {
        return -1
    }
    item := s.items[len(s.items)-1]
    s.items = s.items[:len(s.items)-1]
    return item
}

// 示例用法
s := &Stack{}
s.Push(1)
s.Push(2)
s.Push(3)
fmt.Println(s.Pop())  // 输出：3
  1. 实现一个简单的计算器，支持加减乘除
func calculate(a, b int, operator byte) int {
    switch operator {
    case '+':
        return a + b
    case '-':
        return a - b
    case '*':
        return a * b
    case '/':
        return a / b
    default:
        return 0
    }
}

// 示例用法
result := calculate(2, 3, '+')
fmt.Println(result)  // 输出：5
