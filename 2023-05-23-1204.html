以下是一个实现快速排序算法的Golang代码示例：

package main

import "fmt"

func quickSort(arr []int) []int {
    if len(arr) <= 1 {
        return arr
    }
    pivot := arr[0]
    var left, right []int
    for i := 1; i < len(arr); i++ {
        if arr[i] < pivot {
            left = append(left, arr[i])
        } else {
            right = append(right, arr[i])
        }
    }
    left = quickSort(left)
    right = quickSort(right)
    return append(append(left, pivot), right...)
}

func main() {
    arr := []int{5, 9, 1, 4, 8, 3, 2, 7, 6}
    sortedArr := quickSort(arr)
    fmt.Println(sortedArr)
}
