以下是一个简单的常量和变量的练习代码：

package main

import (
	"fmt"
)

func main() {
	const Pi = 3.14     // 常量定义
	var name string     // 字符串变量
	var age int         // 整型变量
	var isMale bool     // 布尔型变量
	var weight float64  // 浮点型变量
	name = "John Smith" // 字符串变量赋值
	age = 25            // 整型变量赋值
	isMale = true       // 布尔型变量赋值
	weight = 68.5       // 浮点型变量赋值
	fmt.Println("Name:", name)
	fmt.Println("Age:", age)
	fmt.Println("Is Male:", isMale)
	fmt.Println("Weight (kg):", weight)
	fmt.Println("Pi:", Pi)
}
