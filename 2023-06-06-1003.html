1. 编写一个简单的HTTP服务器，返回Hello, world!：
package main

import (
    "io"
    "net/http"
)

func handler(w http.ResponseWriter, r *http.Request) {
    io.WriteString(w, "Hello, world!")
}

func main() {
    http.HandleFunc("/", handler)
    http.ListenAndServe(":8080", nil)
}
  1. 编写一个函数，将int类型转换为字符串：
package main

import (
    "strconv"
)

func intToStr(i int) string {
    return strconv.Itoa(i)
}

func main() {
    i := 123
    s := intToStr(i)
    println(s) // 输出"123"
}
  1. 编写一个函数，将字符串逆序输出：
package main

import "fmt"

func reverse(s string) string {
    r := []rune(s)
    length := len(r)
    for i := 0; i < length/2; i++ {
        j := length - 1 - i
        r[i], r[j] = r[j], r[i]
    }
    return string(r)
}

func main() {
    s := "hello, world!"
    fmt.Println(reverse(s)) // 输出"!dlrow ,olleh"
}
  1. 编写一个函数，将slice中的重复元素去除：
package main

import "fmt"

func removeDuplicates(s []int) []int {
    m := map[int]bool{}
    result := []int{}
    for _, v := range s {
        if _, ok := m[v]; !ok {
            m[v] = true
            result = append(result, v)
        }
    }
    return result
}

func main() {
    s := []int{1, 2, 2, 3, 3, 3}
    s = removeDuplicates(s)
    fmt.Println(s) // 输出[1 2 3]
}
  1. 编写一个函数，找出slice中的最大值和最小值：
package main

import "fmt"

func maxMin(s []int) (max, min int) {
    if len(s) == 0 {
        return
    }
    max, min = s[0], s[0]
    for _, v := range s {
        if v > max {
            max = v
        }
        if v < min {
            min = v
        }
    }
    return
}

func main() {
    s := []int{2, 4, 6, 1, 3, 8}
    max, min := maxMin(s)
    fmt.Printf("max=%d min=%d\n", max, min) // 输出"max=8 min=1"
}
