1. 判断一个数是否为偶数
package main

import "fmt"

func isEven(num int) bool {
    return num%2 == 0
}

func main() {
    num := 10
    if isEven(num) {
        fmt.Println(num, "is even")
    } else {
        fmt.Println(num, "is odd")
    }
}

  1. 字符串倒序输出
package main

import "fmt"

func reverse(s string) string {
    runes := []rune(s)
    for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 {
        runes[i], runes[j] = runes[j], runes[i]
    }
    return string(runes)
}

func main() {
    s := "Hello, world!"
    rev := reverse(s)
    fmt.Println(rev)
}
  1. 计算数组中所有元素的和
package main

import "fmt"

func sum(arr []int) int {
    total := 0
    for _, v := range arr {
        total += v
    }
    return total
}

func main() {
    arr := []int{1, 2, 3, 4, 5}
    total := sum(arr)
    fmt.Println(total)
}
  1. 将一个数组中的元素按照从大到小的顺序排列
package main

import "fmt"

func bubbleSort(arr []int) {
    n := len(arr)

    for i := 0; i < n; i++ {
        for j := 0; j < n-i-1; j++ {
            if arr[j] < arr[j+1] {
                arr[j], arr[j+1] = arr[j+1], arr[j]
            }
        }
    }
}

func main() {
    arr := []int{4, 2, 7, 1, 5, 3, 6}
    bubbleSort(arr)
    fmt.Println(arr)
}
