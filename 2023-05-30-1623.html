  1. 定义一个函数IsPrime(num int) bool，接受一个整数，判断是否是质数，是则返回true，否则返回false。

    func IsPrime(num int) bool {
if num < 2 {
    return false
}
for i := 2; i <= int(math.Sqrt(float64(num))); i++ {
    if num % i == 0 {
        return false
    }
}
return true
}

  2. 定义一个函数Rotate(s []int, k int)，接收一个整数切片和一个k值，将切片中的元素向右旋转k步。

    func Rotate(s []int, k int) {
l := len(s)
if l == 0 {
    return
}
k %= l
reverse(s, 0, l-1)
reverse(s, 0, k-1)
reverse(s, k, l-1)
}
func reverse(s []int, start, end int) {
for start < end {
    s[start], s[end] = s[end], s[start]
    start++
    end--
}
}

  3. 定义一个函数Reverse(s string) string，接收一个字符串，返回该字符串的反转形式。

    func Reverse(s string) string {
runes := []rune(s)
for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 {
    runes[i], runes[j] = runes[j], runes[i]
}
return string(runes)
}
