func main() { // 练习1：打印Hello, World! fmt.Println(“Hello, World!”)

// 练习2：计算两个整数的和并输出
num1 := 10
num2 := 20
sum := num1 + num2
fmt.Println("Sum:", sum)

// 练习3：判断一个数是否为偶数并输出结果
num := 17
isEven := (num % 2) == 0
fmt.Println("Is even:", isEven)

// 练习4：将一个字符串反转并输出结果
str := "Hello, Golang!"
reversedStr := ""
for i := len(str) - 1; i >= 0; i-- {
	reversedStr += string(str[i])
}
fmt.Println("Reversed string:", reversedStr)

// 练习5：使用切片将数组元素去重并输出结果
arr := []int{1, 2, 3, 4, 3, 2, 1}
uniqueArr := []int{}
seen := map[int]bool{}
for _, num := range arr {
	if !seen[num] {
		seen[num] = true
		uniqueArr = append(uniqueArr, num)
	}
}
fmt.Println("Unique array:", uniqueArr)

}
