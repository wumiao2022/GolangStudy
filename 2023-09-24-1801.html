package main

import "fmt"

func main() {
	// 练习1：打印 "Hello, World!"
	fmt.Println("Hello, World!")

	// 练习2：定义一个整数变量，并打印出它的值
	var num1 int = 5
	fmt.Println(num1)

	// 练习3：定义一个字符串变量，并打印出它的值
	var str1 string = "Golang"
	fmt.Println(str1)

	// 练习4：定义一个布尔变量，并打印出它的值
	var isTrue bool = false
	fmt.Println(isTrue)

	// 练习5：定义一个浮点数变量，并打印出它的值
	var floatNum float64 = 3.14
	fmt.Println(floatNum)

	// 练习6：定义一个整数数组，并打印出它的所有元素
	arr := []int{1, 2, 3, 4, 5}
	for _, num := range arr {
		fmt.Println(num)
	}

	// 练习7：定义一个函数，接收两个整数参数，并返回它们的和
	add := func(a, b int) int {
		return a + b
	}
	result := add(3, 4)
	fmt.Println(result)

	// 练习8：定义一个结构体类型，表示一个人的信息，并打印出其中的字段值
	type Person struct {
		Name string
		Age  int
	}
	person := Person{
		Name: "Tom",
		Age:  20,
	}
	fmt.Println(person.Name)
	fmt.Println(person.Age)
}
