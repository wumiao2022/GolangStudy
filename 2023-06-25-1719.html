// 练习3：判断一个正整数是不是质数 package main

import ( “fmt” “math” )

func isPrime(n int) bool { if n <= 1 { return false } if n == 2 { return true } for i := 2; i <= int(math.Sqrt(float64(n))); i++ { if n%i == 0 { return false } } return true }

func main() { n := 7 if isPrime(n) { fmt.Println(n, “是质数”) } else { fmt.Println(n, “不是质数”) } }

// 练习4：使用冒泡排序对数组进行从小到大的排序 package main

import “fmt”

func bubbleSort(arr []int) []int { for i := 0; i < len(arr)-1; i++ { for j := 0; j < len(arr)-i-1; j++ { if arr[j] > arr[j+1] { arr[j], arr[j+1] = arr[j+1], arr[j] } } } return arr }

func main() { arr := []int{9, 4, 10, 3, 7, 2, 1, 5, 8, 6} fmt.Println(bubbleSort(arr)) }

// 练习5：编写一个小程序，实现一个简单的计算器，支持加法、减法、乘法、除法四种运算 package main

import ( “fmt” “strconv” )

func add(a, b float64) float64 { return a + b }

func subtract(a, b float64) float64 { return a - b }

func multiply(a, b float64) float64 { return a * b }

func divide(a, b float64) float64 { return a / b }

func main() { var a, b float64 var operator string

fmt.Print("请输入第一个数：")
fmt.Scanln(&a)

fmt.Print("请输入第二个数：")
fmt.Scanln(&b)

fmt.Print("请选择运算符（+、-、*、/）：")
fmt.Scanln(&operator)

switch operator {
case "+":
    fmt.Println(a, "+", b, "=", add(a, b))
case "-":
    fmt.Println(a, "-", b, "=", subtract(a, b))
case "*":
    fmt.Println(a, "*", b, "=", multiply(a, b))
case "/":
    if b != 0 {
        fmt.Println(a, "/", b, "=", divide(a, b))
    } else {
        fmt.Println("除数不能为0")
    }
default:
    fmt.Println("输入的运算符有误")
}

}
