以下是一个练习实例代码：

package main

import (
	"fmt"
	"math/rand"
	"time"
)

func main() {
	rand.Seed(time.Now().UnixNano())

	// 生成随机数并猜测
	answer := rand.Intn(100) + 1
	fmt.Println("我想了一个1-100之间的整数，请你猜猜看。")
	var guess int
	for {
		fmt.Print("你的猜测是：")
		fmt.Scanln(&guess)
		if guess < answer {
			fmt.Println("你猜的数字太小了，再试试？")
		} else if guess > answer {
			fmt.Println("你猜的数字太大了，再试试？")
		} else {
			fmt.Println("恭喜你，猜中了！答案是", answer)
			break
		}
	}
}

这个程序会生成一个1-100之间的随机数，然后让用户进行猜测，根据用户每次提交的猜测结果，程序会提示用户是猜小了还是猜大了，直到用户猜中为止。
