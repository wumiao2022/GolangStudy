package main

import "fmt"

func main() {
	// 实例1：打印Hello, World!
	fmt.Println("Hello, World!")

	// 实例2：变量和常量
	var a int = 10
	b := 5
	const c = 3.14

	fmt.Println("a:", a)
	fmt.Println("b:", b)
	fmt.Println("c:", c)

	// 实例3：条件判断和循环
	for i := 1; i <= 10; i++ {
		if i%2 == 0 {
			fmt.Println(i, "is even")
		} else {
			fmt.Println(i, "is odd")
		}
	}

	// 实例4：切片和循环
	numbers := []int{1, 2, 3, 4, 5}
	sum := 0
	for _, num := range numbers {
		sum += num
	}
	fmt.Println("Sum of numbers:", sum)

	// 实例5：函数和返回值
	result := add(3, 5)
	fmt.Println("Result:", result)
}

func add(a, b int) int {
	return a + b
}
