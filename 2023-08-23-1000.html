package main

import "fmt"

func main() {
	// 练习1：打印出1到10的所有整数
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}

	// 练习2：创建一个包含5个元素的整数数组，并打印数组中的值
	arr := [5]int{1, 2, 3, 4, 5}
	for _, val := range arr {
		fmt.Println(val)
	}

	// 练习3：定义一个函数，接收两个整数参数，返回它们的和
	sum := add(3, 4)
	fmt.Println(sum)

	// 练习4：使用递归实现阶乘函数
	factorial := calcFactorial(5)
	fmt.Println(factorial)
}

func add(a, b int) int {
	return a + b
}

func calcFactorial(n int) int {
	if n == 0 {
		return 1
	}
	return n * calcFactorial(n-1)
}
