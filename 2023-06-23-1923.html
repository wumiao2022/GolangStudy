随机生成10个数字，分别表示每天的训练时间（单位小时），计算这10天中训练时间最长的一天是第几天。

package main

import (
    "fmt"
    "math/rand"
)

func main() {
    var maxHrs int
    var maxDay int

    // 随机生成10个数字
    hrs := []int{}
    for i := 0; i < 10; i++ {
        hrs = append(hrs, rand.Intn(12))
    }

    fmt.Println("每日训练时间：", hrs)

    // 找出最长训练时间
    for i, hr := range hrs {
        if hr > maxHrs {
            maxHrs = hr
            maxDay = i + 1
        }
    }

    fmt.Printf("第 %d 天是训练时间最长的一天，总共训练了 %d 小时。\n", maxDay, maxHrs)
}
