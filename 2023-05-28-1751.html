// 练习1：求1~100所有数字的和

package main

import “fmt”

func main() { sum := 0 for i := 1; i <= 100; i++ { sum += i } fmt.Println(sum) }

// 练习2：判断一个数字是否是素数

package main

import “fmt”

func isPrime(n int) bool { if n <= 1 { return false } for i := 2; i*i <= n; i++ { if n%i == 0 { return false } } return true }

func main() { fmt.Println(isPrime(11)) // true fmt.Println(isPrime(21)) // false }

// 练习3：将一组数字按照从小到大的顺序排序

package main

import “fmt”

func main() { nums := []int{5, 3, 8, 2, 6, 1} for i := 0; i < len(nums); i++ { for j := i + 1; j < len(nums); j++ { if nums[i] > nums[j] { nums[i], nums[j] = nums[j], nums[i] } } } fmt.Println(nums) // [1 2 3 5 6 8] }

// 练习4：判断一个字符串是否是回文字符串

package main

import “fmt”

func isPalindrome(str string) bool { for i := 0; i < len(str)/2; i++ { if str[i] != str[len(str)-1-i] { return false } } return true }

func main() { fmt.Println(isPalindrome(“level”)) // true fmt.Println(isPalindrome(“hello”)) // false }

// 练习5：统计一个字符串中每个字符出现的次数

package main

import “fmt”

func countChar(str string) map[string]int { res := make(map[string]int) for i := 0; i < len(str); i++ { char := string(str[i]) if _, ok := res[char]; ok { res[char]++ } else { res[char] = 1 } } return res }

func main() { fmt.Println(countChar(“hello”)) // map[e:1 h:1 l:2 o:1] }
