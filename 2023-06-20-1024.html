以下是一个简单的Golang练习代码实例：

package main

import “fmt”

func main() { // 1. 将输入的字符串反转 str := “Hello, world!” reverseStr := “” for _, s := range str { reverseStr = string(s) + reverseStr } fmt.Println(reverseStr)

// 2. 打印1~100之间的所有质数
for i := 2; i <= 100; i++ {
	flag := true
	for j := 2; j < i; j++ {
		if i%j == 0 {
			flag = false
			break
		}
	}
	if flag {
		fmt.Printf("%d ", i)
	}
}
fmt.Println()

// 3. 实现一个简单的计算器（支持加、减、乘、除运算）
a, b := 10, 5
op := "+"
result := 0
switch op {
case "+":
	result = a + b
case "-":
	result = a - b
case "*":
	result = a * b
case "/":
	result = a / b
}
fmt.Printf("%d %s %d = %d\n", a, op, b, result)

}

这个代码包含了三个练习实例：

  1. 将输入的字符串反转；
  2. 打印1~100之间的所有质数；
  3. 实现一个简单的计算器（支持加、减、乘、除运算）。
