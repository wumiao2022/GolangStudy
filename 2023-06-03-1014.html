以下是一个简单的Golang程序，可以输出”Hello, World!“：

package main

import "fmt"

func main() {
	fmt.Println("Hello, World!")
}

以下是一个简单的Golang程序，可以实现一个HTTP服务器，监听本地端口8888，返回”Hello, World!“：

package main

import (
	"fmt"
	"net/http"
)

func main() {
	http.HandleFunc("/", func(w http.ResponseWriter, r *http.Request) {
		fmt.Fprintf(w, "Hello, World!")
	})

	http.ListenAndServe(":8888", nil)
}
