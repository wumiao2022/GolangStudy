// 1. 打印九九乘法表 package main

import “fmt”

func main() { for i := 1; i <= 9; i++ { for j := 1; j <= i; j++ { fmt.Printf(“%d*%d=%d\t”, i, j, i*j) } fmt.Println() } }

// 2. 计算一个整数的阶乘 package main

import “fmt”

func factorial(n int) int { if n <= 1 { return 1 } return n * factorial(n-1) }

func main() { fmt.Println(factorial(5)) }

// 3. 判断一个字符串是否为回文字符串 package main

import “fmt”

func isPalindrome(s string) bool { for i := 0; i < len(s)/2; i++ { if s[i] != s[len(s)-i-1] { return false } } return true }

func main() { fmt.Println(isPalindrome(“racecar”)) }

// 4. 使用冒泡排序对整数数组进行排序 package main

import “fmt”

func bubbleSort(arr []int) { for i := 0; i < len(arr)-1; i++ { for j := i + 1; j < len(arr); j++ { if arr[i] > arr[j] { arr[i], arr[j] = arr[j], arr[i] } } } }

func main() { arr := []int{3, 2, 1, 5, 4} bubbleSort(arr) fmt.Println(arr) }

// 5. 实现一个简单的计算器 package main

import “fmt”

func main() { var num1, num2 float64 var operator string

fmt.Print("Enter the first number: ")
fmt.Scan(&num1)
fmt.Print("Enter the second number: ")
fmt.Scan(&num2)
fmt.Print("Enter the operator (+, -, *, /): ")
fmt.Scan(&operator)

var result float64
switch operator {
case "+":
    result = num1 + num2
case "-":
    result = num1 - num2
case "*":
    result = num1 * num2
case "/":
    result = num1 / num2
default:
    fmt.Println("Invalid operator")
    return
}

fmt.Println("Result:", result)

}
