以下是一个练习示例代码：

package main

import "fmt"

func main() {
    // 1. 反转字符串
    str := "hello"
    bytes := []byte(str)
    for i := 0; i < len(bytes)/2; i++ {
        j := len(bytes) - i - 1
        bytes[i], bytes[j] = bytes[j], bytes[i]
    }
    fmt.Println(string(bytes))

    // 2. 求数组中的最大值
    arr := []int{3, 1, 4, 1, 5, 9, 2, 6, 5, 3, 5}
    max := arr[0]
    for _, num := range arr {
        if num > max {
            max = num
        }
    }
    fmt.Println(max)

    // 3. 判断一个数是否为素数
    num := 7
    isPrime := true
    for i := 2; i < num; i++ {
        if num%i == 0 {
            isPrime = false
            break
        }
    }
    if isPrime {
        fmt.Println(num, "是素数")
    } else {
        fmt.Println(num, "不是素数")
    }
}

这个练习包括了三个小任务：

  1. 将字符串反转输出。
  2. 在一个数组中找到最大值输出。
  3. 判断一个数是否为素数输出。
