  2. 写一个函数，将一个字符串中的大写字母转换成小写字母，将小写字母转换成大写字母

    func ConvertCase(str string) string {
// 定义一个空字符串变量
var result string
// 遍历字符串
for _, char := range str {
    // 判断字符是否为大写字母
    if unicode.IsUpper(char) {
        // 将大写字母转换成小写字母，并添加到结果中
        result += string(unicode.ToLower(char))
    } else if unicode.IsLower(char) { // 判断字符是否为小写字母
        // 将小写字母转换成大写字母，并添加到结果中
        result += string(unicode.ToUpper(char))
    } else { // 非字母字符不变，直接添加到结果中
        result += string(char)
    }
}
return result
}

    func Factorial(num int) int {
if num == 0 {
    return 1
} else {
    return num * Factorial(num-1)
}
}

  4. 写一个函数，计算一个字符串中指定字符的个数

    func CountChar(str string, char byte) int {
count := 0
// 遍历字符串
for i := 0; i < len(str); i++ {
    // 判断当前字符是否和指定字符相等
    if str[i] == char {
        count++
    }
}
return count
}

  5. 写一个函数，判断一个字符串是否为回文字符串

    func IsPalindrome(str string) bool {
// 将字符串转换成小写字母，并去除空格
str = strings.ReplaceAll(strings.ToLower(str), " ", "")
// 遍历字符串
for i := 0; i < len(str)/2; i++ {
    // 判断左右两端的字符是否相等
    if str[i] != str[len(str)-1-i] {
        return false
    }
}
return true
}
