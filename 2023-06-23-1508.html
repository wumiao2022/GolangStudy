很好！这是一个打印斐波那契数列的Go语言程序，你可以用它来进行练习:

package main

import "fmt"

func main() {
    var n int
    fmt.Print("Enter the number of terms: ")
    fmt.Scan(&n)

    fn := make([]int, n)
    fn[0], fn[1] = 0, 1

    for i := 2; i < n; i++ {
        fn[i] = fn[i-1] + fn[i-2]
    }

    fmt.Println("Fibonacci series:")
    for i := 0; i < n; i++ {
        fmt.Print(fn[i], " ")
    }
}
