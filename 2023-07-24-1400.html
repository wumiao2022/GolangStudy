package main

import "fmt"

func main() {
	// 练习1：打印Hello, World!
	fmt.Println("Hello, World!")

	// 练习2：声明一个整数变量，并打印其值
	var num int
	fmt.Println(num)

	// 练习3：给变量赋值，并打印其值
	num = 10
	fmt.Println(num)

	// 练习4：声明一个字符串变量，并打印其值
	var str string
	fmt.Println(str)

	// 练习5：给变量赋值，并打印其值
	str = "Golang"
	fmt.Println(str)

	// 练习6：声明一个浮点数变量，并打印其值
	var floatNum float64
	fmt.Println(floatNum)

	// 练习7：给变量赋值，并打印其值
	floatNum = 3.14
	fmt.Println(floatNum)
}
