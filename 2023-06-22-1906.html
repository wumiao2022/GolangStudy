这是一个简单的Golang练习代码：

package main

import "fmt"

func main() {
    // 1. 输出Hello, World!
    fmt.Println("Hello, World!")

    // 2. 输出数字1-10中的偶数
    for i := 1; i <= 10; i++ {
        if i%2 == 0 {
            fmt.Println(i)
        }
    }

    // 3. 计算1-100中的整数和
    sum := 0
    for i := 1; i <= 100; i++ {
        sum += i
    }
    fmt.Println("1-100中的整数和：", sum)

    // 4. 判断一个字符串是否是回文串
    str := "level"
    isPalindrome := true
    for i := 0; i < len(str)/2; i++ {
        if str[i] != str[len(str)-i-1] {
            isPalindrome = false
            break
        }
    }
    if isPalindrome {
        fmt.Println(str, "是回文串")
    } else {
        fmt.Println(str, "不是回文串")
    }
}
