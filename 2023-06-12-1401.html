以下是Golang的练习实例代码，祝你练习顺利！

  1. 打印”Hello, World!“到控制台

    package main
import "fmt"
func main() {
fmt.Println("Hello, World!")
}

  2. 计算1到10的和并打印结果

    package main
import "fmt"
func main() {
sum := 0
for i := 1; i <= 10; i++ {
sum += i
}
fmt.Println(sum)
}

  3. 判断一个字符串是否为回文串

    package main
import "fmt"
func main() {
str := "racecar"
var reversed string
for i := len(str) - 1; i >= 0; i-- {
reversed += string(str[i])
}
if str == reversed {
fmt.Println("是回文串")
} else {
fmt.Println("不是回文串")
}
}

  4. 找出一个切片中的最大值

    package main
import "fmt"
func main() {
slice := []int{1, 8, 3, 9, 2}
max := slice[0]
for _, num := range slice {
if num > max {
  max = num
}
}
fmt.Println(max)
}

  5. 计算n的阶乘

    package main
import "fmt"
func main() {
n := 5
factorial := 1
for i := 1; i <= n; i++ {
factorial *= i
}
fmt.Println(factorial)
}
