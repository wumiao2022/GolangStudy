func main() { // 1. 打印Hello, World! fmt.Println(“Hello, World!”)

// 2. 计算两个数的和并打印结果
num1 := 10
num2 := 5
sum := num1 + num2
fmt.Println("Sum:", sum)

// 3. 判断一个数是否为奇数，并打印结果
number := 7
if number%2 == 0 {
	fmt.Println("Number is even")
} else {
	fmt.Println("Number is odd")
}

// 4. 使用for循环打印1至10的数字
for i := 1; i <= 10; i++ {
	fmt.Println(i)
}

// 5. 创建一个切片，并在其中添加元素后打印出来
slice := []int{1, 2, 3, 4, 5}
slice = append(slice, 6)
fmt.Println(slice)

// 6. 使用map存储学生的姓名和分数，并打印出来
studentScores := map[string]int{
	"Tom":   90,
	"Jerry": 85,
	"Alice": 95,
}
for name, score := range studentScores {
	fmt.Println(name, ":", score)
}

}
