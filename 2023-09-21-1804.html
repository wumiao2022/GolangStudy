package main

import "fmt"

func main() {
	// 1. 输出Hello, World!
	fmt.Println("Hello, World!")

	// 2. 定义一个整数变量num并赋值为10，输出num的值
	num := 10
	fmt.Println(num)

	// 3. 定义一个字符串变量name并赋值为"John"，输出name的值
	name := "John"
	fmt.Println(name)

	// 4. 定义一个浮点数变量pi并赋值为3.14，输出pi的值
	pi := 3.14
	fmt.Println(pi)

	// 5. 定义一个布尔型变量isTrue并赋值为true，输出isTrue的值
	isTrue := true
	fmt.Println(isTrue)

	// 6. 创建一个整型数组arr，包含元素1，2，3，4，5，并输出数组的长度和每个元素的值
	arr := []int{1, 2, 3, 4, 5}
	fmt.Println("Array length:", len(arr))
	for i := 0; i < len(arr); i++ {
		fmt.Println(arr[i])
	}

	// 7. 定义一个函数add，接收两个整数参数x和y，返回它们的和，并输出结果
	add := func(x, y int) int {
		return x + y
	}
	fmt.Println(add(1, 2))

	// 8. 使用if语句判断num是否大于10，如果是则输出"num大于10"，否则输出"num不大于10"
	if num > 10 {
		fmt.Println("num大于10")
	} else {
		fmt.Println("num不大于10")
	}

	// 9. 使用for循环输出从1到10的所有整数
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}

	// 10. 使用switch语句判断num的值，如果为1则输出"num等于1"，如果为2则输出"num等于2"，其他情况输出"num不等于1或2"
	switch num {
	case 1:
		fmt.Println("num等于1")
	case 2:
		fmt.Println("num等于2")
	default:
		fmt.Println("num不等于1或2")
	}
}

”`
