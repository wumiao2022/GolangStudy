package main

import (
	"fmt"
	"math/rand"
	"time"
)

func main() {
	rand.Seed(time.Now().UnixNano())

	for i := 0; i < 5; i++ {
		num := rand.Intn(100)
		fmt.Printf("随机数：%d\n", num)
	}

	for i := 0; i < 5; i++ {
		num1 := rand.Intn(10)
		num2 := rand.Intn(10)
		result := add(num1, num2)
		fmt.Printf("数字 %d 和 %d 的和是：%d\n", num1, num2, result)
	}
}

func add(a, b int) int {
	return a + b
}
