  1. 实现一个简单的计算器，支持加、减、乘、除四种运算。
package main

import (
    "fmt"
)

func main() {
    var num1 float64
    var num2 float64
    var operator string

    fmt.Print("请输入第一个数字: ")
    fmt.Scanln(&num1)

    fmt.Print("请输入第二个数字: ")
    fmt.Scanln(&num2)

    fmt.Print("请输入运算符号(+、-、*、/): ")
    fmt.Scanln(&operator)

    switch operator {
    case "+":
        fmt.Printf("%.2f + %.2f = %.2f", num1, num2, num1+num2)
    case "-":
        fmt.Printf("%.2f - %.2f = %.2f", num1, num2, num1-num2)
    case "*":
        fmt.Printf("%.2f * %.2f = %.2f", num1, num2, num1*num2)
    case "/":
        if num2 == 0 {
            fmt.Println("除数不能为 0")
            return
        }
        fmt.Printf("%.2f / %.2f = %.2f", num1, num2, num1/num2)
    default:
        fmt.Println("输入的运算符号有误")
    }
}
  1. 实现一个自定义类型的结构体，模拟一个人的基本信息（姓名、年龄、性别、身高、体重），并输出其信息。
package main

import "fmt"

type Person struct {
    Name   string
    Age    int
    Gender string
    Height float64
    Weight float64
}

func main() {
    person := Person{
        Name:   "张三",
        Age:    28,
        Gender: "男",
        Height: 180.0,
        Weight: 70.0,
    }

    fmt.Printf("姓名：%s\n", person.Name)
    fmt.Printf("年龄：%d\n", person.Age)
    fmt.Printf("性别：%s\n", person.Gender)
    fmt.Printf("身高：%.2f cm\n", person.Height)
    fmt.Printf("体重：%.2f kg\n", person.Weight)
}
  1. 实现一个函数，统计一个字符串中出现次数最多的字符及其出现的次数。
package main

import (
    "fmt"
    "strings"
)

func main() {
    str := "abracadabra"
    maxCnt := 0
    maxChar := byte(0)
    for i := 0; i < len(str); i++ {
        cnt := strings.Count(str, string(str[i]))
        if cnt > maxCnt {
            maxCnt = cnt
            maxChar = str[i]
        }
    }
    fmt.Printf("出现次数最多的字符是：%c，出现次数为：%d", maxChar, maxCnt)
}
  1. 实现一个函数，接收一个整数参数n，返回斐波那契数列的前n项。
package main

import "fmt"

func main() {
    fmt.Println(fibonacci(10))
}

func fibonacci(n int) []int {
    var fibs []int
    for i := 0; i < n; i++ {
        var fib int
        if i < 2 {
            fib = 1
        } else {
            fib = fibs[i-1] + fibs[i-2]
        }
        fibs = append(fibs, fib)
    }
    return fibs
}
