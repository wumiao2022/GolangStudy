  1. 题目：求两个数的最大公约数
   package main
   
   import "fmt"
   
   func gcd(a, b int) int {
       if a < b {
           a, b = b, a
       }
       for b != 0 {
           temp := a % b
           a = b
           b = temp
       }
       return a
   }
   
   func main() {
       fmt.Println(gcd(24, 36)) // output: 12
   }
  1. 题目：求从1到n的所有数的和
   package main
   
   import "fmt"
   
   func sum(n int) int {
       if n == 0 {
           return 0
       }
       return n + sum(n-1)
   }
   
   func main() {
       fmt.Println(sum(10)) // output: 55
   }
  1. 题目：使用递归逆序输出一个字符串
   package main
   
   import "fmt"
   
   func reverse(str string) {
       if len(str) == 0 {
           return
       }
       reverse(str[1:])
       fmt.Printf("%c", str[0])
   }
   
   func main() {
       reverse("hello") // output: olleh
   }
