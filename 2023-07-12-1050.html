package main

import "fmt"

func main() {
	// Exercise 1: Print "Hello, World!"
	fmt.Println("Hello, World!")

	// Exercise 2: Calculate the sum of two integers
	num1 := 10
	num2 := 5
	sum := num1 + num2
	fmt.Println("Sum:", sum)

	// Exercise 3: Swap the values of two variables
	a := 10
	b := 5
	a, b = b, a
	fmt.Println("a:", a)
	fmt.Println("b:", b)

	// Exercise 4: Print the odd numbers from 1 to 10
	for i := 1; i <= 10; i++ {
		if i%2 == 1 {
			fmt.Println(i)
		}
	}

	// Exercise 5: Calculate the factorial of a number
	num := 5
	factorial := 1
	for i := 1; i <= num; i++ {
		factorial *= i
	}
	fmt.Println("Factorial:", factorial)
}
