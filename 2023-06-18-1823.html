1. 写一个求和函数，接收任意长度的int类型参数，返回它们的和。
func sum(nums ...int) int {
    ans := 0
    for _, num := range nums {
        ans += num
    }
    return ans
}
  1. 写一个打印99乘法表的函数。
func multiplicationTable() {
    for i := 1; i <= 9; i++ {
        for j := 1; j <= i; j++ {
            fmt.Printf("%d*%d=%d ", j, i, j*i)
        }
        fmt.Println()
    }
}
  1. 写一个函数接收一个字符串作为参数，返回一个新字符串，其中每个字符都是原字符串中它的后面一个字符。
func shiftString(str string) string {
    ans := ""
    for _, char := range str {
        ans += string(char+1)
    }
    return ans
}
  1. 写一个递归的斐波那契数列函数，接收一个整数作为参数，返回斐波那契数列的第n项。
func fibonacci(n int) int {
    if n <= 1 {
        return n
    }
    return fibonacci(n-1) + fibonacci(n-2)
}
  1. 写一个函数，接收任意一个int类型参数，返回它的阶乘。
func factorial(n int) int {
    ans := 1
    for i := 1; i <= n; i++ {
        ans *= i
    }
    return ans
}
