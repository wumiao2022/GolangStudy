package main

import "fmt"

func main() {
	// 示例 1：打印 Hello, World!
	fmt.Println("Hello, World!")

	// 示例 2：计算并打印 1+2 的结果
	fmt.Println(1 + 2)

	// 示例 3：定义一个整型变量 x，并打印它的值
	x := 10
	fmt.Println(x)

	// 示例 4：使用 for 循环打印出数字 0 到 9
	for i := 0; i < 10; i++ {
		fmt.Println(i)
	}

	// 示例 5：定义一个包含 5 个整数的数组，并按照顺序打印出来
	arr := [5]int{1, 2, 3, 4, 5}
	for _, num := range arr {
		fmt.Println(num)
	}

	// 示例 6：定义一个函数 add，接受两个整型参数，并返回它们的和
	add := func(a, b int) int {
		return a + b
	}

	// 调用 add 函数，并打印结果
	result := add(3, 4)
	fmt.Println(result)
}
