给你一个练习：实现一个冒泡排序算法。

package main

import (
    "fmt"
)

func main() {
    nums := []int{1, 5, 3, 2, 4} // 待排序数据

    for i := 0; i < len(nums)-1; i++ {
        for j := 0; j < len(nums)-1-i; j++ {
            if nums[j] > nums[j+1] {
                nums[j+1], nums[j] = nums[j], nums[j+1] // 交换位置
            }
        }
    }

    fmt.Println(nums) // 输出排序后的结果
}
