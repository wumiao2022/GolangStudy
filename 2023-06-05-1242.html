1. 将一个字符串反转并输出:
package main

import "fmt"

func main() {
    str := "hello, world"
    runes := []rune(str)
    for i, j := 0, len(runes)-1; i < len(runes)/2; i, j = i+1, j-1 {
        runes[i], runes[j] = runes[j], runes[i]
    }
    fmt.Println(string(runes))
}
  1. 判断一个整数是否是回文数:
package main

import (
    "fmt"
    "strconv"
)

func main() {
    num := 12321
    s := strconv.Itoa(num)
    runes := []rune(s)
    for i, j := 0, len(runes)-1; i <= j; i, j = i+1, j-1 {
        if runes[i] != runes[j] {
            fmt.Printf("%d is not a palindrome number.\n", num)
            return
        }
    }
    fmt.Printf("%d is a palindrome number.\n", num)
}
  1. 计算两个整数的最大公约数:
package main

import "fmt"

func gcd(a, b int) int {
    if b == 0 {
        return a
    }
    return gcd(b, a%b)
}

func main() {
    num1, num2 := 12, 18
    fmt.Printf("The GCD of %d and %d is %d.\n", num1, num2, gcd(num1, num2))
}
  1. 实现十进制转二进制的函数:
package main

import "fmt"

func decimalToBinary(n int) int {
    if n == 0 {
        return 0
    }
    return n%2 + 10*decimalToBinary(n/2)
}

func main() {
    num := 10
    fmt.Printf("The binary representation of %d is %d.\n", num, decimalToBinary(num))
}
  1. 判断一个字符串是否是回文字符串:
package main

import "fmt"

func isPalindrome(s string) bool {
    runes := []rune(s)
    for i, j := 0, len(runes)-1; i <= j; i, j = i+1, j-1 {
        if runes[i] != runes[j] {
            return false
        }
    }
    return true
}

func main() {
    str1 := "abcba"
    str2 := "hello"
    fmt.Printf("%s is palindrome: %v\n", str1, isPalindrome(str1))
    fmt.Printf("%s is palindrome: %v\n", str2, isPalindrome(str2))
}
