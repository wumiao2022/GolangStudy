package main

import "fmt"

func main() {
	// 练习1：打印Hello, World!
	fmt.Println("Hello, World!")

	// 练习2：定义一个整型变量x，并打印出其值
	var x int
	fmt.Println(x)

	// 练习3：定义一个字符串变量name，并给它赋值为你的名字，然后打印出来
	name := "John Doe"
	fmt.Println(name)

	// 练习4：定义一个切片slice，并初始化为包含5个元素的字符串切片，然后打印出来
	slice := []string{"apple", "banana", "cherry", "date", "elderberry"}
	fmt.Println(slice)

	// 练习5：定义一个map，并初始化为包含两个键值对（字符串到整型的映射），然后打印出来
	myMap := map[string]int{
		"apple":  1,
		"banana": 2,
	}
	fmt.Println(myMap)
}
