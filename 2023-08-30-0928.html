package main

import "fmt"

func main() {
    // 练习1: 打印 "Hello, World!"
    fmt.Println("Hello, World!")

    // 练习2: 声明一个字符串变量，赋值为 "Golang"
    var str string = "Golang"
    fmt.Println(str)

    // 练习3: 声明一个整数变量，赋值为 10，再声明一个整数常量，赋值为 5
    var num int = 10
    const constant int = 5
    fmt.Println(num, constant)

    // 练习4: 实现一个简单的计算器，接收用户输入的两个数和操作符，并输出计算结果
    var number1, number2 int
    var operator string

    fmt.Print("Enter number 1: ")
    fmt.Scanln(&number1)

    fmt.Print("Enter operator: ")
    fmt.Scanln(&operator)

    fmt.Print("Enter number 2: ")
    fmt.Scanln(&number2)

    result := 0

    switch operator {
    case "+":
        result = number1 + number2
    case "-":
        result = number1 - number2
    case "*":
        result = number1 * number2
    case "/":
        result = number1 / number2
    }

    fmt.Println("Result:", result)

    // 练习5: 声明一个切片，包含 5 个整数，然后使用 for 循环遍历并输出每个元素
    numbers := []int{1, 2, 3, 4, 5}

    for _, number := range numbers {
        fmt.Println(number)
    }
}
