1. 查找两个slice中的共同元素
func FindCommonElements(s1, s2 []int) []int {
    m := make(map[int]bool) // 创建一个map用于存储s1中的元素
    for _, v := range s1 {
        m[v] = true
    }

    res := []int{} // 创建结果slice
    for _, v := range s2 {
        if _, ok := m[v]; ok { // 如果s2中存在s1中的元素，将其存入结果slice中
            res = append(res, v)
        }
    }

    return res
}
  1. 找出字符串中的所有数字并返回它们的和
func SumOfNumbers(s string) int {
    re := regexp.MustCompile("\\d+") // 创建正则表达式，匹配所有数字
    nums := re.FindAllString(s, -1)

    sum := 0
    for _, num := range nums { // 将找出来的数字转为int型，求和
        n, _ := strconv.Atoi(num)
        sum += n
    }

    return sum
}
  1. 判断一个字符串是否是回文字符串
func IsPalindrome(s string) bool {
    l, r := 0, len(s)-1
    for l < r { // 从字符串两端向中间比较是否相等
        if s[l] != s[r] {
            return false
        }
        l++
        r--
    }
    return true
}
  1. 实现二分查找算法
func BinarySearch(arr []int, target int) int {
    lo, hi := 0, len(arr)
    for lo < hi {
        mid := lo + (hi-lo)/2 // 计算中间下标
        if arr[mid] == target {
            return mid // 找到目标元素，返回下标
        } else if arr[mid] < target {
            lo = mid + 1 // 目标元素在右半部分，更新下边界
        } else {
            hi = mid // 目标元素在左半部分，更新上边界
        }
    }
    return -1 // 没有找到目标元素
}
  1. 找出数组中的最大值和最小值
func FindMinMax(arr []int) (int, int) {
    min, max := arr[0], arr[0]

    for _, v := range arr {
        if v < min {
            min = v // 更新最小值
        }
        if v > max {
            max = v // 更新最大值
        }
    }

    return min, max
}
