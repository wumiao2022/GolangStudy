package main

import "fmt"

func main() {
    // 练习1：打印1到10的数字
    for i := 1; i <= 10; i++ {
        fmt.Println(i)
    }

    // 练习2：计算1到100之间的偶数之和
    sum := 0
    for i := 1; i <= 100; i++ {
        if i%2 == 0 {
            sum += i
        }
    }
    fmt.Println(sum)

    // 练习3：判断一个数是否为质数
    number := 17
    isPrime := true
    for i := 2; i < number; i++ {
        if number%i == 0 {
            isPrime = false
            break
        }
    }
    fmt.Println("Is", number, "a prime number? ", isPrime)

    // 练习4：遍历一个字符串的每个字符并打印
    str := "Hello, Golang!"
    for i := 0; i < len(str); i++ {
        fmt.Println(string(str[i]))
    }

    // 练习5：定义一个包含5个元素的整数切片，并计算它们的和
    numbers := []int{1, 2, 3, 4, 5}
    sum = 0
    for _, num := range numbers {
        sum += num
    }
    fmt.Println("Sum:", sum)
}
