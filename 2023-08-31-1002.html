package main

import "fmt"

func main() {
	// 练习1：打印出Hello World!
	fmt.Println("Hello World!")

	// 练习2：定义一个整型变量并赋值为10，将其打印出来
	num := 10
	fmt.Println(num)

	// 练习3：定义一个字符串变量并赋值为"Go语言编程"，将其打印出来
	str := "Go语言编程"
	fmt.Println(str)

	// 练习4：计算两个整型数的和并将结果打印出来
	a := 5
	b := 7
	sum := a + b
	fmt.Println(sum)

	// 练习5：定义一个字符串数组并打印数组的长度
	arr := []string{"apple", "banana", "orange"}
	fmt.Println(len(arr))

	// 练习6：定义一个函数，传入一个整型数组，返回数组中的最大值
	nums := []int{10, 5, 8, 3, 6}
	max := findMax(nums)
	fmt.Println(max)
}

func findMax(arr []int) int {
	max := arr[0]
	for _, num := range arr {
		if num > max {
			max = num
		}
	}
	return max
}
