下面是一些不同难度和主题的 Golang 练习实例代码：

  1. 输出 “Hello, World!” 字符串。
package main

import "fmt"

func main() {
  fmt.Println("Hello, World!")
}
  1. 接收用户输入两个整数并输出它们的和。
package main

import "fmt"

func main() {
  var a, b int
  fmt.Print("Enter two integers separated by space: ")
  fmt.Scan(&a, &b)
  fmt.Printf("The sum of %d and %d is %d\n", a, b, a+b)
}
  1. 定义一个函数，接收一个整数切片并返回该切片的最大值。
package main

import (
  "fmt"
  "sort"
)

func main() {
  nums := []int{8, 2, 5, 1, 9, 3, 4, 7, 6}
  fmt.Printf("The maximum value in %v is %d\n", nums, max(nums))
}

func max(nums []int) int {
  sort.Ints(nums)
  return nums[len(nums)-1]
}
  1. 打印斐波那契数列中前 10 个数字。
package main

import "fmt"

func main() {
  a, b := 0, 1
  for i := 0; i < 10; i++ {
    fmt.Printf("%d ", a)
    a, b = b, a+b
  }
  fmt.Println()
}
  1. 定义一个结构体表示一个人员信息，包括名字、年龄和工资，然后创建两个该结构体类型的实例并比较它们的工资是否相等。
package main

import "fmt"

type Person struct {
  name string
  age int
  salary int
}

func main() {
  p1 := Person{name: "Alice", age: 30, salary: 5000}
  p2 := Person{name: "Bob", age: 35, salary: 5000}
  if p1.salary == p2.salary {
    fmt.Println("The two persons have the same salary.")
  } else {
    fmt.Println("The two persons have different salaries.")
  }
}
