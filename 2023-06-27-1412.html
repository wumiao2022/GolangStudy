2. 在一个切片中随机选择一个元素并删除
```go
package main

import (
    "fmt"
    "math/rand"
    "time"
)

func main() {
    rand.Seed(time.Now().UnixNano()) // 初始化随机数种子
    arr := []string{"apple", "banana", "orange", "grape", "watermelon"}
    index := rand.Intn(len(arr)) // 随机生成索引
    fmt.Println("随机选择的元素为:", arr[index])
    arr = append(arr[:index], arr[index+1:]...) // 删除元素
    fmt.Println("删除后的切片为:", arr)
}

4. 判断一个切片中是否包含某个元素
```go
package main

import (
    "fmt"
)

func main() {
    arr := []int{1, 2, 3, 4, 5}
    isContain := false
    for _, item := range arr {
        if item == 3 {
            isContain = true
            break
        }
    }
    fmt.Println(isContain)
}
