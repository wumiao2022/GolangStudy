package main

import (
	"fmt"
	"math/rand"
	"time"
)

func main() {
	rand.Seed(time.Now().Unix()) // 设置随机数种子为当前时间戳

	for i := 0; i < 10; i++ {
		fmt.Print(rand.Intn(100), " ")
	}
	fmt.Println()

	for i := 0; i < 10; i++ {
		fmt.Print(rand.Float64()*100, " ")
	}
	fmt.Println()

	for i := 0; i < 10; i++ {
		fmt.Print(randInt(-50, 50), " ")
	}
	fmt.Println()
}

// 生成一个在[min, max]之间的整数
func randInt(min int, max int) int {
	return min + rand.Intn(max-min+1)
}
