// 1. 打印99乘法表
func printMultiplicationTable() {
    for i:=1;i<10;i++{
        for j:=1;j<=i;j++{
            fmt.Printf("%v*%v=%v\t", i, j, i*j)
        }
        fmt.Println()
    }
}

// 2. 打印斐波那契数列（输出前20位）
func printFibonacci() {
    num1, num2 := 0, 1
    for i:=0;i<20;i++{
        fmt.Printf("%v\t", num2)
        num1, num2 = num2, num1+num2
    }
}

// 3. 判断是否是质数
func isPrime(n int) bool {
    if n<2 {
        return false
    }
    for i:=2;i<=int(math.Sqrt(float64(n)));i++{
        if n%i == 0 {
            return false
        }
    }
    return true
}

// 4. 打印100以内的质数
func printPrime() {
    for i:=2;i<=100;i++{
        if isPrime(i) {
            fmt.Printf("%v ", i)
        }
    }
}

// 5. 打印100-999中的水仙花数
func printNarcissistic() {
    for i:=100;i<1000;i++{
        num := i
        sum := 0
        for num > 0 {
            digit := num % 10
            sum += digit * digit * digit
            num /= 10
        }
        if sum == i {
            fmt.Printf("%v ", i)
        }
    }
}
