package main

import "fmt"

func main() {
	// 练习1: 输出 "Hello, world!"
	fmt.Println("Hello, world!")

	// 练习2: 声明一个名为 x 的整数变量并赋值为 10，然后打印出 x 的值
	x := 10
	fmt.Println(x)

	// 练习3: 声明一个名为 str 的字符串变量并赋值为 "Golang is awesome!"，然后打印出 str 的值
	str := "Golang is awesome!"
	fmt.Println(str)

	// 练习4: 声明一个名为 arr 的整型数组，并初始化为包含数值 1, 2, 3 的数组，然后打印出 arr 的值
	arr := [3]int{1, 2, 3}
	fmt.Println(arr)

	// 练习5: 声明一个名为 sum 的函数，接收两个整数参数，返回它们的和，并在 main 函数中调用 sum 函数并打印结果
	sum := func(a, b int) int {
		return a + b
	}
	fmt.Println(sum(3, 4))
}
