// 练习1：变量声明与打印输出
package main

import "fmt"

func main() {
    var num1 int = 10
    var num2 float32 = 3.14
    var str string = "Hello, Golang!"

    fmt.Printf("num1 = %d\n", num1)
    fmt.Printf("num2 = %f\n", num2)
    fmt.Printf("str = %s\n", str)
}

// 练习2：基本算术运算
package main

import "fmt"

func main() {
    var num1 int = 10
    var num2 int = 3

    fmt.Printf("num1 + num2 = %d\n", num1+num2)
    fmt.Printf("num1 - num2 = %d\n", num1-num2)
    fmt.Printf("num1 * num2 = %d\n", num1*num2)
    fmt.Printf("num1 / num2 = %d\n", num1/num2)
    fmt.Printf("num1 %% num2 = %d\n", num1%num2)
}

// 练习3：for循环和if条件判断
package main

import "fmt"

func main() {
    for i := 1; i <= 10; i++ {
        if i%2 == 0 {
            fmt.Printf("%d is an even number\n", i)
        } else {
            fmt.Printf("%d is an odd number\n", i)
        }
    }
}

// 练习4：switch语句
package main

import "fmt"

func main() {
    var grade string = "A"

    switch {
    case grade == "A":
        fmt.Printf("Excellent!\n")
    case grade == "B":
        fmt.Printf("Good!\n")
    case grade == "C":
        fmt.Printf("Passed!\n")
    default:
        fmt.Printf("Failed!\n")
    }
}

// 练习5：数组和切片的使用
package main

import "fmt"

func main() {
    // 数组的使用
    var a [5]int
    for i := 0; i < len(a); i++ {
        a[i] = i + 1
    }
    fmt.Println(a)

    // 切片的使用
    b := []int{1, 2, 3, 4, 5}
    fmt.Println(b)
    fmt.Println(b[1:3])
    fmt.Println(b[:3])
    fmt.Println(b[3:])
}
