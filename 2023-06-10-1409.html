package main

import "fmt"

func main() {
    var num int
    fmt.Print("Enter a number: ")
    fmt.Scan(&num)

    for i := 1; i <= num; i++ {
        for j := 1; j <= i; j++ {
            fmt.Print("* ")
        }
        fmt.Println()
    }
}

        for j := 1; j <= num-i; j++ {

            fmt.Print("* ")
        }
        fmt.Println()
    }
}

package main

import "fmt"

func main() {
    var num int
    fmt.Print("Enter a number: ")
    fmt.Scan(&num)
    fib1 := 0
    fib2 := 1
    for i := 0; i < num; i++ {
        fmt.Print(fib1, " ")
        temp := fib1
        fib1 = fib2
        fib2 = temp + fib2
    }
}
