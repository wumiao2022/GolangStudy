package main

import (
	"fmt"
)

func main() {
	// Example 1: Print "Hello, World!"
	fmt.Println("Hello, World!")

	// Example 2: Calculate the sum of two numbers
	num1 := 10
	num2 := 16
	sum := num1 + num2
	fmt.Println("The sum of", num1, "and", num2, "is", sum)

	// Example 3: Calculate the square of a number
	number := 5
	square := number * number
	fmt.Println("The square of", number, "is", square)

	// Example 4: Check if a number is even or odd
	checkNumber := 7
	if checkNumber%2 == 0 {
		fmt.Println(checkNumber, "is an even number.")
	} else {
		fmt.Println(checkNumber, "is an odd number.")
	}

	// Example 5: Find the largest number in an array
	numbers := []int{55, 78, 32, 98, 12}
	largest := numbers[0]
	for i := 1; i < len(numbers); i++ {
		if numbers[i] > largest {
			largest = numbers[i]
		}
	}
	fmt.Println("The largest number in the array is", largest)
}

注意：这些示例代码可以直接在Golang的开发环境中运行，以练习和理解Golang的基本语法和特性。
