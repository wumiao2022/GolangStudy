package main

import (
    "fmt"
    "math/rand"
    "time"
)

func main() {
    rand.Seed(time.Now().UnixNano()) // 设置随机数种子

    guess := -1
    number := rand.Intn(100) + 1 // 生成1到100之间的随机整数
    fmt.Println("我想了一个1到100之间的整数，你猜猜是多少？")

    for guess != number {
        fmt.Print("请猜一个整数： ")
        fmt.Scan(&guess)

        if guess > number {
            fmt.Println("猜大了！")
        } else if guess < number {
            fmt.Println("猜小了！")
        } else {
            fmt.Println("恭喜你猜对了！")
        }
    }
}

这个程序会生成一个1到100之间的随机整数，然后让玩家猜这个数是多少，每次猜完都会给出提示，直到猜对为止。这个程序可以帮助你练习随机数生成、if语句、循环、格式化输入输出等技能。
