package main

import (
	"fmt"
)

func main() {
	// 练习1：打印Hello, World!
	fmt.Println("Hello, World!")

	// 练习2：计算1到10的和
	sum := 0
	for i := 1; i <= 10; i++ {
		sum += i
	}
	fmt.Println("Sum from 1 to 10:", sum)

	// 练习3：判断一个数是否为素数
	num := 17
	isPrime := true
	for i := 2; i < num; i++ {
		if num%i == 0 {
			isPrime = false
			break
		}
	}
	if isPrime {
		fmt.Println(num, "is a prime number.")
	} else {
		fmt.Println(num, "is not a prime number.")
	}

	// 练习4：打印斐波那契数列前20个数
	n1 := 0
	n2 := 1
	fmt.Println(n1)
	fmt.Println(n2)
	for i := 2; i < 20; i++ {
		nextNum := n1 + n2
		fmt.Println(nextNum)
		n1, n2 = n2, nextNum
	}
}

这是一个简单的练习示例包括：打印Hello, World!、计算1到10的和、判断一个数是否为素数、打印斐波那契数列前20个数。但这只是一个起点，希望你能不断扩展自己的练习，并不断提高你的编程技能。加油！
