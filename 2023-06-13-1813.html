以下是一个Golang练习示例：

package main

import "fmt"

func main() {
    // 练习1：打印从0到9的数字
    for i := 0; i < 10; i++ {
        fmt.Println(i)
    }

    // 练习2：输出一个10行10列的*组成的矩阵
    for i := 0; i < 10; i++ {
        for j := 0; j < 10; j++ {
            fmt.Print("* ")
        }
        fmt.Println()
    }

    // 练习3：输出1~100的数，如果能被3整除，输出Fizz；如果能被5整除，输出Buzz；如果既能被3也能被5整除，输出FizzBuzz
    for i := 1; i <= 100; i++ {
        switch {
        case i%3 == 0 && i%5 == 0:
            fmt.Println("FizzBuzz")
        case i%3 == 0:
            fmt.Println("Fizz")
        case i%5 == 0:
            fmt.Println("Buzz")
        default:
            fmt.Println(i)
        }
    }
}
