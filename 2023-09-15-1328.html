以下是一个示例的Golang练习实例代码：

package main

import "fmt"

func main() {
	// 练习1：打印Hello, World!
	fmt.Println("Hello, World!")

	// 练习2：计算两个数的和并打印结果
	num1 := 10
	num2 := 20
	sum := num1 + num2
	fmt.Println("Sum:", sum)

	// 练习3：判断一个数是否为偶数并打印结果
	num := 15
	if num%2 == 0 {
		fmt.Println(num, "is even.")
	} else {
		fmt.Println(num, "is odd.")
	}

	// 练习4：使用循环打印出1到10之间的所有奇数
	for i := 1; i <= 10; i += 2 {
		fmt.Println(i)
	}

	// 练习5：使用数组存储学生的成绩，并计算平均分
	grades := [5]float64{98.5, 87.2, 95.1, 91.7, 89.3}
	var sum float64
	for _, grade := range grades {
		sum += grade
	}
	average := sum / float64(len(grades))
	fmt.Println("Average:", average)
}

请注意，这只是一个简单的示例练习，你可以根据自己的需求和兴趣进行更多的练习。练习的关键是多动手写代码，并通过实践来学习和提升你的编程技能。
