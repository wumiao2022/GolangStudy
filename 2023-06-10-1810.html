下面是一些Golang练习题的示例代码：

  1. 打印出所有1到100之间的奇数。
for i := 1; i <= 100; i++ {
    if i%2 != 0 {
        fmt.Println(i)
    }
}
  1. 编写一个函数，将一个字符串中的每个单词的首字母转换为大写。
func toUpperCase(s string) string {
    words := strings.Fields(s)
    for i := 0; i < len(words); i++ {
        words[i] = strings.ToUpper(words[i][:1]) + words[i][1:]
    }
    return strings.Join(words, " ")
}
  1. 编写一个函数，判断一个字符串是否是回文字符串（正读和倒读都一样，例如”level”）。
func isPalindrome(s string) bool {
    for i := 0; i < len(s)/2; i++ {
        if s[i] != s[len(s)-i-1] {
            return false
        }
    }
    return true
}
  1. 编写一个函数，将一个字符串中出现的所有数字相加后返回。
func addNumbers(s string) int {
    sum := 0
    for _, c := range s {
        if unicode.IsDigit(c) {
            n, _ := strconv.Atoi(string(c))
            sum += n
        }
    }
    return sum
}
  1. 编写一个程序，实现斐波那契数列（前两个数为1，之后每个数为前两个数之和）。
func fibonacci(n int) int {
    if n == 1 || n == 2 {
        return 1
    } else {
        return fibonacci(n-1) + fibonacci(n-2)
    }
}

func main() {
    for i := 1; i <= 10; i++ {
        fmt.Print(fibonacci(i), " ")
    }
}
