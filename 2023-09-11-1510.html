package main

import "fmt"

func main() {
	// 练习一：打印 Hello, World!
	fmt.Println("Hello, World!")

	// 练习二：计算并打印数组中元素的和
	numbers := []int{1, 2, 3, 4, 5}
	sum := 0
	for _, num := range numbers {
		sum += num
	}
	fmt.Println("Sum:", sum)

	// 练习三：判断一个数是否为素数
	number := 7
	isPrime := true
	for i := 2; i < number; i++ {
		if number%i == 0 {
			isPrime = false
			break
		}
	}
	fmt.Println("Is Prime:", isPrime)

	// 练习四：实现一个简单的计算器函数
	result := calculate(8, 4, "+")
	fmt.Println("Result:", result)
}

func calculate(a int, b int, operator string) int {
	switch operator {
	case "+":
		return a + b
	case "-":
		return a - b
	case "*":
		return a * b
	case "/":
		return a / b
	default:
		return 0
	}
}
