package main

import "fmt"

func main() {
	// 练习1: 交换两个变量的值，并输出结果
	a := 10
	b := 20
	a, b = b, a
	fmt.Println("a =", a, "b =", b)

	// 练习2: 判断一个数字是否为偶数，并输出结果
	num := 15
	if num%2 == 0 {
		fmt.Println(num, "是偶数")
	} else {
		fmt.Println(num, "不是偶数")
	}

	// 练习3: 计算斐波那契数列前 10 个数，并输出结果
	fib := make([]int, 10)
	fib[0] = 0
	fib[1] = 1
	for i := 2; i < 10; i++ {
		fib[i] = fib[i-1] + fib[i-2]
	}
	fmt.Println("斐波那契数列前 10 个数:", fib)

	// 练习4: 打印九九乘法表
	for i := 1; i <= 9; i++ {
		for j := 1; j <= i; j++ {
			fmt.Printf("%d*%d=%d\t", j, i, i*j)
		}
		fmt.Println()
	}
}

这是一个简单的练习代码示例，包括经典的交换变量值、判断奇偶数、计算斐波那契数列和打印九九乘法表等。你可以将代码复制到 Golang 编辑器中运行并观察结果。
