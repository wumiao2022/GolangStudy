package main

import (
	"fmt"
)

func main() {
	// 练习1：打印Hello, World!
	fmt.Println("Hello, World!")

	// 练习2：求两个整数的和并输出结果
	num1 := 10
	num2 := 20
	sum := num1 + num2
	fmt.Println("Sum:", sum)

	// 练习3：判断一个数的奇偶性
	number := 15
	if number%2 == 0 {
		fmt.Println(number, "is even")
	} else {
		fmt.Println(number, "is odd")
	}

	// 练习4：通过for循环计算1到100的和并输出结果
	total := 0
	for i := 1; i <= 100; i++ {
		total += i
	}
	fmt.Println("Total:", total)

	// 练习5：通过切片反转字符串并输出结果
	word := "Hello, World!"
	reversedWord := ""
	for _, char := range word {
		reversedWord = string(char) + reversedWord
	}
	fmt.Println("Reversed word:", reversedWord)
}
