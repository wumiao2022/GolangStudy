package main

import (
	"fmt"
)

func main() {
	// 练习1：打印1到10的数字
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}

	// 练习2：计算1到100之间所有整数的和
	sum := 0
	for i := 1; i <= 100; i++ {
		sum += i
	}
	fmt.Println("和为：", sum)

	// 练习3：判断一个数是否为素数
	num := 17
	isPrime := true
	for i := 2; i < num; i++ {
		if num%i == 0 {
			isPrime = false
			break
		}
	}
	if isPrime {
		fmt.Println(num, "是素数")
	} else {
		fmt.Println(num, "不是素数")
	}

	// 练习4：找出数组中的最大值
	arr := [5]int{10, 20, 30, 40, 50}
	max := arr[0]
	for _, num := range arr {
		if num > max {
			max = num
		}
	}
	fmt.Println("数组中的最大值为：", max)

	// 练习5：计算斐波那契数列的前20个数
	num1, num2 := 0, 1
	fmt.Println("斐波那契数列前20个数：")
	for i := 0; i < 20; i++ {
		fmt.Println(num1)
		num1, num2 = num2, num1+num2
	}
}
