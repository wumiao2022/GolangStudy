这是一个简单的Golang练习，它利用HTTP GET请求获取给定URL的HTML内容，并将它们输出到控制台。

package main

import (
	"fmt"
	"io/ioutil"
	"net/http"
)

func main() {
	url := "https://www.example.com"
	response, err := http.Get(url)
	if err != nil {
		fmt.Printf("Failed to fetch HTML for URL %s: %s", url, err)
	}

	defer response.Body.Close()

	html, err := ioutil.ReadAll(response.Body)
	if err != nil {
		fmt.Printf("Failed to read response body for URL %s: %s", url, err)
	}

	fmt.Println(string(html))
}
