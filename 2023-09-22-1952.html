package main

import (
	"fmt"
	"time"
)

func main() {
	fmt.Println("Hello, Go!")
	fmt.Println("Today's exercise: ")

	// Exercise 1: Hello World
	fmt.Println("Exercise 1:")
	fmt.Println("Hello, World!")

	// Exercise 2: Current Time
	fmt.Println("Exercise 2:")
	currentTime := time.Now()
	fmt.Println("Current time is:", currentTime.Format("2006-01-02 15:04:05"))

	// Exercise 3: Calculate the sum of two numbers
	fmt.Println("Exercise 3:")
	num1 := 10
	num2 := 20
	sum := num1 + num2
	fmt.Println(num1, "+", num2, "=", sum)

	// Exercise 4: Print a pattern
	fmt.Println("Exercise 4:")
	for i := 1; i <= 5; i++ {
		for j := 1; j <= i; j++ {
			fmt.Print("* ")
		}
		fmt.Println()
	}

	// Exercise 5: Check if a number is even or odd
	fmt.Println("Exercise 5:")
	number := 7
	if number%2 == 0 {
		fmt.Println(number, "is even")
	} else {
		fmt.Println(number, "is odd")
	}

	// Exercise 6: Find the maximum number in an array
	fmt.Println("Exercise 6:")
	numbers := []int{3, 9, 2, 7, 5}
	max := numbers[0]
	for _, num := range numbers {
		if num > max {
			max = num
		}
	}
	fmt.Println("The maximum number is:", max)
}

请运行以上代码并观察输出结果。它包含了六个练习实例，分别是： 1. 输出 “Hello, World!“。 2. 输出当前时间。 3. 计算两个数的和。 4. 打印一个星号的图案。 5. 判断一个数是奇数还是偶数。 6. 找出数组中的最大数。
