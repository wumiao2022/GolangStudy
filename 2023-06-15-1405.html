package main

import (
    "fmt"
    "time"
)

func main() {
    // 打印当前时间
    fmt.Println("当前时间：", time.Now())

    // 定义一个切片
    arr := []int{1, 2, 3, 4, 5}

    // 遍历切片并打印元素
    for i := range arr {
        fmt.Println(arr[i])
    }

    // 定义一个map
    m := map[string]int{
        "apple":  1,
        "banana": 2,
        "orange": 3,
    }

    // 遍历map并打印键值对
    for k, v := range m {
        fmt.Printf("%s -> %d\n", k, v)
    }

    // 使用defer关键字延迟执行函数
    defer fmt.Println("deferred function")

    // 使用panic关键字抛出异常
    panic("oh no!")
}
