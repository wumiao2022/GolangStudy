  1. 编写一个程序，输入一个数字n，然后输出1~n之间所有的奇数。

    package main
import "fmt"
func main() {
var n int
fmt.Println("请输入一个数字n：")
fmt.Scanln(&n)
for i := 1; i <= n; i += 2 {
    fmt.Println(i)
}
}

  2. 编写一个程序，输入一串字符串，检查其中是否存在任何重复的字符。

    package main
import "fmt"
func main() {
var str string
fmt.Println("请输入一个字符串：")
fmt.Scanln(&str)
m := make(map[rune]bool)
for _, c := range str {
    if m[c] {
        fmt.Println("存在重复字符！")
        return
    }
    m[c] = true
}
fmt.Println("不存在重复字符。")
}

  3. 编写一个程序，输入一个数字n，然后输出斐波那契数列的前n项。

    package main
import "fmt"
func main() {
var n int
fmt.Println("请输入一个数字n：")
fmt.Scanln(&n)
fib1, fib2 := 0, 1
for i := 0; i < n; i++ {
    fmt.Println(fib2)
    fib1, fib2 = fib2, fib1+fib2
}
}

  4. 编写一个程序，输入两个数字a和b，然后输出它们的最大公约数和最小公倍数。

    package main
import "fmt"
func gcd(a, b int) int {
if b == 0 {
    return a
}
return gcd(b, a%b)
}
func lcm(a, b int) int {
return a * b / gcd(a, b)
}
func main() {
var a, b int
fmt.Println("请输入两个数字a和b：")
fmt.Scanln(&a, &b)
fmt.Println("最大公约数：", gcd(a, b))
fmt.Println("最小公倍数：", lcm(a, b))
}

  5. 编写一个程序，输入一个浮点数x和一个整数n，然后输出x的n次方。

    package main
import "fmt"
func pow(x float64, n int) float64 {
if n == 0 {
    return 1
}
if n < 0 {
    return 1 / pow(x, -n)
}
if n%2 == 0 {
    return pow(x*x, n/2)
}
return x * pow(x*x, n/2)
}
func main() {
var x float64
var n int
fmt.Println("请输入一个浮点数x和一个整数n：")
fmt.Scanln(&x, &n)
fmt.Println(x, "的", n, "次方为：", pow(x, n))
}
