将字符串转为大写字母：

package main

import "fmt"
import "strings"

func main() {
    str := "hello, world!"
    fmt.Println(strings.ToUpper(str))
}

计算数组元素之和：

package main

import "fmt"

func main() {
    nums := []int{1, 2, 3, 4, 5}
    sum := 0
    for _, num := range nums {
        sum += num
    }
    fmt.Println("sum:", sum)
}

翻转字符串：

package main

import "fmt"

func main() {
    str := "hello, world!"
    runes := []rune(str)
    for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 {
        runes[i], runes[j] = runes[j], runes[i]
    }
    fmt.Println(string(runes))
}

检查切片中是否包含指定元素：

package main

import "fmt"

func main() {
    fruits := []string{"apple", "banana", "orange"}
    containsBanana := false
    for _, fruit := range fruits {
        if fruit == "banana" {
            containsBanana = true
            break
        }
    }
    fmt.Println("contains banana:", containsBanana)
}
