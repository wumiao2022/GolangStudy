package main

import "fmt"

func main() {
	// 练习1: 打印Hello, World!
	fmt.Println("Hello, World!")

	// 练习2: 交换两个变量的值
	a := 3
	b := 5
	a, b = b, a
	fmt.Println("a =", a, "b =", b)

	// 练习3: 判断一个数是否为奇数
	num := 7
	if num%2 == 0 {
		fmt.Println(num, "is even")
	} else {
		fmt.Println(num, "is odd")
	}

	// 练习4: 求两个数的最大公约数
	num1 := 12
	num2 := 18
	gcd := 1
	for i := 1; i <= num1 && i <= num2; i++ {
		if num1%i == 0 && num2%i == 0 {
			gcd = i
		}
	}
	fmt.Println("GCD of", num1, "and", num2, "is", gcd)

	// 练习5: 计算一个整数的阶乘
	n := 5
	fact := 1
	for i := 1; i <= n; i++ {
		fact *= i
	}
	fmt.Println("Factorial of", n, "is", fact)
}
