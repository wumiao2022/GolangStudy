	fmt.Scanln(&n)

package main

import "fmt"

func main() {
	var number int
	fmt.Print("Enter any number: ")
	fmt.Scanln(&number)

	if number > 0 {
		fmt.Println(number, "is a positive number")
	} else if number < 0 {
		fmt.Println(number, "is a negative number")
	} else {
		fmt.Println(number, "is zero")
	}
}

package main

import "fmt"

func main() {
	var grade int
	fmt.Print("Enter your grade: ")
	fmt.Scanln(&grade)

	switch {
	case grade >= 90:
		fmt.Println("Grade: A")
	case grade >= 80 && grade < 90:
		fmt.Println("Grade: B")
	case grade >= 70 && grade < 80:
		fmt.Println("Grade: C")
	case grade >= 60 && grade < 70:
		fmt.Println("Grade: D")
	default:
		fmt.Println("Grade: F")
	}
}
