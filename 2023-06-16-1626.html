package main

import (
	"fmt"
)

func main() {
	// 实现一个计算器，支持加减乘除四则运算
	var num1, num2 float64
	var operator string
	fmt.Println("请输入第一个数：")
	fmt.Scanln(&num1)
	fmt.Println("请输入运算符（+、-、*、/）：")
	fmt.Scanln(&operator)
	fmt.Println("请输入第二个数：")
	fmt.Scanln(&num2)

	switch operator {
	case "+":
		fmt.Printf("%.2f + %.2f = %.2f\n", num1, num2, num1+num2)
	case "-":
		fmt.Printf("%.2f - %.2f = %.2f\n", num1, num2, num1-num2)
	case "*":
		fmt.Printf("%.2f * %.2f = %.2f\n", num1, num2, num1*num2)
	case "/":
		if num2 == 0 {
			fmt.Println("除数不能为0")
		} else {
			fmt.Printf("%.2f / %.2f = %.2f\n", num1, num2, num1/num2)
		}
	default:
		fmt.Println("不支持的运算符")
	}

	// 实现一个简单的字符串反转
	var str string
	fmt.Println("请输入一个字符串：")
	fmt.Scanln(&str)
	reverseStr := ""
	for i := len(str) - 1; i >= 0; i-- {
		reverseStr += string(str[i])
	}
	fmt.Println("反转后的字符串为：", reverseStr)
}
