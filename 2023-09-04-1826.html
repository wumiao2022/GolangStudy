package main

import "fmt"

func main() {
    // 1. 打印出 "Hello, World!"
    fmt.Println("Hello, World!")

    // 2. 声明一个整数变量并赋值为10，打印出变量的值
    var num int = 10
    fmt.Println(num)

    // 3. 声明一个布尔变量并赋值为true，打印出变量的值
    var isTrue bool = true
    fmt.Println(isTrue)

    // 4. 使用if语句判断一个整数变量是否大于等于5，如果是则打印"大于等于5"，否则打印"小于5"
    if num >= 5 {
        fmt.Println("大于等于5")
    } else {
        fmt.Println("小于5")
    }

    // 5. 使用for循环打印出10到1的整数，每个数字一行
    for i := 10; i >= 1; i-- {
        fmt.Println(i)
    }

    // 6. 声明一个字符串变量并赋值为"Hello, Golang!"，通过for循环遍历打印出每个字符
    str := "Hello, Golang!"
    for _, char := range str {
        fmt.Printf("%c ", char)
    }
    fmt.Println()

    // 7. 声明一个字符串切片，包含5个元素，分别为"apple", "banana", "orange", "grape", "watermelon"，通过for循环遍历打印出每个元素
    fruits := []string{"apple", "banana", "orange", "grape", "watermelon"}
    for _, fruit := range fruits {
        fmt.Println(fruit)
    }
}

这些练习代码包括了打印输出、变量声明与赋值、条件语句、循环语句的使用。你可以将以上代码复制到你的Go编辑器中执行，通过查看输出来验证结果。加油！
