  1. 计算两个整数的和：
package main

import "fmt"

func main() {
    var a, b int
    a = 10
    b = 20
    fmt.Println(a+b)
}
  1. 判断一个整数是否为偶数：
package main

import "fmt"

func main() {
    var a int
    a = 6
    if a%2 == 0 {
        fmt.Println(a, "is even")
    } else {
        fmt.Println(a, "is odd")
    }
}
  1. 遍历一个字符串并打印每个字符的ASCII码值：
package main

import "fmt"

func main() {
    str := "Hello World"
    for i := 0; i < len(str); i++ {
        fmt.Printf("%c: %d\n", str[i], str[i])
    }
}
  1. 找到一个整数切片中的最大值和最小值：
package main

import "fmt"

func main() {
    arr := []int{4, 12, 7, 2, 9, 20, 5}
    max := arr[0]
    min := arr[0]
    for _, v := range arr {
        if v > max {
            max = v
        }
        if v < min {
            min = v
        }
    }
    fmt.Println("Max:", max)
    fmt.Println("Min:", min)
}
  1. 反转一个字符串：
package main

import "fmt"

func main() {
    str := "Hello World"
    var rev string
    for i := len(str)-1; i >= 0; i-- {
        rev += string(str[i])
    }
    fmt.Println(rev)
}
