  4. 控制流：if 语句

    package main
import "fmt"
func main() {
var num int = 10
if num > 0 {
    fmt.Println(num, "is a positive number")
} else if num == 0 {
    fmt.Println(num, "is zero")
} else {
    fmt.Println(num, "is a negative number")
}
}

  10. 接口声明与实现

    package main
import "fmt"
// 声明一个接口
type Shape interface {
area() float64
}
// 声明一个 Circle 结构体，并实现 Shape 接口
type Circle struct {
radius float64
}
func (c Circle) area() float64 {
return 3.14 * c.radius * c.radius
}
func main() {
// 声明一个 Circle 类型的变量
circle := Circle{radius: 5}
// 将 Circle 类型的变量赋值给 Shape 接口类型的变量
var shape Shape
shape = circle
// 调用 Shape 接口中的方法
fmt.Println(shape.area())
}
