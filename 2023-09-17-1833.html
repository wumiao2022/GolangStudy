package main

import (
	"fmt"
	"math/rand"
	"time"
)

func main() {
	rand.Seed(time.Now().UnixNano())

	// 练习1: 生成并打印一个随机整数
	randomInt := rand.Intn(100)
	fmt.Println("Random Integer:", randomInt)

	// 练习2: 生成并打印一个随机浮点数
	randomFloat := rand.Float64()
	fmt.Println("Random Float:", randomFloat)

	// 练习3: 生成并打印一个指定范围内的随机整数
	min := 10
	max := 50
	randomInRange := rand.Intn(max-min+1) + min
	fmt.Println("Random Integer in Range:", randomInRange)

	// 练习4: 生成并打印一个UUID(Universally Unique Identifier)
	uuid := fmt.Sprintf("%x-%x-%x-%x-%x",
		rand.Uint32(), rand.Uint16(), rand.Uint16(), rand.Uint16(), rand.Uint32())
	fmt.Println("UUID:", uuid)
}

这是一个每日多练的练习集合，其中包含了生成随机整数、随机浮点数、指定范围内的随机整数和UUID的练习。你可以每天完成一个练习来提升你的Golang编程技能。
