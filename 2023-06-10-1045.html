  1. 判断一个数是否为素数（质数） “`go package main

for i := 2; i <= int(math.Sqrt(float64(num))); i++ {
    if num%i == 0 {
        isPrime = false
        break
    }
}

if isPrime {
    fmt.Printf("%d是素数", num)
} else {
    fmt.Printf("%d不是素数", num)
}

}


4. 实现冒泡排序
```go
package main

import (
    "fmt"
)

func main() {
    nums := []int{3, 1, 4, 1, 5, 9, 2, 6, 5, 3, 5, 8, 9, 7, 9}

    for i := 0; i < len(nums)-1; i++ {
        for j := 0; j < len(nums)-1-i; j++ {
            if nums[j] > nums[j+1] {
                nums[j], nums[j+1] = nums[j+1], nums[j]
            }
        }
    }

    fmt.Println(nums)
}
