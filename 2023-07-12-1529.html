package main

import (
	"fmt"
)

func main() {
	// 练习1：输出Hello, World!
	fmt.Println("Hello, World!")

	// 练习2：输出1到10之间的整数
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}

	// 练习3：计算两个整数的和并输出
	num1 := 10
	num2 := 5
	sum := num1 + num2
	fmt.Println("Sum:", sum)

	// 练习4：打印出1到100之间所有的偶数
	for i := 1; i <= 100; i++ {
		if i%2 == 0 {
			fmt.Println(i)
		}
	}

	// 练习5：计算20的阶乘并输出结果
	factorial := 1
	for i := 1; i <= 20; i++ {
		factorial *= i
	}
	fmt.Println("20的阶乘:", factorial)
}

这是一个包含5个练习的示例代码，每个练习都有一个简单的任务，并且输出结果。您可以运行代码来观察结果。
