  1. 打印Hello World
package main
 
import "fmt"
 
func main() {
    fmt.Println("Hello, World!")
}
  1. 计算1-100的和
package main
 
import "fmt"
 
func main() {
    sum := 0
    for i := 1; i <= 100; i++ {
        sum += i
    }
    fmt.Println(sum)
}
  1. 判断一个数是否为质数
package main
 
import "fmt"
 
func isPrime(num int) bool {
    if num <= 1 {
        return false
    }
    for i := 2; i*i <= num; i++ {
        if num%i == 0 {
            return false
        }
    }
    return true
}
 
func main() {
    fmt.Println(isPrime(17))
}
  1. 反转一个字符串
package main
 
import "fmt"
 
func reverseString(s string) string {
    r := []rune(s)
    for i, j := 0, len(r)-1; i < len(r)/2; i, j = i+1, j-1 {
        r[i], r[j] = r[j], r[i]
    }
    return string(r)
}
 
func main() {
    str := "hello world"
    fmt.Println(reverseString(str))
}
  1. 计算斐波那契数列的第n项
package main
 
import "fmt"
 
func fibonacci(n int) int {
    if n == 0 {
        return 0
    } else if n == 1 {
        return 1
    } else {
        return fibonacci(n-1) + fibonacci(n-2)
    }
}
 
func main() {
    fmt.Println(fibonacci(7))
}
