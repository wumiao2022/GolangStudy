  1. 判断一个数是不是质数
package main

import (
	"fmt"
	"math"
)

func isPrime(x int) bool {
	if x < 2 {
		return false
	} else if x == 2 {
		return true
	} else {
		for i := 2; i <= int(math.Sqrt(float64(x))); i++ {
			if x%i == 0 {
				return false
			}
		}
	}
	return true
}

func main() {
	fmt.Println(isPrime(7))
	fmt.Println(isPrime(12))
}
  1. 计算一个整数序列的和
package main

import "fmt"

func sum(nums []int) int {
	total := 0
	for _, num := range nums {
		total += num
	}
	return total
}

func main() {
	nums := []int{2, 3, 4, 5, 6}
	fmt.Println(sum(nums))
}
  1. 判断一个字符串是否是回文字符串
package main

import (
	"fmt"
	"strings"
)

func isPalindrome(s string) bool {
	s = strings.ToLower(s)
	for i := 0; i < len(s)/2; i++ {
		if s[i] != s[len(s)-1-i] {
			return false
		}
	}
	return true
}

func main() {
	fmt.Println(isPalindrome("racecar"))
	fmt.Println(isPalindrome("hello"))
}
