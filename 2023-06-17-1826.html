1. 反转字符串
package main

import "fmt"

func main() {
	str := "hello world"
	bytes := []byte(str)
	for i, j := 0, len(bytes)-1; i < j; i, j = i+1, j-1 {
		bytes[i], bytes[j] = bytes[j], bytes[i]
	}
	fmt.Println(string(bytes))
}
  1. 斐波那契数列
package main

import "fmt"

func main() {
	n := 10
	a, b := 0, 1
	for i := 0; i < n; i++ {
		fmt.Println(a)
		a, b = b, a+b
	}
}
  1. 冒泡排序
package main

import "fmt"

func main() {
	arr := []int{3, 1, 4, 1, 5, 9, 2, 6, 5, 3}
	for i := 0; i < len(arr)-1; i++ {
		for j := 0; j < len(arr)-1-i; j++ {
			if arr[j] > arr[j+1] {
				arr[j], arr[j+1] = arr[j+1], arr[j]
			}
		}
	}
	fmt.Println(arr)
}
  1. 判断素数
package main

import (
	"fmt"
	"math"
)

func main() {
	n := 47
	if isPrime(n) {
		fmt.Printf("%d is prime\n", n)
	} else {
		fmt.Printf("%d is not prime\n", n)
	}
}

func isPrime(n int) bool {
	if n == 1 {
		return false
	}
	end := int(math.Sqrt(float64(n)))
	for i := 2; i <= end; i++ {
		if n%i == 0 {
			return false
		}
	}
	return true
}
  1. 求最大公约数
package main

import "fmt"

func main() {
	a, b := 72, 48
	fmt.Printf("gcd of %d and %d: %d\n", a, b, gcd(a, b))
}

func gcd(a, b int) int {
	if b == 0 {
		return a
	} else {
		return gcd(b, a%b)
	}
}
