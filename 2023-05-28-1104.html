完整代码如下：

package main

import "fmt"

func main() {
    // 打印 1~10 的所有偶数
    for i := 1; i <= 10; i++ {
        if i%2 == 0 {
            fmt.Println(i)
        }
    }

    // 判断一个数是否为奇数
    num := 7
    if num%2 == 0 {
        fmt.Println("偶数")
    } else {
        fmt.Println("奇数")
    }

    // 计算两个数的和并返回
    res := sum(2, 4)
    fmt.Println(res)
}

func sum(num1, num2 int) int {
    return num1 + num2
}
