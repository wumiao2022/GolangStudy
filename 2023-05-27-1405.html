以下是一段Golang多练习代码实例：

package main
 
import (
    "fmt"
)
 
func main() {
    //1. 输出“Hello, World!”
    fmt.Println("Hello, World!")
 
    //2. 求1到100的和
    sum := 0
    for i := 1; i <= 100;i++ {
        sum += i
    }
    fmt.Println("1到100的和：", sum)
 
    //3. 判断一个数是否是素数
    num := 37
    isPrime := true
    for i := 2;i < num; i++ {
        if num % i == 0 {
            isPrime = false
            break
        }
    }
    if isPrime {
        fmt.Println(num, "是素数")
    } else {
        fmt.Println(num, "不是素数")
    }
 
    //4. 将一个整数转换为二进制数字符串
    num = 345
    binaryStr := ""
    for num > 0 {
        remainder := num % 2
        binaryStr = fmt.Sprintf("%d", remainder) + binaryStr
        num /= 2
    }
    fmt.Println("345的二进制字符串为：", binaryStr)
 
    //5. 打印九九乘法表
    for i := 1;i <= 9;i++ {
        for j := 1;j <= i;j++ {
            fmt.Printf("%d*%d=%d\t", i,j,i*j)
        }
        fmt.Println()
    }
}
