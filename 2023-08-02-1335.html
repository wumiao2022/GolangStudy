package main

import (
	"fmt"
)

func main() {
	// 练习1：打印从1到10的所有整数
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}

	// 练习2：计算并打印从1到10的所有整数的和
	sum := 0
	for i := 1; i <= 10; i++ {
		sum += i
	}
	fmt.Println("Sum:", sum)

	// 练习3：判断一个整数是否为奇数或偶数
	num := 5
	if num%2 == 0 {
		fmt.Println(num, "is even")
	} else {
		fmt.Println(num, "is odd")
	}

	// 练习4：打印斐波那契数列的前n个数 (n > 0)
	n := 10
	a, b := 0, 1
	fmt.Println("Fibonacci Series:")
	for i := 0; i < n; i++ {
		fmt.Println(a)
		a, b = b, a+b
	}

	// 练习5：计算并打印n的阶乘 (n > 0)
	n = 5
	factorial := 1
	for i := 1; i <= n; i++ {
		factorial *= i
	}
	fmt.Println(n, "! =", factorial)
}
