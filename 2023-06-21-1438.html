这是一道Golang练习题，实现计算一个数的阶乘：

package main

import "fmt"

func Factorial(n int) int {
    if n == 0 {
        return 1
    } else {
        return n * Factorial(n-1)
    }
}

func main() {
    var n int
    fmt.Print("请输入一个正整数：")
    fmt.Scan(&n)

    result := Factorial(n)
    fmt.Printf("%d 的阶乘是 %d\n", n, result)
}
