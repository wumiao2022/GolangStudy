package main

import "fmt"

func main() {
	// 将数组中的元素按照奇偶性分别放到两个新的数组中
	nums := []int{1, 2, 3, 4, 5, 6}
	var oddNums, evenNums []int
	for i := 0; i < len(nums); i++ {
		if nums[i]%2 == 0 {
			evenNums = append(evenNums, nums[i])
		} else {
			oddNums = append(oddNums, nums[i])
		}
	}
	fmt.Println("oddNums:", oddNums)
	fmt.Println("evenNums:", evenNums)

	// 判断一个字符串是否是回文字符串
	str1 := "abcdeedcba"
	str2 := "hello"
	strLen := len(str1)
	for i := 0; i < strLen/2; i++ {
		if str1[i] != str1[strLen-1-i] {
			fmt.Printf("%s 不是回文字符串\n", str1)
			break
		}
		if i == strLen/2-1 {
			fmt.Printf("%s 是回文字符串\n", str1)
		}
	}
	strLen = len(str2)
	for i := 0; i < strLen/2; i++ {
		if str2[i] != str2[strLen-1-i] {
			fmt.Printf("%s 不是回文字符串\n", str2)
			break
		}
		if i == strLen/2-1 {
			fmt.Printf("%s 是回文字符串\n", str2)
		}
	}
}
