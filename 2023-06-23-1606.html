1. 题目：计算并输出1-1000中的所有素数。
package main

import (
    "fmt"
)

func main() {
    for i := 2; i <= 1000; i++ {
        flag := true
        for j := 2; j*j <= i; j++ {
            if i%j == 0 {
                flag = false
                break
            }
        }
        if flag {
            fmt.Printf("%d ", i)
        }
    }
}
  1. 题目：对数组进行冒泡排序。
package main

import (
    "fmt"
)

func bubbleSort(arr []int) {
    n := len(arr)
    for i := 0; i < n-1; i++ {
        for j := 0; j < n-1-i; j++ {
            if arr[j] > arr[j+1] {
                arr[j], arr[j+1] = arr[j+1], arr[j]
            }
        }
    }
}

func main() {
    arr := []int{5, 4, 3, 2, 1}
    bubbleSort(arr)
    fmt.Println(arr) // output: [1 2 3 4 5]
}
  1. 题目：将字符串反转。
package main

import (
    "fmt"
)

func reverse(s string) string {
    r := []rune(s)
    for i, j := 0, len(r)-1; i < j; i, j = i+1, j-1 {
        r[i], r[j] = r[j], r[i]
    }
    return string(r)
}

func main() {
    s := "hello"
    fmt.Println(reverse(s)) // output: olleh
}
  1. 题目：求出两个数的最大公约数。
package main

import (
    "fmt"
)

func gcd(a, b int) int {
    if b == 0 {
        return a
    }
    return gcd(b, a%b)
}

func main() {
    a, b := 12, 16
    fmt.Println(gcd(a, b)) // output: 4
}
  1. 题目：判断一个字符串是否是回文串。
package main

import (
    "fmt"
)

func isPalindrome(s string) bool {
    n := len(s)
    for i := 0; i < n/2; i++ {
        if s[i] != s[n-i-1] {
            return false
        }
    }
    return true
}

func main() {
    s1 := "level"
    s2 := "hello"
    fmt.Println(isPalindrome(s1)) // output: true
    fmt.Println(isPalindrome(s2)) // output: false
}
