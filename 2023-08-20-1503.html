package main

import "fmt"

func main() {
    // 练习1：打印Hello, World!
    fmt.Println("Hello, World!")

    // 练习2：将两个整数相加并打印结果
    num1 := 10
    num2 := 5
    sum := num1 + num2
    fmt.Println("Sum:", sum)

    // 练习3：判断一个数是否为偶数，并打印结果
    num := 7
    if num%2 == 0 {
        fmt.Println(num, "is even")
    } else {
        fmt.Println(num, "is odd")
    }

    // 练习4：使用循环打印1到10之间的所有整数
    for i := 1; i <= 10; i++ {
        fmt.Println(i)
    }

    // 练习5：使用函数计算并返回两个整数的乘积
    multiply(6, 3)
}

func multiply(a, b int) int {
    product := a * b
    return product
}
