package main

import "fmt"

func main() {
	// 练习1：打印Hello, World!
	fmt.Println("Hello, World!")

	// 练习2：计算两个数的和
	num1 := 1
	num2 := 2
	sum := num1 + num2
	fmt.Println("The sum of", num1, "and", num2, "is", sum)

	// 练习3：判断字符串是否是回文字符串
	str := "level"
	isPalindrome := true
	for i := 0; i < len(str)/2; i++ {
		if str[i] != str[len(str)-1-i] {
			isPalindrome = false
			break
		}
	}
	fmt.Println("Is", str, "a palindrome?", isPalindrome)

	// 练习4：计算斐波那契数列第n项的值
	n := 10
	fib := []int{0, 1}
	for i := 2; i <= n; i++ {
		fib = append(fib, fib[i-1]+fib[i-2])
	}
	fmt.Println("The", n, "th number in Fibonacci sequence is", fib[n])

	// 练习5：使用递归计算阶乘
	n = 5
	factorial := calcFactorial(n)
	fmt.Println("The factorial of", n, "is", factorial)
}

func calcFactorial(n int) int {
	if n <= 0 {
		return 1
	}
	return n * calcFactorial(n-1)
}
