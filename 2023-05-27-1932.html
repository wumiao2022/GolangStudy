这是一道基于Golang的练习题，可以帮助您提升对goroutine和chan的理解：

package main

import (
	"fmt"
	"time"
)

func main() {
	ch := make(chan int)
	go func() {
		for i := 0; i < 5; i++ {
			fmt.Println("send:", i)
			ch <- i
			time.Sleep(time.Second)
		}
		close(ch)
	}()

	for {
		v, ok := <-ch
		if ok {
			fmt.Println("receive:", v)
		} else {
			fmt.Println("channel closed")
			break
		}
	}
}
