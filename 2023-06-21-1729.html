1. 反转字符串：

    func reverseString(s string) string {
r := []rune(s)
for i, j := 0, len(r)-1; i < len(r)/2; i, j = i+1, j-1 {
    r[i], r[j] = r[j], r[i]
}
return string(r)
}

  2. 求一个切片中的最大值：

    func getMax(s []int) int {
max := s[0]
for _, v := range s {
    if v > max {
        max = v
    }
}
return max
}

  3. 斐波那契数列：

    func fibonacci(n int) int {
if n < 2 {
    return n
}
return fibonacci(n-1) + fibonacci(n-2)
}

  4. 冒泡排序：

    func bubbleSort(s []int) []int {
for i := 0; i < len(s)-1; i++ {
    for j := i + 1; j < len(s); j++ {
        if s[i] > s[j] {
            s[i], s[j] = s[j], s[i]
        }
    }
}
return s
}

  5. 判断一个数是否为素数：

    func isPrime(n int) bool {
if n < 2 {
    return false
}
for i := 2; i*i <= n; i++ {
    if n%i == 0 {
        return false
    }
}
return true
}
