package main

import (
	"fmt"
)

func main() {
	// 练习1: 打印数字1到10
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}

	// 练习2: 计算两个整数的和并输出结果
	a := 10
	b := 5
	sum := a + b
	fmt.Println(sum)

	// 练习3: 判断一个整数是否是偶数
	num := 8
	if num%2 == 0 {
		fmt.Println("偶数")
	} else {
		fmt.Println("奇数")
	}

	// 练习4: 遍历数组并打印每个元素
	arr := []int{1, 2, 3, 4, 5}
	for _, value := range arr {
		fmt.Println(value)
	}

	// 练习5: 将字符串反转并输出结果
	str := "Hello, Golang"
	reversedStr := ""
	for _, char := range str {
		reversedStr = string(char) + reversedStr
	}
	fmt.Println(reversedStr)
}
