// 练习1：打印1到100之间的所有整数
for i := 1; i <= 100; i++ {
    fmt.Println(i)
}

// 练习2：计算1到100之间所有整数的和
sum := 0
for i := 1; i <= 100; i++ {
    sum += i
}
fmt.Println(sum)

// 练习3：判断一个数是否为素数
num := 17
isPrime := true
for i := 2; i < num; i++ {
    if num%i == 0 {
        isPrime = false
        break
    }
}
fmt.Println(isPrime)

// 练习4：计算斐波那契数列的前10个数
fib := []int{0, 1}
for i := 2; i < 10; i++ {
    fib = append(fib, fib[i-1]+fib[i-2])
}
fmt.Println(fib)

// 练习5：实现一个冒泡排序算法
arr := []int{5, 3, 8, 2, 1}
for i := 0; i < len(arr)-1; i++ {
    for j := 0; j < len(arr)-1-i; j++ {
        if arr[j] > arr[j+1] {
            arr[j], arr[j+1] = arr[j+1], arr[j]
        }
    }
}
fmt.Println(arr)
