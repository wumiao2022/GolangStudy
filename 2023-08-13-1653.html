package main

import (
	"fmt"
)

func main() {
	// 练习1：使用for循环打印出1到10的数字
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}

	// 练习2: 声明一个字符串变量name并赋值为"GoLang"
	name := "GoLang"
	fmt.Println(name)

	// 练习3: 声明一个整数变量num1并赋值为10，声明一个整数变量num2并赋值为5，将num1和num2相加并打印出结果
	num1 := 10
	num2 := 5
	sum := num1 + num2
	fmt.Println(sum)

	// 练习4: 声明一个切片slice1，并添加三个整数元素：1, 2, 3，然后打印出切片长度和容量
	slice1 := []int{1, 2, 3}
	fmt.Printf("切片长度：%d, 切片容量：%d\n", len(slice1), cap(slice1))

	// 练习5: 声明一个字典map1，键的类型为字符串，值的类型为整数，并添加三个键值对：{"a": 1, "b": 2, "c": 3}，然后打印出字典的长度
	map1 := map[string]int{
		"a": 1,
		"b": 2,
		"c": 3,
	}
	fmt.Println("字典长度：", len(map1))

	// 练习6：声明一个函数add，接受两个整数参数并返回它们相加的结果，然后调用该函数并打印出结果
	add := func(a, b int) int {
		return a + b
	}
	result := add(5, 3)
	fmt.Println(result)
}

这是一个简单的练习示例代码，包括使用for循环打印数字、定义变量、操作切片和字典、声明函数等基础知识。如果你对每个练习的结果感到好奇，可以运行代码并观察输出结果。不断练习可以帮助你更熟悉Golang的语法和特性。希望对你的学习有所帮助！
