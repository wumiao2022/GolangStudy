package main

import "fmt"

func main() {
	// Example 1: Hello World
	fmt.Println("Hello, World!")

	// Example 2: Addition
	fmt.Println(1 + 2)

	// Example 3: Variables
	name := "John"
	age := 30
	fmt.Printf("My name is %s and I am %d years old.\n", name, age)

	// Example 4: Loops
	for i := 0; i < 5; i++ {
		fmt.Println(i)
	}

	// Example 5: Conditions
	x := 10
	if x > 5 {
		fmt.Println("x is greater than 5")
	} else {
		fmt.Println("x is less than or equal to 5")
	}
}
