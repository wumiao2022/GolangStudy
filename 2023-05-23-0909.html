以下是一道Golang练习题的答案代码：

package main

import (
	"fmt"
)

func main() {
	// 给定一个整数数组nums和一个目标值target，在数组中找出和为目标值的两个整数
	nums := []int{2, 7, 11, 15}
	target := 9

	// 创建一个map，用于存储目标值和当前值的差值及其所在的下标
	diffMap := make(map[int]int)

	for i, num := range nums {
		if j, ok := diffMap[target-num]; ok {
			// 如果map中存在target-num这个差值，那么说明当前值和之前的某个值的和为目标值
			fmt.Printf("[%d, %d]\n", j, i)
			break
		} else {
			// 否则将当前值和目标值的差值及其所在的下标存入map中，等待后续的遍历
			diffMap[num-target] = i
		}
	}
}
