以下是一个 Golang 练习实例代码：

package main

import (
    "fmt"
)

func main() {
    // 打印 1~10 的整数
    for i := 1; i <= 10; i++ {
        fmt.Println(i)
    }

    // 打印一个切片中的元素
    fruits := []string{"apple", "banana", "cherry"}
    for _, fruit := range fruits {
        fmt.Println(fruit)
    }

    // 遍历一个 map，打印 key 和 value
    scores := map[string]int{"Alice": 90, "Bob": 80, "Charlie": 70}
    for name, score := range scores {
        fmt.Printf("%s: %d\n", name, score)
    }

    // 定义一个结构体，并使用结构体初始化
    type Person struct {
        name string
        age  int
    }
    p := Person{name: "John", age: 30}
    fmt.Printf("My name is %s, I'm %d years old.\n", p.name, p.age)
}

这个代码片段包含了以下练习：

  1. 使用 for 循环打印 1~10 的整数。
  2. 使用 for range 遍历一个切片，并打印其中的元素。
  3. 使用 for range 遍历一个 map，并打印其中的 key 和 value。
  4. 定义一个结构体，并使用结构体初始化。
