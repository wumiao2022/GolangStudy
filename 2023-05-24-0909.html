  3. 实现一个函数，对给定的整数数组进行去重操作，返回去重后的切片。

    func uniqSlice(s []int) []int {
n := len(s)
if n == 0 {
    return s
}
m := make(map[int]bool)
j := 0
for i := 0; i < n; i++ {
    if !m[s[i]] {
        m[s[i]] = true
        s[j] = s[i]
        j++
    }
}
return s[:j]
}
