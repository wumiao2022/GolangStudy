1. 编写一个程序，将一个整数切片按从小到大的顺序排序。
package main

import (
	"fmt"
	"sort"
)

func main() {
	nums := []int{5, 2, 9, 3, 7, 6}
	sort.Ints(nums)
	fmt.Println(nums)
}
  1. 编写一个程序，比较两个字符串是否相等，不区分大小写。
package main

import (
	"fmt"
	"strings"
)

func main() {
	str1 := "Hello, world!"
	str2 := "hELLo, WoRLD!"
	if strings.EqualFold(str1, str2) {
		fmt.Println("Equal")
	} else {
		fmt.Println("Not equal")
	}
}
  1. 编写一个程序，找到一个整数切片中的最大值和最小值。
package main

import "fmt"

func main() {
	nums := []int{5, 2, 9, 3, 7, 6}
	min := nums[0]
	max := nums[0]
	for _, num := range nums {
		if num < min {
			min = num
		} else if num > max {
			max = num
		}
	}
	fmt.Printf("Min: %d, Max: %d\n", min, max)
}
  1. 编写一个程序，将一个整数切片的元素顺序颠倒。
package main

import "fmt"

func main() {
	nums := []int{5, 2, 9, 3, 7, 6}
	for i := 0; i < len(nums)/2; i++ {
		j := len(nums) - i - 1
		nums[i], nums[j] = nums[j], nums[i]
	}
	fmt.Println(nums)
}
  1. 编写一个程序，计算一个整数切片中所有元素的和。
package main

import "fmt"

func main() {
	nums := []int{5, 2, 9, 3, 7, 6}
	sum := 0
	for _, num := range nums {
		sum += num
	}
	fmt.Printf("Sum: %d\n", sum)
}
