1. 翻转字符串
func reverseString(s string) string {
    runes := []rune(s)
    for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 {
        runes[i], runes[j] = runes[j], runes[i]
    }
    return string(runes)
}
  1. 计算一组整数的平均值
func calculateAverage(nums []int) float64 {
    sum := 0
    for _, n := range nums {
        sum += n
    }
    return float64(sum) / float64(len(nums))
}
  1. 检查一个字符串是否是回文字符串
func isPalindrome(s string) bool {
    runes := []rune(s)
    for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 {
        if runes[i] != runes[j] {
            return false
        }
    }
    return true
}
  1. 给定一个整数数组，找到两个数的和等于给定目标值，返回它们的索引
func findTwoSum(nums []int, target int) []int {
    m := make(map[int]int)
    for i, n := range nums {
        diff := target - n
        if j, ok := m[diff]; ok {
            return []int{j, i}
        }
        m[n] = i
    }
    return nil
}
  1. 给定一个链表，反转它
type ListNode struct {
    Val  int
    Next *ListNode
}

func reverseList(head *ListNode) *ListNode {
    var prev *ListNode
    curr := head
    for curr != nil {
        next := curr.Next
        curr.Next = prev
        prev = curr
        curr = next
    }
    return prev
}
