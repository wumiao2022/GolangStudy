  1. 统计一个字符串中每个字符出现的次数，返回一个map，key为字符，value为出现次数。
package main

import "fmt"

func count(str string) map[rune]int {
    result := make(map[rune]int)
    for _, c := range str {
        result[c]++
    }
    return result
}

func main() {
    str := "hello world"
    fmt.Println(count(str))
}
  1. 给定一个整数数组，返回去重后的数组。
package main

import "fmt"

func unique(arr []int) []int {
    result := make([]int, 0)
    seen := make(map[int]bool)
    for _, num := range arr {
        if !seen[num] {
            result = append(result, num)
            seen[num] = true
        }
    }
    return result
}

func main() {
    arr := []int{1, 2, 3, 2, 1, 4}
    fmt.Println(unique(arr))
}
  1. 编写一个函数，输出斐波那契数列的前n个数字。
package main

import "fmt"

func Fibonacci(n int) []int {
    if n <= 0 {
        return nil
    }
    if n == 1 {
        return []int{0}
    }
    result := []int{0, 1}
    for i := 2; i < n; i++ {
        result = append(result, result[i-1]+result[i-2])
    }
    return result
}

func main() {
    fmt.Println(Fibonacci(10))
}
  1. 编写一个函数，判断一个字符串是否为回文字符串。
package main

import "fmt"

func isPalindrome(str string) bool {
    for i := 0; i < len(str)/2; i++ {
        if str[i] != str[len(str)-1-i] {
            return false
        }
    }
    return true
}

func main() {
    fmt.Println(isPalindrome("racecar"))
    fmt.Println(isPalindrome("hello"))
}
  1. 编写一个函数，输出1-100之间的所有素数。
package main

import "fmt"

func isPrime(n int) bool {
    if n <= 1 {
        return false
    }
    for i := 2; i*i <= n; i++ {
        if n%i == 0 {
            return false
        }
    }
    return true
}

func findPrimes() []int {
    result := make([]int, 0)
    for i := 2; i <= 100; i++ {
        if isPrime(i) {
            result = append(result, i)
        }
    }
    return result
}

func main() {
    fmt.Println(findPrimes())
}
