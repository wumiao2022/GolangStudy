package main

import "fmt"

func main() {
	// Exercise 1: Print "Hello, World!"
	fmt.Println("Hello, World!")

	// Exercise 2: Print the sum of two numbers
	num1 := 5
	num2 := 10
	sum := num1 + num2
	fmt.Println("Sum:", sum)

	// Exercise 3: Print the product of three numbers
	num3 := 2
	num4 := 3
	product := num1 * num2 * num3 * num4
	fmt.Println("Product:", product)

	// Exercise 4: Determine if a number is even or odd
	num5 := 7
	if num5%2 == 0 {
		fmt.Println("Even")
	} else {
		fmt.Println("Odd")
	}

	// Exercise 5: Calculate the factorial of a number
	num6 := 5
	factorial := 1
	for i := 1; i <= num6; i++ {
		factorial *= i
	}
	fmt.Println("Factorial:", factorial)
}
