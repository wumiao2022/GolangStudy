// 1. 判断一个数是否为偶数 func isEven(num int) bool { return num%2 == 0 }

// 2. 判断一个字符串是否为回文串 func isPalindrome(str string) bool { for i := 0; i < len(str)/2; i++ { if str[i] != str[len(str)-1-i] { return false } } return true }

// 3. 找出数组中的最大值 func findMax(nums []int) int { max := nums[0] for i := 1; i < len(nums); i++ { if nums[i] > max { max = nums[i] } } return max }

// 4. 实现阶乘函数 func factorial(n int) int { if n <= 1 { return 1 } return n * factorial(n-1) }

// 5. 判断一个数是否为素数 func isPrime(num int) bool { if num <= 1 { return false } for i := 2; i*i <= num; i++ { if num%i == 0 { return false } } return true }
