1. 生成10个随机数，输出其中最大值和最小值
package main

import (
    "fmt"
    "math/rand"
    "time"
)

func main() {
    rand.Seed(time.Now().UnixNano())

    var max, min int = rand.Intn(100), rand.Intn(100)

    for i := 1; i < 10; i++ {
        num := rand.Intn(100)
        fmt.Printf("随机数%d：%d\n", i, num)

        if num > max {
            max = num
        }

        if num < min {
            min = num
        }
    }

    fmt.Printf("最大值：%d，最小值：%d", max, min)
}

  1. 打印出1024到2048之间的素数
package main

import (
    "fmt"
    "math"
)

func main() {
    for i := 1024; i <= 2048; i++ {
        if isPrime(i) {
            fmt.Println(i)
        }
    }
}

func isPrime(num int) bool {
    if num <= 1 {
        return false
    }

    if num == 2 || num == 3 {
        return true
    }

    if num%2 == 0 || num%3 == 0 {
        return false
    }

    for i := 5; i <= int(math.Sqrt(float64(num))); i += 6 {
        if num%i == 0 || num%(i+2) == 0 {
            return false
        }
    }

    return true
}
  1. 打印出九九乘法表
package main

import (
    "fmt"
)

func main() {
    for i := 1; i < 10; i++ {
        for j := 1; j <= i; j++ {
            fmt.Printf("%d*%d=%-2d ", j, i, i*j)
        }
        fmt.Println()
    }
}
  1. 冒泡排序
package main

import (
    "fmt"
)

func main() {
    arr := []int{5, 3, 8, 6, 10, 7, 2, 1}

    for i := 0; i < len(arr)-1; i++ {
        for j := 0; j < len(arr)-1-i; j++ {
            if arr[j] > arr[j+1] {
                arr[j], arr[j+1] = arr[j+1], arr[j]
            }
        }
    }

    fmt.Println(arr)
}
