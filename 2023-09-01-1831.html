package main

import (
	"fmt"
	"time"
)

func main() {
	// 练习1：打印当前时间
	fmt.Println(time.Now())

	// 练习2：判断一个数字是奇数还是偶数
	num := 23
	if num%2 == 0 {
		fmt.Println("偶数")
	} else {
		fmt.Println("奇数")
	}

	// 练习3：计算阶乘
	factorial := 1
	n := 5
	for i := 1; i <= n; i++ {
		factorial *= i
	}
	fmt.Printf("%d的阶乘为%d\n", n, factorial)
}
