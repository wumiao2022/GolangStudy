  2. 反转字符串

    func reverse(s string) string {
runes := []rune(s)
for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 {
    runes[i], runes[j] = runes[j], runes[i]
}
return string(runes)
}
    1. 判断一个字符串是否为回文串 go func isPalindrome(s string) bool { runes := []rune(s) for i := 0; i < len(runes)/2; i++ { if runes[i] != runes[len(runes)-i-1] { return false } } return true }

  3. 实现冒泡排序

    func bubbleSort(nums []int) {
for i := 0; i < len(nums)-1; i++ {
    for j := 0; j < len(nums)-i-1; j++ {
        if nums[j] > nums[j+1] {
            nums[j], nums[j+1] = nums[j+1], nums[j]
        }
    }
}
