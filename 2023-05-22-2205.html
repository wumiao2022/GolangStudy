以下是一个练习实例代码，可以用于练习Golang并且提高编程能力。

package main

import "fmt"

func main() {
    // 打印1-10的数字
	for i := 1; i <= 10; i++ {
		fmt.Printf("%d ", i)
	}
	fmt.Println()

    // 打印0-100内的偶数
	for i := 0; i <= 100; i += 2 {
		fmt.Printf("%d ", i)
	}
	fmt.Println()

    // 打印1-100内的能被3整除但不能被5整除的数
	for i := 1; i <= 100; i++ {
		if i%3 == 0 && i%5 != 0 {
			fmt.Printf("%d ", i)
		}
	}
	fmt.Println()

    // 打印10以内的斐波那契数列
	a, b := 0, 1
	for i := 0; i < 10; i++ {
		fmt.Printf("%d ", a)
		a, b = b, a+b
	}
	fmt.Println()

    // 计算1-100内的奇数之和
	sum := 0
	for i := 1; i <= 100; i += 2 {
		sum += i
	}
	fmt.Printf("Sum of odd numbers between 1 to 100 is %d\n", sum)
}
