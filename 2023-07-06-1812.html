func main() { // 练习一：打印Hello, World! fmt.Println(“Hello, World!”)

// 练习二：计算两个整数的和
num1 := 10
num2 := 20
sum := num1 + num2
fmt.Printf("The sum of %d and %d is: %d\n", num1, num2, sum)

// 练习三：判断一个数是否为偶数

	fmt.Printf("%d is even.\n", num)
} else {
	fmt.Printf("%d is odd.\n", num)
}

// 练习四：计算字符串的长度
