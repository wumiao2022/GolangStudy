package main

import "fmt"

func main() {
	// 1. 声明一个整数变量并赋值为10
	var num int = 10

	// 2. 声明一个字符串变量并赋值为"Hello, Go"
	var str string = "Hello, Go"

	// 3. 声明一个浮点数变量并赋值为3.14
	var pi float64 = 3.14

	// 4. 声明一个布尔变量并赋值为true
	var isTrue bool = true

	// 5. 声明一个切片变量并初始化为包含整数1、2、3的切片
	var slice []int = []int{1, 2, 3}

	// 6. 声明一个Map变量并初始化为包含键值对的Map
	var m map[string]int = map[string]int{"one": 1, "two": 2, "three": 3}

	// 7. 声明一个函数并调用它
	printHello()

	// 8. 使用for循环打印1到10的整数
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}

	// 9. 声明一个结构体类型，并创建一个结构体对象
	type Person struct {
		Name string
		Age  int
	}

	p := Person{Name: "John", Age: 30}

	// 10. 使用if语句判断一个变量的值，并打印不同的结果
	if num > 5 {
		fmt.Println("Number is greater than 5")
	} else {
		fmt.Println("Number is less than or equal to 5")
	}
}

func printHello() {
	fmt.Println("Hello, Go!")
}
