package main

import "fmt"

func main() {
	// 寻找素数
	primeNumbers := findPrimeNumbers(20)
	fmt.Println(primeNumbers)

	// 反转字符串
	reversedString := reverseString("Hello, world!")
	fmt.Println(reversedString)

	// 计算斐波那契数列
	fibonacciNumbers := calculateFibonacci(10)
	fmt.Println(fibonacciNumbers)

	// 判断是否是回文字符串
	isPalindrome := checkPalindrome("level")
	fmt.Println(isPalindrome)

	// 冒泡排序
	numbers := []int{10, 7, 3, 9, 5}
	bubbleSort(numbers)
	fmt.Println(numbers)
}

// 寻找素数
func findPrimeNumbers(n int) []int {
	primeNumbers := make([]int, 0)
	isPrime := func(num int) bool {
		for i := 2; i*i <= num; i++ {
			if num%i == 0 {
				return false
			}
		}
		return num > 1
	}
	for i := 2; i <= n; i++ {
		if isPrime(i) {
			primeNumbers = append(primeNumbers, i)
		}
	}
	return primeNumbers
}

// 反转字符串
func reverseString(s string) string {
	runes := []rune(s)
	for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 {
		runes[i], runes[j] = runes[j], runes[i]
	}
	return string(runes)
}

// 计算斐波那契数列
func calculateFibonacci(n int) []int {
	fibonacciNumbers := make([]int, n)
	fibonacciNumbers[0] = 0
	fibonacciNumbers[1] = 1
	for i := 2; i < n; i++ {
		fibonacciNumbers[i] = fibonacciNumbers[i-1] + fibonacciNumbers[i-2]
	}
	return fibonacciNumbers
}

// 判断是否是回文字符串
func checkPalindrome(s string) bool {
	runes := []rune(s)
	for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 {
		if runes[i] != runes[j] {
			return false
		}
	}
	return true
}

// 冒泡排序
func bubbleSort(arr []int) {
	n := len(arr)
	for i := 0; i < n-1; i++ {
		for j := 0; j < n-1-i; j++ {
			if arr[j] > arr[j+1] {
				arr[j], arr[j+1] = arr[j+1], arr[j]
			}
		}
	}
}
