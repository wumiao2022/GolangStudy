package main

    fmt.Println("Hello, World!")
}

这是一个使用 for 循环输出九九乘法表的程序：

package main

import "fmt"

func main() {
    for i := 1; i <= 9; i++ {
        for j := 1; j <= i; j++ {
            fmt.Printf("%d*%d=%d ", j, i, i*j)
        }
        fmt.Println()
    }
}

这是一个判断一个数是否是素数的程序：

package main

import "fmt"

func main() {
    var n int
    fmt.Print("Enter a positive integer: ")
    fmt.Scan(&n)

    if n <= 1 {
        fmt.Printf("%d is not a prime number.\n", n)
        return
    }

    for i := 2; i <= n/2; i++ {
        if n%i == 0 {
            fmt.Printf("%d is not a prime number.\n", n)
            return
        }
    }

    fmt.Printf("%d is a prime number.\n", n)
}
