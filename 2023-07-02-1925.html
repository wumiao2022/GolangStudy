package main

import "fmt"

func main() {
    // Exercise: Print the sum of all even numbers from 1 to 100
    
    sum := 0
    
    for i := 1; i <= 100; i++ {
        if i%2 == 0 {
            sum += i
        }
    }
    
    fmt.Println("Sum of even numbers from 1 to 100:", sum)
}
