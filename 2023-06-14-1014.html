下面是一个Golang练习题示例代码：

package main

import "fmt"

func main() {
    // 将一个整数数组中的每个元素乘以2并输出
    nums := []int{1, 2, 3, 4, 5}
    for i := 0; i < len(nums); i++ {
        fmt.Println(nums[i] * 2)
    }

    // 将一个字符串数组中所有元素拼接起来并输出
    strs := []string{"hello", "world", "!"}
    var combinedStr string
    for i := 0; i < len(strs); i++ {
        combinedStr += strs[i]
    }
    fmt.Println(combinedStr)

    // 创建一个包含键值对的map并输出所有键和对应的值
    m := map[string]int{
        "apple":  1,
        "banana": 2,
        "pear":   3,
    }
    for k, v := range m {
        fmt.Printf("%v: %v\n", k, v)
    }
}

以上代码可以用来练习数组、字符串、map等常见数据类型的使用。
