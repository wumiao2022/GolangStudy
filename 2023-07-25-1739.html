package main

import "fmt"

func main() {
	// 练习1: 打印 "Hello, World!"
	fmt.Println("Hello, World!")

	// 练习2: 求两个整数的和
	a, b := 10, 20
	sum := a + b
	fmt.Println("Sum:", sum)

	// 练习3: 判断一个数是否为偶数
	num := 35
	if num%2 == 0 {
		fmt.Println(num, "is even")
	} else {
		fmt.Println(num, "is odd")
	}

	// 练习4: 打印1到100之间的所有偶数
	for i := 1; i <= 100; i++ {
		if i%2 == 0 {
			fmt.Println(i)
		}
	}

	// 练习5: 计算1到100之间的所有奇数的和
	sumOdd := 0
	for i := 1; i <= 100; i++ {
		if i%2 != 0 {
			sumOdd += i
		}
	}
	fmt.Println("Sum of odd numbers:", sumOdd)
}
