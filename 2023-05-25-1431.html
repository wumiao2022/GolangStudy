  1. 计算加法的程序
package main

import "fmt"

func main() {
    var a, b, sum int
    a = 10
    b = 20
    sum = a + b
    fmt.Println("The sum of", a, "and", b, "is", sum)
}
  1. 判断一个数是否是奇数的程序
package main

import "fmt"

func main() {
    var num int
    fmt.Print("Enter a number: ")
    fmt.Scan(&num)
    if num%2 == 0 {
        fmt.Println(num, "is an even number")
    } else {
        fmt.Println(num, "is an odd number")
    }
}
  1. 判断一个数是否是质数的程序
package main

import (
    "fmt"
    "math"
)

func main() {
    var num int
    fmt.Print("Enter a number: ")
    fmt.Scan(&num)

    isPrime := true
    for i := 2; i <= int(math.Sqrt(float64(num))); i++ {
        if num%i == 0 {
            isPrime = false
            break
        }
    }

    if isPrime {
        fmt.Println(num, "is a prime number")
    } else {
        fmt.Println(num, "is not a prime number")
    }
}
