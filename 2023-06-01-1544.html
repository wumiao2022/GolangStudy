	score := 85

  1. 循环语句：
package main

import "fmt"

func main() {
	// for loop
	for i := 0; i < 5; i++ {
		fmt.Println(i)
	}

	// while loop
	j := 0
	for j < 5 {
		fmt.Println(j)
		j++
	}

	// infinite loop
	k := 0
	for {
		fmt.Println(k)
		k++
	}
}
  1. 数组和切片：
package main

import "fmt"

func main() {
	// array
	var arr [5]int
	arr[0] = 1
	arr[1] = 2
	arr[2] = 3
	arr[3] = 4
	arr[4] = 5
	fmt.Println(arr)

	// slice
	slice := make([]int, 3, 5)
	slice[0] = 1
	slice[1] = 2
	slice[2] = 3
	fmt.Println(slice)
	fmt.Println("Length:", len(slice))
	fmt.Println("Capacity:", cap(slice))

	slice = append(slice, 4, 5, 6)
	fmt.Println(slice)
	fmt.Println("Length:", len(slice))
	fmt.Println("Capacity:", cap(slice))

	slice2 := slice[2:4]
	fmt.Println(slice2)
	fmt.Println("Length:", len(slice2))
	fmt.Println("Capacity:", cap(slice2))
}
  1. 结构体和方法：
package main

import "fmt"

type Student struct {
	name   string
	age    int
	gender string
}

func (s *Student) sayHello() {
	fmt.Println("Hello, my name is", s.name)
}

func main() {
	stu := Student{name: "Jack", age: 18, gender: "male"}
	fmt.Println(stu)
	stu.sayHello()
}
  1. 并发和通道：
package main

import (
	"fmt"
	"time"
)

func sayHello() {
	fmt.Println("Hello")
}

func count(id int, ch chan int) {
	for i := 1; i <= 5; i++ {
		fmt.Println(id, ":", i)
		time.Sleep(time.Second)
	}
	ch <- id
}

func main() {
	// goroutine
	go sayHello()

	// channel
	ch := make(chan int)
	go count(1, ch)
	go count(2, ch)
	id1 := <-ch
	id2 := <-ch
	fmt.Println(id1, id2)
}
