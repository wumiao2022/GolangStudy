以下是一道Golang练习题示例：

// 编写一个函数，返回数组中的最大值和次大值
func findMaxAndSecondMax(nums []int) (max, secondMax int) {
    max, secondMax = nums[0], nums[1]
    if secondMax > max {
        max, secondMax = secondMax, max
    }
    for _, num := range nums[2:] {
        if num > max {
            max, secondMax = num, max
        } else if num > secondMax {
            secondMax = num
        }
    }
    return
}

func main() {
    nums := []int{3, 5, 2, 8, 9, 1}
    max, secondMax := findMaxAndSecondMax(nums)
    fmt.Println("Max:", max)
    fmt.Println("Second Max:", secondMax)
}
