  1. 判断一个数是否是质数：
func isPrime(n int) bool {
    if n <= 1 {
        return false
    }
    for i := 2; i*i <= n; i++ {
        if n%i == 0 {
            return false
        }
    }
    return true
}

// 使用示例
fmt.Println(isPrime(7)) // true
fmt.Println(isPrime(8)) // false
  1. 获取切片中的最大值：
func max(slice []int) int {
    if len(slice) == 0 {
        return 0
    }
    result := slice[0]
    for _, value := range slice {
        if value > result {
            result = value
        }
    }
    return result
}

// 使用示例
slice := []int{1, 3, 5, 7, 9, 2, 4, 6, 8}
fmt.Println(max(slice)) // 9
  1. 使用结构体表示一个人的基本信息，并输出：
