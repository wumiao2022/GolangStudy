package main

import "fmt"

func main() {
	// 1. 输出 "Hello, World!"
	fmt.Println("Hello, World!")

	// 2. 声明一个整数变量x，并赋值为10
	x := 10

	// 3. 打印变量x的值
	fmt.Println(x)

	// 4. 声明一个字符串变量name，并赋值为"Alice"
	name := "Alice"

	// 5. 打印变量name的值
	fmt.Println(name)

	// 6. 声明一个布尔变量isTrue，并赋值为true
	isTrue := true

	// 7. 打印变量isTrue的值
	fmt.Println(isTrue)

	// 8. 定义一个整型数组，并赋值为[1, 2, 3, 4, 5]
	arr := []int{1, 2, 3, 4, 5}

	// 9. 打印数组arr的每一个元素
	for i := 0; i < len(arr); i++ {
		fmt.Println(arr[i])
	}

	// 10. 声明一个函数add，接受两个整数参数，并返回它们的和
	add := func(a, b int) int {
		return a + b
	}

	// 11. 调用函数add，并将结果赋值给变量sum
	sum := add(2, 3)

	// 12. 打印变量sum的值
	fmt.Println(sum)
}
