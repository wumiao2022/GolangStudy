  1. 找出给定切片中的最大值
package main

import "fmt"

func main() {
  numbers := []int{3, 7, 2, 8, 1, 9, 4, 6, 5}

  max := numbers[0]
  for _, num := range numbers {
    if num > max {
      max = num
    }
  }

  fmt.Printf("The maximum number in the slice is %d\n", max)
}
  1. 计算斐波那契数列的第n个数
package main

import "fmt"

func main() {
  n := 10

  num1 := 0
  num2 := 1

  for i := 0; i < n; i++ {
    temp := num1
    num1 = num2
    num2 = temp + num2
  }

  fmt.Printf("The %dth number in the Fibonacci sequence is %d\n", n, num1)
}
  1. 判断一个字符串是否为回文数
package main

import (
  "fmt"
  "strings"
)

func main() {
  str := "racecar"

  reverseStr := ""
  for i := len(str) - 1; i >= 0; i-- {
    reverseStr += string(str[i])
  }

  if strings.ToLower(str) == strings.ToLower(reverseStr) {
    fmt.Printf("%s is a palindrome\n", str)
  } else {
    fmt.Printf("%s is not a palindrome\n", str)
  }
}
