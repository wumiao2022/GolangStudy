func main() { // 1. Hello, World! fmt.Println(“Hello, World!”)

// 2. Variables and Constants
var num1 int = 10
fmt.Println(num1)
const num2 float64 = 3.14
fmt.Println(num2)

// 3. Functions
result := add(5, 3)
fmt.Println(result)

// 4. Arrays and Slices
arr := [5]int{1, 2, 3, 4, 5}
slice := arr[1:3]
fmt.Println(slice)

// 5. Loops
for i := 0; i < 5; i++ {
	fmt.Println(i)
}

// 6. Conditionals
num := 10
if num > 5 {
	fmt.Println("Number is greater than 5")
} else {
	fmt.Println("Number is less than or equal to 5")
}

}
