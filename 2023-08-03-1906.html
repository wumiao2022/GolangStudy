func reverseString(str string) string { reversedStr := “” for i := len(str)-1; i >= 0; i– { reversedStr += string(str[i]) } return reversedStr }

func checkPrime(num int) bool { if num < 2 { return false } for i := 2; i <= int(math.Sqrt(float64(num))); i++ { if num%i == 0 { return false } } return true }
