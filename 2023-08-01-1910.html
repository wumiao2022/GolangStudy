func main() { // 数组切片的创建和遍历 nums := []int{1, 2, 3, 4, 5} for i := 0; i < len(nums); i++ { fmt.Println(nums[i]) }

// 使用 range 遍历切片
for _, num := range nums {
    fmt.Println(num)
}

// map 的创建和遍历
info := map[string]string{
    "name":   "John",
    "age":    "25",
    "gender": "male",
}

for key, value := range info {
    fmt.Println(key, value)
}

// 使用 defer 关键字延迟执行
defer fmt.Println("I am deferred")
fmt.Println("I am normal")

// 使用 panic 和 recover 处理错误
defer func() {
    if err := recover(); err != nil {
        fmt.Println(err)
    }
}()
panic("Something went wrong")

}
