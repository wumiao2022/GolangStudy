package main

import "fmt"

func main() {
    // Exercise 1: Hello, World!
    fmt.Println("Hello, World!")

    // Exercise 2: Sum of two numbers
    a := 10
    b := 20
    sum := a + b
    fmt.Println("Sum:", sum)

    // Exercise 3: Fibonacci sequence
    n := 10
    fibonacci := []int{0, 1}
    for i := 2; i < n; i++ {
        fibonacci = append(fibonacci, fibonacci[i-1]+fibonacci[i-2])
    }
    fmt.Println("Fibonacci Sequence:", fibonacci)

    // Exercise 4: Factorial of a number
    num := 5
    factorial := 1
    for i := 1; i <= num; i++ {
        factorial *= i
    }
    fmt.Println("Factorial of", num, "is", factorial)
}

以上是一个示例代码，其中包含了四个练习实例：

  1. 打印 “Hello, World!”
  2. 计算两个数的和
  3. 生成 Fibonacci 数列
  4. 计算一个数的阶乘
