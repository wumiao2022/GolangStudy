// 练习1：打印九九乘法表

package main

import “fmt”

func main() { for i := 1; i <= 9; i++ { for j := 1; j <= i; j++ { fmt.Printf(“%d*%d=%d “, j, i, j*i) } fmt.Println() } }

// 练习2：判断闰年

package main

import “fmt”

func main() { year := 2020 result := “” if year%4 == 0 { if year%100 == 0 { if year%400 == 0 { result = “是闰年” } else { result = “不是闰年” } } else { result = “是闰年” } } else { result = “不是闰年” } fmt.Printf(“%d年%s”, year, result) }

// 练习3：冒泡排序

package main

import “fmt”

func main() { arr := []int{5, 2, 4, 6, 1, 3} for i := 0; i < len(arr)-1; i++ { for j := 0; j < len(arr)-1-i; j++ { if arr[j] > arr[j+1] { arr[j], arr[j+1] = arr[j+1], arr[j] } } } fmt.Println(arr) }

// 练习4：递归求阶乘

package main

import “fmt”

func factorial(n int) int { if n == 0 { return 1 } else { return n * factorial(n-1) } }

func main() { fmt.Println(factorial(5)) }

// 练习5：斐波那契数列

package main

import “fmt”

func Fibonacci(n int) int { if n <= 1 { return n } else { return Fibonacci(n-1) + Fibonacci(n-2) } }

func main() { for i := 0; i < 10; i++ { fmt.Printf(“%d “, Fibonacci(i)) } }
