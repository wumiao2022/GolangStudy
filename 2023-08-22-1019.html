package main

import "fmt"

func main() {
	// 示例1：输出Hello, World!
	fmt.Println("Hello, World!")

	// 示例2：计算两数之和，并输出结果
	num1 := 10
	num2 := 20
	sum := num1 + num2
	fmt.Println("The sum of", num1, "and", num2, "is", sum)

	// 示例3：判断一个数是否为正数，并输出结果
	num := -5
	if num > 0 {
		fmt.Println(num, "is a positive number.")
	} else {
		fmt.Println(num, "is not a positive number.")
	}

	// 示例4：输出1到10的所有偶数
	for i := 1; i <= 10; i++ {
		if i%2 == 0 {
			fmt.Println(i)
		}
	}

	// 示例5：查找一个切片中的最大值，并输出结果
	numbers := []int{1, 5, 3, 9, 7, 2, 8, 4, 6}
	max := numbers[0]
	for _, num := range numbers {
		if num > max {
			max = num
		}
	}
	fmt.Println("The maximum number is", max)
}
