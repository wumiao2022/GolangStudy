以下是一个简单的Golang程序，实现了将字符串翻转输出的功能。

package main

import (
	"fmt"
)

func main() {
	str := "Hello, world!"
	
	// 将字符串转换成rune数组
	strArray := []rune(str)
	
	// 翻转字符串
	for i, j := 0, len(strArray)-1; i < j; i, j = i+1, j-1 {
		strArray[i], strArray[j] = strArray[j], strArray[i]
	}
	
	// 输出翻转后的字符串
	fmt.Println(string(strArray))
}
