以下是一个简单的Golang练习实例，它打印出1到100的数字，但是对于3的倍数，它打印“Fizz”；对于5的倍数，它打印“Buzz”；对于既是3的倍数也是5的倍数的数字，它打印“FizzBuzz”。

package main

import "fmt"

func main() {
    for i := 1; i <= 100; i++ {
        if i % 3 == 0 && i % 5 == 0 {
            fmt.Println("FizzBuzz")
        } else if i % 3 == 0 {
            fmt.Println("Fizz")
        } else if i % 5 == 0 {
            fmt.Println("Buzz")
        } else {
            fmt.Println(i)
        }
    }
}
