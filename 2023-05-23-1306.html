1. 打印Hello, World!

    package main
import "fmt"
func main() {
fmt.Println("Hello, World!")
}

  2. 变量定义和使用

    package main
import "fmt"
func main() {
var message string = "Hello, World!" // 定义一个字符串变量
fmt.Println(message) // 输出：Hello, World!
var num int = 10 // 定义一个整型变量
num = num + 5 // 修改变量值
fmt.Println(num) // 输出：15
var boolVar bool = true // 定义一个布尔类型变量
fmt.Println(boolVar) // 输出：true
}

  3. for循环

    package main
import "fmt"
func main() {
for i := 0; i < 5; i++ {
    fmt.Println(i)
}
// 使用for循环遍历数组
array := [3]int{1, 2, 3}
for i, value := range array {
    fmt.Printf("index: %d, value: %d\n", i, value)
}
}

  4. if语句

    package main
import "fmt"
func main() {
num := 10
if num > 5 {
    fmt.Println("num is greater than 5")
} else {
    fmt.Println("num is less than or equal to 5")
}
}

  5. 函数定义和调用

    package main
import "fmt"
// 定义一个计算两个整数之和的函数
func add(x int, y int) int {
return x + y
}
func main() {
result := add(3, 4) // 调用add函数
fmt.Println(result) // 输出：7
}
