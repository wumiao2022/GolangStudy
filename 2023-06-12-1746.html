)

        fmt.Println("Usage:")
        fmt.Println("  add <item>     Add a new item")
        fmt.Println("  remove <num>   Remove the item at <num>")
        fmt.Println("  show           Show all items")
        return
    }

    switch args[0] {
    case "add":

            fmt.Println("Usage: add <item>")
