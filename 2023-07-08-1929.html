func main() { // 1. 输出Hello, World! fmt.Println(“Hello, World!”)

// 2. 声明变量并赋值
var x int = 5
fmt.Println(x)

// 3. 循环输出1到10
for i := 1; i <= 10; i++ {
	fmt.Println(i)
}

// 4. 判断语句
if x > 10 {
	fmt.Println("x is greater than 10")
} else if x < 10 {
	fmt.Println("x is less than 10")
} else {
	fmt.Println("x is equal to 10")
}

// 5. 数组遍历
numbers := [5]int{1, 2, 3, 4, 5}
for _, num := range numbers {
	fmt.Println(num)
}

// 6. 切片操作
slice := []int{1, 2, 3, 4, 5}
fmt.Println(slice[1:3])

// 7. 函数调用
result := add(3, 4)
fmt.Println(result)

}

// 定义一个函数，实现两数相加 func add(a, b int) int { return a + b }
