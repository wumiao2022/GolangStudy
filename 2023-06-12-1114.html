以下是一道简单的练习题：将一个整数数组按照奇数和偶数分为两个数组，并且要求奇数数组有序排列，偶数数组有序排列，但数组内部顺序不变。实现代码如下：

func splitArray(nums []int) ([]int, []int) {
    even := []int{}
    odd := []int{}

    for _, num := range nums {
        if num%2 == 0 {
            even = append(even, num)
        } else {
            odd = append(odd, num)
        }
    }

    sort.Ints(even)
    sort.Ints(odd)

    return odd, even
}
