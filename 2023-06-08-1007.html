// 练习1：编写一个程序，计算输入的整数之和 package main

import ( “fmt” )

func main() { var n int var sum int

fmt.Println("请输入一个整数：")
fmt.Scanln(&n)

for i := 0; i <= n; i++ {
    sum += i
}

fmt.Println("从0到", n, "的整数之和为：", sum)

}

// 练习2：编写一个程序，统计输入的字符串中字母、数字、空格和其他字符的个数 package main

import ( “bufio” “fmt” “os” )

func main() { var str string var letter, digit, space, other int

fmt.Println("请输入一段字符串：")
reader := bufio.NewReader(os.Stdin)
str, _ = reader.ReadString('\n')

for _, s := range str {
    if s >= '0' && s <= '9' {
        digit++
    } else if (s >= 'a' && s <= 'z') || (s >= 'A' && s <= 'Z') {
        letter++
    } else if s == ' ' {
        space++
    } else {
        other++
    }
}

fmt.Println("字母个数：", letter)
fmt.Println("数字个数：", digit)
fmt.Println("空格个数：", space)
fmt.Println("其他字符个数：", other)

}

// 练习3：编写一个程序，求出一个切片中的最大值和最小值 package main

import ( “fmt” )

func main() { var nums []int var n, max, min int

fmt.Println("请输入需要求最大最小值的切片（以空格分隔）：")
for {
    _, err := fmt.Scan(&n)
    if err != nil {
        break
    }
    nums = append(nums, n)
}

if len(nums) > 0 {
    max = nums[0]
    min = nums[0]
}

for _, num := range nums {
    if num > max {
        max = num
    }
    if num < min {
        min = num
    }
}

fmt.Println("最大值为：", max)
fmt.Println("最小值为：", min)

}
