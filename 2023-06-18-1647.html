  1. 判断一个数是否为素数
func isPrime(num int) bool {
    if num < 2 {
        return false
    }
    for i := 2; i <= int(math.Sqrt(float64(num))); i++ {
        if num%i == 0 {
            return false
        }
    }
    return true
}

}
  1. 求一个正整数的阶乘
func factorial(num int) int {
    if num == 0 {
        return 1
    }
    return num * factorial(num-1)
}
