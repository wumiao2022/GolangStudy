1. 判断一个整数是否为质数
func isPrime(n int) bool {
    if n <= 1 {
        return false
    }
    for i := 2; i <= int(math.Sqrt(float64(n))); i++ {
        if n%i == 0 {
            return false
        }
    }
    return true
}
  1. 冒泡排序
func bubbleSort(nums []int) {
    n := len(nums)
    for i := 0; i < n-1; i++ {
        for j := 0; j < n-i-1; j++ {
            if nums[j] > nums[j+1] {
                nums[j], nums[j+1] = nums[j+1], nums[j]
            }
        }
    }
}
  1. 计算一个字符串中每个字符出现的次数
func countChars(str string) map[rune]int {
    counts := make(map[rune]int)
    for _, c := range str {
        counts[c]++
    }
    return counts
}
  1. 找到两个数组中的重复元素
func findDuplicates(nums1, nums2 []int) []int {
    set := make(map[int]struct{})
    var res []int
    for _, num := range nums1 {
        set[num] = struct{}{}
    }
    for _, num := range nums2 {
        if _, ok := set[num]; ok {
            res = append(res, num)
        }
    }
    return res
}
  1. 反转一个字符串
func reverseString(str string) string {
    n := len(str)
    runes := []rune(str)
    for i := 0; i < n/2; i++ {
        runes[i], runes[n-i-1] = runes[n-i-1], runes[i]
    }
    return string(runes)
}
