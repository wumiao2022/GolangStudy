这里是一道Golang练习题：编写一个函数，实现两个整数相除并返回结果，不使用除法、乘法和mod运算符。

func divide(dividend int, divisor int) int {
    if dividend == 0 {
        return 0
    }
    if divisor == 1 {
        return dividend
    }
    if dividend == math.MinInt32 && divisor == -1 {
        return math.MaxInt32
    }
    
    sign := 1
    if dividend < 0 {
        sign = -sign
        dividend = -dividend
    }
    if divisor < 0 {
        sign = -sign
        divisor = -divisor
    }
    
    res := 0
    for dividend >= divisor {
        tmp, multiple := divisor, 1
        for dividend >= (tmp << 1) {
            tmp <<= 1
            multiple <<= 1
        }
        dividend -= tmp
        res += multiple
    }
    
    if sign < 0 {
        res = -res
    }
    
    return res
}
