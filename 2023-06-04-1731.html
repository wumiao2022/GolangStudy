以下是一个简单的Golang练习示例，用于计算输入数字的平均值：

package main

import (
    "fmt"
)

func main() {
    var total, count, num float64 = 0, 0, 0
    for {
        fmt.Print("请输入一个数字（输入 -1 结束）：")
        fmt.Scanln(&num)
        if num == -1 {
            break
        }
        total += num
        count++
    }
    if count == 0 {
        fmt.Println("没有输入数字")
    } else {
        avg := total / count
        fmt.Printf("平均值为 %.2f\n", avg)
    }
}
