1. 反转字符串
package main

import "fmt"

func main() {
	s := "hello world"
	r := []rune(s)
	for i, j := 0, len(r)-1; i < len(r)/2; i, j = i+1, j-1 {
		r[i], r[j] = r[j], r[i]
	}
	fmt.Println(string(r))
}
  1. 计算斐波那契数列
package main

import "fmt"

func main() {
	fmt.Println(fibonacci(10))
}

func fibonacci(n int) []int {
	res := make([]int, n)
	if n <= 0 {
		return res
	}
	res[0] = 0
	if n == 1 {
		return res
	}
	res[1] = 1
	for i := 2; i < n; i++ {
		res[i] = res[i-1] + res[i-2]
	}
	return res
}
  1. 判断一个数是否为素数
package main

import "fmt"

func main() {
	fmt.Println(isPrime(7))
}

func isPrime(n int) bool {
	if n <= 1 {
		return false
	}
	for i := 2; i*i <= n; i++ {
		if n%i == 0 {
			return false
		}
	}
	return true
}
  1. 计算两个整数的最大公约数和最小公倍数
package main

import "fmt"

func main() {
	a, b := 12, 18
	fmt.Println("最大公约数是：", gcd(a, b))
	fmt.Println("最小公倍数是：", lcm(a, b))
}

func gcd(a, b int) int {
	if b == 0 {
		return a
	}
	return gcd(b, a%b)
}

func lcm(a, b int) int {
	return a * b / gcd(a, b)
}
  1. 判断一个字符串是否回文
package main

import "fmt"

func main() {
	fmt.Println(isPalindrome("level"))
}

func isPalindrome(s string) bool {
	r := []rune(s)
	for i, j := 0, len(r)-1; i < len(r)/2; i, j = i+1, j-1 {
		if r[i] != r[j] {
			return false
		}
	}
	return true
}
