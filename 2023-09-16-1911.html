package main

import "fmt"


// 练习1：输出Hello, World!
func exercise1() {
	fmt.Println("Hello, World!")
}

// 练习2：计算两个整数的和并输出结果
func exercise2() {
	a := 10
	b := 20
	sum := a + b
	fmt.Println("Sum:", sum)
}

// 练习3：判断一个数是否为偶数并输出结果
func exercise3() {
	num := 6
	if num % 2 == 0 {
		fmt.Println(num, "是偶数")
	} else {
		fmt.Println(num, "是奇数")
	}
}

// 练习4：计算一个整数的阶乘并输出结果
func exercise4() {
	num := 5
	factorial := 1
	for i := 1; i <= num; i++ {
		factorial *= i
	}
	fmt.Println("Factorial of", num, "is", factorial)
}

// 练习5：判断一个字符串是否为回文并输出结果
func exercise5() {
	str := "level"
	isPalindrome := true
	for i := 0; i < len(str)/2; i++ {
		if str[i] != str[len(str)-1-i] {
			isPalindrome = false
			break
		}
	}
	if isPalindrome {
		fmt.Println(str, "是回文")
	} else {
		fmt.Println(str, "不是回文")
	}
}

func main() {
	exercise1()
	exercise2()
	exercise3()
	exercise4()
	exercise5()
}
