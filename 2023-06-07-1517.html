package main

import (
    "fmt"
)

func main() {
    // 给定一个整数数组nums和一个目标值target，请在数组中找到两个数，它们的和等于目标值
    nums := []int{2, 7, 11, 15}
    target := 9
    result := twoSum(nums, target)
    fmt.Println(result)
}

// 两数之和
func twoSum(nums []int, target int) []int {
    numMap := make(map[int]int, len(nums))
    for i, num := range nums {
        if j, ok := numMap[target-num]; ok {
            return []int{j, i}
        }
        numMap[num] = i
    }
    return nil
}

此代码实现了一个”两数之和”的算法，可以找到一个整数数组中两个数之和等于目标值的数的下标。
