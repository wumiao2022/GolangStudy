package main

import "fmt"

func main() {
	// 1. 打印 "Hello, World!" 到控制台
	fmt.Println("Hello, World!")

	// 2. 定义一个整数变量 x，并将其赋值为 10
	x := 10

	// 3. 定义一个浮点数变量 y，并将其赋值为 3.14
	y := 3.14

	// 4. 计算 x 和 y 的和，并将结果赋值给变量 sum
	sum := x + int(y)

	// 5. 打印 sum 的值到控制台
	fmt.Println("Sum:", sum)

	// 6. 定义一个字符串变量 message，并将其赋值为 "Hello, Golang!"
	message := "Hello, Golang!"

	// 7. 打印 message 的值到控制台
	fmt.Println(message)

	// 8. 使用 for 循环打印出 1 到 10 的数字
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}

	// 9. 定义一个切片变量 numbers，并初始化为包含 5 个整数 1、2、3、4、5
	numbers := []int{1, 2, 3, 4, 5}

	// 10. 使用 for 循环遍历切片 numbers，并打印出每个元素的值
	for _, number := range numbers {
		fmt.Println(number)
	}
}
