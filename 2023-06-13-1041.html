1. 反转字符串

    func reverseString(s []byte) {
for i:=0; i<len(s)/2; i++ {
    s[i], s[len(s)-1-i] = s[len(s)-1-i], s[i]
}
}

  2. 判断素数

    func isPrime(n int) bool {
if n <= 1 {
    return false
}
for i:=2; i*i<=n; i++ {
    if n%i == 0 {
        return false
    }
}
return true
}

  3. 斐波那契数列

    func fibonacci(n int) []int {
if n <= 0 {
    return []int{0}
} else if n == 1 {
    return []int{0, 1}
}
f := fibonacci(n-1)
return append(f, f[n-1]+f[n-2])
}

  4. 冒泡排序

    func bubbleSort(arr []int) {
for i:=0; i<len(arr)-1; i++ {
    for j:=0; j<len(arr)-1-i; j++ {
        if arr[j] > arr[j+1] {
            arr[j], arr[j+1] = arr[j+1], arr[j]
        }
    }
}
}

  5. 判断回文字符串

    func isPalindrome(s string) bool {
for i:=0; i<len(s)/2; i++ {
    if s[i] != s[len(s)-1-i] {
        return false
    }
}
return true
}
