package main

import "fmt"

func main() {
	// 1. 输出"Hello, World!"到控制台
	fmt.Println("Hello, World!")

	// 2. 定义一个整型变量x，并将其赋值为10
	x := 10
	
	// 3. 将x的值加上5，并打印出结果
	x += 5
	fmt.Println(x)
	
	// 4. 编写一个函数，接收两个整型参数，返回它们之和
	sum := add(3, 5)
	fmt.Println(sum)
}

func add(a, b int) int {
	return a + b
}

package main

import "fmt"

func main() {
	// 1. 声明一个切片slice，长度为5，并初始化为[1, 2, 3, 4, 5]
	slice := []int{1, 2, 3, 4, 5}
	
	// 2. 将切片中的第3个元素修改为10
	slice[2] = 10
	
	// 3. 遍历切片，计算所有元素的和，并打印出结果
	sum := 0
	for _, num := range slice {
		sum += num
	}
	fmt.Println(sum)
	
	// 4. 创建一个新的切片newSlice，包含原切片中的前两个元素
	newSlice := slice[:2]
	fmt.Println(newSlice)
}

package main

import "fmt"

func main() {
	// 1. 声明一个map，使用string作为key，int作为value，并初始化为{"a": 1, "b": 2, "c": 3}
	m := map[string]int{
		"a": 1,
		"b": 2,
		"c": 3,
	}
	
	// 2. 将key为"b"的元素删除
	delete(m, "b")
	
	// 3. 遍历map，打印出所有的key和value
	for key, value := range m {
		fmt.Println(key, value)
	}
	
	// 4. 判断map中是否存在key为"c"的元素，并将结果打印出来
	_, exist := m["c"]
	fmt.Println(exist)
}

package main

import "fmt"

func main() {
	// 1. 声明一个结构体Person，包含name和age两个字段
	type Person struct {
		name string
		age  int
	}
	
	// 2. 创建一个Person类型的变量p，name为"John"，age为25
	p := Person{name: "John", age: 25}
	
	// 3. 打印出p的name和age字段
	fmt.Println(p.name, p.age)
	
	// 4. 创建一个指向Person类型的指针变量ptr，指向p
	ptr := &p
	
	// 5. 通过指针变量ptr修改p的age字段为30
	ptr.age = 30
	
	// 6. 再次打印出p的name和age字段，观察是否有变化
	fmt.Println(p.name, p.age)
}

package main

import "fmt"

func main() {
	// 1. 声明一个通道channel，用于传输字符串类型的数据
	ch := make(chan string)
	
	// 2. 启动一个并发函数，向通道发送5个字符串"Hello"，然后关闭通道
	go func() {
		for i := 0; i < 5; i++ {
			ch <- "Hello"
		}
		close(ch)
	}()
	
	// 3. 从通道接收数据并打印出来，直到通道关闭
	for {
		msg, ok := <-ch
		if !ok {
			break
		}
		fmt.Println(msg)
	}
}
