package main

import "fmt"

func main() {
    // 练习1：打印 "Hello, World!"
    fmt.Println("Hello, World!")

    // 练习2：计算并打印两个整数的和
    a := 5
    b := 3
    sum := a + b
    fmt.Println("Sum:", sum)

    // 练习3：计算并打印一个整数的平方
    c := 4
    square := c * c
    fmt.Println("Square:", square)

    // 练习4：计算并打印两个浮点数的乘积
    x := 2.5
    y := 3.5
    product := x * y
    fmt.Println("Product:", product)

    // 练习5：判断并打印一个数是否为偶数
    num := 6
    if num%2 == 0 {
        fmt.Println(num, "is even")
    } else {
        fmt.Println(num, "is odd")
    }
}
