以下为一份简单的习题，希望能帮助你巩固Golang的基础知识：

  1. 请编写一个函数，计算字符串中每个字母出现的次数，并以map的形式返回结果。
func charCount(s string) map[rune]int {
    count := make(map[rune]int)
    for _, c := range s {
        count[c]++
    }
    return count
}
  1. 请编写一个函数，接收一个字符串参数，将字符串中的每个单词的首字母转为大写，并返回修改后的字符串。
func toUpperFirst(s string) string {
    words := strings.Fields(s)
    for i, word := range words {
        words[i] = strings.ToUpper(string(word[0])) + word[1:]
    }
    return strings.Join(words, " ")
}
  1. 请编写一个函数，接收两个参数（一个整数切片和一个整数k），把整数切片分为多个长度为k的切片，并返回这些切片。
func splitSlice(nums []int, k int) [][]int {
    var result [][]int
    for i := 0; i < len(nums); i += k {
        end := i + k
        if end > len(nums) {
            end = len(nums)
        }
        result = append(result, nums[i:end])
    }
    return result
}
  1. 请编写一个函数，接收一个整数切片并排序，要求使用冒泡排序算法。
func bubbleSort(nums []int) {
    for i := 0; i < len(nums); i++ {
        for j := i + 1; j < len(nums); j++ {
            if nums[i] > nums[j] {
                nums[i], nums[j] = nums[j], nums[i]
            }
        }
    }
}
  1. 请编写一个函数，接收一个字符串和一个整数k，将字符串中的每个字符向右移动k个位置后返回新的字符串。
func rotateString(s string, k int) string {
    n := len(s)
    k %= n
    return s[n-k:] + s[:n-k]
}
