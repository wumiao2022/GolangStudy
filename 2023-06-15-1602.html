好的，这里给你一个示例代码：

package main

import "fmt"

func main() {
    // 练习1：输出Hello, World!
    fmt.Println("Hello, World!")

    // 练习2：输出1到10的整数
    for i := 1; i <= 10; i++ {
        fmt.Println(i)
    }

    // 练习3：输出2的幂次方，幂次方从0到10
    for i := 0; i <= 10; i++ {
        fmt.Println(1 << uint(i))
    }

    // 练习4：计算并输出5的阶乘
    n := 5
    result := 1
    for i := 1; i <= n; i++ {
        result *= i
    }
    fmt.Println(result)

    // 练习5：将一个整数数组反转并输出
    arr := []int{1, 2, 3, 4, 5}
    for i, j := 0, len(arr)-1; i < j; i, j = i+1, j-1 {
        arr[i], arr[j] = arr[j], arr[i]
    }
    fmt.Println(arr)
}
