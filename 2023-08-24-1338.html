package main

import "fmt"

func main() {
	// 练习1: 打印Hello, world!
	fmt.Println("Hello, world!")

	// 练习2: 计算并打印1+2的结果
	result := 1 + 2
	fmt.Println(result)

	// 练习3: 找出数组[1, 2, 3, 4, 5]中的偶数并打印
	numbers := []int{1, 2, 3, 4, 5}
	for _, num := range numbers {
		if num % 2 == 0 {
			fmt.Println(num)
		}
	}

	// 练习4: 编写一个函数，接收一个整数参数，判断该数是奇数还是偶数，并打印结果
	checkNumberType(7)
	checkNumberType(12)
}

func checkNumberType(num int) {
	if num % 2 == 0 {
		fmt.Println(num, "是偶数")
	} else {
		fmt.Println(num, "是奇数")
	}
}
