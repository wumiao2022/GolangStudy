  1. 实现一个函数，输入一个字符串，返回该字符串的长度和该字符串中所有数字字符的个数。
package main

import (
	"fmt"
)

func count(str string) (int, int) {
	length := len(str)
	count := 0
	for _, char := range str {
		if char >= '0' && char <= '9' {
			count++
		}
	}
	return length, count
}

func main() {
	str := "abc123def456"
	length, digitCount := count(str)
	fmt.Printf("字符串长度为%d，数字字符个数为%d\n", length, digitCount)
}
  1. 实现一个函数，输入一个字符串，将其中每个单词的首字母大写，并返回修改后的字符串。
package main

import (
	"fmt"
	"strings"
)

func titleCase(str string) string {
	words := strings.Split(str, " ")
	for i, word := range words {
		if len(word) > 0 {
			words[i] = strings.ToUpper(string(word[0])) + word[1:]
		}
	}
	return strings.Join(words, " ")
}

func main() {
	str := "this is a Test sTrIng"
	title := titleCase(str)
	fmt.Println(title)
}
  1. 实现一个函数，输入一个整数数组和一个目标值，返回数组中和为目标值的两个数的下标。
package main

import (
	"fmt"
)

func twoSum(nums []int, target int) []int {
	hashTable := make(map[int]int)
	for i, num := range nums {
		if j, ok := hashTable[target-num]; ok {
			return []int{j, i}
		}
		hashTable[num] = i
	}
	return nil
}

func main() {
	nums := []int{2, 7, 11, 15}
	target := 9
	result := twoSum(nums, target)
	fmt.Println(result)
}
  1. 实现一个函数，输入一个字符串数组，将其中所有字符串按照字典序升序排列后返回。
package main

import (
	"fmt"
	"sort"
)

func sortString(strs []string) []string {
	sort.Strings(strs)
	return strs
}

func main() {
	strs := []string{"cba", "abc", "bac", "cab"}
	sortedStrs := sortString(strs)
	fmt.Println(sortedStrs)
}
