这是一个简单的Golang练习实例，用于熟悉Go语言的基本语法。

package main

import "fmt"

func main() {
	// 练习1：打印Hello, world!
	fmt.Println("Hello, world!")

	// 练习2：计算两个数的和
	num1 := 10
	num2 := 20
	sum := num1 + num2
	fmt.Printf("The sum of %d and %d is %d\n", num1, num2, sum)

	// 练习3：循环输出1-10内的偶数
	for i := 1; i <= 10; i++ {
		if i%2 == 0 {
			fmt.Println(i)
		}
	}

	// 练习4：判断某年是否是闰年
	year := 2020
	if (year%4 == 0 && year%100 != 0) || year%400 == 0 {
		fmt.Printf("%d is a leap year\n", year)
	} else {
		fmt.Printf("%d is not a leap year\n", year)
	}

	// 练习5：使用递归函数计算阶乘
	num := 5
	factorial := calcFactorial(num)
	fmt.Printf("The factorial of %d is %d\n", num, factorial)
}

func calcFactorial(n int) int {
	if n <= 1 {
		return n
	}
	return n * calcFactorial(n-1)
}
