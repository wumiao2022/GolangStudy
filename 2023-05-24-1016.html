func main() { // 1. 输出 Hello, world! fmt.Println(“Hello, world!”)

// 2. 输出 1 到 10 的数字
for i := 1; i <= 10; i++ {
    fmt.Println(i)
}

// 3. 打印当前时间
fmt.Println(time.Now())

// 4. 字符串反转
s := "Hello, world!"
r := []rune(s)
for i, j := 0, len(r)-1; i < j; i, j = i+1, j-1 {
    r[i], r[j] = r[j], r[i]
}
fmt.Println(string(r))

// 5. 判断一个数是否为质数
num := 17
isPrime := true
for i := 2; i < num; i++ {
    if num%i == 0 {
        isPrime = false
        break
    }
}
if isPrime {
    fmt.Printf("%d is a prime number\n", num)
} else {
    fmt.Printf("%d is not a prime number\n", num)
}

}
