下面是一个练习题，编写一个 Golang 程序，读取一个整数 n，然后输出 n 行星号组成的等腰直角三角形。例如读取数值 4，输出如下图形：

*
**
***
****

package main

import "fmt"

func main() {
    var n int
    fmt.Print("请输入一个整数：")
    fmt.Scan(&n)
    for i := 1; i <= n; i++ {
        for j := 1; j <= i; j++ {
            fmt.Print("*")
        }
        fmt.Println()
    }
}
