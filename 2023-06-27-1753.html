  1. 判断一个数是否为质数：
package main

import (
    "fmt"
    "math"
)

func isPrime(number int) bool {
    if number <= 1 {
        return false
    }
    for i := 2; i <= int(math.Sqrt(float64(number))); i++ {
        if number%i == 0 {
            return false
        }
    }
    return true
}

func main() {
    number := 37
    if isPrime(number) {
        fmt.Println(number, "is a prime number")
    } else {
        fmt.Println(number, "is not a prime number")
    }
}
  1. 实现一个简单的计算器，支持加减乘除四则运算：
package main

import "fmt"

func main() {
    var num1, num2 float64
    var operator string
    fmt.Print("Enter first number: ")
    fmt.Scan(&num1)
    fmt.Print("Enter second number: ")
    fmt.Scan(&num2)
    fmt.Print("Enter operator (+,-,*,/): ")
    fmt.Scan(&operator)
    result := 0.0
    switch operator {
    case "+":
        result = num1 + num2
    case "-":
        result = num1 - num2
    case "*":
        result = num1 * num2
    case "/":
        if num2 == 0 {
            fmt.Println("Error: Can't divide by zero")
            return
        }
        result = num1 / num2
    default:
        fmt.Println("Error: Invalid operator")
        return
    }
    fmt.Println(num1, operator, num2, "=", result)
}
  1. 实现一个简单的HTTP服务，返回”Hello, World!“：
package main

import (
    "fmt"
    "net/http"
)

func handler(w http.ResponseWriter, r *http.Request) {
    fmt.Fprint(w, "Hello, World!")
}

func main() {
    http.HandleFunc("/", handler)
    err := http.ListenAndServe(":8080", nil)
    if err != nil {
        fmt.Println("Error:", err)
    }
}
