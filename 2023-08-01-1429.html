package main

import (
	"fmt"
)

func main() {
	// 练习1：打印Hello, World!
	fmt.Println("Hello, World!")

	// 练习2：变量和常量
	var a int = 10
	b := 5.5
	const c string = "Golang"
	fmt.Println(a)
	fmt.Println(b)
	fmt.Println(c)

	// 练习3：基本数据类型
	var number int = 10
	var pi float64 = 3.14159
	var flag bool = true
	var str string = "Golang"
	fmt.Println(number)
	fmt.Println(pi)
	fmt.Println(flag)
	fmt.Println(str)

	// 练习4：控制流程
	for i := 1; i <= 10; i++ {
		if i%2 == 0 {
			fmt.Println(i, "is even")
		} else {
			fmt.Println(i, "is odd")
		}
	}

	// 练习5：数组和切片
	arr := [3]int{1, 2, 3}
	slice := []int{4, 5, 6}
	fmt.Println(arr)
	fmt.Println(slice)

	// 练习6：函数和方法
	result := add(5, 3)
	fmt.Println("Result:", result)

	p := Person{"Alice", 25}
	p.sayHello()

	// 练习7：并发编程
	ch := make(chan string)

	go printGreeting("Alice", ch)
	go printGreeting("Bob", ch)

	fmt.Println(<-ch)
	fmt.Println(<-ch)
}

func add(a, b int) int {
	return a + b
}

type Person struct {
	name string
	age  int
}

func (p Person) sayHello() {
	fmt.Println("Hello, my name is", p.name)
}

func printGreeting(name string, ch chan string) {
	ch <- "Hello, " + name
}
