1. 计算n的阶乘(循环实现)
func factorial(n int) int {
    result := 1
    for i := 1; i <= n; i++ {
        result *= i
    }
    return result
}
  1. 判断字符串是否为回文串(双指针实现)
func isPalindrome(s string) bool {
    i := 0
    j := len(s) - 1
    for i < j {
        if s[i] != s[j] {
            return false
        }
        i++
        j--
    }
    return true
}
  1. 删除切片中指定元素
func deleteElement(s []int, target int) []int {
    result := []int{}
    for _, val := range s {
        if val != target {
            result = append(result, val)
        }
    }
    return result
}
  1. 获取切片中的最大值和最小值
func getMaxAndMin(s []int) (int, int) {
    if len(s) == 0 {
        return 0, 0
    }
    max := s[0]
    min := s[0]
    for _, val := range s {
        if val > max {
            max = val
        }
        if val < min {
            min = val
        }
    }
    return max, min
}
  1. 查找切片中第k大的元素(快速选择算法)
func quickSelect(s []int, k int) int {
    if len(s) < k {
        return -1
    }
    pivot := s[0]
    left := []int{}
    right := []int{}
    for _, val := range s[1:] {
        if val <= pivot {
            left = append(left, val)
        } else {
            right = append(right, val)
        }
    }
    if len(left) == k-1 {
        return pivot
    } else if len(left) >= k {
        return quickSelect(left, k)
    } else {
        return quickSelect(right, k-1-len(left))
    }
}
