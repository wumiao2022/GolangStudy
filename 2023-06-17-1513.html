下面是一个简单的Golang练习实例，实现了将字符串中的每个字母往后移动n个位置的功能：

package main

import "fmt"

func main() {
	str := "abcd"
	n := 2
	fmt.Println(encrypt(str, n)) // 输出：cdef
}

func encrypt(str string, n int) string {
	result := ""
	for _, char := range str {
		if char >= 'a' && char <= 'z' {
			char = 'a' + (char-'a'+rune(n))%26
		} else if char >= 'A' && char <= 'Z' {
			char = 'A' + (char-'A'+rune(n))%26
		}
		result += string(char)
	}
	return result
}
