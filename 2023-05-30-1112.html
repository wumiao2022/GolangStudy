以下是一个Golang练习代码例子：

package main

import (
    "fmt"
    "math/rand"
    "time"
)

func main() {
    rand.Seed(time.Now().UnixNano())

    // 生成随机数
    randomNum := rand.Intn(100)
    fmt.Printf("随机数为：%d\n", randomNum)

    // 猜数游戏
    var guess int
    for {
        fmt.Print("请输入你猜的数：")
        fmt.Scanf("%d", &guess)

        if guess == randomNum {
            fmt.Println("恭喜你猜对了！")
            break
        } else if guess > randomNum {
            fmt.Println("猜大了，请重新猜！")
        } else {
            fmt.Println("猜小了，请重新猜！")
        }
    }
}

该例子演示了一个猜数游戏，程序会随机生成一个0~100之间的整数。然后用户可以通过输入猜测的数字来尝试猜中这个数字。如果猜对了，程序会输出恭喜语句并结束游戏。如果猜错了，程序会输出提示信息，并让用户重新输入猜测的数字。
