package main

import "fmt"

func main() {
    // 练习1：交换两个整数的值
    a, b := 1, 2
    a, b = b, a
    fmt.Println(a, b)

    // 练习2：判断一个年份是否为闰年
    year := 2020
    if year % 4 == 0 && year % 100 != 0 || year % 400 == 0 {
        fmt.Println(year, "是闰年")
    } else {
        fmt.Println(year, "不是闰年")
    }

    // 练习3：求数组中的最大值和最小值
    arr := []int{1, 2, 3, 4, 5}
    max, min := arr[0], arr[0]
    for _, v := range arr {
        if v > max {
            max = v
        }
        if v < min {
            min = v
        }
    }
    fmt.Println("最大值：", max)
    fmt.Println("最小值：", min)
}
