4. 实现一个简单的计算器
```go
package main

import "fmt"

func add(num1, num2 float64) float64 {
    return num1 + num2
}

func sub(num1, num2 float64) float64 {
    return num1 - num2
}

func mul(num1, num2 float64) float64 {
    return num1 * num2
}

func div(num1, num2 float64) float64 {
    return num1 / num2
}

func main() {
    var num1, num2 float64
    var op string
    fmt.Print("Enter number 1: ")
    fmt.Scan(&num1)
    fmt.Print("Enter number 2: ")
    fmt.Scan(&num2)
    fmt.Print("Enter operation (+, -, *, /): ")
    fmt.Scan(&op)
    var res float64
    switch op {
    case "+":
        res = add(num1, num2)
    case "-":
        res = sub(num1, num2)
    case "*":
        res = mul(num1, num2)
    case "/":
        res = div(num1, num2)
    default:
        fmt.Println("Invalid operation")
        return
    }
    fmt.Printf("Result of %f %s %f is %f", num1, op, num2, res)
}
