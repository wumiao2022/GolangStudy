package main

import "fmt"

func main() {
	// 练习1：输出Hello, World!
	fmt.Println("Hello, World!")

	// 练习2：求两个整数的和
	a := 3
	b := 4
	sum := a + b
	fmt.Println("Sum:", sum)

	// 练习3：判断一个数是否为偶数
	num := 7
	if num%2 == 0 {
		fmt.Println(num, "is even")
	} else {
		fmt.Println(num, "is odd")
	}

	// 练习4：遍历打印切片元素
	numbers := []int{1, 2, 3, 4, 5}
	for _, number := range numbers {
		fmt.Println(number)
	}

	// 练习5：定义一个结构体
	type Person struct {
		Name string
		Age  int
	}
	person := Person{
		Name: "Alice",
		Age:  25,
	}
	fmt.Println("Name:", person.Name)
	fmt.Println("Age:", person.Age)
}
