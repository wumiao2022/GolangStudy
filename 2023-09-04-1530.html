package main

import (
	"fmt"
)

func main() {
	// 练习1：打印1到100的所有偶数
	for i := 1; i <= 100; i++ {
		if i%2 == 0 {
			fmt.Println(i)
		}
	}

	// 练习2：计算1到100的所有奇数之和
	sum := 0
	for i := 1; i <= 100; i++ {
		if i%2 == 1 {
			sum += i
		}
	}
	fmt.Println("奇数之和:", sum)

	// 练习3：判断一个字符串是否是回文字符串（正序与倒序相同）
	str := "level"
	isPalindrome := true
	for i := 0; i < len(str)/2; i++ {
		if str[i] != str[len(str)-1-i] {
			isPalindrome = false
			break
		}
	}
	fmt.Println("是回文字符串:", isPalindrome)

	// 练习4：判断一个数是否是质数
	num := 37
	isPrime := true
	for i := 2; i <= num/2; i++ {
		if num%i == 0 {
			isPrime = false
			break
		}
	}
	fmt.Println("是质数:", isPrime)
}
