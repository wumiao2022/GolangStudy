  1. 反转字符串
package main

import (
    "fmt"
)

func reverseString(s string) string {
    runes := []rune(s)
    for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 {
        runes[i], runes[j] = runes[j], runes[i]
    }
    return string(runes)
}

func main() {
    s := "hello world"
    fmt.Println(reverseString(s))
}
  1. 计算斐波那契数列第n项
package main

import (
    "fmt"
)

func fibonacci(n int) int {
    if n == 0 {
        return 0
    } else if n == 1 {
        return 1
    } else {
        return fibonacci(n-1) + fibonacci(n-2)
    }
}

func main() {
   fmt.Println(fibonacci(10)) 
}
  1. 判断一个字符串是否是回文
package main

import (
    "fmt"
)

func isPalindrome(s string) bool {
    runes := []rune(s)
    for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 {
        if runes[i] != runes[j] {
            return false
        }
    }
    return true
}

func main() {
   s := "racecar"
   fmt.Println(isPalindrome(s)) 
}
  1. 使用二分法查找一个整数是否在一个有序数组中
package main

import (
    "fmt"
)

func binarySearch(arr []int, x int) bool {
    l := 0
    r := len(arr) - 1
    for l <= r {
        mid := l + (r-l)/2
        if arr[mid] == x {
            return true
        } else if arr[mid] < x {
            l = mid + 1
        } else {
            r = mid - 1
        }
    }
    return false
}

func main() {
   arr := []int{1, 2, 4, 7, 10, 19}
   x := 7
   fmt.Println(binarySearch(arr, x)) 
}
