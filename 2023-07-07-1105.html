package main

import (
    "fmt"
    "math/rand"
    "time"
)

func main() {
    // 练习1: 生成一个随机数序列，并按照从小到大的顺序排序
    nums := generateRandomNumbers(10) // 生成10个随机数
    fmt.Println("随机数序列：", nums)
    sort(nums)
    fmt.Println("排序后的序列：", nums)

    // 练习2: 实现一个猜数字的游戏，给出提示信息，让用户猜出正确的数值
    playGuessingGame()

    // 练习3: 实现斐波那契数列的生成函数，并输出前20个斐波那契数
    fmt.Println("斐波那契数列:")
    for i := 1; i <= 20; i++ {
        fmt.Printf("%d ", fibonacci(i))
    }
    fmt.Println()
}

func generateRandomNumbers(n int) []int {
    rand.Seed(time.Now().UnixNano())
    nums := make([]int, n)
    for i := 0; i < n; i++ {
        nums[i] = rand.Intn(100)
    }
    return nums
}

func sort(nums []int) {
    for i := 0; i < len(nums)-1; i++ {
        for j := i + 1; j < len(nums); j++ {
            if nums[i] > nums[j] {
                nums[i], nums[j] = nums[j], nums[i]
            }
        }
    }
}

func playGuessingGame() {
    rand.Seed(time.Now().UnixNano())
    number := rand.Intn(100)
    fmt.Println("猜数字游戏开始！")
    fmt.Println("请猜一个范围在0到100之间的整数：")
    for {
        var guess int
        fmt.Scan(&guess)
        if guess < number {
            fmt.Println("猜小了，请继续猜：")
        } else if guess > number {
            fmt.Println("猜大了，请继续猜：")
        } else {
            fmt.Println("恭喜你，猜对了！")
            break
        }
    }
}

func fibonacci(n int) int {
    if n <= 2 {
        return 1
    }
    return fibonacci(n-1) + fibonacci(n-2)
}
