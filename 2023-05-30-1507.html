这是一个简单的计算器程序：

package main

import "fmt"

func main() {
    var operator string
    var num1, num2 float64

    fmt.Println("请输入第一个数字")
    fmt.Scanln(&num1)

    fmt.Println("请输入运算符 (+, -, *, /)")
    fmt.Scanln(&operator)

    fmt.Println("请输入第二个数字")
    fmt.Scanln(&num2)

    var result float64

    switch operator {
    case "+":
        result = num1 + num2
    case "-":
        result = num1 - num2
    case "*":
        result = num1 * num2
    case "/":
        result = num1 / num2
    default:
        fmt.Println("无效的运算符")
        return
    }

    fmt.Printf("%.2f %s %.2f = %.2f\n", num1, operator, num2, result)
}
