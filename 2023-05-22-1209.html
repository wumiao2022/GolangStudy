以下是一个简单的“hello world”示例，可以用作Golang每日练习的起点：

package main

import "fmt"

func main() {
    fmt.Println("Hello, World!")
}

以下是一个简单的循环示例，可以用于练习Golang的循环控制语句：

package main

import "fmt"

func main() {
    // 循环1-10并打印
    for i := 1; i <= 10; i++ {
        fmt.Printf("%d ", i)
    }
    fmt.Println()

    // 使用range循环打印数组
    arr := []string{"apple", "banana", "cherry"}
    for i, v := range arr {
        fmt.Printf("arr[%d]: %s\n", i, v)
    }
}

以下是一个简单的并发示例，可以用于练习Golang并发编程：

package main

import (
    "fmt"
    "sync"
)

func main() {
    var wg sync.WaitGroup

    wg.Add(2)
    go func() {
        defer wg.Done()
        fmt.Println("This is function 1")
    }()
    go func() {
        defer wg.Done()
        fmt.Println("This is function 2")
    }()

    // 等待两个函数执行完成
    wg.Wait()
}
