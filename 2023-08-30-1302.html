package main

import (
	"fmt"
)

func main() {
	// Example 1: Printing "Hello, World!" to the console
	fmt.Println("Hello, World!")

	// Example 2: Calculating the sum of two numbers
	num1 := 10
	num2 := 5
	sum := num1 + num2
	fmt.Printf("The sum of %d and %d is %d\n", num1, num2, sum)

	// Example 3: Checking if a number is even or odd
	number := 7
	if number%2 == 0 {
		fmt.Printf("%d is even\n", number)
	} else {
		fmt.Printf("%d is odd\n", number)
	}

	// Example 4: Finding the maximum number in an array
	numbers := []int{5, 3, 8, 2, 9}
	max := numbers[0]
	for _, num := range numbers {
		if num > max {
			max = num
		}
	}
	fmt.Printf("The maximum number in the array is %d\n", max)

	// Example 5: Converting a temperature from Celsius to Fahrenheit
	celsius := 30.0
	fahrenheit := (celsius * 9 / 5) + 32
	fmt.Printf("%.2f°C is equal to %.2f°F\n", celsius, fahrenheit)
}
