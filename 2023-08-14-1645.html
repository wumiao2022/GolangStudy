package main

import "fmt"

func main() {
	// 示例 1：打印 Hello, World!
	fmt.Println("Hello, World!")

	// 示例 2：求和函数
	num1, num2 := 10, 20
	sum := add(num1, num2)
	fmt.Println("Sum:", sum)

	// 示例 3：判断奇偶数
	number := 15
	result := isEven(number)
	if result {
		fmt.Println(number, "is even")
	} else {
		fmt.Println(number, "is odd")
	}

	// 示例 4：计算斐波那契数列
	for i := 0; i < 10; i++ {
		fibNum := fibonacci(i)
		fmt.Println(fibNum)
	}
}

// 求和函数
func add(a, b int) int {
	return a + b
}

// 判断奇偶数
func isEven(num int) bool {
	if num%2 == 0 {
		return true
	}
	return false
}

// 计算斐波那契数列
func fibonacci(n int) int {
	if n <= 1 {
		return n
	}
	return fibonacci(n-1) + fibonacci(n-2)
}
