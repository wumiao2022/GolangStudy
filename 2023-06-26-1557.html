  1. 编写一个函数，将字符串进行反转，并输出反转后的字符串。
package main

import "fmt"

func reverseStr(s string) string {
    r := []rune(s)
    for i, j := 0, len(r)-1; i < j; i, j = i+1, j-1 {
        r[i], r[j] = r[j], r[i]
    }
    return string(r)
}

func main() {
    str := "Hello World"
    fmt.Printf("原字符串：%s，反转后：%s", str, reverseStr(str))
}
  1. 编写一个函数，判断给定的年份是否是闰年。
package main

import "fmt"

func isLeapYear(year int) bool {
    if year%400 == 0 || (year%4 == 0 && year%100 != 0) {
        return true
    } else {
        return false
    }
}

func main() {
    year := 2022
    if isLeapYear(year) {
        fmt.Printf("%d是闰年", year)
    } else {
        fmt.Printf("%d不是闰年", year)
    }
}
  1. 编写一个程序，判断给定的字符串是否是回文字符串。
package main

import "fmt"

func isPalindrome(s string) bool {
    for i := 0; i < len(s)/2; i++ {
        if s[i] != s[len(s)-1-i] {
            return false
        }
    }
    return true
}

func main() {
    str1 := "abcba"
    str2 := "abcdefg"
    if isPalindrome(str1) {
        fmt.Printf("%s是回文字符串", str1)
    } else {
        fmt.Printf("%s不是回文字符串", str1)
    }
    fmt.Println()
    if isPalindrome(str2) {
        fmt.Printf("%s是回文字符串", str2)
    } else {
        fmt.Printf("%s不是回文字符串", str2)
    }
}
  1. 编写一个程序，打印杨辉三角的前n行。
package main

import "fmt"

func generateTriangle(n int) [][]int {
    triangle := make([][]int, n)
    for i := 0; i < n; i++ {
        triangle[i] = make([]int, i+1)
        for j := 0; j < i+1; j++ {
            if j == 0 || j == i {
                triangle[i][j] = 1
            } else {
                triangle[i][j] = triangle[i-1][j-1] + triangle[i-1][j]
            }
        }
    }
    return triangle
}

func printTriangle(triangle [][]int) {
    for i := 0; i < len(triangle); i++ {
        for j := 0; j < len(triangle[i]); j++ {
            fmt.Printf("%d ", triangle[i][j])
        }
        fmt.Println()
    }
}

func main() {
    n := 6
    triangle := generateTriangle(n)
    printTriangle(triangle)
}
