1. 编写一个函数，可以对字符串数组进行去重操作。

    func uniqueSlice(input []string) []string {
unique := make(map[string]bool)
result := []string{}
for _, val := range input {
    if !unique[val] {
        unique[val] = true
        result = append(result, val)
    }
}
return result
}

  2. 编写一个函数，可以将一个整数切片按从小到大的顺序进行排序。

    func sortIntSlice(input []int) []int {
sort.Slice(input, func(i, j int) bool {
    return input[i] < input[j]
})
return input
}

  3. 编写一个函数，可以计算两个整数的最大公约数。

    func gcd(a, b int) int {
if a < b {
    a, b = b, a
}
for b != 0 {
    a, b = b, a%b
}
return a
}

  4. 编写一个函数，可以递归地求斐波那契数列的第n个数。

    func fibonacci(n int) int {
if n <= 1 {
    return n
}
return fibonacci(n-1) + fibonacci(n-2)
}

  5. 编写一个函数，可以判断一个整数是否为素数。

    func isPrime(n int) bool {
if n <= 1 {
    return false
}
for i := 2; i*i <= n; i++ {
    if n%i == 0 {
        return false
    }
}
return true
}
