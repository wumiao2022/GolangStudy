以下是Golang 实现常用排序算法的示例代码：

  1. 冒泡排序

    package main
import "fmt"
func bubbleSort(arr []int) {
n := len(arr)
for i := 0; i < n-1; i++ {
    for j := 0; j < n-i-1; j++ {
        if arr[j] > arr[j+1] {
            arr[j], arr[j+1] = arr[j+1], arr[j]
        }
    }
}
}
func main() {
arr := []int{5, 4, 3, 2, 1}
bubbleSort(arr)
fmt.Println(arr)
}

  2. 选择排序

    package main
import "fmt"
func selectionSort(arr []int) {
n := len(arr)
for i := 0; i < n-1; i++ {
    minIndex := i
    for j := i + 1; j < n; j++ {
        if arr[j] < arr[minIndex] {
            minIndex = j
        }
    }
    arr[i], arr[minIndex] = arr[minIndex], arr[i]
}
}
func main() {
arr := []int{5, 4, 3, 2, 1}
selectionSort(arr)
fmt.Println(arr)
}

  3. 插入排序

    package main
import "fmt"
func insertionSort(arr []int) {
n := len(arr)
for i := 1; i < n; i++ {
    key := arr[i]
    j := i - 1
    for j >= 0 && arr[j] > key {
        arr[j+1] = arr[j]
        j--
    }
    arr[j+1] = key
}
}
func main() {
arr := []int{5, 4, 3, 2, 1}
insertionSort(arr)
fmt.Println(arr)
}

  4. 快速排序

    package main
import "fmt"
func quickSort(arr []int, left, right int) {
if left < right {
    pivotIndex := partition(arr, left, right)
    quickSort(arr, left, pivotIndex-1)
    quickSort(arr, pivotIndex+1, right)
}
}
func partition(arr []int, left, right int) int {
pivot := arr[right]
i := left - 1
for j := left; j < right; j++ {
    if arr[j] < pivot {
        i++
        arr[i], arr[j] = arr[j], arr[i]
    }
}
arr[i+1], arr[right] = arr[right], arr[i+1]
return i + 1
}
func main() {
arr := []int{5, 4, 3, 2, 1}
quickSort(arr, 0, len(arr)-1)
fmt.Println(arr)
}
