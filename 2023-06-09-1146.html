func isPalindrome(x int) bool { if x < 0 { return false } if x < 10 { return true } str := strconv.Itoa(x) runes := []rune(str) for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 { if runes[i] != runes[j] { return false } } return true }

// 3. 杨辉三角 func generate(numRows int) [][]int { res := make([][]int, numRows) for i := range res { res[i] = make([]int, i+1) res[i][0], res[i][i] = 1, 1 for j := 1; j < i; j++ { res[i][j] = res[i-1][j-1] + res[i-1][j] } } return res }
