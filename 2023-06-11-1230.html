package main

import "fmt"

func main() {
	// 练习1：输出hello world
	fmt.Println("hello world")

	// 练习2：变量声明和初始化
	var num int = 20
	fmt.Println(num)

	// 练习3：常量声明和使用
	const pi float64 = 3.14
	fmt.Println(pi)

	// 练习4：条件语句if-else
	if num > 10 {
		fmt.Println("大于10")
	} else {
		fmt.Println("不大于10")
	}

	// 练习5：循环语句for
	for i := 0; i < 5; i++ {
		fmt.Println(i)
	}

	// 练习6：数组声明和使用
	var arr [5]int = [5]int{1, 2, 3, 4, 5}
	fmt.Println(arr)

	// 练习7：切片声明和使用
	var slice []int = make([]int, 3)
	slice[0] = 1
	slice[1] = 2
	slice[2] = 3
	fmt.Println(slice)

	// 练习8：map声明和使用
	var m map[string]int = make(map[string]int)
	m["apple"] = 1
	m["banana"] = 2
	m["orange"] = 3
	fmt.Println(m)

	// 练习9：函数声明和使用
	add := func(a, b int) int {
		return a + b
	}
	result := add(1, 2)
	fmt.Println(result)

	// 练习10：结构体声明和使用
	type Person struct {
		Name string
		Age  int
	}
	p := Person{Name: "Tom", Age: 18}
	fmt.Println(p)
}
