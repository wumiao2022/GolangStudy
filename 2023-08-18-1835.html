package main

import "fmt"

func main() {
	// 练习1: 输出Hello, World!
	fmt.Println("Hello, World!")

	// 练习2: 声明变量并赋值，打印变量的值
	var num int = 10
	fmt.Println(num)

	// 练习3: 声明一个切片并初始化，将切片中的元素打印出来
	numbers := []int{1, 2, 3, 4, 5}
	for _, num := range numbers {
		fmt.Println(num)
	}

	// 练习4: 声明一个结构体并初始化，打印结构体的字段值
	type Person struct {
		Name string
		Age  int
	}

	person := Person{
		Name: "John",
		Age:  30,
	}
	fmt.Println(person.Name, person.Age)

	// 练习5: 使用if语句判断一个变量的值是否大于10，如果是则打印"大于10"，否则打印"小于等于10"
	num1 := 12
	if num1 > 10 {
		fmt.Println("大于10")
	} else {
		fmt.Println("小于等于10")
	}
}
