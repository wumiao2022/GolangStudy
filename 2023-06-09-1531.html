下面是一个例子：

package main

import (
    "fmt"
)

func main() {
    // 1. 将字符串转换为字节数组
    str := "hello world"
    bytes := []byte(str)
    fmt.Println(bytes)

    // 2. 将字节数组转换为字符串
    bytes2 := []byte{104, 101, 108, 108, 111, 32, 119, 111, 114, 108, 100}
    str2 := string(bytes2)
    fmt.Println(str2)

    // 3. 使用for循环遍历切片
    nums := []int{1, 2, 3, 4, 5}
    for i := 0; i < len(nums); i++ {
        fmt.Println(nums[i])
    }

    // 4. 利用range关键字遍历切片
    for index, value := range nums {
        fmt.Printf("index: %d, value: %d\n", index, value)
    }

    // 5. 使用make创建切片
    slice := make([]int, 5)
    fmt.Println(slice)

    // 6. 从切片中删除一个元素
    slice = append(slice[:2], slice[3:]...)
    fmt.Println(slice)

    // 7. 复制一个切片
    slice2 := make([]int, len(slice))
    copy(slice2, slice)
    fmt.Println(slice2)
}
