1. 判断字符串是否为回文字符串
func isPalindrome(str string) bool {
    for i := 0; i < len(str)/2; i++ {
        if str[i] != str[len(str)-1-i] {
            return false
        }
    }
    return true
}
  1. 计算斐波那契数列第n项值
func fibonacci(n int) int {
    if n == 0 {
        return 0
    } else if n == 1 {
        return 1
    } else {
        return fibonacci(n-1) + fibonacci(n-2)
    }
}
  1. 将一个整数划分为若干个不同的正整数之和，求不同划分的个数
func partition(n int) int {
    if n <= 1 {
        return 1
    }
    count := 0
    for i := 1; i <= n; i++ {
        count += partition(n-i)
    }
    return count
}
  1. 将一个字符串中的每个单词的首字母大写
func capitalize(str string) string {
    words := strings.Split(str, " ")
    for i := 0; i < len(words); i++ {
        if words[i] != "" {
            words[i] = strings.ToUpper(string(words[i][0])) + words[i][1:]
        }
    }
    return strings.Join(words, " ")
}
  1. 判断一个整数是否为质数
func isPrime(num int) bool {
    if num < 2 {
        return false
    }
    for i := 2; i*i <= num; i++ {
        if num%i == 0 {
            return false
        }
    }
    return true
}
