  1. 将字符串逆序输出：
package main

import "fmt"

func reverseString(str string) string {
    runes := []rune(str)
    for i, j := 0, len(runes)-1; i < j; i, j = i+1, j-1 {
        runes[i], runes[j] = runes[j], runes[i]
    }
    return string(runes)
}

func main() {
    str := "hello world"
    fmt.Println(reverseString(str))
}

  1. 使用冒泡排序对整数数组进行排序：
package main

import "fmt"

func bubbleSort(arr []int) {
    n := len(arr)
    for i := 0; i < n-1; i++ {
        swapped := false
        for j := 0; j < n-i-1; j++ {
            if arr[j] > arr[j+1] {
                arr[j], arr[j+1] = arr[j+1], arr[j]
                swapped = true
            }
        }
        if !swapped {
            break
        }
    }
}

func main() {
    arr := []int{64, 25, 12, 22, 11}
    bubbleSort(arr)
    fmt.Println(arr)
}
  1. 判断一个整数是否为素数：
package main

import "fmt"

func isPrime(num int) bool {
    if num < 2 {
        return false
    }
    for i := 2; i*i <= num; i++ {
        if num%i == 0 {
            return false
        }
    }
    return true
}

func main() {
    num := 17
    if isPrime(num) {
        fmt.Printf("%d is prime", num)
    } else {
        fmt.Printf("%d is not prime", num)
    }
}
  1. 将两个字符串连接成一个字符串并去重：
package main

import "fmt"

func joinStrings(str1 string, str2 string) string {
    result := make(map[rune]bool)
    for _, ch := range str1 {
        result[ch] = true
    }
    for _, ch := range str2 {
        result[ch] = true
    }
    var res string
    for ch := range result {
        res += string(ch)
    }
    return res
}

func main() {
    str1 := "hello"
    str2 := "world"
    fmt.Println(joinStrings(str1, str2))
}
