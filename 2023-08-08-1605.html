package main

import "fmt"

func main() {
	// 实现一个计算两个整数之和的函数
	func add(x, y int) int {
		return x + y
	}

	// 实现一个计算给定整数数组元素之和的函数
	func sum(arr []int) int {
		total := 0
		for _, num := range arr {
			total += num
		}
		return total
	}

	// 实现一个计算给定字符串数组中字符串长度之和的函数
	func stringLengthSum(arr []string) int {
		total := 0
		for _, str := range arr {
			total += len(str)
		}
		return total
	}

	// 实现一个判断给定整数是否为偶数的函数
	func isEven(num int) bool {
		return num%2 == 0
	}

	// 实现一个判断给定字符串是否为回文字符串的函数
	func isPalindrome(str string) bool {
		for i := 0; i < len(str)/2; i++ {
			if str[i] != str[len(str)-1-i] {
				return false
			}
		}
		return true
	}

	// 实现一个计算斐波那契数列前n项的函数
	func fibonacci(n int) []int {
		fib := make([]int, n)
		fib[0], fib[1] = 0, 1
		for i := 2; i < n; i++ {
			fib[i] = fib[i-1] + fib[i-2]
		}
		return fib
	}

	// 实现一个打印给定整数n的所有因数的函数
	func printFactors(n int) {
		for i := 1; i <= n; i++ {
			if n%i == 0 {
				fmt.Println(i)
			}
		}
	}
}
