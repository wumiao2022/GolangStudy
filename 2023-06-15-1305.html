1. 打印出 Hello World

    package main
import "fmt"
func main() {
fmt.Println("Hello World")
}

  2. 计算两个数的和并输出结果

    package main
import "fmt"
func main() {
a := 2
b := 3
c := a + b
fmt.Println(c)
}

  3. 判断一个数是否是偶数并输出结果

    package main
import "fmt"
func main() {
num := 6
if num%2 == 0 {
	fmt.Println(num, "是偶数")
} else {
	fmt.Println(num, "不是偶数")
}
}

  4. 使用循环打印数字 1 到 10

    package main
import "fmt"
func main() {
for i := 1; i <= 10; i++ {
	fmt.Println(i)
}
}

  5. 使用切片操作，将一个数组中的元素倒序

    package main
import "fmt"
func main() {
arr := []int{1, 2, 3, 4, 5}
for i := 0; i < len(arr)/2; i++ {
	j := len(arr) - 1 - i
	arr[i], arr[j] = arr[j], arr[i]
}
fmt.Println(arr)
}
