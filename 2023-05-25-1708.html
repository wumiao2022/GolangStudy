package main

import (
    "fmt"
    "math/rand"
    "time"
)

func main() {
    rand.Seed(time.Now().UnixNano()) // 设置随机数种子

    // 生成长度为10的随机数组
    arr := make([]int, 10)
    for i := 0; i < 10; i++ {
        arr[i] = rand.Intn(100)
    }

    // 打印数组
    fmt.Println("数组:", arr)

    // 计算数组中的最大值和最小值
    var max, min int
    max, min = arr[0], arr[0]
    for i := 1; i < len(arr); i++ {
        if arr[i] > max {
            max = arr[i]
        }
        if arr[i] < min {
            min = arr[i]
        }
    }

    // 打印最大值和最小值
    fmt.Println("最大值:", max)
    fmt.Println("最小值:", min)

    // 对数组进行排序
    for i := len(arr) - 1; i > 0; i-- {
        for j := 0; j < i; j++ {
            if arr[j] > arr[j+1] {
                arr[j], arr[j+1] = arr[j+1], arr[j]
            }
        }
    }

    // 打印排序后的数组
    fmt.Println("排序后的数组:", arr)
}
