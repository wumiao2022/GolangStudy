  5. 编写一个函数，接受一个字符串作为参数并返回该字符串的反转形式：

    func reverseString(str string) string {
runes := []rune(str)
for i, j := 0, len(runes)-1; i < len(runes)/2; i, j = i+1, j-1 {
runes[i], runes[j] = runes[j], runes[i]
}
return string(runes)
}
