package main

import (
	"fmt"
	"time"
)

// 给定一个整数数组，返回数组中的两个数字使其和为指定的目标值。
func twoSum(numbers []int, target int) []int {
	for i := 0; i < len(numbers); i++ {
		for j := i + 1; j < len(numbers); j++ {
			if numbers[i]+numbers[j] == target {
				return []int{i, j}
			}
		}
	}
	return nil
}

func main() {
	numbers := []int{2, 7, 11, 15}
	target := 9
	result := twoSum(numbers, target)
	fmt.Println(result) // Output: [0 1]
}

// 反转字符串中的字符顺序。
func reverseString(s string) string {
	str := []rune(s)
	i := 0
	j := len(str) - 1
	for i < j {
		str[i], str[j] = str[j], str[i]
		i++
		j--
	}
	return string(str)
}

func main() {
	s := "hello world"
	fmt.Println(reverseString(s)) // Output: dlrow olleh
}

// 给定一个字符串，返回字符串中单词的个数。
// 单词是指仅由字母或数字组成的连续字符的序列。
func countSegments(s string) int {
	count := 0
	for i := 0; i < len(s); i++ {
		if (i == 0 || s[i-1] == ' ') && s[i] != ' ' {
			count++
		}
	}
	return count
}

func main() {
	s := "Hello, my name is John"
	fmt.Println(countSegments(s)) // Output: 5
}
