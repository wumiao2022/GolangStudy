下面是一个简单的练习代码，用来统计一个字符串中每个单词出现的次数：

package main

import (
    "fmt"
    "strings"
)

func main() {
    str := "This is a sample string. This string contains sample words."
    words := strings.Fields(str)
    wordCount := make(map[string]int)

    for _, word := range words {
        wordCount[word]++
    }

    fmt.Println("Word count for each word:")
    for word, count := range wordCount {
        fmt.Printf("%s: %d\n", word, count)
    }
}
