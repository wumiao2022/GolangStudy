以下是一道Golang练习题，实现一个函数，接收一个字符串，返回该字符串中出现出现次数最多的字母和该字母出现的次数：

func maxChar(str string) (byte, int) {
    charCount := make(map[byte]int)
    for i := 0; i < len(str); i++ {
        charCount[str[i]]++
    }
    maxChar, maxCount := byte(0), 0
    for k, v := range charCount {
        if v > maxCount {
            maxChar, maxCount = k, v
        }
    }
    return maxChar, maxCount
}
