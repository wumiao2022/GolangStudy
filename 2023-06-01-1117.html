《每日一题：在数组中找到两个数之和等于目标值》

func twoSum(nums []int, target int) []int {
    m := make(map[int]int)

    for i := 0; i < len(nums); i++ {
        complement := target - nums[i]
        if _, ok := m[complement]; ok {
            return []int{m[complement], i}
        }
        m[nums[i]] = i
    }

    return []int{-1, -1}
}
