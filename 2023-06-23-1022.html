以下是一个用Golang编写的简单HTTP服务器，打印出”Hello, World!“：

package main

import (
	"fmt"
	"net/http"
)

func main() {
	http.HandleFunc("/", func(w http.ResponseWriter, r *http.Request) {
		fmt.Fprintf(w, "Hello, World!")
	})
	http.ListenAndServe(":8080", nil)
}

以下是一个Golang中如何使用goroutine并发处理的示例：

package main

import (
	"fmt"
	"time"
)

func main() {
	for i := 0; i < 10; i++ {
		go func(i int) {
			fmt.Println("Hello from goroutine", i)
		}(i)
	}
	time.Sleep(time.Second)
}

以下是一个Golang中如何使用channel进行通信的示例：

package main

import "fmt"

func main() {
	ch := make(chan int)
	go func() {
		ch <- 10
	}()
	i := <-ch
	fmt.Println(i)
}
