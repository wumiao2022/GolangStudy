package main

import (
	"fmt"
)

func main() {
	// 练习1：声明一个整型变量x并赋值为10，打印出该变量的值

	x := 10
	fmt.Println(x)

	// 练习2：声明一个浮点型变量y并赋值为3.14，打印出该变量的值

	y := 3.14
	fmt.Println(y)

	// 练习3：声明一个字符串变量name并赋值为"John"，打印出该变量的值

	name := "John"
	fmt.Println(name)

	// 练习4：声明一个布尔型变量isTrue并赋值为true，打印出该变量的值

	isTrue := true
	fmt.Println(isTrue)

	// 练习5：声明一个整型数组a，并初始化它为包含5个元素的数组，其中的元素依次为1、2、3、4、5。打印数组中的每个元素的值

	a := [5]int{1, 2, 3, 4, 5}
	for i := 0; i < len(a); i++ {
		fmt.Println(a[i])
	}
}
