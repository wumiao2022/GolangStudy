  1. 从命令行接收一个整数参数，并输出该整数的阶乘。
package main

import (
    "fmt"
    "os"
    "strconv"
)

func main() {
    n, _ := strconv.Atoi(os.Args[1])
    result := 1
    for i := 2; i <= n; i++ {
        result *= i
    }
    fmt.Println(result)
}
  1. 将一个字符串反转输出。
package main

import "fmt"

func main() {
    str := "hello world"
    var reversed string
    for i := len(str) - 1; i >= 0; i-- {
        reversed += string(str[i])
    }
    fmt.Println(reversed)
}
  1. 冒泡排序算法。
package main

import "fmt"

func main() {
    arr := []int{4, 3, 1, 5, 2}
    for i := 0; i < len(arr)-1; i++ {
        for j := 0; j < len(arr)-i-1; j++ {
            if arr[j] > arr[j+1] {
                arr[j], arr[j+1] = arr[j+1], arr[j]
            }
        }
    }
    fmt.Println(arr)
}
  1. 判断一个字符串是否为回文字符串。
package main

import "fmt"

func main() {
    str := "racecar"
    isPalindrome := true
    for i := 0; i < len(str)/2; i++ {
        if str[i] != str[len(str)-1-i] {
            isPalindrome = false
            break
        }
    }
    fmt.Println(isPalindrome)
}
  1. 二分查找算法。
package main

import "fmt"

func binarySearch(arr []int, key int) int {
    low := 0
    high := len(arr) - 1
    for low <= high {
        mid := (low + high) / 2
        if arr[mid] == key {
            return mid
        } else if arr[mid] < key {
            low = mid + 1
        } else {
            high = mid - 1
        }
    }
    return -1
}

func main() {
    arr := []int{1, 3, 5, 7, 9}
    key := 5
    index := binarySearch(arr, key)
    fmt.Println(index)
}
