1. 实现一个简单的HTTP服务器，返回 Hello, World! 字符串。
package main

import (
    "fmt"
    "net/http"
)

func main() {
    http.HandleFunc("/", func(w http.ResponseWriter, r *http.Request) {
        fmt.Fprintf(w, "Hello, World!")
    })

    if err := http.ListenAndServe(":8080", nil); err != nil {
        panic(err)
    }
}

  1. 实现一个函数，给定一个字母 A-Z，返回该字母在字母表中的位置（1-26）。
package main

import "fmt"

func main() {
    fmt.Println(position('A'))
    fmt.Println(position('B'))
    fmt.Println(position('Z'))
}

func position(alpha byte) int {
    return int(alpha - 'A' + 1)
}
  1. 实现一个函数，将一个字符串转换为数字。
package main

import (
    "fmt"
    "strconv"
)

func main() {
    num := toInt("12345")
    fmt.Println(num)
}

func toInt(str string) int {
    num, err := strconv.Atoi(str)
    if err != nil {
        panic(err)
    }
    return num
}
  1. 实现一个简单的排序算法，例如冒泡排序。
package main

import "fmt"

func main() {
    arr := []int{3, 1, 4, 2, 5}
    bubbleSort(arr)
    fmt.Println(arr)
}

func bubbleSort(arr []int) {
    n := len(arr)
    for i := 0; i < n-1; i++ {
        for j := 0; j < n-1-i; j++ {
            if arr[j] > arr[j+1] {
                arr[j], arr[j+1] = arr[j+1], arr[j]
            }
        }
    }
}
