package main

import "fmt"

func main() {
	// 1. 打印出 "Hello, World!" 字符串
	fmt.Println("Hello, World!")

	// 2. 将两个整数相加并打印出结果
	a := 5
	b := 10
	sum := a + b
	fmt.Println("Sum:", sum)

	// 3. 使用循环打印出从 1 到 10 的所有整数
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}

	// 4. 根据用户输入的数字判断奇偶性
	var num int
	fmt.Println("请输入一个整数:")
	fmt.Scan(&num)
	if num%2 == 0 {
		fmt.Println("该数字是偶数")
	} else {
		fmt.Println("该数字是奇数")
	}

	// 5. 将一个字符串反转并打印出结果
	str := "Hello, Go!"
	reversedStr := ""
	for i := len(str) - 1; i >= 0; i-- {
		reversedStr += string(str[i])
	}
	fmt.Println("Reversed:", reversedStr)
}

注意: 这些代码只是示例练习，可能还有一些其他的解法和更好的实现方式。
