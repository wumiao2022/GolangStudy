func main() { // Practice 1: Hello World fmt.Println(“Hello, World!”)

// Practice 2: Variables
var name string = "John"
age := 25
fmt.Printf("My name is %s and I am %d years old.\n", name, age)

// Practice 3: Constants
const pi = 3.14159
fmt.Printf("The value of pi is %.2f\n", pi)

// Practice 4: Arrays
numbers := [5]int{1, 2, 3, 4, 5}
fmt.Println(numbers)

// Practice 5: Slices
fruits := []string{"apple", "banana", "orange"}
fmt.Println(fruits)

// Practice 6: Loops
for i := 1; i <= 10; i++ {
	fmt.Println(i)
}

// Practice 7: Functions
result := add(3, 5)
fmt.Println(result)

// Practice 8: Pointers
value := 10
fmt.Println("Before: ", value)
changeValue(&value)
fmt.Println("After: ", value)

// Practice 9: Structs
person := Person{Name: "Alice", Age: 30}
fmt.Println(person)

// Practice 10: Recursion
fmt.Println(factorial(5))

}

func add(a, b int) int { return a + b }

func changeValue(val *int) { *val = 20 }

type Person struct { Name string Age int }

func factorial(n int) int { if n == 0 { return 1 } return n * factorial(n-1) }
