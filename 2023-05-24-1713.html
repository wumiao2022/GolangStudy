package main

import (
    "fmt"
    "math/rand"
    "time"
)

func main() {
    rand.Seed(time.Now().UnixNano()) // 初始化随机数种子

    // 生成10个随机数，范围在1到100之间
    var nums [10]int
    for i := 0; i < len(nums); i++ {
        nums[i] = rand.Intn(100) + 1
    }

    // 打印生成的随机数
    fmt.Println("生成的随机数：", nums)

    // 冒泡排序
    for i := 0; i < len(nums)-1; i++ {
        for j := 0; j < len(nums)-1-i; j++ {
            if nums[j] > nums[j+1] {
                nums[j], nums[j+1] = nums[j+1], nums[j]
            }
        }
    }

    // 打印排序后的结果
    fmt.Println("排序后的结果：", nums)
}
