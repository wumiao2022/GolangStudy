package main

import "fmt"

func main() {
	// 练习1: 输出 "Hello, World!"
	fmt.Println("Hello, World!")

	// 练习2: 声明一个整数变量并赋值为10，然后输出该变量的值
	num := 10
	fmt.Println(num)

	// 练习3: 声明一个字符串变量并赋值为"Go语言是最好的编程语言！"，然后输出该变量的值
	str := "Go语言是最好的编程语言！"
	fmt.Println(str)

	// 练习4: 声明两个整数变量a和b，并分别赋值为10和5，然后输出它们的和、差、积、商（a除以b的结果）
	a := 10
	b := 5
	fmt.Println("和:", a+b)
	fmt.Println("差:", a-b)
	fmt.Println("积:", a*b)
	fmt.Println("商:", a/b)

	// 练习5: 声明一个布尔变量isTrue，并赋值为true，然后输出该变量的值
	isTrue := true
	fmt.Println(isTrue)
}
