// 编写一个函数，将字母中间的数字转换为中文数字，并去掉非字母、数字字符。

func NumToCN(s string) string { cn := make(map[rune]string) cn[‘0’] = “零” cn[‘1’] = “一” cn[‘2’] = “二” cn[‘3’] = “三” cn[‘4’] = “四” cn[‘5’] = “五” cn[‘6’] = “六” cn[‘7’] = “七” cn[‘8’] = “八” cn[‘9’] = “九”

var res string
for _, char := range s {
    if (char >= 'a' && char <= 'z') ||
        (char >= 'A' && char <= 'Z') ||
        (char >= '0' && char <= '9') {
        if _, ok := cn[char]; ok {
            res += cn[char]
        } else {
            res += string(char)
        }
    }
}
return res

}
