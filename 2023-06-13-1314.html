这是一个实现计算器的简单代码：

package main

import "fmt"

func main() {
	var num1, num2 float64
	var operator string

	fmt.Print("请输入第一个数：")
	fmt.Scanln(&num1)
	fmt.Print("请输入第二个数：")
	fmt.Scanln(&num2)
	fmt.Print("请输入运算符：")
	fmt.Scanln(&operator)

	switch operator {
	case "+":
		fmt.Printf("%.2f + %.2f = %.2f\n", num1, num2, num1+num2)
	case "-":
		fmt.Printf("%.2f - %.2f = %.2f\n", num1, num2, num1-num2)
	case "*":
		fmt.Printf("%.2f * %.2f = %.2f\n", num1, num2, num1*num2)
	case "/":
		fmt.Printf("%.2f / %.2f = %.2f\n", num1, num2, num1/num2)
	default:
		fmt.Println("无法识别的运算符")
	}
}
