1. 打印1到100的数字，但是如果数字是3的倍数，就输出Fizz，如果数字是5的倍数，就输出Buzz，如果同时是3和5的倍数，就输出FizzBuzz。
package main

import "fmt"

func main() {
    for i := 1; i <= 100; i++ {
        if i%3 == 0 && i%5 == 0 {
            fmt.Println("FizzBuzz")
        } else if i%3 == 0 {
            fmt.Println("Fizz")
        } else if i%5 == 0 {
            fmt.Println("Buzz")
        } else {
            fmt.Println(i)
        }
    }
}
  1. 编写一个函数，用于将一个字符串中的单词倒序排列。
package main

import (
    "fmt"
    "strings"
)

func reverseWords(str string) string {
    words := strings.Fields(str)
    for i, j := 0, len(words)-1; i < j; i, j = i+1, j-1 {
        words[i], words[j] = words[j], words[i]
    }
    return strings.Join(words, " ")
}

func main() {
    str := "hello world, this is a test"
    fmt.Println(reverseWords(str))
}
  1. 编写一个函数，用于计算斐波那契数列中第N个数字的值。
package main

import "fmt"

func fibonacci(n int) int {
    if n < 2 {
        return n
    }
    return fibonacci(n-1) + fibonacci(n-2)
}

func main() {
    n := 10
    fmt.Printf("斐波那契数列第%d个数字为%d", n, fibonacci(n))
}
