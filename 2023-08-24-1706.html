package main

import "fmt"

func main() {
	// 示例1：打印Hello, World!
	fmt.Println("Hello, World!")

	// 示例2：定义一个整型变量num并赋值为10
	num := 10
	fmt.Println(num)

	// 示例3：使用for循环打印1到10的整数
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}

	// 示例4：定义一个字符串切片并打印切片的长度和容量
	strSlice := []string{"apple", "banana", "cherry"}
	fmt.Println("Length:", len(strSlice))
	fmt.Println("Capacity:", cap(strSlice))
}
