1. 打印100到200之间的偶数：

    package main
import "fmt"
func main() {
for i := 100; i <= 200; i++ {
if i % 2 == 0 {
  fmt.Println(i)
}
}
}

  2. 获取一个字符串的长度：

    package main
import "fmt"
func main() {
str := "Hello, world!"
fmt.Println(len(str))
}

  3. 根据年龄判断是否为成年人：

    package main
import "fmt"
func main() {
age := 20
if age >= 18 {
fmt.Println("成年人")
} else {
fmt.Println("未成年人")
}
}

  4. 判断一个数是否为质数：

    package main
import "fmt"
func main() {
num := 17
isPrime := true
for i := 2; i < num; i++ {
if num % i == 0 {
  isPrime = false
  break
}
}
if isPrime {
fmt.Println(num, "是质数")
} else {
fmt.Println(num, "不是质数")
}
}

  5. 判断一个字符串是否为回文串：

    package main
import "fmt"
func main() {
str := "level"
isPalindrome := true
for i := 0; i < len(str) / 2; i++ {
if str[i] != str[len(str) - i - 1] {
  isPalindrome = false
  break
}
}
if isPalindrome {
fmt.Println(str, "是回文串")
} else {
fmt.Println(str, "不是回文串")
}
}
