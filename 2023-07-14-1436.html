package main

import (
	"fmt"
	"time"
)

func main() {
	// 练习1：打印当前时间
	printCurrentTime()

	// 练习2：获取当前时间的年、月、日
	printCurrentDate()

	// 练习3：计算两个时间的时间差
	calculateTimeDifference()

	// 练习4：使用定时器实现定时任务
	runTimerTask()
}

// 练习1：打印当前时间
func printCurrentTime() {
	currentTime := time.Now()
	fmt.Println("Current Time:", currentTime)
}

// 练习2：获取当前时间的年、月、日
func printCurrentDate() {
	currentTime := time.Now()
	year, month, day := currentTime.Date()
	fmt.Println("Current Date:")
	fmt.Printf("Year: %d\n", year)
	fmt.Printf("Month: %s\n", month.String())
	fmt.Printf("Day: %d\n", day)
}

// 练习3：计算两个时间的时间差
func calculateTimeDifference() {
	now := time.Now()
	past := time.Date(2022, time.January, 1, 0, 0, 0, 0, time.UTC)

	difference := now.Sub(past)
	fmt.Printf("Time difference: %v\n", difference)
}

// 练习4：使用定时器实现定时任务
func runTimerTask() {
	fmt.Println("Start task...")
	timer := time.NewTimer(5 * time.Second)
	<-timer.C
	fmt.Println("Task completed!")
}

这是一个包含四个练习的示例代码，具体每个练习的内容如下：

  1. 打印当前时间
  2. 获取当前时间的年、月、日
  3. 计算两个时间的时间差
  4. 使用定时器实现定时任务
