package main

import (
	"fmt"
)

func main() {
	// 练习1：水仙花数
	for i := 100; i <= 999; i++ {
		a := i / 100      // 百位数
		b := i / 10 % 10  // 十位数
		c := i % 10       // 个位数
		if a*a*a+b*b*b+c*c*c == i {
			fmt.Println(i)
		}
	}

	// 练习2：选择排序
	arr := []int{2, 1, 4, 7, 5, 8, 6, 3}
	for i := 0; i < len(arr); i++ {
		minIndex := i
		for j := i + 1; j < len(arr); j++ {
			if arr[j] < arr[minIndex] {
				minIndex = j
			}
		}
		arr[i], arr[minIndex] = arr[minIndex], arr[i]
	}
	fmt.Println(arr)
}
