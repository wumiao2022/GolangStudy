以下是一道Golang的字符串操作练习题：

package main

import (
    "fmt"
    "strings"
)

func ReverseString(s string) string {
    var output strings.Builder

    for i := len(s) - 1; i >= 0; i-- {
        fmt.Fprintf(&output, "%c", s[i])
    }

    return output.String()
}

func main() {
    s := "hello world"
    fmt.Printf("Original: %v\n", s)
    fmt.Printf("Reversed: %v\n", ReverseString(s))
}
