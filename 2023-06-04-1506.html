以下是一个简单的Golang程序，用于将输入的整数打印到控制台：

package main

import "fmt"

func main() {
    var num int
    fmt.Print("Enter an integer: ")
    fmt.Scanf("%d", &num)
    fmt.Println("You entered:", num)
}

以下是一个Golang程序，用于使用递归方式计算斐波那契数列中的第n个数字：

package main

import "fmt"

func fibonacci(n int) int {
    if n <= 1 {
        return n
    }
    return fibonacci(n-1) + fibonacci(n-2)
}

func main() {
    fmt.Println(fibonacci(10))
}

以下是一个Golang程序，用于遍历数组并打印所有元素：

package main

import "fmt"

func main() {
    nums := [5]int{1, 2, 3, 4, 5}
    for i := 0; i < len(nums); i++ {
        fmt.Println(nums[i])
    }
}

以下是一个Golang程序，用于使用for循环和switch语句打印数字1到10中的偶数：

package main

import "fmt"

func main() {
    for i := 1; i <= 10; i++ {
        switch i % 2 {
        case 0:
            fmt.Println(i)
        }
    }
}
