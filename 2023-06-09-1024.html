1. 实现一个简单的HTTP服务器，监听本地8080端口，并在访问根路径时返回”Hello, World!“的字符串。

    package main
import (
"fmt"
"net/http"
)
func main() {
http.HandleFunc("/", func(w http.ResponseWriter, r *http.Request) {
	fmt.Fprintf(w, "Hello, World!")
})
http.ListenAndServe(":8080", nil)
}

  2. 实现一个函数，接受一个整数切片，并返回切片中的最大值和最小值。如果切片为空，则返回0和0。

    package main
import (
"fmt"
)
func minMax(numbers []int) (int, int) {
if len(numbers) == 0 {
	return 0, 0
}
min := numbers[0]
max := numbers[0]
for _, num := range numbers {
	if num < min {
		min = num
	} else if num > max {
		max = num
	}
}
return min, max
}
func main() {
numbers := []int{3, 2, 5, 1, 4}
min, max := minMax(numbers)
fmt.Printf("Min: %d, Max: %d\n", min, max)
}

  3. 实现一个二叉查找树，并实现插入和查找功能。二叉查找树是一种有序树结构，在树中的每个节点上都有一个值，且每个节点的左子树中的所有值都小于该节点的值，右子树中的所有值都大于该节点的值。

    package main
import (
"fmt"
)
type Node struct {
value int
left  *Node
right *Node
}
func Insert(root *Node, value int) *Node {
if root == nil {
	return &Node{value: value}
}
if value < root.value {
	root.left = Insert(root.left, value)
} else if value > root.value {
	root.right = Insert(root.right, value)
}
return root
}
func Search(root *Node, value int) bool {
if root == nil {
	return false
}
if value == root.value {
	return true
} else if value < root.value {
	return Search(root.left, value)
} else {
	return Search(root.right, value)
}
}
func main() {
root := &Node{value: 6}
root = Insert(root, 4)
root = Insert(root, 8)
root = Insert(root, 2)
root = Insert(root, 5)
root = Insert(root, 7)
root = Insert(root, 9)
fmt.Println(Search(root, 7))
fmt.Println(Search(root, 3))
}
