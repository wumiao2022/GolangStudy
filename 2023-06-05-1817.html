// 练习1：打印九九乘法表（for循环实现）

package main

import "fmt"

func main() {
	for i := 1; i <= 9; i++ {
		for j := 1; j <= i; j++ {
			fmt.Printf("%d x %d = %d\t", j, i, i*j)
		}
		fmt.Println()
	}
}

// 练习2：判断一个数是否为质数

package main

import (
	"fmt"
	"math"
)

// 判断一个数是否为质数
func IsPrime(num int) bool {
	if num <= 1 {
		return false
	}
	if num == 2 {
		return true
	}
	if num%2 == 0 {
		return false
	}
	for i := 3; i <= int(math.Sqrt(float64(num))); i += 2 {
		if num%i == 0 {
			return false
		}
	}
	return true
}

func main() {
	fmt.Println(IsPrime(11)) // true
	fmt.Println(IsPrime(12)) // false
}

// 练习3：使用递归实现斐波那契数列

package main

import "fmt"

// 使用递归实现斐波那契数列
func Fibonacci(n int) int {
	if n <= 1 {
		return n
	}
	return Fibonacci(n-1) + Fibonacci(n-2)
}

func main() {
	fmt.Println(Fibonacci(10)) // 输出：55
}
