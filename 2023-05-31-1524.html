1. 判断一个字符串是否是回文字符串
func isPalindrome(str string) bool {
    for i := 0; i < len(str)/2; i++ {
        if str[i] != str[len(str)-i-1] {
            return false
        }
    }
    return true
}
  1. 获取一个切片中的最大值和最小值
func getMinMax(s []int) (int, int) {
    if len(s) == 0 {
        return 0, 0
    }
    min := s[0]
    max := s[0]
    for _, v := range s {
        if v < min {
            min = v
        }
        if v > max {
            max = v
        }
    }
    return min, max
}
  1. 计算一个数的阶乘
func factorial(n int) int {
    if n == 0 {
        return 1
    }
    return n * factorial(n-1)
}
  1. 找到两个切片中的公共元素
func findCommon(s1, s2 []int) []int {
    m := make(map[int]bool)
    for _, v := range s1 {
        m[v] = true
    }
    res := []int{}
    for _, v := range s2 {
        if m[v] {
            res = append(res, v)
        }
    }
    return res
}
  1. 实现冒泡排序
func bubbleSort(s []int) {
    for i := 0; i < len(s)-1; i++ {
        for j := 0; j < len(s)-i-1; j++ {
            if s[j] > s[j+1] {
                s[j], s[j+1] = s[j+1], s[j]
            }
        }
    }
}
