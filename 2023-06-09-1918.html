随机生成10个整数，并将其从小到大排序，最后输出排好序的数组。

package main

import (
    "fmt"
    "math/rand"
    "sort"
    "time"
)

func main() {
    // 设置随机数种子
    rand.Seed(time.Now().Unix())

    // 随机生成10个整数
    nums := make([]int, 10)
    for i := 0; i < 10; i++ {
        nums[i] = rand.Intn(100)
    }

    // 从小到大排序
    sort.Ints(nums)

    // 输出排好序的数组
    fmt.Println(nums)
}
