示例代码：

package main

import "fmt"

func main() {
    // 打印 Hello, World!
    fmt.Println("Hello, World!")
    
    // 声明和输出变量
    var num1 int = 10
    var str1 string = "Golang"
    fmt.Printf("%d\n", num1)
    fmt.Printf("%s\n", str1)
    
    // 数组
    var arr1 [5]int
    arr2 := [3]string{"a", "b", "c"}
    fmt.Println(arr1)
    fmt.Println(arr2)
    
    // 切片
    slice1 := []string{"x", "y", "z"}
    slice2 := make([]int, 2, 4)
    slice2 = append(slice2, 1, 2, 3)
    fmt.Println(slice1)
    fmt.Println(slice2)
    
    // map
    map1 := make(map[string]int)
    map1["a"] = 1
    map1["b"] = 2
    fmt.Println(map1)
}
