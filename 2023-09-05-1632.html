package main

import "fmt"

func main() {
	// 练习1: 打印"Hello, World!"
	fmt.Println("Hello, World!")

	// 练习2: 计算两个数的和并打印结果
	num1 := 10
	num2 := 20
	sum := num1 + num2
	fmt.Println("Sum:", sum)

	// 练习3: 编写一个函数，接收一个数组并返回该数组的最大值
	arr := []int{5, 8, 1, 9, 3}
	max := findMax(arr)
	fmt.Println("Max:", max)

	// 练习4: 编写一个递归函数，计算斐波那契数列的第 n 个数字
	n := 8
	fib := fibonacci(n)
	fmt.Println("Fibonacci:", fib)

	// 练习5: 创建一个结构体表示学生，包含姓名和年龄字段，使用指针类型接收函数修改学生的年龄并打印
	student := Student{"Alice", 20}
	modifyAge(&student, 21)
	fmt.Println("Modified Age:", student.Age)
}

// 练习3: 查找数组的最大值
func findMax(arr []int) int {
	max := arr[0]
	for _, num := range arr {
		if num > max {
			max = num
		}
	}
	return max
}

// 练习4: 计算斐波那契数列
func fibonacci(n int) int {
	if n <= 1 {
		return n
	}
	return fibonacci(n-1) + fibonacci(n-2)
}

// 练习5: 学生结构体
type Student struct {
	Name string
	Age  int
}

// 练习5: 修改学生年龄
func modifyAge(student *Student, age int) {
	student.Age = age
}
