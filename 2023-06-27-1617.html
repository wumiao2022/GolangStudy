这里给您提供一份简单的练习实例代码：

package main

import "fmt"

func main() {
    // 1. 输出10以内的奇数
    for i := 1; i < 10; i += 2 {
        fmt.Println(i)
    }

    // 2. 计算1到100的和
    sum := 0
    for i := 1; i <= 100; i++ {
        sum += i
    }
    fmt.Println(sum)

    // 3. 判断一个数是否是素数
    num := 59
    prime := true
    for i := 2; i < num; i++ {
        if num%i == 0 {
            prime = false
            break
        }
    }
    if prime {
        fmt.Println(num, "是素数")
    } else {
        fmt.Println(num, "不是素数")
    }

    // 4. 取出一个数组中的最大值
    arr := []int{1, 3, 5, 2, 4}
    max := arr[0]
    for i := 1; i < len(arr); i++ {
        if arr[i] > max {
            max = arr[i]
        }
    }
    fmt.Println(max)
}

以上代码提供了几个简单的练习例子，分别为：

  1. 输出10以内的奇数；
  2. 计算1到100的和；
  3. 判断一个数是否是素数；
  4. 取出一个数组中的最大值。
