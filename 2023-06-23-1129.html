package main

import (
	"fmt"
)

func main() {
	// 练习1：打印100以内的所有偶数
	for i := 1; i <= 100; i++ {
		if i%2 == 0 {
			fmt.Println(i)
		}
	}

	// 练习2：计算1到100的和，输出结果
	sum := 0
	for i := 1; i <= 100; i++ {
		sum += i
	}
	fmt.Println(sum)

	// 练习3：判断用户输入的数是否是质数
	var num int
	fmt.Println("请输入一个整数:")
	fmt.Scanln(&num)
	isPrime := true
	for i := 2; i < num; i++ {
		if num%i == 0 {
			isPrime = false
			break
		}
	}
	if isPrime {
		fmt.Println(num, "是质数")
	} else {
		fmt.Println(num, "不是质数")
	}
}
