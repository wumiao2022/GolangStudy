package main

import "fmt"

func main() {
    // 练习1：打印出从1到100的数字
    for i := 1; i <= 100; i++ {
        fmt.Println(i)
    }
    
    // 练习2：计算5的阶乘
    factorial := 1
    for i := 1; i <= 5; i++ {
        factorial *= i
    }
    fmt.Println("5的阶乘为:", factorial)
    
    // 练习3：计算斐波那契数列前20项
    fib := []int{0, 1}
    for i := 2; i < 20; i++ {
        fib = append(fib, fib[i-1]+fib[i-2])
    }
    fmt.Println("斐波那契数列前20项:", fib)
    
    // 练习4：判断一个数是否是素数
    num := 17
    isPrime := true
    for i := 2; i < num; i++ {
        if num%i == 0 {
            isPrime = false
            break
        }
    }
    if isPrime {
        fmt.Println(num, "是素数")
    } else {
        fmt.Println(num, "不是素数")
    }
    
    // 练习5：计算整数切片的平均值
    numbers := []int{5, 10, 15, 20, 25}
    sum := 0
    for _, num := range numbers {
        sum += num
    }
    average := float64(sum) / float64(len(numbers))
    fmt.Println("整数切片的平均值为:", average)
}
