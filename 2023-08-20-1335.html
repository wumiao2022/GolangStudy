package main

import "fmt"

func main() {
	// 实现一个函数，接收两个整数参数，返回它们的和
	func sum(a, b int) int {
		return a + b
	}

	// 实现一个函数，接收一个整数参数n，打印出从1到n的所有奇数
	func printOddNumbers(n int) {
		for i := 1; i <= n; i++ {
			if i%2 != 0 {
				fmt.Println(i)
			}
		}
	}

	// 实现一个函数，接收一个字符串参数，将其反转并返回结果
	func reverseString(s string) string {
		result := ""
		for _, char := range s {
			result = string(char) + result
		}
		return result
	}

	// 实现一个函数，接收一个整数切片参数，返回切片中的最大值和最小值
	func findMaxAndMin(numbers []int) (int, int) {
		max := numbers[0]
		min := numbers[0]
		for _, num := range numbers {
			if num > max {
				max = num
			}
			if num < min {
				min = num
			}
		}
		return max, min
	}
}
