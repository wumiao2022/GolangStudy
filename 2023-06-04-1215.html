1. 判断一个整数是否是奇数：
func isOdd(n int) bool {
    return n%2 != 0
}
  1. 取一个整数的绝对值：
func abs(n int) int {
    if n < 0 {
        return -n
    }
    return n
}
  1. 计算两个整数之和：
func add(a, b int) int {
    return a + b
}
  1. 判断一个字符串是否是回文字符串：
func isPalindrome(s string) bool {
    for i := 0; i < len(s)/2; i++ {
        if s[i] != s[len(s)-i-1] {
            return false
        }
    }
    return true
}
  1. 计算一个数的阶乘：
func factorial(n int) int {
    res := 1
    for i := 2; i <= n; i++ {
        res *= i
    }
    return res
}
  1. 计算一个字符串中的单词数量：
func countWords(s string) int {
    count := 0
    for i := 0; i < len(s); i++ {
        if (i == 0 || !unicode.IsLetter(rune(s[i-1]))) && unicode.IsLetter(rune(s[i])) {
            count++
        }
    }
    return count
}
  1. 判断一个字符串是否是数字：
func isDigit(s string) bool {
    _, err := strconv.Atoi(s)
    return err == nil
}
  1. 将一个字符串中所有 \r\n 替换为 \n：
func normalizeLineEndings(s string) string {
    return strings.Replace(strings.Replace(s, "\r\n", "\n", -1), "\r", "\n", -1)
}
  1. 打印出所有的斐波那契数列：
func fibonacci(n int) []int {
    fib := make([]int, n)
    fib[0], fib[1] = 0, 1
    for i := 2; i < n; i++ {
        fib[i] = fib[i-1] + fib[i-2]
    }
    return fib
}
  1. 打印出 1~100 中的所有质数：
func isPrime(n int) bool {
    if n < 2 {
        return false
    }
    for i := 2; i*i <= n; i++ {
        if n%i == 0 {
            return false
        }
    }
    return true
}

func primes(n int) []int {
    res := []int{}
    for i := 2; i <= n; i++ {
        if isPrime(i) {
            res = append(res, i)
        }
    }
    return res
}
