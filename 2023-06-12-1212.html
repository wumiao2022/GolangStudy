以下是一个简单的练习实例代码，用于从命令行获取两个整数并将它们相加：

package main

import (
    "fmt"
    "os"
    "strconv"
)

func main() {
    if len(os.Args) < 3 {
        fmt.Println("Please enter two numbers to add.")
        return
    }

    num1, err1 := strconv.Atoi(os.Args[1])
    num2, err2 := strconv.Atoi(os.Args[2])

    if err1 != nil || err2 != nil {
        fmt.Println("Please enter valid numbers.")
        return
    }

    result := num1 + num2

    fmt.Printf("The sum of %d and %d is %d.", num1, num2, result)
}
