1. 打印1到100的数字，但是对于3的倍数输出”Fizz”，对于5的倍数输出”Buzz”，对于既是3又是5的倍数输出”FizzBuzz”。
package main

import "fmt"

func main() {
    for i := 1; i <= 100; i++ {
        if i%3 == 0 && i%5 == 0 {
            fmt.Println("FizzBuzz")
        } else if i%3 == 0 {
            fmt.Println("Fizz")
        } else if i%5 == 0 {
            fmt.Println("Buzz")
        } else {
            fmt.Println(i)
        }
    }
}
  1. 使用递归计算n的阶乘。
package main

import "fmt"

func factorial(n int) int {
    if n == 0 {
        return 1
    }
    return n * factorial(n-1)
}

func main() {
    fmt.Println(factorial(5))
}
  1. 对一个int数组进行冒泡排序。
package main

import "fmt"

func bubbleSort(arr []int) {
    n := len(arr)
    for i := 0; i < n; i++ {
        for j := 0; j < n-i-1; j++ {
            if arr[j] > arr[j+1] {
                arr[j], arr[j+1] = arr[j+1], arr[j]
            }
        }
    }
}

func main() {
    arr := []int{64, 25, 12, 22, 11}
    bubbleSort(arr)
    fmt.Println(arr)
}
  1. 实现斐波那契数列，打印前n个数。
package main

import "fmt"

func fibonacci(n int) {
    a, b := 0, 1
    for i := 0; i < n; i++ {
        fmt.Println(a)
        a, b = b, a+b
    }
}

func main() {
    fibonacci(10)
}
  1. 实现一个函数，输入n值，生成n个长度为6的随机字符串，并将它们存储在一个字符串切片中。
package main

import (
    "fmt"
    "math/rand"
    "time"
)

func generateRandomString(n int) []string {
    const charset = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789"
    b := make([]byte, 6)
    rand.Seed(time.Now().UnixNano())

    res := make([]string, n)
    for i := 0; i < n; i++ {
        for j := range b {
            b[j] = charset[rand.Intn(len(charset))]
        }
        res[i] = string(b)
    }
    return res
}

func main() {
    strings := generateRandomString(5)
    fmt.Println(strings)
}
