  1. 编写一个程序，将输入的字符串反转输出
package main

import "fmt"

func main() {
    str := "Hello World"
    var reverse string
    for i := len(str) - 1; i >= 0; i-- {
        reverse += string(str[i])
    }
    fmt.Println(reverse)
}
  1. 编写一个程序，判断输入的字符串是否为回文字符串
package main

import (
    "fmt"
    "strings"
)

func main() {
    str := "level"
    reverse := ""
    for i := len(str) - 1; i >= 0; i-- {
        reverse += string(str[i])
    }
    if strings.EqualFold(str, reverse) {
        fmt.Println("The string is palindrome")
    } else {
        fmt.Println("The string is not palindrome")
    }
}
  1. 编写一个程序，找到输入的整数数组中最大的元素并输出
package main

import "fmt"

func main() {
    arr := []int{1, 3, 2, 4, 5}
    max := arr[0]
    for _, elem := range arr {
        if elem > max {
            max = elem
        }
    }
    fmt.Println("The maximum element is", max)
}
  1. 编写一个程序，在输入的整数数组中找到两个元素的和为指定值，并输出这两个元素的下标
package main

import "fmt"

func main() {
    arr := []int{2, 4, 7, 9, 11}
    target := 13
    for i := 0; i < len(arr)-1; i++ {
        for j := i + 1; j < len(arr); j++ {
            if arr[i]+arr[j] == target {
                fmt.Println("The two elements are", arr[i], "and", arr[j], "at indexes", i, "and", j)
            }
        }
    }
}
  1. 编写一个程序，输入两个整数并计算它们的最大公约数
package main

import "fmt"

func main() {
    num1 := 12
    num2 := 16
    var gcd int
    for i := 1; i <= num1 && i <= num2; i++ {
        if num1%i == 0 && num2%i == 0 {
            gcd = i
        }
    }
    fmt.Println("The GCD of", num1, "and", num2, "is", gcd)
}
