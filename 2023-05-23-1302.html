以下是一个简单的Golang程序，它定义了一个函数，该函数接受一个字符串，将其反转，并返回结果：

package main

import "fmt"

// ReverseString takes a string and returns its reverse
func ReverseString(str string) string {
    runes := []rune(str)
    for i, j := 0, len(runes)-1; i < len(runes)/2; i, j = i+1, j-1 {
        runes[i], runes[j] = runes[j], runes[i]
    }
    return string(runes)
}

func main() {
    s := "Hello, world!"
    fmt.Println(ReverseString(s))
}
