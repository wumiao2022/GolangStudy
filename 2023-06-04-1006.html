// 练习2：计算两个数的和并输出 package main

import “fmt”

func main() { var a, b int fmt.Scan(&a, &b) fmt.Println(a + b) }

// 练习3：计算整数数组的平均值并输出 package main

import “fmt”

func main() { var nums []int = []int{1, 2, 3, 4, 5} var sum int for _, num := range nums { sum += num } fmt.Println(sum / len(nums)) }

// 练习4：判断一个数是否为偶数并输出 package main

import “fmt”

func main() { var num int fmt.Scan(&num) if num%2 == 0 { fmt.Println(“是偶数”) } else { fmt.Println(“不是偶数”) } }

// 练习5：在命令行中输入一个字符串，输出其中的元音字母个数 package main

import ( “fmt” “strings” )

func main() { var str string fmt.Scanln(&str) str = strings.ToLower(str) var count int for _, char := range str { switch char { case ‘a’, ‘e’, ‘i’, ‘o’, ‘u’: count++ } } fmt.Println(count) }
